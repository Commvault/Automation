    @adminconsole_handler()
    def create_alert_definition(self,
                                alert_name,
                                alert_type,
                                to_recipients,
                                cc_recipients,
                                bcc_recipients,
                                value_of_x,
                                alert_entities,
                                alert_target,
                                ind_notification=None,
                                alert_locale=None,
                                mail_subject=None):
        """
        Function to create Alert Definition

        Args:
            alert_name      (str)       : Name of alert to be created

            alert_type      (str)       : Type of alert to be created

            to_recipients  (dict)       : user, user group or email to be added to alert
                                            recipients list
                Eg. to_recipients = {"Group":["master"],
                                     "Email":["jrana@commvault.com"],
                                     "User":["admin"]}

            bcc_recipients  (dict)      : user, user group or email to be added to alert
                                            recipients list
                Eg. to_recipients = {"Group":["master"],
                                     "Email":["jrana@commvault.com"],
                                     "User":["admin"]}

            cc_recipients    (dict)     : user, user group or email to be added to alert
                                            recipients list
                Eg. to_recipients = {"Group":["master"],
                                     "Email":["jrana@commvault.com"],
                                     "User":["admin"]}

            value_of_x        (integer) : value for days/time/percent based on alert selected

            alert_entities    (list)    : server groups or clients for which alert is to be
                                            defined
                Eg. alert_entities = ["Server groups", "Clients"]

            alert_target      (dict)    : dict containing values if alert targets be
                                            enabled/disabled
                Eg. alert_target = {'Email':True, 'Event viewer':True,
                                    'Console':True, 'SNMP':True}

            ind_notification  (str)    :  individual notification to be enabled/disabled
                Eg. - ind_notification = 'ON'

            alert_locale      (str)    :   alert locale to be selected
                Eg. - 'English'

            mail_subject      (str)    :   subject line for the mailer to be updated
                Eg. - 'Test Alert'

        Returns:
            None

        Raises:
            Exception:
                -- if fails to create alert definition
        """

        to_list = []
        cc_list = []
        bcc_list = []

        for group in to_recipients['Group']:
            to_list.append(group)
        for user in to_recipients['User']:
            to_list.append(user)
        for email in to_recipients['Email']:
            to_list.append(email)
        for group in cc_recipients['Group']:
            cc_list.append(group)
        for user in cc_recipients['User']:
            cc_list.append(user)
        for email in cc_recipients['Email']:
            cc_list.append(email)
        for group in bcc_recipients['Group']:
            bcc_list.append(group)
        for user in bcc_recipients['User']:
            bcc_list.append(user)
        for email in bcc_recipients['Email']:
            bcc_list.append(email)

        self.log.info("Creating alert definition with name %s", alert_name)
        self.select_hyperlink("Add alert definition")
        self.wait_for_completion()

        self.log.info("Populating Alert Name")
        self.fill_form_by_id("alertName", alert_name)

        self.log.info("Setting Alert criteria")
        self.cv_single_select('Alert type', alert_type)

        if self.check_if_entity_exists("id", "alertParam"):
            self.log.info("Populating value of X")
            self.fill_form_by_id("alertParam", value_of_x)

        self.log.info("Setting individual Notification")
        if ind_notification == 'ON':
            self.toggle_enable('Send individual notification for this alert')
        else:
            self.toggle_disable('Send individual notification for this alert')

        self.log.info("Navigating to next page")
        self.button_next()

        self.log.info("Setting Servers and Server groups")
        if self.check_if_entity_exists(
                "xpath", "//button[@class='collapsed']"):
            collapsed_elems = self.driver.find_elements_by_xpath(
                "//button[@class='collapsed']")
            for elem in collapsed_elems:
                elem.click()
            entities = self.driver.find_element_by_xpath(
                "//div[@class='browse-tree alerts-browse']")
        for value in alert_entities['server_group_list']:
            entities.find_element_by_xpath(
                "//span[./text()='"+str(value)+"']").click()
        for value in alert_entities['server_list']:
            entities.find_element_by_xpath(
                "//span[./text()='"+str(value)+"']").click()
        self.wait_for_completion()

        self.log.info("Navigating to next page")
        self.button_next()

        self.log.info("Setting alert targets and recipients")
        if alert_target:
            if alert_target['Email']:
                self.checkbox_select('alertNotifTypeEmail')

                if to_list:
                    self.log.info("Adding recipients in TO LIST")
                    for value in to_list:
                        self.driver.find_element_by_xpath(
                            "//div[@id='s2id_toUserList']/ul/li/input").send_keys(value)
                        self.wait_for_completion()
                        if self.check_if_entity_exists(
                                "xpath", "//span[@class='group-type plan-user-suggestion' and \
                            contains(text(),' ("+str(value)+")"+"')]"):
                            self.driver.find_element_by_xpath(
                                "//span[@class='group-type plan-user-suggestion' and contains\
                                (text(),' ("+str(value)+")"+"')]").click()
                        elif self.check_if_entity_exists(
                                "xpath", "//span[@class='user-type plan-user-suggestion' and \
                            contains(text(),' ("+str(value)+")"+"')]"):
                            self.driver.find_element_by_xpath(
                                "//span[@class='user-type plan-user-suggestion' and contains\
                                (text(),' ("+str(value)+")"+"')]").click()
                        elif self.check_if_entity_exists(
                                "xpath", "//span[@class='email-type plan-user-suggestion' and \
                            contains(text(),'"+str(value)+"')]"):
                            self.driver.find_element_by_xpath(
                                "//span[@class='email-type plan-user-suggestion' and contains\
                                (text(),'"+str(value)+"')]").click()
                        self.wait_for_completion()

                if cc_list:
                    self.log.info("Adding recipients in CC LIST")
                    for value in cc_list:
                        self.driver.find_element_by_xpath(
                            "//div[@id='s2id_ccUserList']/ul/li/input").send_keys(value)
                        self.wait_for_completion()
                        if self.check_if_entity_exists(
                                "xpath", "//span[@class='group-type plan-user-suggestion' and \
                            contains(text(),' ("+str(value)+")"+"')]"):
                            self.driver.find_element_by_xpath(
                                "//span[@class='group-type plan-user-suggestion' and contains\
                                (text(),' ("+str(value)+")"+"')]").click()
                        elif self.check_if_entity_exists(
                                "xpath", "//span[@class='user-type plan-user-suggestion' and \
                            contains(text(),' ("+str(value)+")"+"')]"):
                            self.driver.find_element_by_xpath(
                                "//span[@class='user-type plan-user-suggestion' and contains\
                                (text(),' ("+str(value)+")"+"')]").click()
                        elif self.check_if_entity_exists(
                                "xpath", "//span[@class='email-type plan-user-suggestion' and \
                            contains(text(),'"+str(value)+"')]"):
                            if self.check_if_entity_exists(
                                    "xpath", "//div[@id='select2-drop-mask']"):
                                self.driver.find_element_by_xpath(
                                    "//div[@id='select2-drop-mask']").click()
                            self.driver.find_element_by_xpath(
                                "//span[@class='email-type plan-user-suggestion' and contains\
                                (text(),'"+str(value)+"')]").click()
                        self.wait_for_completion()

                if bcc_list:
                    self.log.info("Adding recipients in BCC LIST")
                    for value in bcc_list:
                        self.driver.find_element_by_xpath(
                            "//div[@id='s2id_bccUserList']/ul/li/input").send_keys(value)
                        self.wait_for_completion()
                        if self.check_if_entity_exists(
                                "xpath", "//span[@class='group-type plan-user-suggestion' and \
                            contains(text(),' ("+str(value)+")"+"')]"):
                            self.driver.find_element_by_xpath(
                                "//span[@class='group-type plan-user-suggestion' and contains\
                                (text(),' ("+str(value)+")"+"')]").click()
                        elif self.check_if_entity_exists(
                                "xpath", "//span[@class='user-type plan-user-suggestion' and \
                            contains(text(),' ("+str(value)+")"+"')]"):
                            self.driver.find_element_by_xpath(
                                "//span[@class='user-type plan-user-suggestion' and contains\
                                (text(),' ("+str(value)+")"+"')]").click()
                        elif self.check_if_entity_exists(
                                "xpath", "//span[@class='email-type plan-user-suggestion' and \
                            contains(text(),'"+str(value)+"')]"):
                            if self.check_if_entity_exists(
                                    "xpath", "//div[@id='select2-drop-mask']"):
                                self.driver.find_element_by_xpath(
                                    "//div[@id='select2-drop-mask']").click()
                            self.driver.find_element_by_xpath(
                                "//span[@class='email-type plan-user-suggestion' and contains\
                                (text(),'"+str(value)+"')]").click()
                        self.wait_for_completion()
            else:
                self.checkbox_deselect('alertNotifTypeEmail')

            if alert_target['Event viewer']:
                self.checkbox_select('alertNotifTypeEV')
            else:
                self.checkbox_deselect('alertNotifTypeEV')

            if alert_target['Console']:
                self.checkbox_select('alertNotifTypeCA')
                if not alert_target['Email']:
                    if to_list:
                        self.log.info("Adding recipients in TO LIST")
                        for value in to_list:
                            self.driver.find_element_by_xpath(
                                "//div[@id='s2id_toUserList']/ul/li/input").send_keys(value)
                            self.wait_for_completion()
                            if self.check_if_entity_exists(
                                    "xpath", "//span[@class='group-type plan-user-suggestion' and \
                                contains(text(),' ("+str(value)+")"+"')]"):
                                self.driver.find_element_by_xpath(
                                    "//span[@class='group-type plan-user-suggestion' and \
                                    contains(text(),' ("+str(value)+")"+"')]").click()
                            elif self.check_if_entity_exists(
                                    "xpath", "//span[@class='user-type plan-user-suggestion' and \
                                contains(text(),' ("+str(value)+")"+"')]"):
                                self.driver.find_element_by_xpath(
                                    "//span[@class='user-type plan-user-suggestion' and \
                                    contains(text(),' ("+str(value)+")"+"')]").click()
                            elif self.check_if_entity_exists(
                                    "xpath", "//span[@class='email-type plan-user-suggestion' and \
                                contains(text(),'"+str(value)+"')]"):
                                if self.check_if_entity_exists(
                                        "xpath", "//div[@id='select2-drop-mask']"):
                                    self.driver.find_element_by_xpath(
                                        "//div[@id='select2-drop-mask']").click()
                                self.driver.find_element_by_xpath(
                                    "//span[@class='email-type plan-user-suggestion' and \
                                    contains(text(),'"+str(value)+"')]").click()
                            self.wait_for_completion()
            else:
                self.checkbox_deselect('alertNotifTypeCA')

            if alert_target['SNMP']:
                self.checkbox_select('alertNotifTypeSNMP')
            else:
                self.checkbox_deselect('alertNotifTypeSNMP')

        self.log.info("Navigating to next page")
        self.button_next()

        if alert_locale:
            self.select_value_from_dropdown("alertLocale", alert_locale)
        if mail_subject:
            self.fill_form_by_id("alertNotifEmailSub", mail_subject)

        self.submit_form()
        self.wait_for_completion()

        if self.check_if_entity_exists(
                "xpath", "//span[@class='server-message error']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='server-message error']").text
            if "doesn't have [Create Alert] permission" in exp:
                self.log.info(exp)
                self.driver.find_element_by_xpath(
                    "//button[@class='btn btn-default cvBusyOnAjax']").click()
            else:
                self.log.exception(exp)
                self.driver.find_element_by_xpath(
                    "//button[@class='btn btn-default cvBusyOnAjax']").click()
                raise Exception(exp)
        self.check_error_message()

    @adminconsole_handler()
    def delete_alert_definition(self, alert_name):
        """
        Method to delete alert definition

        Args:
           alert_name (string): Name of the alert to be deleted

        Returns:
            None

        Raises:
            Exception:
                -- if fails to delete the alert
        """
        self.log.info("Deleting Alert Definition")
        self.__table.access_action_item(alert_name, 'Delete')
        self.click_button_using_text('Yes')
        if self.check_if_entity_exists(
                "xpath", "//div[@class='growl-item alert ng-scope alert-success']"):
            self.log.info("%s has been deleted successfully.", alert_name)
        elif self.check_if_entity_exists(
                "xpath", "//div[@class='growl-item alert ng-scope alert-error alert-danger']"):
            exp = self.driver.find_element_by_xpath(
                "//div[@class='growl']/div[@class=\
                'growl-item alert ng-scope alert-error alert-danger']/div/div/div").text
            raise Exception(exp)
        else:
            exp = "Delete Alert confirmation message is not displayed"
            self.log.exception(exp)
            raise Exception(exp)
