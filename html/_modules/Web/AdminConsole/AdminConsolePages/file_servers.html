"""
This module provides the function or operations that can be performed on the File servers page in admin console

Class:
      FileServers() -> AdminPage() -> login_page() -> AdminConsoleBase() ->object()

Functions :

    backup_subclient()                      -> Method to run backup operation from file servers page
    run_check_readiness()                   -> Method to run check readiness operation from file servers page
    restore_client()                        -> Method to run restore operation from file servers page
    enable_toggle()                         -> enable toggle if it is disabled
    add_server_new_windows_or_unix_server   -> Add new unix/windows server
    retire_server                           -> retire client
    action_uninstall_software               -> uninstall all/selected packages
    action_update_software                  -> Update client
    action_add_software                     -> add packages to existing client
"""

    """class for file servers page"""

    def __init__(self, driver):
        super().__init__(driver)
        self.__table = Table(self)
        self.__drop_down = DropDown(self)

    @WebAction()
    def __get_job_id_dialog(self):
        """
        Returns jobid

        """
        if self.check_if_entity_exists("xpath", '//div[contains(text(), "Job ID:")]'):
            return self.driver.find_element_by_xpath('//div[contains(text(), "Job ID:")]').text
        raise Exception("Job was not started. Check logs")

    @WebAction()
    def __get_job_id_page(self):
        """
        Returns jobid

        """
        if self.check_if_entity_exists(
                "xpath", '//div[@data-ng-controller="pushInstallController"]//a[contains(@href,"jobs")]'):
            return self.driver.find_element_by_xpath(
                '//div[@data-ng-controller="pushInstallController"]//a[contains(@href,"jobs")]').text
        raise Exception("Job was not started. Check logs")

    @WebAction()
    def __is_toggle_enabled(self, element):
        """ Checks if toggle is enabled

        Args:
            element: WebElement corresponding to the toggle option.

        Returns (bool) : True if toggle is enabled

        """
        return 'enabled' in element.get_attribute('class')

    @WebAction()
    def __get_toggle_element(self, label):
        """ Gets WebElement corresponding to the label.

        Args:
            label   (basestring):   Label corresponding to the toggle option.

        Returns : toggle WebElement

        """
        return self.driver.find_element_by_xpath(
            "//span[contains(text(), '" + label + "')]")

    @PageService()
    def enable_toggle(self, label):
        """
        Enables the toggle bar if disabled,
        Args:
            label   (basestring):   Label corresponding to the toggle option.

        """
        element = self.__get_toggle_element(label)

        if not self.__is_toggle_enabled(element):
            element.click()
            self.wait_for_completion()

    @adminconsole_handler()
    def backup_subclient(self, client_name, backupset_name, subclient_name, backup_level):
        """
        Method to backup the given subclient

        Args:
             client_name (str)      : Name of the client to be backed up

             backupset_name (str)   : backup set name of the client

             subclient_name (str)   : subclient to be backed up

             backup_level   (str)   : type of backup

        Returns :
             None

        Raises:
            Exception :

             -- if fails to run the backup
        """

        self.__table.access_action_item(client_name, 'Backup')
        if self.check_if_entity_exists(
                'xpath',
                f'//label[text()="{backupset_name}"]/../..//label[text()="{subclient_name}"]'):
            self.driver.find_element_by_xpath(
                f'//label[text()="{backupset_name}"]/../..//label[text()="{subclient_name}"]'
                "/../preceding-sibling::span[1]//input"
            ).click()
            self.wait_for_completion()
            self.submit_form()
            self.select_radio(backup_level)
            self.driver.find_element_by_xpath('//button[@data-ng-click="startBackup()"]').click()
            self.wait_for_completion()
        else:
            raise Exception('Unable to locate the given subclient')

    @adminconsole_handler()
    def restore_client(self, client_name, backupset_name, subclient_name, restore_file):
        """
                Method to Restore the given client data

                Args:
                     client_name (str)      : Name of the client

                     backupset_name (str)   : backup set name of the client

                     subclient_name (str)   : subclient name

                     restore_file   (str)   : Data to be restored

                Returns :
                     None

                Raises:
                    Exception :

                     -- if fails to run the restore operation
                """
        self.__table.access_action_item(client_name, 'Restore')
        if self.check_if_entity_exists(
                'xpath',
                f'//label[text()="{backupset_name}"]/../..//label[text()="{subclient_name}"]'):
            self.driver.find_element_by_xpath(
                f'//label[text()="{backupset_name}"]/../..//label[text()="{subclient_name}"]'
                "/../preceding-sibling::span[1]//input"
            ).click()
        else:
            raise Exception('Unable to locate the given subclient')

        self.wait_for_completion()
        self.submit_form()
        self.__table.search_for(restore_file)
        self.driver.find_element_by_xpath("//div[@ng-if='grid.options.enableSelectAll']").click()
        self.select_hyperlink("Restore")
        self.submit_form()
        self.check_error_message()

    @PageService()
    def add_server_new_windows_or_unix_server(
            self,
            hostname=None,
            username=None,
            password=None,
            os_type='windows',
            reboot=False):
        """To create a new server

        Args:

            hostname   (list)  -- list of servers to install packages on

            username    (str)   -- username of the server machine

            password    (str)   -- password of the server machine

            os_type     (str)   -- os type of the server machine
                                    default: windows

            reboot      (bool)  -- set to True to reboot if required
                                    default: False
        """
        self.select_hyperlink(self.props['label.installSoftware'])
        self.select_hyperlink(self.props['label.windowsUnixSystem'])
        hostname = hostname.split(',')
        hostname = [f"{name.strip()}\n" for name in hostname]
        self.fill_form_by_id('hostName', hostname)
        self.driver.find_element_by_id('add-host-name').click()
        self.fill_form_by_id('fakeusernameremembered', username)
        self.fill_form_by_id('fakepasswordremembered', password)
        self.fill_form_by_id('confirmPasswordPushInstall', password)

        if 'windows' in os_type.lower():
            self.select_radio('osTypeWINDOWS')
        else:
            self.select_radio('osTypeUNIX')

        if reboot:
            self.enable_toggle(self.props['label.foreceReboot'])
        self.submit_form()
        jobid = self.__get_job_id_page()
        self.submit_form()
        return jobid

    @PageService()
    def action_add_software(
            self,
            client_name=None,
            select_all_packages=False,
            packages=None,
            reboot=False):
        """selects the Add software option for the given client

        Args:
            client_name     (str)       -- client to add software on

            select_all_packages  (bool)  -- selects all the packages if set True
                                            default: False

            packages        (list)      -- list of packages to be installed

            reboot          (bool)      -- set to True if reboot required
                                            default: False
        """
        self.__table.access_action_item(client_name, self.props['action.addSoftware'])
        if select_all_packages:
            self.__drop_down.select_drop_down_values(0, select_all=True)
        elif packages:
            self.__drop_down.select_drop_down_values(0, packages)
        else:
            raise Exception('Packages list is not provided')
        if reboot:
            self.enable_toggle(self.props['label.foreceReboot'])
        self.submit_form()
        jobid = self.__get_job_id_page()
        self.click_button(self.props['OK'])
        return jobid

    @PageService()
    def action_update_software(self, client_name=None, reboot=False):
        """selects the update software option for the given client

        Args:
            client_name     (str) -- client to update software on

            reboot      (bool)    -- set to True if reboot required
         """

        self.__table.access_action_item(client_name, self.props['label.updateSoftware'])
        if reboot:
            self.enable_toggle(self.props['label.foreceReboot'])

        self.click_button_using_text(self.props['button.yes'])

        jobid = self.__get_job_id_dialog().split(': ')[1]
        self.click_button(self.props['OK'])
        return jobid

    @PageService()
    def action_uninstall_software(self, client_name=None, packages=None):
        """
        uninstalls selected packages for a client
        Args:
            client_name     (str)       -- client to uninstall software on
            packages        (list)      -- list of packages to be uninstalled

        """
        self.__table.access_action_item(client_name, self.props['label.uninstallSoftware'])
        if packages:
            self.__drop_down.select_drop_down_values(0, packages)
            self.click_button(self.props['action.uninstall'])
        else:
            raise Exception('Packages list is not provided')
        self.click_button_using_text(self.props['button.yes'])
        return self.get_jobid_from_popup()

    @PageService()
    def retire_server(self, server_name):
        """Performs retire action for the given server

        Args:
                server_name     (str) -- server name to retire
        """
        self.__table.access_action_item(server_name, self.props['action.commonAction.retire'])
        self.click_button_using_text(self.props['button.yes'])
        self.check_error_message()
        return self.get_jobid_from_popup()
