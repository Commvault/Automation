    def __init__(self, admin_console):

        """Method to initiate Array Details Class

                        admin_console   (Object) :   Admin Console Class object"""

        self.__driver = admin_console.driver

    @PageService()
    def array_info(self):
        """
        Displays all the information about the array
        :return array_info  (dict)  --  info about the array
        """
        panel_details = PanelInfo(self)
        return panel_details.get_details()

    @PageService()
    def delete_array(self):
        """
        Deletes the array
        """

        self.__admin_console.click_button_using_text('Delete')
        self.__admin_console.wait_for_completion()
        self.__admin_console.click_button_using_text('Yes')

    @WebAction
    def edit_array(self, user_name, password):
        """
        Editing the array
        :param user_name    (str)   --  the new username of te array
        :param password     (str)   --  the new password of the array
        """
        if self.check_if_entity_exists(
                "xpath", "//span[contains(text(), 'General')]/../div/div[2]/div/a"):
            self.driver.find_element_by_xpath(
                "//span[contains(text(),'General')]/../div/div[2]/div/a").click()
            self.wait_for_completion()
            self.fill_form_by_id("userName", user_name)
            self.fill_form_by_id("password", password)
            self.driver.find_element_by_xpath(
                "//button[contains(text(), 'Save')]").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to edit the array")

    @WebAction()
    def associate_existing_server(self, servers, all_subclient=False, plan=None):
        """
        Associating a server with the array
        :param servers          (list)  --  the list of all servers to be associated
        :param all_subclient    (bool)  --  if all_subclient should be selected
        :param plan             (str)   --  name of the plan
        """
        self.log.info("Associating a server with the array")
        if self.check_if_entity_exists("xpath", "//span[contains(text(),'Associate server')]"):
            self.driver.find_element_by_xpath(
                "//span[contains(text(), 'Associate server')]").click()
            if self.check_if_entity_exists("link", "Existing server"):
                self.driver.find_element_by_link_text("Existing server").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath(
                    "//isteven-multi-select[@id='servers']/span/button").click()
                for server in servers:
                    self.driver.find_element_by_xpath(
                        "//div[@class='line-search']/button[@class='clearButton']").click()
                    self.driver.find_element_by_xpath(
                        "//div[@class='line-search']/input").send_keys(server)
                    self.driver.find_element_by_xpath(
                        "//div[@class='checkBoxContainer']//span[contains(text(),\
                        '"+server+"')]").click()
                self.driver.find_element_by_xpath("//div[@class='line-searchOk ng-scope']").click()
                if all_subclient:
                    if not self.driver.find_element_by_xpath(
                            "//input[@id='enablePlan]").is_selected():
                        self.driver.find_element_by_xpath(
                            "//input[@id='enablePlan]/following-sibling::label").click()
                        self.wait_for_completion()
                    self.driver.find_element_by_xpath(
                        "//isteven-multi-select[@id='plans']/span/button").click()
                    self.driver.find_element_by_xpath(
                        "//isteven-multi-select[@id='plans']//div[@class='line-search']/"
                        "button[@class='clearButton']").click()
                    self.driver.find_element_by_xpath(
                        "//isteven-multi-select[@id='plans']//div[@class='line-search']/"
                        "input").send_keys(plan)
                    self.driver.find_element_by_xpath(
                        "//isteven-multi-select[@id='plans']//div[@class='checkBoxContainer'"
                        "]//span[contains(text(),'"+plan+"')]").click()
                self.driver.find_element_by_xpath("//button[contains(text(), 'Save')]").click()
                self.wait_for_completion()
            else:
                raise Exception("There is no option to attach to existing server")
        else:
            raise Exception("There is no option to associate the array with a server")

    @WebAction()
    def action_disassociate_server(self, server_name):
        """
        Disassociating the server from the array
        :param server_name  (str)   --  the name of the server to be disassociated from the array
        """
        self.log.info("Disassociating the server from the array")
        while True:
            if self.check_if_entity_exists("xpath", "//a[text()='"+server_name+"']"):
                self.driver.find_element_by_xpath(
                    "//a[text()='"+server_name+"']/../../div[2]/div/a/span").click()
                if self.check_if_entity_exists(
                        "xpath", "//a[text()='"+server_name+"']/../../div[2]/div/ul//a[ \
                        contains(text(),'Disassociate')]"):
                    self.driver.find_element_by_xpath(
                        "//a[text()='"+server_name+"']/../../div[2]/div/ul//a[contains( \
                        text(),'Disassociate')]").click()
                    self.wait_for_completion()
                    self.driver.find_element_by_xpath(
                        "//button[2]").click()
                    self.wait_for_completion()
                    break
                else:
                    raise Exception("There is no option to disassociate the server")
            else:
                if self.cv_table_next_button_exists():
                    if self.driver.find_element_by_xpath(
                            "//button[@ng-disabled='cantPageForward()']").is_enabled():
                        self.cv_table_click_next_button()
                        self.wait_for_completion()
                        continue
                    else:
                        raise Exception("There is no server with the given name to "
                                        "disassociate")
                else:
                    raise Exception("There is no server with the given name to disassociate")
