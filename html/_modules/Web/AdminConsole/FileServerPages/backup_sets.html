from Web.AdminConsole.Components.table import Table

[docs]class BackupSets(AdminPage):
    """class for backup sets page"""

    def __init__(self, driver):

        super(BackupSets, self).__init__(driver)
        self.__modal_dialog = ModalDialog(self)
        self.__table = Table(self)

    @adminconsole_handler()
    def add_backup_set(self,
                       backupset_name=None,
                       client_name=None,
                       backup_set_content=None,
                       backup_plan=None,
                       storage_policy=None,
                       subclient_policy=None,
                       schedule_policies=None,
                       override_backup_content=False):
        """
        Method to create a backup set

        Args:
            backupset_name (str)            : Name of the backup set to be created

            client_name (str)               : Client name

            backup_set_content (str)        : Content for the default subclient created by this backup set

            backup_plan(str)                : Plan name to be used as policy for default subclient

            storage_policy(str)             : Storage policy name

            subclient_policy(str)           : Subclient policy name

            schedule_policies(list)         : List containing schedule policies to be used by subclient

            override_backup_content(boolean): Boolean values to determine if backup content to be overwritten when
                                                Plan is selected

        Returns:
            None

        Raises:

            Exception:

                --if failed to create a backup set

        """
        self.select_hyperlink("Add backup set")
        self.fill_form_by_id("backupSetName", backupset_name)
        self.cv_single_select("Server name", client_name)

        if backup_plan:
            self.toggle_enable("Use backup plan")
            self.cv_single_select("Plan", backup_plan)

            if override_backup_content:
                self.toggle_enable("Override backup content")
                self.select_hyperlink("Content")
                path = self.driver.find_element_by_xpath("//input[@placeholder='Enter custom path']")
                path.clear()
                path.send_keys(backup_set_content)
                self.wait_for_completion()
                self.driver.find_element_by_xpath("//i[@title='Add'][@data-ng-click='addCustomPath(0)']").click()
                self.wait_for_completion()
            else:
                self.toggle_disable("Override backup content")

        else:
            self.toggle_disable("Use backup plan")

            if subclient_policy:
                self.select_value_from_dropdown("selectBackupOptions", "Policy driven")

                self.cvselect_from_dropdown("Schedule policy", schedule_policies)
                self.cv_single_select("Subclient policy", subclient_policy)

            else:
                self.select_value_from_dropdown("selectBackupOptions", "User defined")

                self.cv_single_select("Storage policy", storage_policy)
                self.cvselect_from_dropdown("Schedule policy", schedule_policies)

                self.select_hyperlink("Content")
                path = self.driver.find_element_by_xpath("//input[@placeholder='Enter custom path']")
                path.clear()
                path.send_keys(backup_set_content)
                self.wait_for_completion()
                self.driver.find_element_by_xpath("//i[@title='Add'][@data-ng-click='addCustomPath(0)']").click()
                self.wait_for_completion()

        self.submit_form()
        self.check_error_message()

    @adminconsole_handler()
    def add_subclient(self, subclient_name, subclient_content, plan_name, backupset_name):
        """
        Method to create a subclient

        Args:
            subclient_name (str)    : Name of the subclient

            subclient_content (str) : Subclient content

            plan_name (str)         : Name of the plan

            backupset_name (str)    : backup set name

        Returns:
            None

        Raises:

            Exception:

                --if failed to create a subclient

        """
        self.__table.access_action_item(backupset_name, 'Add subclient')
        self.fill_form_by_id("subclientName", subclient_name)
        self.cv_single_select("Plan", plan_name)
        self.select_hyperlink("Content")
        path = self.driver.find_element_by_xpath("//input[@placeholder='Enter custom path']")
        path.clear()
        path.send_keys(subclient_content)
        self.wait_for_completion()
        self.driver.find_element_by_xpath("//i[@title='Add'][@data-ng-click='addCustomPath(0)']").click()
        self.wait_for_completion()
        self.click_button("Save")
        self.check_error_message()

    @PageService()
    def select_backup_set(self, client_name, backupset_name):
        """
        Method to open the details page of the given backup set

        Args:

              client_name (str)         : Client name

              backupset_name (str)      : backup set name of the client

         Returns :
              None

         Raises:
             Exception :

              -- if fails to run the backup operation

        """
        self.__table.search_for(client_name)
        self.select_hyperlink(backupset_name)
        self.check_error_message()

    @adminconsole_handler()
    def restore_client(self, backupset_name, restore_file):
        """
        Method to Restore the given client data

        Args:

             backupset_name (str)   : backup set name of the client

             restore_file   (str)   : Data to be restored

        Returns :

              job id  (int) : Job id of the restore job

        Raises:
            Exception :

             -- if fails to run the restore operation

        """
        self.__table.access_action_item(backupset_name, 'Restore')
        self.__table.search_for(restore_file)
        self.driver.find_element_by_xpath("//div[@ng-if='grid.options.enableSelectAll']").click()
        self.select_hyperlink("Restore")
        self.submit_form()
        return int(self.get_jobid_from_popup())

    @PageService()
    def delete_backup_set(self, backupset_name):
        """
         Method to delete the given backup set

        Args:

             backupset_name (str)   : backup set name of the client

        Returns :
             None

        Raises:
            Exception :

             -- if fails to run the delete operation

        """
        self.__table.access_action_item(backupset_name, 'Delete')
        self.__modal_dialog.type_text_and_delete("DELETE")
        self.check_error_message()
