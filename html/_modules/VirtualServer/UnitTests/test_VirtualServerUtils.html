[docs]    def test_get_os_flavor(self):
        os_to_tests = [
            {"Microsoft Windows server 2012": "Windows"},
            {"CentOS": "Unix"}
        ]

        for os_type in os_to_tests:
            os = list(os_type)[0]
            result = get_os_flavor(os)
            self.assertEqual(result, os_type[os])
