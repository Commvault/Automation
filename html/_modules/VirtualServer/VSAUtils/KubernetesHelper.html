[docs]    def backup(self, backup_option, type_of_content="VM", msg="", special_vm_drive=None):
        """
        Submit VSA backup job

        Args:
                backup_option           (object):   object of Backup Options class in options
                                                    helper containes all backup options

                type_of_content         (string):   type of content of a subclient

                msg                     (string):  Log line to be printed

                special_vm_drive        (dict):     Vm and drive where more data needs to be copied
        Raise Exception:
                if job does not complete successfully

        """
        try:
            VirtualServerUtils.decorative_log(msg)
            self.log.info("Starting Backup Job")
            if 'SYNTHETIC_FULL' in backup_option.backup_type:
                _backup_type = ['INCREMENTAL', 'SYNTHETIC_FULL']
            else:
                _backup_type = [backup_option.backup_type]
            for _bc_type in _backup_type:
                _backup_jobs = self.subclient.backup(_bc_type,
                                                     backup_option.run_incr_before_synth,
                                                     backup_option.incr_level,
                                                     backup_option.collect_metadata,
                                                     backup_option.advance_options)
                if not isinstance(_backup_jobs, list):
                    _backup_jobs = [_backup_jobs]
                for _backup_job in _backup_jobs:
                    self.backup_job = _backup_job
                    self.log.info("Starting backup Job : {0}".format(self.backup_job.job_id))
                    if not _backup_job.wait_for_completion():
                        pass
                        raise Exception("Failed to run backup with error: {0}"
                                        .format(_backup_job.delay_reason))
                    if "errors" in _backup_job.status:
                        raise Exception("Backup Job completed with one or more errors")
                    self.log.info("Backup Job {0} completed successfully"
                                  "Checking if Job type is Expected for job ID {0}".
                                  format(self.backup_job.job_id))
                    if _bc_type == 'SYNTHETIC_FULL':
                        _bc_type = 'Synthetic Full'
                    self.auto_commcell.check_backup_job_type_expected(
                        self.backup_job.job_id, _bc_type)

        except Exception as err:
            self.log.exception(
                "Exception while submitting Backup job:" + str(err))
            raise err



[docs]    def restore_out_of_place(self, client_name, storage_type):
        """

        perform  Kubernetes pod/ VM level restore

        Args:
                client_name             (String):   client

                storage_type            (string):  Storage type to used with restore

        Exception:
                        if job fails
                        if validation fails

        """
        browse_output = self.subclient.browse()
        source_paths = []
        for path in browse_output[0]:
            source_paths.append(path.strip("\\"))
        self.log.info("sourcepaths are {0}".format(source_paths))
        for source_path in source_paths:
            job = self.subclient.full_vm_restore_out_of_place(vm_to_restore=source_path,
                                                              restored_vm_name=source_path,
                                                              vcenter_client=client_name,
                                                              esx_host=client_name,
                                                              datastore=storage_type)
            self.log.info("Started restore out of place job with job id: %s", str(job.job_id))
            if not job.wait_for_completion():
                raise Exception(
                    "Failed to run restore out of place job with error: {0}".format(
                        job.delay_reason))

            if not job.status.lower() == "completed":
                raise Exception(
                    "Job status is not Completed, job has status: {0}".format(
                        job.status))

            self.log.info("Successfully finished restore out of place job")
            time.sleep(100)
        return job

[docs]def create_name_space(testcase, delete=False):
    """
    Function to create namespace and setup things for Automation.
    Args:
        Testcase        (obj):  Testcase object
        delete          (bool): Switch to delete existing /not.

    Raises:
            Exception:
                if it fails to create namespace

    """
    try:
        log = logger.get_log()
        cmd = "kubectl get namespace "
        output = testcase.master_machine.execute_command(cmd)
        log.info("Getting pod information by running command {} ".format(cmd))
        list_namespaces = []
        for element in output.formatted_output:
            list_namespaces.append(element[0].lower())
        for i in list_namespaces:
            if testcase.namespace in i:
                log.info("Found namespace already exists Removing it {}".format(i))
                cmd = "kubectl delete namespace  {}".format(testcase.namespace)
                output = testcase.master_machine.execute_command(cmd)
                log.info("running command {} ".format(cmd))
                if output.exit_code != 0:
                    log.exception("cmd failed to execute")
                    raise Exception
            if testcase.restore_namespace in i:
                log.info("Found namespace already exists Removing it {}".format(i))
                cmd = "kubectl delete namespace  {}".format(testcase.restore_namespace)
                output = testcase.master_machine.execute_command(cmd)
                log.info("running command {} ".format(cmd))
                if output.exit_code != 0:
                    log.exception("cmd failed to execute")
                    raise Exception
        if delete:
            return

        cmd = "kubectl create namespace {}".format(testcase.namespace)
        output = testcase.master_machine.execute_command(cmd)
        cmd = "kubectl create namespace {}".format(testcase.restore_namespace)
        output = testcase.master_machine.execute_command(cmd)
        log.info("Getting pod information by running command {} ".format(cmd))
        if output.exit_code != 0:
            log.exception("cmd failed to execute")
            raise Exception

    except Exception as exp:
        logger.get_log().error('get_pods_services_error: {} '.format(str(exp)))
        return False
