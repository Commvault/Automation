[docs]    def run(self):
        """Run function of this test case"""
        
        try:
            hyperscale_helper_object = HyperScaleHelper(self.commcell, self.csdb, self.log, self.server)
            status = hyperscale_helper_object.check_if_storage_pool_is_present(self.storagepoolname)
            if status:
                self.log.info("Storage pool by the name: {0} is present, attempting deletion".format(
                    self.storagepoolname))
                hyperscale_helper_object.clean_up_storage_pool(self.storagepoolname, 
                                                               self.sql_sq_password, self.ma1,
                                                               self.ma2, self.ma3, self.ma4,
                                                               self.ma5, self.ma6)
                time.sleep(30)
            else:
                self.log.info(
                    "Storage pool by the name: {0} is not present".format(self.storagepoolname))
            # Create a fresh storage pool
            self.log.info("Creating Storage Pool named {0}".format(self.storagepoolname))
            status = hyperscale_helper_object.create_storage_pool(self.storagepoolname, self.ma1, self.ma2, self.ma3)
            self.log.info(status)
            attempts = 5

            while not status and attempts != 0:
                status = hyperscale_helper_object.check_if_storage_pool_is_present(self.storagepoolname)
                if not status:
                    self.log.info("Storage pool not present, waiting for entry to be added")
                time.sleep(30)
                attempts = attempts - 1
            if not status:
                raise Exception("Storage Pool creation failed")
            else:
                self.log.info("Storage Pool creation Successful")
                
            rows = hyperscale_helper_object.get_associated_mas(self.storagepoolname)
            self.log.info("Number of nodes in the current storage pool = {0}".format(str(len(rows))))
            oldcount = len(rows)
            self.log.info("Current storage pool contains the following nodes: ")
            
            for row in rows:
                self.log.info(row[0])
            self.log.info("Additional time for the MAs to populate the CSDB with brick health and blcok device health "
                          "info ")

            time.sleep(30)
            hyperscale_helper_object.check_brick_flag_status(self.ma1)
            hyperscale_helper_object.check_brick_flag_status(self.ma2)
            hyperscale_helper_object.check_brick_flag_status(self.ma3)
            self.log.info("Horizontally expanding setup")
            status = hyperscale_helper_object.add_nodes(self.storagepoolname, self.ma4, self.ma5, self.ma6)
            self.log.info(status)

            if status:
                self.log.info("Storage Pool expansion Successful")
            else:
                raise Exception("Storage Expansion failed")

            rows = hyperscale_helper_object.get_associated_mas(self.storagepoolname)
            self.log.info("Number of nodes in the expanded storage pool = {0}".format(str(len(rows))))
            newcount = len(rows)
            self.log.info("Current storage pool contains the following nodes: ")
            
            for row in rows:
                self.log.info(row[0])
            added_nodes = newcount - oldcount
            
            if added_nodes == 3:
                self.log.info("Storage pool has been successfully expanded by 3 nodes")
            else:
                raise Exception("Storage pool expansion failed with incorrect number of nodes")
            time.sleep(60)
            hyperscale_helper_object.check_brick_flag_status(self.ma1)
            time.sleep(10)
            hyperscale_helper_object.check_brick_flag_status(self.ma2)
            time.sleep(10)
            hyperscale_helper_object.check_brick_flag_status(self.ma3)
            time.sleep(10)
            hyperscale_helper_object.check_brick_flag_status(self.ma4)
            time.sleep(10)
            hyperscale_helper_object.check_brick_flag_status(self.ma5)
            time.sleep(10)
            hyperscale_helper_object.check_brick_flag_status(self.ma6)
        except Exception as exp:
            self.result_string = str(exp)
            self.status = constants.FAILED
            self.log.exception("Exception message while executing test case: %s",
                               self.result_string)
