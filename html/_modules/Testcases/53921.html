    def _run_backup_job(self, subclient_id, status_list, next_interval, scan_type, job_id, clbackup_log, zero, six):
        """ Add incremental data and wait for backup completed successfully

        Args:

            subclient_id (int)       -- Subclient id

            status_list (list)       -- Run status list of the client

            next_interval (int)      -- min_interval_minutes of the automatic schedule

            scan_type (str)          -- scan type used in cloud laptop --> always recursive

            job_id(int)              -- Current job id

            clbackup_log(str)        -- Clbackup log name

            zero(str)                -- '0' status constant value

            six(str)                 -- '6' status constant value

        """

        while(True):
            # read the registry to check the status
            status_value = self.cloud_object.get_reg_key(subclient_id, "LaptopCache", "RunStatus")
            if status_value in status_list:  # Status: 1,2,3,4,5 -job is running
                self._log.info("Backup is running on client.Waiting for {0} seconds as per automatic job to trigger"
                               .format(next_interval))
                self._utility.sleep_time(next_interval)

            elif status_value == zero:  # Status: 0 -waiting for next attempt
                self.cloud_object.rename_remove_logfile(self.machine, clbackup_log)
                self.path = self.tcinputs["TestPath"] + self.slash_format + "incr_bsynth"
                self.helper.add_new_data_incr(
                    self.path, str(self.slash_format), scan_type, hlinks=False, slinks=False, sparse=False, file_size=2
                )
                self.cloud_object.run_backup(clbackup_log, cloudlaptop_constants.LOGSTRING, job_id, subclient_id)
                self._log.info("Incr backup job {0} completed successfully".format(job_id))
                backupexecuted = True
                break
            elif status_value == six:   # Status: 6 - job failed
                raise Exception("Last backup job {0} failed on client {1}, Please check the logs"
                                .format(job_id, self.client.client_name))
            else:
                raise Exception("Unknown phase status on client {0}" .format(self.client.client_name))
        if not backupexecuted:
            raise Exception("previous backup stuck on client{0}, please check logs"
                            .format(self.client.client_name))


[docs]    def run(self):
        """Main function for test case execution"""
        try:
            self._log.info("Started executing {0} testcase".format(self.id))
            self._utility = OptionsSelector(self._commcell)
            self.serverbase = CommonUtils(self.commcell)
            self.machine = machine.Machine(self.client, self.commcell)
            subclient_id = self.subclient.subclient_id
            self.server_obj = serverhelper.ServerTestCases(self)
            subclient_obj = self.subclient
            self.cloud_object = cloudlaptophelper.CloudLaptopHelper(self)
            logfile_to_check = ["clBackup.log"]
            _log = os.path.join(self.client.log_directory, logfile_to_check[0])
            clbackup_log = self.machine.get_unc_path(_log)  # get the uncpath of clbackup.log to read
            job_id = self.cloud_object.get_reg_key(subclient_id, "LaptopCache", "JobID")
            status_list = cloudlaptop_constants.STATUS
            zero = cloudlaptop_constants.RUNSTATUSZERO
            six = cloudlaptop_constants.RUNSTATUSSIX
            tmp_path = None
            # Initialize test case inputs
            compare_source = self.tcinputs["TestPath"]
            FSHelper.populate_tc_inputs(self, False)
            scan_type = ScanType.RECURSIVE

            test_path = self.test_path
            if test_path.endswith(str(self.slash_format)):
                test_path = str(test_path).rstrip(str(self.slash_format))

            next_interval = self.cloud_object.get_automatic_schedule_wait()

        # -------------------------------------------------------------------------------------
            self.cloud_object.server_obj.log_step("""
              1. Add new data for the incremental verify data is added
              2. Verify the incremental backup triggered as per automatic schedule interval
                        and completes without failures.
              3. Start new job from the registry and wait for new job to start
                        so that previous incremental job can be committed and able to see from gui
              4. Run Synthfull for the subclient and verify it completes without any failures.
              5. Run a restore of the latest data and verify correct data is restored
            """, 200)
        # -------------------------------------------------------------------------------------
            # run incr job
            self._run_backup_job(subclient_id, status_list, next_interval, scan_type, job_id, clbackup_log, zero, six)
            # Commit the existing job and start new new job
            self.cloud_object.update_reg_key(subclient_id, "LaptopCache", "StartNewJob")
            # wait for next incremental job to trigger
            self._utility.sleep_time(next_interval, "Waiting for new job to trigger")

            self._log.info("Run a Synthetic Full backup and verify it completes without failures.")
            # Run Synth full job
            self.helper.run_backup_verify(
                scan_type, backup_level="Synthetic_full", incremental_backup=False, incremental_level='AFTER_SYNTH'
            )
            self._log.info("Run a restore of the latest data and verify correct data is restored.")

            if tmp_path is None:
                tmp_path = self._utility.create_directory(self.machine)

            data_path_leaf = ntpath.basename(str(compare_source))

            self.dest_path = self.machine.os_sep.join([tmp_path, data_path_leaf + "_restore"])
            if len(subclient_obj.content) is 0:
                _paths = [each['subclientPolicyPath'] for each in subclient_obj._content]
            else:
                _paths = subclient_obj.content

            # out-of-Place restore
            self.serverbase.subclient_restore_out_of_place(
                self.dest_path, _paths, client=self.client.client_name, subclient=subclient_obj, wait=True
            )
            self._log.info("Subclient Restore Job completed successfully")
            compare_destination = self.machine.os_sep.join([self.dest_path, data_path_leaf])
            # validation of restore data
            self.cloud_object.validate_restore(compare_source, compare_destination)

            self._log.info("***TEST CASE COMPLETED SUCCESSFULLY AND PASSED***")

        except Exception as exp:
            self.server_obj.fail(exp)

        finally:
            self.machine.remove_directory(self.path)
            if self.machine.check_directory_exists(self.dest_path):
                self._utility.remove_directory(self.machine, self.dest_path)
