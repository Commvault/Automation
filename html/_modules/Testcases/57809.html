[docs]    def init_tc(self):
        """Initial configuration for the test case"""
        try:
            self.browser = BrowserFactory().create_browser_object()
            self.browser.open()
            self.admin_console = AdminConsole(self.browser, self.commcell.webconsole_hostname)
            self.admin_console.login()

            self.activate = GovernanceApps(self.admin_console)
            self.case_manager = CaseManager(self.admin_console)
            self.jobs = Jobs(self.browser.driver)
            self.table = Table(self.admin_console)
            self.ex_app_helper = ExchangeAppHelper(self.admin_console)

            self.navigator = self.admin_console.navigator
            self.navigator.navigate_to_governance_apps()
            self.activate.select_case_manager()

            self.do_not_collect_case = (self.tcinputs['CaseName'] +
                                        'DoNotCollect' + str(int(time.time())))
            self.one_time_only_case = (self.tcinputs['CaseName'] +
                                       'OneTimeOnly' + str(int(time.time())))
            self.continuous_case = (self.tcinputs['CaseName'] +
                                    'Continuous' + str(int(time.time())))
        except Exception as exception:
            raise CVTestCaseInitFailure(exception) from exception


    @test_step
    def create_do_not_collect_client(self):
        """Create a case with data collection as 'Do not collect'"""
        try:
            self.case_manager.select_add_case()
            self.case_manager.enter_case_details(
                self.do_not_collect_case,
                self.tcinputs['DataType'],
                'Do not collect',
                self.tcinputs['DNC_Custodians'],
                self.tcinputs['DCPlan'],
                self.tcinputs['ServerPlan'],
                self.tcinputs['Keyword']
            )
            self.log.info('Case Added')
        except Exception:
            raise CVTestStepFailure("Error creating case")

    @test_step
    def trigger_collection_job(self):
        """Verifying whether the Case Manager Index Copy Job has been submitted"""
        try:
            self.case_manager.submit_collection_job()
            self.job_id = str(self.case_manager.get_index_copy_job_id())

            self.jobs.job_completion(self.job_id)
        except:
            raise CVTestStepFailure(f"Error Verifying whether job has been submitted")


[docs]    def get_email_num(self, case_name):
        """Get the number of emails"""
        emails_num = 0
        self.admin_console.logout()
        self.admin_console.login()

        self.navigator.navigate_to_governance_apps()
        self.activate.select_case_manager()

        if self.table.is_entity_present_in_column('Name', case_name):
            self.case_manager.select_case(case_name)
            self.case_manager.open_search_tab()
            try:
                emails_num = int(self.table.get_total_rows_count())
            except IndexError:
                emails_num = 0
        return emails_num


    @test_step
    def verify_do_not_collect_case(self):
        """Verify that the case has not collected any emails"""
        try:
            self.emails_num = self.get_email_num(self.do_not_collect_case)
            self.log.info('Identified number of emails as %s', self.emails_num)

            if self.emails_num != 0:
                raise CVTestStepFailure('Data collected for case with datatype "Do not collect"')
        except Exception:
            raise CVTestStepFailure("Error getting the number of emails")

    @test_step
    def edit_one_time_only_definition(self):
        """Edit the case definition to add one more custodian"""
        try:
            self.case_manager.open_overview_tab()
            self.case_manager.add_custodian_to_definition(
                self.one_time_only_case + '-definition',
                self.tcinputs['OTO_Add_Custodians']

    @test_step
    def verify_one_time_only_case(self):
        """Verify that data is collected only once"""
        try:
            if self.emails_num != self.get_email_num(self.one_time_only_case):
                raise CVTestStepFailure('Data is collected more than once for "One time only"')
        except Exception:
            raise CVTestStepFailure('Error verifying that data is collected only once')

[docs]    def run(self):
        try:
            self.init_tc()
            self.create_do_not_collect_client()
            self.trigger_collection_job()
            self.verify_do_not_collect_case()
            self.create_one_time_only_client()
            self.trigger_collection_job()
            self.get_one_time_only_emails()
            self.edit_one_time_only_definition()
            self.trigger_collection_job()
            self.verify_one_time_only_case()
            self.create_continuous_client()
            self.verify_continuous_case()
        except Exception as err:
            handle_testcase_exception(self, err)
        finally:
            AdminConsole.logout_silently(self.admin_console)
            Browser.close_silently(self.browser)
