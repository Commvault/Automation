[docs]    def get_reg_key(self, subclient_id, reg_key, flag):
        """ Check/Gets the registry key value for given registry key on a client

        Args:

            subclient_id  (number)     --  Sub Client id

            reg_key    (str)           --  Registry key path. e.g -> 'LaptopCache'

            flag       (str)           --  Reg key name to get the value. e.g -> 'RunStatus'

        Returns:
            (str/int)                  --  return value of registry.

        Raises:
            Exception:

                - if module failed to execute due to some error

        """
        try:
            # read the registry to check the status
            path = self.machin.join_path(reg_key, subclient_id)

            self.log.info("Getting {0} reg key value on [{1}]" .format(flag, self.machin))

            if self.machin.check_registry_exists(path, flag):
                value = self.machin.get_registry_value(path, flag)
                self.log.info("{0} value: [{1}]".format(flag, value))
                return value
            else:
                raise Exception("{0} does not exist in registry".format(flag))

        except Exception as excp:
            raise Exception(
                "\n {0} validation failed {1} {2}".format(flag, inspect.stack()[0][3], str(excp))
            )

[docs]    def validate_restore(self, compare_source, compare_destination):
        """
        Method to validate restore jobs with meta data and checksum information

           Args:

                compare_source          (str)    --   source path that data needs to be
                                                      validated on restore completion

                compare_destination     (str)    --   destination path from data needs to be
                                                      validated on restore completion

            Raises:
                Exception:

                    if Meta data comparison failed

                    if Checksum comparison failed

        """
        machine_obj = self.utility.get_machine_object(self.client.client_name)

        self.log.info("Client: [{0}], Source path [{1}]".format(self.client, compare_source))
        self.log.info("Client: [{0}], Destination path [{1}]".format(self.client, compare_destination))
        self.log.info("Comparing meta data for source and destination paths")
        result, diff_output = machine_obj.compare_meta_data(compare_source, compare_destination)
        self.log.info("Performing meta data comparison on source and destination")
        if not result:
            self.log.error("Meta data comparison failed")
            self.log.error("Diff output: \n{0}".format(diff_output))
            raise Exception("Meta data comparison failed")
        self.log.info("Meta data comparison successful")

        self.log.info("Performing checksum comparison on source and destination")
        result, diff_output = machine_obj.compare_checksum(compare_source, compare_destination)
        if not result:
            self.log.error("Checksum comparison failed")
            self.log.error("Diff output: \n{0}".format(diff_output))
            raise Exception("Checksum comparison failed")
        self.log.info("Checksum comparison successful")



[docs]    def validate_logs(self, client_machine, validatelog, success):
        """
        Method to validate log with required log line

           Args:

                client_machine   (str)   -- Client name where Log file exists

                validatelog     (str)    -- Log file name which need to be validated

                success         (str)    -- Log line which we want to verify

            Raises:
                Exception:

                    if backup does not start within time

        """
        try:
            _wait = self.get_automatic_schedule_wait()
            for _count in range(self.retry_count):
                if client_machine.check_file_exists(validatelog):
                    with open(validatelog, 'r', encoding='utf-8') as file_obj:
                        self.utility.sleep_time(self.wait_secs, "Waiting for backup log to be completed")
                        for line in file_obj:
                            if success in line:
                                return True
                        return False
                self.utility.sleep_time(_wait, "Waiting for backup log to be generated")
            raise Exception("Backup phase did not start, Please check logs")

        except Exception as excp:
            raise Exception("\n [{0}] [{1}]".format(inspect.stack()[0][3], str(excp)))

        """ Add testdata and check for backup triggered and completed successfully

        Args:

            clbackup_log  (str)         -- Clbackup log to be validated after backup

            logstring   (str)         --  Logstring which we want to verify in log.

            job_id (str)              -- Current running job id

            input_path(str)           -- input path for subclient content

            subclient_id (id)         -- subclient id of the client

        Raises:
            Exception:

                - if unable to find required Logstring

        """

        try:

            # step4: perform CLbackup log file validation
            value = self.validate_logs(self.machin, clbackup_log, logstring)
            if value:
                self.log.info("job {0} completed successfully on client {1}"
                              .format(job_id, self.client.client_name))
            else:
                raise Exception("job {0} failed on client: {1}, Please check the logs"
                                .format(job_id, self.client.client_name))
            full_jobid = self.get_reg_key(subclient_id, "LaptopCache", "LastFullJobID")
            # validate backup is running as incremental
            assert job_id != full_jobid, "Expected Incremental job but converted to full job"
            self.log.info("Current running backup job is incremental as expected")
            self.log.info("Backup testcase validation completed successfully on client: {0}"
                          .format(self.client.client_name))
        except Exception as excp:
            raise Exception("\n [{0}] [{1}]".format(inspect.stack()[0][3], str(excp)))



[docs]    def get_automatic_schedule_wait(self):
        """
        Method to get automatic schedule's minimum backupinterval minutes

            Raises:
                Exception:

                    if module failed to execute due to some error

        """
        try:
            schedule_name = list(self.schedule_obj.schedules.keys())[0]
            schedule_obj = Schedule(self._testcase.subclient, schedule_name)
            min_backup_interval = schedule_obj.automatic['min_interval_minutes']
            interval = min_backup_interval*60
            return interval

        except Exception as excp:
            raise Exception("\n [{0}] [{1}]".format(inspect.stack()[0][3], str(excp)))

        """ removes and renames old logfile

        Args:

            Client_machine (object)     -- Machine object

            validatelog (string)        -- Log file to rename / remove.

        """

        if client_machine.check_file_exists(validatelog.replace(".log", "_1.log")):
            client_machine.remove_directory(validatelog.replace(".log", "_1.log"))
        if client_machine.check_file_exists(validatelog):
            client_machine.rename_file_or_folder(
                validatelog, validatelog.replace(".log", "_1.log"))
