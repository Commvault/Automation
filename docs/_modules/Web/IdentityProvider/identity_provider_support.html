[docs]    def login_to_adfs(self, saml_user_name, saml_password, adfs_app_name=None):
        """
        Logs in to the ADFS site in a new tab

        Args:

            saml_user_name  (str)-- user name to log in

            saml_password   (str)-- password of the user

            adfs_app_name   (str)-- the  AD FS app name to sign in

       Returns:
            True    --  If the login is successful

            False   --  If the login is not successful

            loaded_url  (str)   -- URL loaded in the browser

        """
        if "/adfs/ls/?SAMLRequest=" not in self.driver.current_url:
            if adfs_app_name:
                self.driver.find_element_by_id("idp_OtherRpRadioButton").click()
                select = Select(self.driver.find_element_by_id('idp_RelyingPartyDropDownList'))
                select.select_by_visible_text(adfs_app_name)
                self.driver.find_element_by_name('SignInSubmit').click()
                self.wait_for_completion()
        self.log.info("Page loaded %s", self.driver.current_url)
        self.driver.find_element_by_id('userNameInput').send_keys(saml_user_name)
        self.driver.find_element_by_id('passwordInput').send_keys(saml_password)
        self.driver.find_element_by_id('submitButton').click()
        self.wait_for_completion()
        if self.check_if_entity_is_displayed('id', 'errorText'):
            error_msg = self.driver.find_element_by_id('errorText').text
            raise Exception("Error::"+error_msg)
        loaded_url = self.driver.current_url
        self.log.info("Page loaded %s", loaded_url)
        return True, loaded_url



[docs]    def logout_from_adfs(self, ad_name, local_sign_out=False):
        """
        Logs out from the ADFS site

        Args:
            ad_name         (str)-- AD machine name IP that resolved

            local_sign_out  (bool)--True when we need to sign out only from  AD site

        Returns:
            True    --  If the logout is successful

            False   --  If the logout is not successful
        """
        idp_url = "https://" + ad_name + "/adfs/ls/idpinitiatedsignon.aspx"
        if self.driver.current_url != idp_url:
            self.open_url_in_new_tab(idp_url)

        if self.check_logged_in_state_adfs():
            if local_sign_out:
                self.log.info("Logging out from this site")
                self.driver.find_element_by_id("idp_LocalSignOutRadioButton").click()
            else:
                self.log.info("Logging out from all the sites")
                self.driver.find_element_by_id("idp_SingleSignOutRadioButton").click()
            self.driver.find_element_by_id('idp_SignOutButton').click()
            self.wait_for_completion()
            if self.check_logged_in_state_adfs():
                return False
        else:
            self.log.info("No user is logged in to the site")
        return True



[docs]    def open_url_in_new_tab(self, url):
        """
        Opens the given url in a new tab and change focus on the new tab opened

        Args:
            url     (str)--the url to be opened

        Returns:None
        """
        self.driver.execute_script("window.open('" + url + "');")
        window_list = self.driver.window_handles
        for window_id in window_list:
            self.driver.switch_to.window(window_id)
            if self.driver.current_url == url:
                self.log.info("Loaded the new tab")
                self.wait_for_completion()
                return

[docs]    def identity_provider_initiated_login(self, ad_name, web_console_url, adfs_app_name, user,
                                          password, verify_sso):
        """
        Initiates a SAML login from IDP

        Args:
            ad_name         (str)-- AD machine name/ IP that resolved

            web_console_url (str)-- webconsole URL of the client

            adfs_app_name   (str)-- ADFS app name

            user            (str)-- AD user name

            password        (str)-- AD user password

            verify_sso      (bool)--verifies if SSO is successful

        Returns:
            True    --  If the login is successful

            False   --  If the login is not successful

        """
        status = False
        idp_url = "https://" + ad_name + "/adfs/ls/idpinitiatedsignon.aspx"
        self.open_url_in_new_tab(idp_url)
        self.logout_from_adfs(ad_name, local_sign_out=False)
        self.wait_for_completion()
        flag, loaded_url = self.login_to_adfs(user, password, adfs_app_name)
        if loaded_url == web_console_url + "/applications/":
            self.log.info("Logged in to webconsole successfully")
            status = True
            if verify_sso:
                status = self.check_single_sign_on_status(ad_name, user, web_console_url)
        if loaded_url == web_console_url + "/samlAcsIdpInitCallback.do?samlAppKey=":
            self.log.info("Error during SAML login")
            error_seen = self.driver.find_element_by_xpath("//div[@class='detailsDiv']/p").text
            self.log.info("FAILURE REASON::" + error_seen)
            status = False
        return status

                                         tab_off_approach, verify_sso, sp_initiated_link=None):
        """
        Initiates a SAML login from SP
        Args:
            ad_name             (str)-- AD machine name /IP that resolved

            web_console_url     (str)-- web cosnole URl of the client

            user                (str)-- AD user name

            tab_off_approach    (bool)--True , if tab will be pressed during saml login

            sp_initiated_link   (str)--link to initiate SAML login.Either supply tab off or SP link

        Returns:
            True    --  If the login is successful

            False   --  If the login is not successful
        """
        status = False
        if sp_initiated_link:
            self.open_url_in_new_tab(sp_initiated_link)

        if tab_off_approach and not sp_initiated_link:
            self.open_url_in_new_tab(web_console_url)
            self.driver.find_element_by_id('username').send_keys(user)
            self.driver.find_element_by_id('continuebtn').click()
            self.wait_for_completion()
        flag, loaded_url = self.login_to_adfs(user, password)
        if loaded_url == web_console_url + "/applications/":
            self.log.info("Logged in to webconsole successfully")
            status = True
            if verify_sso:
                status = self.check_single_sign_on_status(ad_name, user, web_console_url)
        if loaded_url == web_console_url + "/samlAcsIdpInitCallback.do?samlAppKey=":
            self.log.info("Error during SAML login")
            error_seen = self.driver.find_element_by_xpath("//div[@class='detailsDiv']/p").text
            self.log.info("FAILURE REASON::" + error_seen)
            status = False
        return status



[docs]    def check_single_sign_on_status(self, ad_name, user_name, web_console_url):
        """
        checks the single sign on status with SAML logged in page
        Args:
            ad_name        (str)-- name of the AD machine

            user_name       (str)-- complete user name with the company

            web_console_url(str)-- web console URL of the client

        Returns:
            True    --  If the login is successful

            False   --  If the login is not successful

        """
        idp_url = "https://" + ad_name + "/adfs/ls/idpinitiatedsignon.aspx"
        applications_page = web_console_url + "/applications/"
        if self.switch_to_tab(applications_page):
            logged_in_user = self.driver.find_element_by_xpath('//li[@class='
                                                               '"rightMenu hideOnExportFriendly'
                                                               ' userNameValue '
                                                               'vw-contains-drop-down"]'
                                                               '/span').text
            if self.get_user_name(logged_in_user) == self.get_user_name(user_name):
                found_app = False
                apps_listed = self.driver.find_elements_by_xpath("//div[@class='displayText "
                                                                 "vw-app-text']/span")
                for app in apps_listed:
                    if app.text == "CommCell Console GUI":
                        found_app = True
                        self.log.info("Commcell console GUI listed for SAML user")
                        break
                if not found_app:
                    self.log.info("Commcell console GUI tile not listed for SAML user")
                    return False
            else:
                self.log.info("Logged in user is not the given user")
                return False

            self.log.info("Loaded page is webconsole applicaitons page, so loading IDP in "
                          "new tab")
            self.open_url_in_new_tab(idp_url)

            if self.check_logged_in_state_adfs():
                self.log.info("SSO successful at IDP")
                return True
            self.log.info("SSO failed at IDP")
            return False

        """
        Initiates a SAML logout from IDP

            Args:

                ad_name             (str)-- AD machine name/ IP that resolved

                web_console_url     (str)-- webconsole URL of the client

                verify_single_logout(bool): verifies if single logout is successful

            Returns:

            True    --  If the logout is successful

        """
        status = False
        if self.logout_from_adfs(ad_name, local_sign_out=False):
            status = True
            if verify_single_logout:
                self.open_url_in_new_tab(web_console_url)
                if self.driver.current_url == web_console_url + "/login/index.jsp":
                    self.log.info("Login screen loaded after log out at IDP")
                else:
                    self.log.info("Logout unsuccessful at SP")
                    status = False
        return status



[docs]    def service_provider_initiated_logout(self, ad_name, web_console_url, verify_single_logout):
        """
        Initiates a SAML logout from SP

            Args:
                ad_name             (str)-- AD machine name/ IP that resolved

                web_console_url     (str)-- webconsole URL of the client

                verify_single_logout(bool)-- verifies if single logout is successful

            Returns:
                True    --  If the login is successful

                False   --  If the login is not successful

        """
        idp_url = "https://" + ad_name + "/adfs/ls/idpinitiatedsignon.aspx"
        applications_page = web_console_url + "/applications/"
        if not self.switch_to_tab(applications_page):
            self.open_url_in_new_tab(web_console_url)

        if self.check_if_entity_is_displayed('xpath', '//span[@class="vw-sprite-arrow '
                                             'vw-sprite-arrow-down"]'):
            self.log.info("Applications page found, logging out from the page")
            self.driver.find_element_by_xpath('//span[@class="vw-sprite-arrow '
                                              'vw-sprite-arrow-down"]').click()
            logout = self.driver.find_element_by_xpath("//li[@class='logout']/a")
            logout.click()
            self.wait_for_completion()

            if self.check_if_entity_is_displayed('id', 'logout-message'):
                self.log.info("Logout from SP successful")
                status = True
                if verify_single_logout:
                    self.open_url_in_new_tab(idp_url)

                    if not self.check_logged_in_state_adfs():
                        self.log.info("Logged out at IDP too")
                    else:
                        status = False
                        self.log.info("Logging out at IDP failed")
            else:
                raise Exception("No option to log out from webconsole")
        else:
            raise Exception("No option to logout from SP")
        return status



[docs]    def check_if_entity_is_displayed(self, entity_name, entity_value):
        """Check if a particular element is displayed or not

        Args:

             entity_name      (str)   --  the entity attribute to check for presence

             entity_value     (str)   --  the entity  value to be checked

        Returns:
                True    --  If the entity is available

                False   --  If the entity is not available
        """
        try:
            if entity_name == "id":
                self.driver.find_element_by_id(entity_value)
                return True
            elif entity_name == "xpath":
                self.driver.find_element_by_xpath(entity_value)
                return True
        except NoSuchElementException:
            return False



[docs]    def switch_to_tab(self, page_to_load):
        """

        Switches to the given page in browser

        Args:

            page_to_load (str)   --  Page to be loaded in the browser

        Returns: True if the page is found otherwise False
        """
        windows_list = self.driver.window_handles
        for window_id in range(len(windows_list)):
            self.driver.switch_to.window(windows_list[window_id])
            if self.driver.current_url == page_to_load:
                return True
        return False



[docs]    def wait_for_completion(self):
        """Checks for the notification bar at the top of the browser.

        Returns:    None """

        exists = True
        completion_time = 0
        interval = 1
        total_time = 600
        while exists:
            time.sleep(interval)
            completion_time += interval
            if completion_time > total_time:
                break
            exists = self.check_if_entity_is_displayed(
                "xpath", "//div[@id='loading-bar']")



[docs]    def get_user_name(self, given_string):
        """
        Parses the given string and gets the username from mail id or AD username format
        Args:
            given_string:   (str)   user name in mail address or domain\\username format

        Returns:  (str)    user name
        """

        if '\\' in given_string:
            return given_string.split('\\')[1]
        return given_string.split('@')[0]
