    @adminconsole_handler()
    def open_folder(self, folder):
        """Opens the folder to browse contents for restore.
            folder: a string, folder name we want to open
        """
        self.log.info("Opening " + folder + " folder")
        if self.check_if_entity_exists("link", folder):
            self.driver.find_element_by_link_text(folder).click()
            self.wait_for_completion()
        else:
            exp = "There is no folder called " + folder
            self.log.error(exp)
            raise exp

    @adminconsole_handler()
    def select_for_restore(self, file_folders):
        """Selects files, folders for restore.

            file_folders   : a list, list of files and folders we want to select to restore
        """
        self.log.info("Selecting items to submit for restore")
        selected = []
        while True:
            elements = self.driver.find_elements_by_xpath(
                "//div[1]/div[2]/div[2]/div[@class='ui-grid-canvas']/div")
            index = 1
            flag = []
            for elem in elements:
                f_elem = elem.find_element_by_xpath("./div/div[1]").text
                for file in file_folders:
                    if f_elem == file:
                        flag.append(index)
                        selected.append(f_elem)
                    else:
                        continue
                index = index + 1
            for flg in flag:
                flg = str(flg)
                self.driver.find_element_by_xpath(
                    "//div[1]/div/div[2]/div[@class='ui-grid-canvas']/div[" +
                    flg +
                    "]/div/div/div/div/div").click()
                file_folders = list(set(file_folders) - set(selected))
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    continue
                else:
                    if file_folders != []:
                        exp = "Could not find the item " + str(file_folders)
                        self.log.error(exp)
                        break
                    else:
                        break
            else:
                if file_folders != []:
                    exp = "Could not find the items " + str(file_folders)
                    self.log.error(exp)
                    break
                else:
                    break

    @adminconsole_handler()
    def submit_restore(self, items, destination, path, overwrite=False):
        """Submits a restore job of the given items.

            items       : a list,    list of files and folders we want to select to restore
            destination : a string,  destanation server we want to restore to
            path        : a string,  path of the directory we want to restore to
            overwrite   : a boolean, to decide whether to overwrite the files or not; default=
        """
        self.log.info("Submitting Restore")
        self.select_for_restore(items)
        self.driver.find_element_by_xpath(
            "//a[contains(text(),'Restore')]").click()
        self.wait_for_completion()
        Select(self.driver.find_element_by_id("destinationServer")
              ).select_by_visible_text(destination)
        if self.driver.find_element_by_xpath(
                "//input[@id='inplace']").is_selected():
            self.driver.find_element_by_xpath(
                "//label[contains(text(),'Restore to original folder')]").click()
        self.driver.find_element_by_id("restorePath").send_keys(path)
        if overwrite:
            if self.driver.find_element_by_xpath(
                    "//input[@type='checkbox' and @name='overwrite']").is_selected():
                self.driver.find_element_by_xpath(
                    "//label[contains(text(),'Unconditionally overwrite if it already exists')]").click()
        self.driver.find_element_by_xpath(
            "//form/div[3]/button[contains(text(),'Submit')]").click()
        self.wait_for_completion()
        job_id = self.driver.find_element_by_xpath(
            "//div[@class='growl']/div/div/div[@class='ng-binding ng-scope']").text
        job_id = job_id.split(":")[1].strip()
        return job_id
