    @WebAction()
    def __check_entity_exists(self):
        """Checks if entity exists for restore"""
        return self.check_if_entity_exists("xpath",
                                           "//div[@class='modal fade ng-isolate-scope in']")

    @WebAction()

        """Initiates restore process by clicking on restore"""
        self.driver.find_element_by_xpath(
            "//*[@id='browseActions']//*[contains(.,'Restore')]").click()

    @WebAction()

        """Enters the VM Restore Name in the input element

    @PageService()
    def full_vm_restore(
            self,
            vms_to_restore,
            inplace=False,
            restore_as="VMware vCenter",
            proxy=None,
            destination_server=None,
            vm_info=None,
            different_vcenter=False,
            different_vcenter_info={},
            disk_prov=None,
            transport_mode=None,
            power_on=True,
            over_write=True,
            live_recovery=False,
            live_recovery_options={},
            hv_restore_options={},
            dest_target=False
    ):
        """
        Submits a VMware full VM restore to a specified destination server

        Args:
            vms_to_restore           (list)  --  list of all Vms to restore

            inplace                  (bool)  --  if the VM needs to be restored in place

            restore_as               (basestring)   --  the hypervisor to be restored to
                    default :   VMware vCenter

            proxy                    (basestring)   --  the proxy to be used for restore

            destination_server       (basestring)   --  the name of the destination hypervisor

            vm_info                  (dict)  --  dict containing each restore VM details
                        vm_info = { 'VM1':  {   'host':'abc', 'datastore':'ds1', 'respool':'rp1',
                                                'network': {'source': 'network1',
                                                            'destination':'nw2'},
                                                'IP': { 'sourceIP':'','sourceSubnet':'',
                                                        'sourceGateway':'',
                                                        'DHCP':False,
                                                        'destinationIP':'', 'desitnaionSubnet':'',
                                                        'destinationGateway':'',
                                                        'destinationPreDNS':'',
                                                        'destinationAltDNS':'',
                                                        'destinationPrefWins':'',
                                                        'destinationAltWins':''}

            different_vcenter        (bool)  --  True / False to create a new vcenter client

            different_vcenter_info   (dict)  --  dict containing the new vcenter info
                        different_vcenter_info = {  'vcenter_hostname':'newvCenter',
                                                    'vcenter_username':'username',
                                                    'vcenter_password':'password'   }

            disk_prov                (basestring) --  the disk provisioning of the restore vm disks

            transport_mode           (basestring)   --  the transport mode to be used for restore

            power_on                 (bool)  --  if the restored VM needs to be powered on

            over_write               (bool)  --  if the restored VM needs to be overwritten

            live_recovery            (bool)  --  if the restore needs to use live recovery

            live_recovery_options    (dict)  --  live recovery options
                        live_recovery_options = {   'redirect_datastore':'ds1',
                                                    'delay_migration':'2'   }

            hv_restore_options       (dict)  --  all the restore options needed to restore
                                                    VMware VM as Hyper-V VM

            dest_target              (bool) -- if restore has to be done to restore target

        Returns:
            job_id      (basestring)   --  the restore job ID

        """
        self.log.info("Selecting VMs for performing Full VM restore")
        vm_list = vms_to_restore

        if not self.check_if_entity_exists("xpath",
                                           "//div[@class='modal fade ng-isolate-scope in']"):
            self.__browse_obj.select_for_restore(vm_list)
            self.driver.execute_script("window.scrollTo(0,0)")
            self.driver.find_element_by_xpath(
                "//*[@id='browseActions']//*[contains(.,'"+self.props['header.restore']+"')]").click()
            self.wait_for_completion()

        if not inplace:
            self.select_radio('outOfPlace')
            if restore_as != "VMware vCenter":
                self.__panel_dropdown_obj.select_drop_down_values(0, [restore_as])
                job_id = self.hv_full_vm_restore(hv_restore_options['vm_dict'],
                                                 hv_restore_options['proxy'],
                                                 hv_restore_options['server'],
                                                 hv_restore_options['location'])
                return job_id

            self.__panel_dropdown_obj.select_drop_down_values(1, [proxy])

            if different_vcenter:
                self.__panel_dropdown_obj.select_drop_down_values(1, ['Select a different vCenter (optional)'])
                self.fill_form_by_id("vcHostName", different_vcenter_info['vcenter_hostname'])
                self.fill_form_by_id("vcLoginName", different_vcenter_info['vcenter_username'])
                self.fill_form_by_id("vcPassword", different_vcenter_info['vcenter_password'])

            if dest_target:
                target = "Target\\"+destination_server
                self.__panel_dropdown_obj.select_drop_down_values(0, [target])


            for vms in vm_list:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']"
                                                  "//a[text()='" + vms + "']").click()
                self.wait_for_completion()

                if vm_info:
                    if 'prefix' in vm_info[vms].keys():
                        display_name = vm_info[vms]['prefix'] + vms
                    elif 'suffix' in vm_info[vms].keys():
                        display_name = vms + vm_info[vms]['suffix']
                    self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                      "@for='displayName']/input").clear()
                    self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                      "@for='displayName']"
                                                      "/input").send_keys(display_name)
                    if not dest_target:
                        self.driver.find_element_by_xpath("//label[@for='destinationHost']/"
                                                          "following-sibling::span/button["
                                                          "@class='btn btn-default browse-btn']"
                                                         ).click()
                        self.wait_for_completion()
                        self.check_error_message()

                        self.select_destination_host(vm_info[vms]['host'])
                        self.driver.find_element_by_xpath(
                            "//form[@name='selectVmForm']/div/button[2]").click()
                        self.wait_for_completion()
                        self.log.info("Destination host selected")
                        self.__panel_dropdown_obj.select_drop_down_values(2, [vm_info[vms]['datastore']])
                        self.log.info("Datastore selected successfully")
                        self.__panel_dropdown_obj.select_drop_down_values(3, [vm_info[vms]['respool']])
                        self.log.info("Resource pool selected successfully")

                        if 'network' in vm_info[vms].keys():
                            self.driver.find_element_by_xpath(
                                "//uib-accordion[@class='restore-network-settings ng-scope' \
                                ]/div/div[1]/div[1]/h4/a").click()
                            xpath = "//cv-grid/div/div[2]/div/div[1]/div[1]/div[2]/div[" \
                                    "@class='ui-grid-canvas']/div"
                            elements = self.driver.find_elements_by_xpath(xpath)
                            if elements:
                                self.scroll_into_view(xpath)
                                for elem in elements:
                                    elem.find_element_by_xpath(
                                        "./div/div[3]//a[@title='Remove setting']").click()
                            self.driver.find_element_by_xpath(
                                "//uib-accordion/div/div[1]/div[2]/div/div/ul/li/a").click()
                            self.wait_for_completion()
                            if 'source' in vm_info[vms]['network'].keys():
                                self.__panel_dropdown_obj.select_drop_down_values(
                                    0, [vm_info[vms]['network']['source']])
                                self.__panel_dropdown_obj.select_drop_down_values(
                                    1, [vm_info[vms]['network']['destination']])
                            self.click_button(self.props['OK'])
                        if 'IP' in vm_info[vms].keys():
                            self.driver.find_element_by_xpath(
                                "//uib-accordion[@class='restore-network-settings ng-scope']/div/"
                                "div[2]/div[1]/h4/a").click()
                            xpath = "//cv-grid/div/div[2]/div/div[1]/div[1]/div[2]/div[@class=" \
                                    "'ui-grid-canvas']/div"
                            self.scroll_into_view(xpath)
                            elements = self.driver.find_elements_by_xpath(xpath)
                            for elem in elements:
                                elem.find_element_by_xpath(
                                    "./div/div[3]/div/span/a[2]").click()
                            self.driver.find_element_by_xpath(
                                "//uib-accordion/div/div[2]/div[2]/div/div/ul/li/a").click()
                            self.wait_for_completion()
                            ip_settings = vm_info[vms]['IP']
                            self.fill_form_by_id("sourceIPAddress", ip_settings['sourceIP'])
                            self.fill_form_by_id("sourceSubnetMask", ip_settings['sourceSubnet'])
                            self.fill_form_by_id("sourceDefaultGateway",
                                                 ip_settings['sourceGateway'])
                            if 'DHCP' in ip_settings.keys():
                                self.toggle_enable("Use DHCP")
                            else:
                                self.fill_form_by_id("destinationIpAddress",
                                                     ip_settings['destinationIP'])
                                self.fill_form_by_id("destinationSubnetMask",
                                                     ip_settings['destinationSubnet'])
                                self.fill_form_by_id("destinationDefaultGateway",
                                                     ip_settings['destinationGateway'])

                                if 'destinationPrefDNS' in ip_settings.keys():
                                    self.fill_form_by_id("destinationPrefDnsServer",
                                                         ip_settings['destinationPrefDNS'])
                                if 'destinationAltDNS' in ip_settings.keys():
                                    self.fill_form_by_id("destinationAltDnsServer",
                                                         ip_settings['destinationAltDNS'])
                                if 'destinationPrefWins' in ip_settings.keys():
                                    self.fill_form_by_id("destinationPrefWinsServer",
                                                         ip_settings['destinationPrefWins'])
                                if 'destinationAltWins' in ip_settings.keys():
                                    self.fill_form_by_id("destinationAltWinsServer",
                                                         ip_settings['destinationAltWins'])
                        if live_recovery:
                            self.toggle_enable("Restore virtual machine using live recovery (vMotion)")
                            self.__panel_dropdown_obj(6, [live_recovery_options['redirect_datastore']])
                            self.fill_form_by_id("delayMigration", live_recovery_options['delay_migration'])
        if power_on:
            self.toggle_enable("Power on VMs after restore")
        else:
            self.toggle_disable("Power on VMs after restore")

        if over_write:
            self.toggle_enable("Unconditionally overwrite if it already exists")
            self.click_button(self.props['button.delete.yes'])
        else:
            self.toggle_disable("Unconditionally overwrite if it already exists")

        self.__panel_dropdown_obj.select_drop_down_values(2, [disk_prov])
        self.__panel_dropdown_obj.select_drop_down_values(3, [transport_mode])
        self.submit_form()
        return self.get_jobid_from_popup()

    @PageService()
    def hv_full_vm_restore(
            self,
            vm_list,
            proxy,
            server,
            location,
            name=None,
            path=None,
            power_on=False,
            over_write=False,
            register=False,
            network=None):
        """
        Submits a hyper V full VM restore to a hyperV server

        Args:
            vm_list      (basestring)   --  the name of the VM to restore

            proxy        (basestring)   --  the name of the proxy to be used for restore

            server       (basestring)   --  the hypervisor server to restore to

            location     (basestring)   --  the location where the disks and VM are to be restored

            name         (basestring)   --  the name of the restored VM

            path         (basestring)   --  the path to the restored VM location if not default

            power_on     (bool)  --  if the restored VM has to be powered on

            over_write   (bool)  --  if the restored VM has to overwrite an existing VM

            register     (bool)  --  if the restored VM has to be registered to cluster

            network      (basestring)   --  if a network card has to be attached to the VM

        Returns:
            job_id      (basestring)   --  the restore job ID


        """
        if not self.check_if_entity_exists("xpath",
                                           "//div[@class='modal fade ng-isolate-scope in']"):
            self.__browse_obj.select_for_restore(vm_list)
            self.driver.find_element_by_xpath(
                "//*[@id='browseActions']//*[contains(.,'Restore')]").click()
            self.wait_for_completion()
        self.cv_single_select("Virtualization client", server)
        self.cv_single_select("Destination client", proxy)
        if power_on:
            self.checkbox_select("powerOn")
        else:
            self.checkbox_deselect("powerOn")

        if over_write:
            self.checkbox_select("overwrite")
            self.wait_for_completion()
            self.click_button(self.props['button.delete.yes'])
        else:
            self.checkbox_deselect("overwrite")

        if register:
            self.checkbox_select("failover")
        else:
            self.checkbox_deselect("failover")

        for vms in vm_list:
            if len(vm_list) != 1:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']//a["
                                                  "contains(text(),'" + vms + "')]").click()
                self.wait_for_completion()
            self.select_value_from_dropdown("restoreLocation", location)
            self.wait_for_completion()
            if location == "Hyper-V default folder":
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']").clear()
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']")\
                    .send_keys(name + "".join(vms))
            elif location == "Select a folder":
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']").clear()
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']")\
                    .send_keys(name + "".join(vms))
                self.fill_form_by_id("destinationFolder", path)

            if network:
                self.select_value_from_dropdown("networkAdapter", network)

        self.log.info("Submitting full VM restore")
        self.wait_for_completion()
        self.driver.find_element_by_xpath("//div[@class='local-options']/form/div[2]/button["
                                          "contains(text(),'Submit')]").click()

        return self.get_jobid_from_popup()

    @PageService()
    def opc_full_vm_restore(self, vm_list, proxy, server, power_on, restore_prefix,
                            user_account, shape, network_list=None,
                            security_groups=None, ssh_keys=None):
        """
        Full instance restore method for Oracle Public Cloud
        Args:
            vm_list             (list)  --  the list of all VMs to restore

            proxy               (basestring)   --  the proxy to be used for restore

            server              (basestring)   --  the oracle cloud server to restore to

            power_on            (bool)  --  True / False to power on the instance

            restore_prefix      (basestring)   --  The basestringing to be prefixed to the VM name

            user_account        (basestring)   --  The user account where instances are to restored

            shape               (basestring)   --  The shape of the restore instances

            network_list        (list)  --  List of all networks to be associated to instances

            security_groups     (list)  --  List of all security groups to be associated

            ssh_keys            (list)  --  List of all ssh keys to be associated

        Returns:
            job_id  (basestring)   --  the restore job ID

        """
        if not self.check_if_entity_exists("xpath",
                                           "//div[@class='modal fade ng-isolate-scope in']"):
            self.__browse_obj.select_for_restore(vm_list)
            self.driver.find_element_by_xpath(
                "//*[@id='browseActions']//*[contains(.,'Restore')]").click()
            self.wait_for_completion()

        self.select_value_from_dropdown("Destination hypervisor", server, search=True)
        self.select_value_from_dropdown("destinationProxy", proxy)

        if power_on:
            self.checkbox_select("powerOn")
        else:
            self.checkbox_deselect("powerOn")

        for vm_name in vm_list:
            restore_vm_name = restore_prefix + vm_name
            if len(vm_list) != 1:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']//a["
                                                  "contains(text(),'" + vm_name + "')]").click()
                self.wait_for_completion()

                # Since there are multiple instances to restore, over writing the shape to Auto
                shape = "Auto"

            self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                              "@for='displayName']/input").clear()
            self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                              "@for='displayName']"
                                              "/input").send_keys(restore_vm_name)
            self.wait_for_completion()

            self.driver.find_element_by_xpath(
                "//button[@class='btn btn-default browse-btn']").click()
            self.wait_for_completion()
            self.check_error_message()

            self.select_destination_host(user_account)
            self.submit_form()
            self.log.info("User account selected")

            self.select_value_from_dropdown("instanceType", shape)
            self.log.info("Instance shape selected")

            if network_list:
                self.cvselect_from_dropdown('Network', network_list)
                self.log.info("Selected the networks")

            if security_groups:
                self.cvselect_from_dropdown('Security groups', security_groups)
                self.log.info("Selected the security groups")

            if ssh_keys:
                self.cvselect_from_dropdown('SSH keys', ssh_keys)
                self.log.info("Selected the ssh keys")

            self.log.info("Going to submit restore job")
            self.wait_for_completion()
            self.driver.find_element_by_xpath("//div[@class='local-options']/form/div[2]/button["
                                              "contains(text(),'Submit')]").click()

            return self.get_jobid_from_popup()

    @PageService()
    def ali_cloud_full_vm_restore(self, instance_list, destination_client, destination_proxy,
                                  power_on, overwrite_instance, in_place=False,
                                  restore_prefix=None, restore_suffix=None,
                                  instance_options=None):
        """
        Performs a full instance restore to Ali Cloud

        Args:
            instance_list       (list):         list of all instances to restore

            destination_client  (basestring):   the destination client where instances are to
                                                be restored

            destination_proxy   (basestring):   the proxy to be used for restore

            power_on            (bool):         if the instances are to be powered on after restore

            overwrite_instance  (bool):         to overwrite existing instances

            restore_prefix      (basestring):   to prepend a prefix to source instance name

            restore_suffix      (basestring):   to append a suffix to a source instance name

            in_place            (bool):         to restore the instances in-place

            instance_options    (dict):         restore options for each individual instance

                    Sample dict:    {'instance1':   {   'availability_zone': '',
                                                        'instance_type: '',
                                                        'network':  '',
                                                        'security_groups':  ['group1', 'group2']
                                                    },
                                    'instance2':   {   'availability_zone': '',
                                                        'instance_type: '',
                                                        'network':  '',
                                                        'security_groups':  ['group1', 'group2']
                                                    }
                                    }

        Raises:
            Exception:
                if there is an error with submitting Ali Cloud Full instance restore

        Returns:
            job_id  (basestring):   the restore job ID

        """
        display_name = None

        if not self.check_if_entity_exists("xpath",
                                           "//div[@class='modal fade ng-isolate-scope in']"):
            self.__browse_obj.select_for_restore(instance_list)
            self.driver.find_element_by_xpath(
                "//*[@id='browseActions']//*[contains(.,'Restore')]").click()
            self.wait_for_completion()

        self.select_value_from_dropdown("Destination client", destination_client, search=True)
        if self.check_if_entity_exists("id", "destinationProxy"):
            self.select_value_from_dropdown("destinationProxy", destination_proxy)

        if power_on:
            self.checkbox_select("powerOn")
        else:
            self.checkbox_deselect("powerOn")

        if overwrite_instance:
            self.checkbox_select("overwrite")
            self.wait_for_completion()
            self.click_button(self.props['button.delete.yes'])
        else:
            self.checkbox_deselect("overwrite")

        if len(instance_list) != 1:
            self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']//a["
                                              "contains(text(),'All Instances')]"
                                             ).click()
            self.wait_for_completion()

        if not in_place:
            self.checkbox_deselect("showInPlaceRestore")
            self.wait_for_completion()

            for instance_name in instance_list:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']"
                                                  "//a[text()='" + instance_name + "']").click()
                self.wait_for_completion()

                if restore_prefix:
                    display_name = restore_prefix + instance_name
                elif restore_suffix:
                    display_name = instance_name + restore_suffix

                self.driver.find_element_by_xpath("//*[@id='perVmOptions']//input["
                                                  "@name='displayName']").clear()
                self.driver.find_element_by_xpath("//*[@id='perVmOptions']//input["
                                                  "@name='displayName']"
                                                 ).send_keys(display_name)

                options = instance_options[instance_name]
                # Browse for availability zones

                self.driver.find_element_by_xpath("//*[@id='perVmOptions']/div[3]//button"
                                                 ).click()
                self.wait_for_completion()

                self.select_destination_host(options['availability_zone'])
                self.submit_form()

                self.select_value_from_dropdown("instanceType", options['instance_type'], True)

                self.driver.find_element_by_xpath("//*[@id='perVmOptions']/div[5]//button"
                                                 ).click()
                self.wait_for_completion()
                self.select_destination_host(options['network'], contains_host_name=True)
                self.submit_form()

                self.cv_single_select("Security groups (browse network to populate)",
                                      options['security_groups'][0])

        self.log.info("Submitting full instance restore")
        self.wait_for_completion()
        self.driver.find_element_by_xpath("//div[@class='local-options']/form/div[2]/button["
                                          "contains(text(),'Submit')]").click()
        return self.get_jobid_from_popup()

    def azure_full_vm_restore(self, vm_list, proxy, server, in_place, vm_info,
                              create_public_ip=False, over_write=False, power_on=False,
                              managed_vm=False, restore_prefix=None):

        """
        Performs a full VM restore for Azure RM
        Args:
            instance_list    (list):     list of all instances to restore

            vm_list          (list):     the name of the VM to restore

            proxy            (str):      the name of the proxy to be used for restore

            power_on         (bool):     if the restored VM has to be powered on

            param over_write (bool):     if the restored VM has to overwrite an
                                         existing VM
            server           (str) :     server name with each restore to be done

            vm_info          (list):     contains:azure_container,azure_vmsize,
                                        storageAccount,azure_vmnetwork

            managed_vm       (str):      False

            create_public_ip (str):      False

            restore_prefix   (str):     Restore prefix

        Returns:
            job_id  (basestring):   the restore job ID

        """
        if not self.check_if_entity_exists("xpath",
                                           "//div[@class='modal fade ng-isolate-scope in']"):
            self.__browse_obj.select_for_restore(vm_list)
            self.driver.find_element_by_xpath("//span[@id='browseActions']/a").click()
            self.wait_for_completion()
        select = Select(self.driver.find_element_by_xpath(
            "//div[@class='global-options']//div[1]//label[1]//select[1]"))
        pattern = re.compile(server, re.IGNORECASE)
        for option in select.options:
            value = option.get_attribute('value')
            if pattern.search(value):
                option.click()
                break
        self.wait_for_completion()
        self.select_value_from_dropdown("destinationProxy", proxy)
        self.wait_for_completion()

        if power_on:

            self.checkbox_select("powerOn")
            self.log.info("Selectd im place option")

        if over_write:
            self.checkbox_select("overwrite")
            self.wait_for_completion()
            self.click_button(self.props['button.delete.yes'])

        for vms in vm_list:

            if not in_place:

                self.checkbox_deselect("showInPlaceRestore")
                self.log.info("Deselectd im place option")

                if len(vm_list) != 1:
                    self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']//a["
                                                      "contains(text(),'" + vms + "')]").click()
                    self.wait_for_completion()

                self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                  "@for='displayName']/input").clear()
                self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                  "@for='displayName']"
                                                  "/input").send_keys(restore_prefix + vms)
                self.log.info("Entered the name of restored VM")

                self.log.info("Entered the name of restored VM")

                select_id = "azureContainer"
                azure_container = vm_info[vms]['resource_group']
                self.select_value_from_dropdown(select_id, azure_container, False, True)
                self.log.info("Entered the name of resource_group")

                select_id = "storageAccount"
                storage_account = vm_info[vms]['storage_account']
                self.select_value_from_dropdown(select_id, storage_account, False, True)
                self.log.info("Entered the name of storage Account")

                select_azure_vmsize = Select(self.driver.find_element_by_id("azureVmSize"))
                azure_vmsize = vm_info[vms]['vm_size']
                pattern = re.compile(azure_vmsize, re.IGNORECASE)
                for option in select_azure_vmsize.options:
                    value = option.get_attribute('value')
                    if pattern.search(value):
                        option.click()
                        break
                self.log.info("Entered the name of storage Account")

                select_id = "Network interface"
                azure_vmnetwork = vm_info[vms]['network_interface']
                self.cvselect_from_dropdown(select_id, azure_vmnetwork, False)
                self.log.info("Entered the name of NetworkInterface")

                if create_public_ip:
                    self.checkbox_select("createPublicIp")
                else:
                    self.checkbox_deselect("createPublicIp")

                select_id = "securityGroup"
                security_group = vm_info[vms]['security_group']
                self.select_value_from_dropdown(select_id, security_group, False, True)
                if managed_vm:
                    self.checkbox_select("restoreAsManagedVM")
            else:
                if not self.driver.find_element_by_id("showInPlaceRestore").is_selected():
                    self.checkbox_select("showInPlaceRestore")

        self.log.info("Submitting full instance restore")
        self.wait_for_completion()
        self.driver.find_element_by_xpath("//div[@class='local-options']/form/div[2]/button["
                                          "contains(text(),'Submit')]").click()
        return self.get_jobid_from_popup()

    @PageService()
    def vcloud_full_vm_restore(self, vm_list, proxy, destination_server, in_place, vm_info, org_vdc,
                               power_on=True, overwrite=False,
                               restore_vapp=True, restore_prefix=None):

        """
        Performs a full VM restore for Azure RM
        Args:
            vm_list                 (list):     the name of the VM to restore

            proxy                   (str):      the name of the proxy to be used for restore

            destination_server      (str) :     server name with each restore to be done

            in_place                (str):      the type of restore to be performed

            vm_info                 (list):     contains:vapp_name ,source network

            org_vdc                 (str) :     VDC name to be used for destination VM

            power_on                (bool):     if the restored VM has to be powered on

            overwrite               (bool):     if the restored VM has to overwrite an
                                                existing VM

           restore_prefix           (str):     Restore prefix

        Returns:
            job_id  (basestring):   the restore job ID

        """
        if not self.check_if_entity_exists("xpath",
                                           "//div[@class='modal fade ng-isolate-scope in']"):
            self.__browse_obj.select_for_restore(vm_list)
            self.driver.find_element_by_xpath("//span[@id='browseActions']/a").click()
            self.wait_for_completion()
        self.select_value_from_dropdown("Destination hypervisor", destination_server)
        self.wait_for_completion()
        self.select_value_from_dropdown("destinationProxy", proxy)
        self.wait_for_completion()

        for vms in vm_list:

            if len(vm_list) != 1:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']//a["
                                                  "contains(text(),'" + vms + "')]").click()
                self.wait_for_completion()

            if not in_place:

                self.checkbox_deselect("showInPlaceRestore")
                self.log.info("Deselectd im place option")

                self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                  "@for='displayName']/input").clear()
                self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                  "@for='displayName']"
                                                  "/input").send_keys(restore_prefix + vms)
                self.log.info("Entered the name of restored VM")

                self.select_value_from_dropdown("orgvDC", org_vdc, False, True)
                self.log.info("Entered the name of Organization VDC")

                self.fill_form_by_id("VAppName", vm_info[vms]['vapp_name'])
                self.log.info("Entered the name of storage Account")

                if 'source_network' in vm_info[vms].keys():
                    self.driver.find_element_by_xpath(
                        "//i[@class='pull-right glyphicon ng-scope glyphicon-chevron-down']"
                        ).click()
                    self.wait_for_completion()

                    xpath = "//cv-grid/div/div[3]/div/div[1]/div[1]/div[2]/div[@class='ui-grid-canvas']/div"
                    elements = self.driver.find_elements_by_xpath(xpath)
                    if elements:
                        self.scroll_into_view(xpath)
                        for elem in elements:
                            elem.find_element_by_xpath(
                                "./div/div[3]//a[@title='Remove setting']").click()
                    self.driver.find_element_by_xpath(
                        "//a[@data-ng-click='openAddNetworkSettingsDialog()']").click()
                    self.wait_for_completion()
                    self.select_value_from_dropdown("networkSettingsSource",
                                                    vm_info[vms]['source_network'])
                    self.select_value_from_dropdown("networkSettingsDestination",
                                                    vm_info[vms]['destination_network'])

                    self.driver.find_element_by_xpath(
                        "//button[@type='submit'][contains(text(),'OK')]").click()
                    self.wait_for_completion()
            else:
                self.checkbox_select("showInPlaceRestore")
                self.wait_for_completion()

        self.log.info("Going to the next page")
        self.button_next()

        if power_on:
            self.checkbox_select("powerOn")
            self.log.info("Selectd Power on option")
        else:
            self.checkbox_deselect("powerOn")
            self.log.info("Deselectd Power on option")

        if overwrite:
            self.checkbox_select("overwrite")
            self.wait_for_completion()
            self.click_button(self.props['button.delete.yes'])
        else:
            self.checkbox_deselect("overwrite")

        if restore_vapp:
            self.checkbox_select("vAppConfiguration")
            self.wait_for_completion()
            self.click_button(self.props['button.delete.yes'])

        self.wait_for_completion()
        self.log.info("Submitting full instance restore")

        if self.check_if_entity_exists("xpath", "//button[contains(text(),'Submit')]"):
            self.driver.find_element_by_xpath("//button[contains(text(),'Submit')]").click()
        return self.get_jobid_from_popup()

    @PageService()
    def oci_full_vm_restore(
            self,
            vm_list,
            proxy,
            server,
            power_on,
            in_place,
            vm_instance,
            over_write=False,
            restore_prefix=""):
        """
                Performs a full VM restore for OCI VM
                Args:
                    vm_list                 (list):     the name of the VM to restore

                    proxy                   (str):      the name of the proxy to be used for restore

                    server      (str) :     server name with each restore to be done

                    power_on                (bool):     if the restored VM has to be powered on

                    in_place                (str):      the type of restore to be performed

                    vm_instance             (list):     contains:compartment, AD, VCN

                    over_write              (bool):     if the restored VM has to overwrite an
                                                        existing VM

                   restore_prefix           (str):     Restore prefix

                Returns:
                    job_id  (basestring):   the restore job ID

                """
        if not self.__check_entity_exists():
            self.__browse_obj.select_for_restore(vm_list)
            self.__click_restore()
            self.wait_for_completion()
        self.select_value_from_dropdown("Destination hypervisor", server)
        self.select_value_from_dropdown("Access node", proxy)
        if power_on:
            self.checkbox_select("powerOn")
        else:
            self.checkbox_deselect("powerOn")

        if over_write:
            self.checkbox_select("overwrite")
            self.wait_for_completion()
            self.click_button(self.props['button.delete.yes'])
        else:
            self.checkbox_deselect("overwrite")
        if not in_place:
            compartment_name = vm_instance.compartment_name
            subnet_id = vm_instance.subnet_id
            datastore = vm_instance.datastore
            vcn_name = vm_instance.vcn_name

            self.checkbox_deselect("showInPlaceRestore")
            for vm_name in vm_list:
                vm_restore_name = restore_prefix + vm_name
                self.__send_vm_name(vm_restore_name)
                self.__select_vm_options(vm_name)
                self.select_value_from_dropdown("compartment", compartment_name)
                self.select_value_from_dropdown("availabilityDomain", datastore)
                self.select_value_from_dropdown("networkSettings", vcn_name)
                self.wait_for_completion()
                self.__select_subnet_id(subnet_id)

        else:
            self.checkbox_select("showInPlaceRestore")

        self.log.info("Submitting full VM restore")
        self.wait_for_completion()
        self.__click_final_restore()
        return self.get_jobid_from_popup()
