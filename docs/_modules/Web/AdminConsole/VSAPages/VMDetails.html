    """
    This class contains methods to get details about a virtual machine like last backup,
    vm jobs, restore, etc.
    """
    @adminconsole_handler()
    def last_backup(self):
        """
        Opens the last backup job details
        """
        if self.check_if_entity_exists("link", "Last backup"):
            self.driver.find_element_by_link_text("Last backup").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option called Last Backup for this VM.Check if the"
                            " VM has been backed up.")

    @adminconsole_handler()
    def vm_jobs(self):
        """
        Opens the jobs of the vm
        """
        if self.check_if_entity_exists("link", "Jobs"):
            self.driver.find_element_by_link_text("Jobs").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no link called jobs")

    @adminconsole_handler()
    def vm_restore(self):
        """
        Opens the restore options page for the vm
        """
        if self.check_if_entity_exists("link", "Restore"):
            self.driver.find_element_by_link_text("Restore").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to restore the VM. Check if the VM has"
                            " been backed up")

    @adminconsole_handler()
    def vm_remove(self):
        """
        Removes the VM from the backed up list
        """
        if self.check_if_entity_exists("xpath", "//span[@data-ng-click='removeVM()']"):
            self.driver.find_element_by_xpath("//span[@data-ng-click='removeVM()']/a").click()
            self.wait_for_completion()
            self.driver.find_element_by_xpath("//div[3]/button[2]").click()
            self.wait_for_completion()
            if self.check_if_entity_exists(
                    "xpath",
                    "//div[@class='modal-body text-danger ng-binding ng-scope']"):
                exp = self.driver.find_element_by_xpath("//div[@class='"
                                                        "modal-body text-danger ng-binding "
                                                        "ng-scope']").text
                if self.check_if_entity_exists("xpath", "//button[@ng-click='close()']"):
                    self.driver.find_element_by_xpath("//button[@ng-click='close()']").click()
                    self.wait_for_completion()
                raise Exception(exp)
        else:
            raise Exception("There is no option to remove the VM")

    @adminconsole_handler()
    def vm_backup_now(self):
        """
        Backs up the VM
        """
        if self.check_if_entity_exists("link", "Back up now"):
            self.driver.find_element_by_link_text("Back up now").click()
            self.wait_for_completion()
            self.driver.find_element_by_xpath(
                "//div/div/div[2]/div/button[ \
                @class='btn btn-primary cvBusyOnAjax']").click()
            self.wait_for_completion()
            job_id = None
            if self.driver.find_element_by_xpath(
                    "//div[@class='setup-content backup-options']/span[1]").is_displayed():
                # Job started so the tag that displays the Job ID stays Valid
                job_text = self.driver.find_element_by_xpath(
                    "//div[contains(text(),'Backup started for')]").text
                vm_details = job_text.split("Job")[-1].split(".")[0].strip()
                job_id = int(vm_details)
                self.log.info("Backup job " + str(job_id) + " has started")
                self.driver.find_element_by_xpath(
                    "//div[2]/div/button[contains(text(),'OK')]").click()
                self.wait_for_completion()
            if not job_id:
                raise Exception("The Job ID could not be obtained")
        else:
            raise Exception("There is no option to backup the VM. Check if the user has "
                            "sufficient permissions to back up this VM")
        return job_id

    @adminconsole_handler()
    def vm_summary(self):
        """Returns the dictionary of all the  vm summary infomation listed

            Ex: {
                'Size (GB)': '',
                'Backup status': 'NOT_PROTECTED',
                'Backup size (GB)': '',
                'Last backup time': '',
                'Server': 'ADMINCONSOLECB',
                'Guest size': '',
                'Host': '',
                'Guest OS': ''
            }
        """
        details = self.get_values_for_entity()
        return details
