    @adminconsole_handler()
    def full_vm_restore(
            self,
            vms_to_restore,
            inplace=False,
            restore_as="VMware vCenter",
            proxy=None,
            destination_server=None,
            vm_info=None,
            different_vcenter=False,
            different_vcenter_info={},
            disk_prov=None,
            transport_mode=None,
            power_on=True,
            over_write=True,
            live_recovery=False,
            live_recovery_options={},
            hv_restore_options={}
    ):
        """
        Submits a VMware full VM restore to a specified destination server

        Args:
            vms_to_restore           (list)  --  list of all Vms to restore

            inplace                  (bool)  --  if the VM needs to be restored in place

            restore_as               (str)   --  the hypervisor to be restored to
                    default :   VMware vCenter

            proxy                    (str)   --  the proxy to be used for restore

            destination_server       (str)   --  the name of the destination hypervisor client

            vm_info                  (dict)  --  dict containing each restore VM details
                        vm_info = { 'VM1':  {   'host':'abc', 'datastore':'ds1', 'respool':'rp1',
                                                'network': {'source': 'network1',
                                                            'destination':'nw2'},
                                                'IP': { 'sourceIP':'','sourceSubnet':'',
                                                        'sourceGateway':'',
                                                        'DHCP':False,
                                                        'destinationIP':'', 'desitnaionSubnet':'',
                                                        'destinationGateway':'',
                                                        'destinationPreDNS':'',
                                                        'destinationAltDNS':'',
                                                        'destinationPrefWins':'',
                                                        'destinationAltWins':''}

            different_vcenter        (bool)  --  True / False to create a new vcenter client

            different_vcenter_info   (dict)  --  dict containing the new vcenter info
                        different_vcenter_info = {  'vcenter_hostname':'newvCenter',
                                                    'vcenter_username':'username',
                                                    'vcenter_password':'password'   }

            disk_prov                (str)   --  the disk provisioning of the restore vm disks

            transport_mode           (str)   --  the transport mode to be used for restore

            power_on                 (bool)  --  if the restored VM needs to be powered on

            over_write               (bool)  --  if the restored VM needs to be overwritten

            live_recovery            (bool)  --  if the restore needs to use live recovery

            live_recovery_options    (dict)  --  live recovery options
                        live_recovery_options = {   'redirect_datastore':'ds1',
                                                    'delay_migration':'2'   }

            hv_restore_options       (dict)  --  all the restore options needed to restore
                                                    VMware VM as Hyper-V VM

        Returns:
            job_id      (str)   --  the restore job ID

        """
        self.log.info("Selecting VMs for performing Full VM restore")
        vm_list = vms_to_restore
        self.select_for_restore(vm_list)
        self.driver.find_element_by_xpath(
            "//span[@id='browseActions']/a").click()
        self.wait_for_completion()

        if not inplace:
            if restore_as != "VMware vCenter":
                self.select_value_from_dropdown("restoreDestinationType", restore_as)
                job_id = self.hv_full_vm_restore(hv_restore_options['vm_dict'],
                                                 hv_restore_options['proxy'],
                                                 hv_restore_options['server'],
                                                 hv_restore_options['location'])
                return job_id

            Select(self.driver.find_element_by_xpath(
                "//div[@class='global-options']/div[1]/div[1]/label/select")
                  ).select_by_visible_text(destination_server)
            self.select_value_from_dropdown("destinationProxy", proxy)

            if different_vcenter:
                self.checkbox_select("owmVcenterSelected")
                self.fill_form_by_id("vcHostName", different_vcenter_info['vcenter_hostname'])
                self.fill_form_by_id("vcLoginName", different_vcenter_info['vcenter_username'])
                self.fill_form_by_id("vcPassword", different_vcenter_info['vcenter_password'])

            if self.driver.find_element_by_id("showInPlaceRestore").is_enabled():
                self.checkbox_deselect("showInPlaceRestore")

            for vms in vm_list:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']"
                                                  "//a[text()='" + vms + "']").click()
                self.wait_for_completion()

                if vm_info:
                    if 'prefix' in vm_info[vms].keys():
                        display_name = vm_info[vms]['prefix'] + vms
                    elif 'suffix' in vm_info[vms].keys():
                        display_name = vms + vm_info[vms]['suffix']
                    self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                      "@for='displayName']/input").clear()
                    self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                                      "@for='displayName']"
                                                      "/input").send_keys(display_name)
                    self.driver.find_element_by_xpath(
                        "//button[@class='btn btn-default browse-btn']").click()
                    self.wait_for_completion()
                    self.check_error_message()

                    self.select_destination_host(vm_info[vms]['host'])
                    self.submit_form()
                    self.log.info("Destination host selected")

                    self.select_value_from_dropdown("data-store", vm_info[vms]['datastore'], True)
                    self.log.info("Datastore selected successfully")

                    self.select_value_from_dropdown("resource-pool", vm_info[vms]['respool'])
                    self.log.info("Resource pool selected successfully")

                    if 'network' in vm_info[vms].keys():
                        self.driver.find_element_by_xpath(
                            "//uib-accordion[@class='restore-network-settings ng-scope' \
                            ]/div/div[1]/div[1]/h4/a").click()
                        elements = self.driver.find_elements_by_xpath(
                            "//cv-grid/div/div[2]/div/div[1]/div[1]/div[2]/div["
                            "@class='ui-grid-canvas']/div")
                        if elements:
                            for elem in elements:
                                elem.find_element_by_xpath("./div/div[3]/div/span/a[2]").click()
                        self.driver.find_element_by_xpath(
                            "//uib-accordion/div/div[1]/div[2]/div/div/ul/li/a").click()
                        self.wait_for_completion()
                        if 'source' in vm_info[vms]['network'].keys():
                            self.select_value_from_dropdown("networkSettingsSource", vm_info[vms]
                                                            ['network']['source'])
                        self.select_value_from_dropdown("networkSettingsDestination", vm_info[vms]
                                                        ['network']['destination'])

                        self.driver.find_element_by_xpath("/html/body/div[1]/div/div/div[2]/form/"
                                                          "div/button[2]").click()
                        self.wait_for_completion()
                    if 'IP' in vm_info[vms].keys():
                        self.driver.find_element_by_xpath(
                            "//uib-accordion[@class='restore-network-settings ng-scope']/div/"
                            "div[2]/div[1]/h4/a").click()
                        elements = self.driver.find_elements_by_xpath(
                            "//cv-grid/div/div[2]/div/div[1]/div[1]/div[2]/div[@class="
                            "'ui-grid-canvas']/div")
                        for elem in elements:
                            elem.find_element_by_xpath(
                                "./div/div[3]/div/span/a[2]").click()
                        self.driver.find_element_by_xpath(
                            "//uib-accordion/div/div[2]/div[2]/div/div/ul/li/a").click()
                        self.wait_for_completion()
                        ip_settings = vm_info[vms]['IP']
                        self.fill_form_by_id("sourceIPAddress", ip_settings['sourceIP'])
                        self.fill_form_by_id("sourceSubnetMask", ip_settings['sourceSubnet'])
                        self.fill_form_by_id("sourceDefaultGateway",
                                             ip_settings['sourceGateway'])
                        if 'DHCP' in ip_settings.keys():
                            self.checkbox_select("useDHCP")
                        else:
                            self.fill_form_by_id("destinationIpAddress",
                                                 ip_settings['destinationIP'])
                            self.fill_form_by_id("destinationSubnetMask",
                                                 ip_settings['destinationSubnet'])
                            self.fill_form_by_id("destinationDefaultGateway",
                                                 ip_settings['destinationGateway'])

                            if 'destinationPrefDNS' in ip_settings.keys():
                                self.fill_form_by_id("destinationPrefDnsServer",
                                                     ip_settings['destinationPrefDNS'])
                            if 'destinationAltDNS' in ip_settings.keys():
                                self.fill_form_by_id("destinationAltDnsServer",
                                                     ip_settings['destinationAltDNS'])
                            if 'destinationPrefWins' in ip_settings.keys():
                                self.fill_form_by_id("destinationPrefWinsServer",
                                                     ip_settings['destinationPrefWins'])
                            if 'destinationAltWins' in ip_settings.keys():
                                self.fill_form_by_id("destinationAltWinsServer",
                                                     ip_settings['destinationAltWins'])

                        self.submit_form()

        self.log.info("Going to the next page")
        self.button_next()

        if power_on:
            self.checkbox_select("powerOn")

        if over_write:
            self.checkbox_select("overwrite")
            self.click_yes_button()

        self.select_value_from_dropdown("diskProvOption", disk_prov)
        self.select_value_from_dropdown("transportMode", transport_mode)

        if live_recovery:
            self.checkbox_select("liveRecoveryOption")
            self.select_value_from_dropdown("redirect-write-data-store", live_recovery_options[
                'redirect_datastore'
            ], True)
            self.fill_form_by_id("delayMigration", live_recovery_options['delay_migration'])

        self.log.info("Going to submit restore job")
        self.wait_for_completion()
        self.submit_form(False)

        return self.get_jobid_from_popup()

    @adminconsole_handler()
    def hv_full_vm_restore(
            self,
            vm_list,
            proxy,
            server,
            location,
            name=None,
            path=None,
            power_on=False,
            over_write=False,
            register=False):
        """
        Submits a hyper V full VM restore to a hyperV server

        Args:
            vm_list      (str)   --  the name of the VM to restore

            proxy        (str)   --  the name of the proxy to be used for restore

            server       (str)   --  the hypervisor server to restore to

            location     (str)   --  the location where the disks and VM are to be restored

            name         (str)   --  the name of the restored VM

            path         (str)   --  the path to the restored VM location if not default

            power_on     (bool)  --  if the restored VM has to be powered on

            over_write   (bool)  --  if the restored VM has to overwrite an existing VM

            register     (bool)  --  if the restored VM has to be registered to cluster

        Returns:
            job_id      (str)   --  the restore job ID


        """
        self.select_for_restore(vm_list)
        self.driver.find_element_by_xpath("//span[@id='browseActions']/a").click()
        self.wait_for_completion()
        select = Select(self.driver.find_element_by_xpath(
            "//label[@for='serverId']/select"))
        pattern = re.compile(server, re.IGNORECASE)
        for option in select.options:
            value = option.get_attribute('value')
            if pattern.search(value):
                option.click()
                break
        self.wait_for_completion()
        self.select_value_from_dropdown("destinationProxy", proxy)
        if power_on:
            self.checkbox_select("powerOn")

        if over_write:
            self.checkbox_select("overwrite")
            self.click_yes_button()

        if register:
            if not self.driver.find_element_by_name("failover").is_selected():
                self.driver.find_element_by_xpath("//label[@for='failover']").click()
        for vms in vm_list:
            if len(vm_list) != 1:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']//a["
                                                  "contains(text(),'" + vms + "')]").click()
                self.wait_for_completion()
            self.select_value_from_dropdown("restoreLocation", location)
            self.wait_for_completion()
            if location == "Hyper-V default folder":
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']").clear()
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']")\
                    .send_keys(name + "".join(vms))
            elif location == "Select a folder":
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']").clear()
                self.driver.find_element_by_xpath(
                    "//form/div[1]/label[3]/input[@name='displayName']")\
                    .send_keys(name + "".join(vms))
                self.fill_form_by_id("destinationFolder", path)

        self.log.info("Submitting full VM restore")
        self.wait_for_completion()
        self.submit_form(False)

        return self.get_jobid_from_popup()

    @adminconsole_handler()
    def opc_full_vm_restore(self, vm_list, proxy, server, power_on, restore_prefix,
                            user_account, shape, network_list=None,
                            security_groups=None, ssh_keys=None):
        """
        Full instance restore method for Oracle Public Cloud
        Args:
            vm_list             (list)  --  the list of all VMs to restore

            proxy               (str)   --  the proxy to be used for restore

            server              (str)   --  the oracle cloud server to restore to

            power_on            (bool)  --  True / False to power on the instance

            restore_prefix      (str)   --  The string to be prefixed to the VM name

            user_account        (str)   --  The user account where the instances are to restored

            shape               (str)   --  The shape of the restore instances

            network_list        (list)  --  List of all networks to be associated to instances

            security_groups     (list)  --  List of all security groups to be associated

            ssh_keys            (list)  --  List of all ssh keys to be associated

        Returns:
            job_id  (str)   --  the restore job ID

        """
        self.select_for_restore(vm_list)
        self.driver.find_element_by_xpath("//span[@id='browseActions']/a").click()
        self.wait_for_completion()

        select = Select(self.driver.find_element_by_xpath(
            "//label[@for='serverId']/select"))
        pattern = re.compile(server, re.IGNORECASE)
        for option in select.options:
            value = option.get_attribute('value')
            if pattern.search(value):
                option.click()
                break
        self.wait_for_completion()
        self.select_value_from_dropdown("destinationProxy", proxy)

        if power_on:
            self.checkbox_select("powerOn")
        else:
            self.checkbox_deselect("powerOn")

        for vm_name in vm_list:
            restore_vm_name = restore_prefix + vm_name
            if len(vm_list) != 1:
                self.driver.find_element_by_xpath("//ul[@class='vm-full-restore-list']//a["
                                                  "contains(text(),'" + vm_name + "')]").click()
                self.wait_for_completion()

                # Since there are multiple instances to restore, over writing the shape to Auto
                shape = "Auto"

            self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                              "@for='displayName']/input").clear()
            self.driver.find_element_by_xpath("//*[@id='perVmOptions']//label["
                                              "@for='displayName']"
                                              "/input").send_keys(restore_vm_name)
            self.wait_for_completion()

            self.driver.find_element_by_xpath(
                "//button[@class='btn btn-default browse-btn']").click()
            self.wait_for_completion()
            self.check_error_message()

            self.select_destination_host(user_account)
            self.submit_form()
            self.log.info("User account selected")

            self.select_value_from_dropdown("instanceType", shape)
            self.log.info("Instance shape selected")

            if network_list:
                self.cvselect_from_dropdown('Network', network_list)
                self.log.info("Selected the networks")

            if security_groups:
                self.cvselect_from_dropdown('Security groups', security_groups)
                self.log.info("Selected the security groups")

            if ssh_keys:
                self.cvselect_from_dropdown('SSH keys', ssh_keys)
                self.log.info("Selected the ssh keys")

            self.log.info("Going to submit restore job")
            self.wait_for_completion()
            self.submit_form(False)

            return self.get_jobid_from_popup()
