    """
    This class contains methods for submitting virtual files machine restores.

    @adminconsole_handler()
    def vm_files_restore(
            self,
            vm_name,
            files,
            destination_server,
            path,
            over_write=False):
        """
        Restores the virtual machine files of the backedup VM to the path in the destination server.
        Same definition can be used for both vmware and hyperV restore.

        Args:
            vm_name             (basestring)    :  name of the VM whose files are to be restored

            files               (list)          :  list of VM files to restore

            destination_server  (basestring)    :  destination server to restore to

            path                (basestring)    :  path to restore the file to

            over_write          (bool)          :  if files are to be overwritten during restore

        Raises:
            Exception:
                if there is an error while submitting a virtual machine files restore

        Returns:
            job_id      (basestring)   :  the restore job ID

        """
        self.log.info("Submitting Virtual Machine files Restore to server {0}".format(
            destination_server))
        if not self.check_if_entity_exists(
                "xpath", "//div[1]/ul/li[2]/span[contains(text(), '" + vm_name + "')]"):
            self.select_vm(vm_name)
        self.select_for_restore(files)
        self.driver.find_element_by_xpath(
            "//span[@id='browseActions']/a").click()
        self.wait_for_completion()
        self.select_value_from_dropdown("destinationServer", destination_server)

        self.fill_form_by_id("restorePath", path)

        if over_write:
            self.checkbox_select("overwrite")
        self.log.info("Submitting virtual  machines files restore")
        self.wait_for_completion()
        self.submit_form(False)

        return self.get_jobid_from_popup()
