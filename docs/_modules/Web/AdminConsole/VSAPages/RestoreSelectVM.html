    @adminconsole_handler()
    def select_vm(self, vm_name):
        """
        Opens the given VM
        """
        self.log.info("Open VM %s to browse content", str(vm_name))
        if self.check_if_entity_exists(
                "xpath", "//div[2]/div/span[@class='serverMessage  error']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//div[2]/div/span[@class='serverMessage  error']").text)
        else:
            self.driver.find_element_by_xpath(
                "//a[@val='" + vm_name + "']").click()
            self.wait_for_completion()

    @adminconsole_handler()
    def select_for_restore(self, file_folders):
        """
        Selects files, folders for restore
        :param file_folders (list)  --  the list of files and folders to select for restore
        """
        self.log.info("Selecting items to submit for restore")
        selected = []
        while True:
            elements = self.driver.find_elements_by_xpath(
                "//div[1]/div[2]/div[2]/div[@class='ui-grid-canvas']/div")
            index = 1
            flag = []
            for elem in elements:
                if elem.find_element_by_xpath(
                        "./div/div[1]/a").is_displayed():
                    file_element = elem.find_element_by_xpath("./div/div[1]/a").text
                else:
                    file_element = elem.find_element_by_xpath(
                        "./div/div[1]/span/label").text
                for file in file_folders:
                    if file_element == file:
                        flag.append(index)
                        selected.append(file_element)
                    else:
                        continue
                index += 1
            for flg in flag:
                flg = str(flg)
                self.driver.find_element_by_xpath(
                    "//div[1]/div/div[2]/div[@class='ui-grid-canvas']/div[" +
                    flg +
                    "]/div/div/div/div/div").click()
                file_folders = list(set(file_folders) - set(selected))
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    continue
                else:
                    break
            else:
                break
        if file_folders:
            raise Exception("Could not find the items " + str(file_folders))

    @adminconsole_handler()
    def backup_for_specific_date(
            self,
            time_value):
        """
        Shows backup content as of a specific date
        :param time_value   (dict)  --  dictionary containing the value to set as date range
        """
        self.log.info("Displaying backup data as of a specific date")
        self.driver.find_element_by_xpath(
            "//div[@data-ng-controller='browseOptionsController']/span/a").click()
        self.driver.find_element_by_xpath(
            "//div[@data-ng-controller='browseOptionsController']/span/ul/li[1]/a").click()

        self._date_picker(time_value, "to-picker")

        self.driver.find_element_by_xpath(
            "//div[3]/button[contains(text(),'OK')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def backup_for_date_range(
            self,
            from_time,
            to_time):
        """
        Shows backup content in a time range
        :param from_time    (dict)  --  dictionary containing the value to be set as from time
        :param to_time      (dict)  --  dictionary containing the value to be set as to time
        """
        self.driver.find_element_by_xpath(
            "//div[@data-ng-controller='browseOptionsController']/span/a").click()
        self.driver.find_element_by_xpath(
            "//div[@data-ng-controller='browseOptionsController']/span/ul/li[1]/a").click()

        self._date_picker(from_time, "from-picker")

        # To Time
        self._date_picker(to_time, "to-picker")

        self.driver.find_element_by_xpath(
            "//div[3]/button[contains(text(),'OK')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def latest_backups(self):
        """
        Shows the latest backup
        """
        self.log.info("Showing the latest backups")
        self.driver.find_element_by_xpath(
            "//a[@class='dropdown-toggle ng-binding']/span").click()
        self.driver.find_element_by_xpath(
            "//a[contains(text(),'Show latest backups')]").click()
        self.wait_for_completion()
