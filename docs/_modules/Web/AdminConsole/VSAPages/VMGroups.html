    @adminconsole_handler()
    def select_vm_group(self, vm_group_name):
        """
        Opens the VM group with the given name
        :param vm_group_name     (str)   --  the name of the VM group to open
        """
        self.search_for(vm_group_name)
        if self.check_if_entity_exists("link", vm_group_name):
            self.driver.find_element_by_link_text(vm_group_name).click()
            self.wait_for_completion()
            self.log.info("The VM group {0} has been opened".format(vm_group_name))
        else:
            exp = "There is no server with the name {0}".format(vm_group_name)
            raise Exception(exp)

    @adminconsole_handler()
    def add_vm_group(self, hypervisor_name, vm_group_name, vm_content, plan=None):
        """
        Adds a new VM group
        :param hypervisor_name      (str)   --  the hypervisor whose VMs should be grouped
        :param vm_group_name         (str)   --  the name of the vm group to create
        :param vm_content           (dict)  --  the content to be grouped
        :param plan                 (str)   --  the plan to be associated
        """
        if self.check_if_entity_exists("link", "Add VM group"):
            self.driver.find_element_by_link_text("Add VM group").click()
            self.wait_for_completion()
            self.select_value_from_dropdown("client", hypervisor_name)
            self.fill_form_by_id("name", vm_group_name)

            from Web.AdminConsole.VSAPages.HypervisorDetails import HypervisorDetails
            server_details_obj = HypervisorDetails(self.driver)
            server_details_obj.select_vm_from_browse_tree(vm_content)

            self.driver.find_element_by_xpath(
                "//span[@class='multiSelect inlineBlock']/button").click()
            self.driver.find_element_by_xpath(
                "//span[contains(text(), '" + plan + "')]").click()
            self.driver.find_element_by_xpath(
                "//form/div[5]/button[@type='submit']").click()
            self.wait_for_completion()

            if self.check_if_entity_exists(
                    "xpath", "//div[2]/p[@class='serverMessage error']"):
                exp = self.driver.find_element_by_xpath(
                    "//div[2]/p[@class='serverMessage error']").text
                raise Exception(exp)
            else:
                self.driver.find_element_by_xpath("//span/ul[1]/li[2]/a").click()
                self.wait_for_completion()
        else:
            exp = "There is no option to Add VM group"
            raise Exception(exp)

    @adminconsole_handler()
    def action_backup_vm_groups(self, vm_group_name, backup_type):
        """
        Starts a backup of the given type for the specified collection
        :param vm_group_name     (str)   --  the name of the VM group to backup
        :param backup_type       (str)   --  the type of backup to run
        :return job_id           (int)   --  the backup job ID
        """
        self.search_for(vm_group_name)
        self.driver.find_element_by_xpath(
            "//a[contains(text(),'" + vm_group_name + "')]/../../div[5]/div/a/span").click()
        self.wait_for_completion()
        if self.check_if_entity_exists("link", "Back up now"):
            self.driver.find_element_by_xpath(
                "//a[contains(text(),'" +
                vm_group_name +
                "')]/../../div[5]/div/ul//a[contains(text(),'Back up now')]").click()
            self.wait_for_completion()

            from Web.AdminConsole.VSAPages.HypervisorDetails import HypervisorDetails
            server_details_obj = HypervisorDetails(self.driver)
            server_details_obj.submit_backup(backup_type)
            return server_details_obj.job_id()
        else:
            exp = "There is no option to backup the vmgroup"
            raise Exception(exp)

    @adminconsole_handler()
    def action_delete_vm_groups(self, vm_group_name):
        """
        Deletes a vm group with the given name
        :param vm_group_name    (str)   --  the VM group to delete
        """
        self.search_for(vm_group_name)
        if self.check_if_entity_exists("link", vm_group_name):
            self.driver.find_element_by_xpath(
                "//a[contains(text(),'" + vm_group_name + "')]/../../div[5]/div/a/span").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("link", "Delete"):
                self.driver.find_element_by_xpath(
                    "//a[contains(text(),'" + vm_group_name + "')]/../../div[5]/div/ul//a["
                                                              "contains(text(),'Delete')]").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath(
                    "//div[3]/button[contains(text(),'Yes')]").click()
                self.wait_for_completion()
            else:
                exp = "There is no Delete option for deleting the vmgroup"
                raise Exception(exp)

            if self.check_if_entity_exists(
                    "xpath",
                    "//div[4]/div/div[@class='growl-item alert ng-scope alert-error"
                    " alert-danger']"):
                exp = self.driver.find_element_by_xpath(
                    "//div[4]/div/div[@class='growl-item alert ng-scope"
                    " alert-error alert-danger']/div/div").text
                raise Exception(exp)
        else:
            exp = "There is no vm group with the name {0}".format(vm_group_name)
            raise Exception(exp)

    @adminconsole_handler()
    def action_restore_vm_groups(self, vm_group_name):
        """
        Opens the restore page of the vm group from the server details page
        :param vm_group_name    (str)   --  the VM group to restore
        """
        self.search_for(vm_group_name)
        if self.check_if_entity_exists("link", vm_group_name):
            self.driver.find_element_by_xpath(
                "//a[contains(text(),'" +
                vm_group_name +
                "')]/../../div[5]/div/a/span").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("link", "Restore"):
                if self.check_if_entity_exists("xpath",
                                               "//div/a[contains(text(),'" + vm_group_name +
                                               "')]/ancestor::div[@class='ng-isolate-scope']/"
                                               "div[5]/div/ul//a[contains(text(),'Restore')]"):
                    self.driver.find_element_by_xpath(
                        "//div/a[contains(text(),'" +
                        vm_group_name +
                        "')]/ancestor::div[@class='ng-isolate-scope']/div[5]/div/ul//a["
                        "contains(text(),'Restore')]").click()
                    self.wait_for_completion()
            else:
                exp = "The vm group has not been backed up yet or " \
                      "there is no vm group by that name"
                self.log.error(exp)

    @adminconsole_handler()
    def list_vm_groups(self):
        """
        Lists all the VM groups
        :return vm_group_list (list)    --  list of all VM groups
        """
        vm_group_list = []
        elements = self.driver.find_elements_by_xpath("//div[@class='ui-grid-canvas']/div")
        for element in elements:
            vm_group = {'VM Group Name': element.find_element_by_xpath("./div/div[1]/a").text,
                        'Hypervisor': element.find_element_by_xpath("./div/div[2]/a").text,
                        'Proxy': element.find_element_by_xpath("./div/div[3]/span").text,
                        'Group type': element.find_element_by_xpath("./div/div[4]/span").text}
            vm_group_list.append(vm_group)
        self.log.info("The list of all VM groups are {0}".format(str(vm_group_list)))
        return vm_group_list
