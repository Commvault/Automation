    @PageService()
    def subclient_storage_pool_info(self):
        """
        Displays the storage pool info of the subclient

        Returns:
            pool_info   (list):  lists the storage pool info

        """
        self.select_configuration_tab()
        elements = self.driver.find_elements_by_xpath("//cv-tile-component[@data-title='"
                                                      "Storage pool']/div/div[3]/div/div[2]/ul/li")
        pool_info = {}
        for element in elements:
            pool_info[element.find_element_by_xpath(
                "./span[1]").text] = element.find_element_by_xpath("./span[2]").text
        self.log.info("The storage pool associated with plan of the subclient is %s",
                      str(pool_info))
        return pool_info

    @PageService()
    def schedule_info(self):
        """
        Displays all the schedules associated with the subclient

        Returns:
            schedules  (list):  list of all schedules associated to the subclient

        """
        self.select_configuration_tab()
        elements = self.driver.find_elements_by_xpath("//cv-tile-component[@data-title="
                                                      "'Schedules']/div/div[3]/div/div[2]/ul")
        schedules = []
        for element in elements:
            schedules.append(element.find_element_by_xpath("./li").text)
        self.log.info("The schedules associated with the subclient are %s", str(schedules))
        return schedules

    @PageService()
    def restore(self, recovery_time=None):
        """
        Opens the select restore page

        Args:
            recovery_time   (basestring):   the backup date in 01-September-1960 format

        Raises:
            Exception:
                if there is no option to restore

        """
        self.select_overview_tab()
        if recovery_time:
            calender = {}
            calender['date'], calender['month'], calender['year'] = recovery_time.split("-")
            self.date_picker(calender)
        panel_info = PanelInfo(self, self.props['label.recoveryPoint'])
        panel_info.open_hyperlink_on_tile(self.props['header.restore'])
        self.wait_for_completion()

    @PageService()
    def backup_now(self, bkp_type):
        """
        Starts a backup job for the collection

        Args:

            bkp_type (BackupType): the backup level, among the type in Backup.BackupType enum

        Returns:

    @PageService()
    def subclient_settings(self, reader=2, collect_file_details=False, bkp_type=None, cbt=True,
                           auto_detect_owner=False, transport_mode=None,
                           free_space=None, ds_free_space_check=True):
        """
        Edits the no of readers assigned to the collection and the backup type
        of the virtual machine

        Args:
            reader                   (int):         the number of readers

            collect_file_details     (bool):        to enable or disable metadata collection

            bkp_type                 (basestring):  app / crash consistent backup

            cbt                      (bool):        enable / disable CBT

            auto_detect_owner        (bool):        to auto detect the owner of the VM

            transport_mode           (basestring):  the transport mode to be used for backup

            free_space               (int):         the amount of free space to be set

            ds_free_space_check      (bool):        to enable / disable datastore freespace check

        Raises:
            Exception:
                when the subclient setting could not be edited or
                when there is an error while updating the settings

        """
        self.select_configuration_tab()
        if cbt:
            self.toggle_enable("Use changed block tracking")
        else:
            self.toggle_disable("Use changed block tracking")

        if auto_detect_owner:
            self.toggle_enable("Auto detect VM Owner")
        else:
            self.toggle_disable("Auto detect VM Owner")

        if collect_file_details:
            self.toggle_enable("Collect file details")
        else:
            self.toggle_disable("Collect file details")

        if self.check_if_entity_exists("xpath", "//cv-tile-component[@data-title='Settings']/"
                                                "div/div[3]/div/div[2]/div/a["
                                                "contains(text(),'Edit')]"):
            self.tile_select_hyperlink("Settings", "Edit")
            self.fill_form_by_id("noOfReaders", reader)

            if bkp_type:
                self.driver.find_element_by_xpath(
                    "//div[@class='frequency']//span[contains("
                    "text(),'" + bkp_type + "')]").click()

            if transport_mode:
                self.select_value_from_dropdown("transportMode", transport_mode)

            if self.check_if_entity_exists("xpath", "//input[@id='datastoreFreespaceCheckCB']"):
                if not ds_free_space_check:
                    self.checkbox_deselect("datastoreFreespaceCheckCB")
                else:
                    self.checkbox_select("datastoreFreespaceCheckCB")
                    if free_space:
                        self.fill_form_by_id("datastoreFreespaceRequired", free_space)

            self.submit_form()
            self.check_error_message()
        else:
            raise Exception("There is no option to edit the subclient properties")

    @PageService()
    def subclient_summary(self):
        """
        Provides the summary info of the subclient like last backup time, next backup time

        Returns:
            summary_info    (dict): info about the subclient

        """
        self.select_overview_tab()
        elements = self.driver.find_elements_by_xpath("//cv-tile-component[@data-title='Summary']"
                                                      "/div/div[3]/div/div[2]/ul/li")
        summary_info = {}
        for element in elements:
            key = element.find_element_by_xpath("./span[1]").text
            if key == "Data management":
                class_value = element.find_element_by_xpath("./span[2]/toggle-control/div"
                                                           ).get_attribute("class")
                if "enabled" in class_value:
                    value = "Enabled"
                else:
                    value = "Disabled"
            else:
                try:
                    value = element.find_element_by_xpath("./span[2]").text
                except Exception:
                    value = None
            summary_info[key] = value
        self.log.info("Summary of the subclient is %s", str(summary_info))
        return summary_info

    @PageService()
    def get_storage_policy(self):
        """
        Gets the storage policy or plan associated with the subclient

        Returns:
            storage_policy  (basestring):   the storage policy associated with the subclient

        """
        policies_tile = "//cv-entity-plan-policy-tiles//span[contains(., 'Policies')]/../.."
        if self.check_if_entity_exists("xpath", policies_tile):
            policy = self.driver.find_element_by_xpath(policies_tile + "//ul/li[1]/a").text
        else:
            plan_tile = "//cv-entity-plan-policy-tiles//span[contains(., 'Plan')]/../.."
            if not self.check_if_entity_exists("xpath", plan_tile):
                raise Exception("The Plan tile or Storage policy tile is not present")
            policy = self.driver.find_element_by_xpath(plan_tile + "/div/div[2]//a").text
            policy = re.split(" \( Server \)", policy, flags=re.IGNORECASE)[0].strip()
        return policy

    @PageService()
    def set_backup_validation(self, backup_validation_options):
        """
        Sets backup Validation details

        Args:
            backup_validation_options    (dict):     Credential details of the user

        Returns:

        """
        self.select_configuration_tab()
        panel_details = PanelInfo(self, self.props['label.backupValidation'])
        panel_details.enable_toggle(self.props['label.enableBackupValidation'])
        if panel_details.check_if_hyperlink_exists_on_tile('Edit'):
            panel_details.edit_tile()
        my_modal = ModalPanel(self)
        if backup_validation_options.get('recovery_target'):
            self.disable_toggle(toggle_id='liveMountUsingSourceEsx')
            drop_details = DropDown(self)
            drop_details.select_drop_down_values(values=[backup_validation_options['recovery_target']],
                                                 drop_down_id='lifeCyclePolicyModal_isteven-multi-select_#7626')
        else:
            self.enable_toggle(toggle_id='liveMountUsingSourceEsx')
        if backup_validation_options.get('schedule', False):
            self.schedule_backup_validation()
        else:
            self.disable_toggle(toggle_id='toggleBackupValidationSchedule')
        #custom validation
        self.driver.find_element_by_xpath(f"//span[contains(text(),"
                                          f"'{self.props['label.customApplication']}')]").click()
        # self.expand_accordin(self.props['label.customApplication'])
        if backup_validation_options.get('path_win') or backup_validation_options.get('path_unix'):
            if backup_validation_options.get('path_win'):
                if self.check_if_entity_exists('id', 'cv_windowsCustomScript_add'):
                    self.access_tile('cv_windowsCustomScript_add')
                else:
                    if self.check_if_entity_exists('id', 'cv_windowsCustomScript_edit'):
                        self.access_tab_from_dropdown('Edit', 4)
                    else:
                        raise Exception("Not able to add windows custom script")
                self.fill_form_by_id("uncNetworkPath", backup_validation_options.get('path_win'))
                if backup_validation_options.get('script_argument_win'):
                    self.fill_form_by_id("scriptArguments", backup_validation_options.get('script_argument_win'))
                my_modal = ModalPanel(self)
                my_modal.submit()
            if backup_validation_options.get('path_unix'):
                if self.check_if_entity_exists('id', 'cv_unixCustomScript_add'):
                    self.access_tile('cv_unixCustomScript_add')
                else:
                    if self.check_if_entity_exists('id', 'cv_windowsCustomScript_edit'):
                        self.access_tab_from_dropdown('Edit', 6)
                    else:
                        raise Exception("Not able to add unix custom script")
                self.fill_form_by_id("uncNetworkPath", backup_validation_options.get('path_unix'))
                if backup_validation_options.get('script_argument_unix'):
                    self.fill_form_by_id("scriptArguments", backup_validation_options.get('script_argument_unix'))
                my_modal = ModalPanel(self)
                my_modal.submit()
        else:
            self.log.info("Clearing custom scripts, if any")
            if self.check_if_entity_exists('id', 'cv_windowsCustomScript_edit'):
                self.access_tab_from_dropdown('Delete', 4)
            if self.check_if_entity_exists('id', 'cv_unixCustomScript_edit'):
                self.access_tab_from_dropdown('Delete', 6)
        # Guest credentials
        self.driver.find_element_by_xpath(f"//span[contains(text(),'{self.props['label.guestCredentials']}')]").click()
        # self.expand_accordin(self.props['label.guestCredentials'])
        self.select_radio('user')
        self.fill_form_by_id("uname", backup_validation_options.get('user', ''))
        self.fill_form_by_id("pass", backup_validation_options.get('password', ''))
        my_modal.submit()


[docs]    def schedule_backup_validation(self, only_schedule=False, after_time=None):
        """
        Scheduling backup validation
        Args:
            only_schedule           (bool): If we are only editing the schedule of the backup validation

            after_time:             (integer): set the schedule after number of seconds

        """
        # can be used schedules.py to enhance this
        if only_schedule:
            self.select_configuration_tab()
            panel_details = PanelInfo(self, self.props['label.backupValidation'])
            panel_details.enable_toggle(self.props['label.enableBackupValidation'])
            if panel_details.check_if_hyperlink_exists_on_tile('Edit'):
                panel_details.open_hyperlink_on_tile('Edit')
        self.enable_toggle(toggle_id='toggleBackupValidationSchedule')
        if self.check_if_entity_exists(
                "xpath",
                '//label[@class="col-xs-12 col-md-6 col-lg-6 padding-top-5"]'):
            _text = self.driver.find_element_by_xpath(
                '//label[@class="col-xs-12 col-md-6 col-lg-6 padding-top-5"]').text
            self.driver.find_element_by_xpath("//a[contains(text(),'"+_text+"')]").click()
        self.fill_form_by_id('name', self.driver.session_id)
        drop_details = DropDown(self)
        drop_details.select_drop_down_values(values=['Daily'],
                                             drop_down_id='addScheduleModal_isteven-multi-select_#5119')
        import time
        current_time = time.time()
        # adding 3 hours in the schedule
        if not after_time:
            after_time = 3 * 60 * 60
        schedule_time = current_time + after_time
        schedule_time = {
            'hour': time.strftime("%I", time.localtime(schedule_time)),
            'min': time.strftime("%M", time.localtime(schedule_time)),
            'meridiem': time.strftime("%p", time.localtime(schedule_time))
        }
        input_val = self.driver.find_element_by_xpath('//div[@id="scheduleTime"]//input[@placeholder="HH"]')
        input_val.clear()
        input_val.send_keys(schedule_time['hour'])
        input_val = self.driver.find_element_by_xpath('//div[@id="scheduleTime"]//input[@placeholder="MM"]')
        input_val.clear()
        input_val.send_keys(schedule_time['min'])
        input_val = self.driver.find_element_by_xpath(
            '//button[@class="btn btn-default text-center ng-binding"]').text
        if schedule_time['meridiem'] != input_val:
            self.driver.find_element_by_xpath(
                '//button[@class="btn btn-default text-center ng-binding"]').click()
        my_modal = ModalPanel(self)
        my_modal.submit()
        if only_schedule:
            my_modal = ModalPanel(self)
            my_modal.submit()
