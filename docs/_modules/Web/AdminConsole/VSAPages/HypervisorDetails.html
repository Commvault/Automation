    @adminconsole_handler()
    def edit_proxy(self, proxy_dict):
        """
        Adds / Removes a proxy to the given server

        Args:
            proxy_dict   (dict):    dictionary with proxy groups and proxies to be selected
                                        The other existing proxies will be removed.
            Sample value:   {'proxyGroup' : [proxy1,proxy2], 'proxy' : [proxy3,proxy4]}

        Raises:
            Exception:
                if the proxy list could not be edited

        """
        if self.check_if_entity_exists(
                "xpath", "//cv-tile-component[@data-ac-id='label_proxy-servers']/div/div[3]"
                         "/div/div[2]/div[2]//a[contains(text(),'Edit')]"):
            self.driver.find_element_by_xpath(
                "//cv-tile-component[@data-ac-id='label_proxy-servers']/div/div[3]/div/div[2]/"
                "div[2]//a[contains(text(),'Edit')]").click()
            self.log.info("Clicked on Edit Proxy")
            self.wait_for_completion()
            proxy_list = []
            # checking for proxy group
            if not self.driver.find_element_by_xpath(
                    "//form/div[1]/uib-accordion/div/div[1]/div[2]").is_displayed():
                self.driver.find_element_by_xpath(
                    "//form/div[1]/uib-accordion/div/div[1]/div[1]/h4/a/span/span").click()
            proxy_list.extend(self.driver.find_elements_by_xpath(
                "//form/div[1]/uib-accordion/div/div[1]/div[2]/div/ul"))
            # checking for proxies
            if not self.driver.find_element_by_xpath(
                    "//form/div[1]/uib-accordion/div/div[2]/div[2]").is_displayed():
                self.driver.find_element_by_xpath(
                    "//form/div[1]/uib-accordion/div/div[2]/div[1]/h4/a/span/span").click()
            proxy_list.extend(self.driver.find_elements_by_xpath(
                "//form/div[1]/uib-accordion/div/div[2]/div[2]/div/ul"))
            # unchecking all the elements
            for item in proxy_list:
                self.driver.execute_script("arguments[0].scrollIntoView();", item)
                if item.find_element_by_xpath("./li/input").is_selected():
                    item.find_element_by_xpath("./li/label[@class='ng-binding']").click()
                    self.wait_for_completion()
            for key in proxy_dict.keys():
                proxies = proxy_dict[key]
                for proxy in proxies:
                    if key == "proxyGroup":
                        if not self.driver.find_element_by_xpath(
                                "//label[contains(text(),'"+proxy+"')]/span[contains(text(), \
                                '(group)')]/../preceding-sibling::input").is_selected():
                            self.driver.find_element_by_xpath(
                                "//label[contains(text(),'"+proxy+"')]").click()
                            self.wait_for_completion()
                    elif key == "proxy":
                        elements = self.driver.find_elements_by_xpath(
                            "//label[text()='" + proxy + "']")
                        for elem in elements:
                            try:
                                elem.find_element_by_xpath("./span")
                            except:
                                if not elem.find_element_by_xpath(
                                        "./preceding-sibling::input").is_selected():
                                    elem.click()
                                    self.wait_for_completion()
                                    break
            self.driver.find_element_by_xpath(
                "//button[@type='submit']").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to edit proxy")

    @adminconsole_handler()
    def reorder_proxy(self, proxy_order):
        """
        Reorders the proxies for the server

        Args:
            proxy_order  (dict):    the order in which the proxies have to be arranged
                Sample value:   {'proxy1': 3, 'proxy2': 1, 'proxy3': 2}

        Raises:
            Exception:
                if the proxies could not be re-ordered

        """
        if self.check_if_entity_exists("link", "Reorder"):
            proxies = self.driver.find_elements_by_xpath(
                "//div[@id='scroll-box']/ul/li[@class='group ng-scope']")
            index = 1
            old_proxy_order = {}
            for proxy in proxies:
                old_proxy_order[proxy.find_element_by_xpath("./span/span").text] = index
                index += 1
            self.driver.find_element_by_link_text("Reorder").click()
            for key in proxy_order.keys():
                source = self.driver.find_element_by_xpath("//span[contains(text(),'" + key +
                                                           "')]")
                destination = self.driver.find_element_by_xpath("//div[@id='scroll-box']/ul/li"
                                                                "[" + str(proxy_order[key]) +
                                                                "]/span/span")
                ActionChains(self.driver).drag_and_drop(source, destination).perform()
                self.wait_for_completion()
            self.driver.find_element_by_link_text("Save").click()
            self.wait_for_completion()
            index = 1
            old_proxy_order = {}
            proxies = self.driver.find_elements_by_xpath(
                "//div[@id='scroll-box']/ul/li[@class='group ng-scope']")
            for proxy in proxies:
                old_proxy_order[proxy.find_element_by_xpath("./span/span").text] = index
                index += 1
            if old_proxy_order == proxy_order:
                self.log.info("Proxy reorder was successful")
            else:
                raise Exception("The proxy was not reordered properly.")

    @adminconsole_handler()
    def proxy_info(self):
        """
        Lists all the proxy of the server

        Returns:
            proxy   (list): list of all proxies that has been assigned to server

        """
        self.log.info("Getting the list of all selected proxies")
        elements = self.driver.find_elements_by_xpath(
            "//div[@id='scroll-box']/ul/li[@class='group ng-scope']")
        proxy = []
        for elem in elements:
            proxy.append(elem.find_element_by_xpath("./span/span").text)
        self.log.info(proxy)
        return proxy

    @adminconsole_handler()
    def action_backup(self, bkp_type, subclient):
        """
        Starts a backup of the given type for the specified collection

        Args:
            bkp_type    (basestring):    the backup level

            subclient   (basestring):   the name of the subclient to be backed up

        Returns:
            job_id  (int):  the backup job ID

        """
        self.log.info("Performing %s backup for subclient %s", bkp_type, subclient)
        self.search_for(subclient)
        self.select_action_item(subclient, "Back up now")
        self.submit_backup(bkp_type)
        return self.backup_job_id()

    @adminconsole_handler()
    def add_subclient(
            self,
            subclient_name,
            vm_content,
            plan=None
    ):
        """
        Adds a new subclient with the given VMs as content

        Args:
            subclient_name  (basestring)    :   the name of the subclient to be created

            vm_content      (dict)          :   the content to be added to the subclient

                Sample value:   {'Datastores and datastore clusters':[ds1,ds2],
                                 'Hosts and clusters':[h1,h2],
                                 'Tags and categories':[tag1,category1],
                                 'VMs and templates':[vm1,vm2],

            plan            (basestring)    :   the plan to be associated with the subclient

        self.select_hyperlink("Add VM group")
        self.fill_form_by_id("name", subclient_name)
        self.select_vm_from_browse_tree(vm_content)
        self.cv_single_select("Plan", plan)
        self.submit_form()
        self.check_error_message()

    @adminconsole_handler()
    def select_vm_from_browse_tree(self, vm_content):
        """
        Select content for subclient from the browse tree

        Args:
            vm_content  (dict):     the content to be selected
                Sample value:   {'Datastores and datastore clusters':[ds1,ds2],
                                 'Hosts and clusters':[h1,h2],
                                 'Tags and categories':[tag1,category1],
                                 'VMs and templates':[vm1,vm2],
                                 'Storage':[strg1, strg2],
                                 'By region': [region1, region2],
                                 'By zone': [zone1, zone2]
                                }

        Raises:
            Exception:
                if the content could not be selected properly

        """
        selected = []
        all_content = []
        for key, value in vm_content.items():
            all_content += value

            self.select_value_from_dropdown("vmBrowseView", key)

            if key == "VMs and templates":
                for _vm in value:
                    self.search_vm(_vm)
                    selected.append(_vm)
            else:
                for elem in value:
                    self.select_destination_host(elem)
                    selected.append(elem)

            difference = list(set(all_content) - set(selected))
            if difference:
                raise Exception("Exception during selection of content. Some of the content "
                                "could not be selected")

    @adminconsole_handler()
    def edit_hypervisor_details(self, vs_hostname, vs_username, vs_password,
                                vs_old_password=None):
        """
        This definition edits the server credentials

        Args:
            vs_hostname          (basestring):  the hostname of the server

            vs_username          (basestring):  the username / access key of the server

            vs_password          (basestring):  the password / secret key of the server

            vs_old_password      (basestring):  the old password of the server

        Raises:
            Exception:
                if the hypervisor details could not be edited

        """
        self.log.info("Editing server credentials")
        if self.check_if_entity_exists("link", "Edit hypervisor details"):
            self.select_hyperlink("Edit hypervisor details")
            if vs_hostname:
                self.fill_form_by_id("vsHostname", vs_hostname)

            if vs_username:
                if self.check_if_entity_exists("id", "vsUserName"):
                    self.fill_form_by_id("vsUserName", vs_username)
                elif self.check_if_entity_exists("name", "vsAccessKey"):
                    self.fill_form_by_id("vsAccessKey", vs_username)

            if vs_old_password:
                self.fill_form_by_id("vsCurrPassword", vs_old_password)

            if vs_password:
                self.fill_form_by_id("vsPassword", vs_password)
                if self.check_if_entity_exists("id", "confirmServerPassword"):
                    self.fill_form_by_id("confirmServerPassword", vs_password)

            self.submit_form()
            self.check_error_message()
        else:
            raise Exception("The edit hypervisor details is not present")

    @adminconsole_handler()
    def server_settings(self, fbr=None, vcloud_hostname=None,
                        vcloud_username=None, vcloud_password=None):
        """
        Sets the FBR MA and other settings like vCloud

        Args:
            fbr                  (basestring):  the name of the FBR MA to be set

            vcloud_hostname      (basestring):  the host name of the vcloud director

            vcloud_username      (basestring):  the vcloud user name

            vcloud_password      (basestring):  the vcloud password

        """
        if fbr:
            self.driver.find_element_by_xpath("//cv-tile-component[@data-title='Settings']"
                                              "/div/div[3]/div/div/ul/li[2]/span[2]/a/i").click()
            self.wait_for_completion()
            self.select_value_from_dropdown("fbrMAUnix", fbr)
            self.submit_form()

        if vcloud_hostname:
            self.driver.find_element_by_xpath("//cv-tile-component[@data-title='Settings']"
                                              "/div/div[3]/div/div/ul/li[1]/span[2]/a/i").click()
            self.wait_for_completion()
            self.fill_form_by_id("vcloudHostName", vcloud_hostname)
            self.fill_form_by_id("vcloudUserName", vcloud_username)
            self.fill_form_by_id("vcloudPassword", vcloud_password)
            self.fill_form_by_id("confirmVcloudPassword", vcloud_password)
            self.submit_form()

    @adminconsole_handler()
    def server_analytics(self, enable_analysis=True, plan=None):
        """
        Enables the data analysis on the server

        Args:
            enable_analysis  (bool):        enable / disable data analysis on the server

            plan             (basestring):  the plan to collect analysis info

        Raises:
            Exception:
                if a plan is not specified or
                if analytics could not be enabled / disabled

        """
        self.tile_select_hyperlink("Analytics", "Edit")
        self.wait_for_completion()
        if enable_analysis:
            self.checkbox_select("enableAnalytics")
            if not plan:
                raise Exception("Please specify a plan")
            self.select_value_from_dropdown("plan", plan)
        else:
            self.checkbox_deselect("enableAnalytics")
        self.driver.find_element_by_xpath("//*[@id='editAnalyticsForm']/div[3]/button[2]").click()
        self.wait_for_completion()
