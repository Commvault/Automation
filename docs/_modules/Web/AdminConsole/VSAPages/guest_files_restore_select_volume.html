from Web.Common.handlers import adminconsole_handler
from Web.AdminConsole.AdminConsolePages.AdminPage import AdminPage
from Web.AdminConsole.VSAPages.restore_select_vm import RestoreSelectVM



[docs]class GuestFilesRestoreSelectVolume(AdminPage):
    """
    This class contains methods for submitting virtual files machine restores.
    """
    def __init__(self, driver):
        """
        Init method to create objects of classes used in the file.

        Args:
            driver      (object)   :  the browser object
        """
        super(GuestFilesRestoreSelectVolume, self).__init__(driver)
        self.res_select_vm_obj = RestoreSelectVM(driver)

    @adminconsole_handler()
    def select_volume(self, vm_name, vol):
        """
        Opens the drives of the backed up VM to browse content

        Args:
            vm_name (basestring): vm name to be selected
            vol (basestring)    : volume to be selected
        """
        if not self.check_if_entity_exists(
                "xpath", "//div[1]/ul/li[1]/span[contains(text(),'" + vm_name + "')]"):
            # //*[@id="wrapper"]/div[2]/div[2]/div/div[2]/span/div[2]/div/div[1]/ul/li/span
            self.res_select_vm_obj.select_vm(vm_name)
        self.log.info("Opening volume %s", vol)
        if "/" in vol:
            folder_to_browse = vol.split("/")[1:]
            if not folder_to_browse[0]:
                return
        else:
            folder_to_browse = [vol]
        for folder in folder_to_browse:
            while True:
                if not self.check_if_entity_exists("xpath", "//a[@val='" + folder + "']"):
                    if self.cv_table_next_button_exists():
                        if self.driver.find_element_by_xpath(
                                "//button[@ng-disabled='cantPageForward()']").is_enabled():
                            self.cv_table_click_next_button()
                            continue
                    else:
                        self.log.error("{} not found".format(vol))
                        break
                self.driver.find_element_by_xpath("//a[@val='" + folder + "']").click()
                self.wait_for_completion()
                break
