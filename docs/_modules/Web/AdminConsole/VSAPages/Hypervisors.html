    @adminconsole_handler()
    def add_hypervisor(
            self,
            vendor,
            server_name,
            proxy_list,
            host_name=None,
            vs_password=None,
            vs_username=None,
            access_key=None,
            secret_key=None,
            domain=None,
            project_id=None,
            service_account_id=None,
            p12_key=None,
            private_key_password=None,
            subscription_id=None,
            thumbprint=None,
            tenant_id=None,
            application_id=None,
            application_password=None
    ):
        """
        Adds a new server with the specified vendors and proxies.

        Args:
            project_id           (basestring):  the project ID of google cloud

            service_account_id   (basestring):  the service account ID of google cloud

            p12_key              (basestring):  the P12 Key of Google Cloud

            private_key_password (basestring):  the private key password of Google Cloud

            subscription_id      (basestring):  the subscription ID of Azure account

            thumbprint           (basestring):  the thumbprint of the Azure account

            tenant_id            (basestring):  the tenant ID of the Azure account

            application_id       (basestring):  the application ID of the Azure account

            application_password (basestring):  the application password of the Azure account

            domain               (basestring):  the domain name of the openstack server

            access_key           (basestring):  the access key required for Amazon client

            secret_key           (basestring):  the secret key required for Amazon client

            vendor               (basestring):  the vendor type of server to be added

            server_name          (basestring):  the name of the server to be added

            host_name            (basestring):  the hostname of the server to be added

            vs_username          (basestring):  the username of the server to be added

            proxy_list           (list):        the list of proxies to be associated with server

            vs_password          (basestring):  the password of the server to be added

        Raises:
            Exception:
                if the hypervisor could not be created

        """
        self.log.info("Adding a %s server %s as %s", vendor, host_name, server_name)
        self.driver.execute_script("window.scrollTo(0,0)")
        if self.check_if_entity_exists("id", "tour_add_server"):
            self.driver.find_element_by_id(
                "tour_add_server").click()
            self.wait_for_completion()
            self.select_value_from_dropdown("vendorType", vendor)
            if vendor in ["VMware", "Nutanix AHV", "OpenStack", "Oracle Cloud Classic",
                          "Oracle VM"]:
                self.fill_form_by_id("hostname", host_name)
                self.fill_form_by_id("serverName", server_name)
                self.driver.find_element_by_xpath("//label[@for='vsUserName']/input").send_keys(
                    vs_username)
                self.driver.find_element_by_xpath("//label[@for='vsPassword']/input").send_keys(
                    vs_password)
                if vendor == "OpenStack":
                    if domain:
                        self.driver.find_element_by_xpath("//label[@for='domainName']/input"
                                                         ).send_keys(domain)
            elif vendor == "Hyper-V":
                self.fill_form_by_id("hostname", host_name)
                self.fill_form_by_id("serverName", server_name)
                self.driver.find_element_by_xpath("//label[@for='vsUserName']/input").send_keys(
                    vs_username)
                self.driver.find_element_by_xpath("//label[@for='vsPassword']/input").send_keys(
                    vs_password)
                self.driver.find_element_by_xpath(
                    "//form/div[2]/button[2]").click()
                self.wait_for_completion()
                # check if at all the discovery is successful.
                if self.check_if_entity_exists(
                        "xpath", "//div[@class='serverMessage error']"):
                    err_msg = self.driver.find_element_by_xpath(
                        "//div[@class='serverMessage error']").text
                    raise Exception(err_msg)
            elif vendor in ["Amazon", "Alibaba Cloud"]:
                self.fill_form_by_id("serverName", server_name)
                self.driver.find_element_by_xpath("//label[@for='accessKey']/input"
                                                 ).send_keys(access_key)
                self.driver.find_element_by_xpath("//label[@for='secretKey']/input"
                                                 ).send_keys(secret_key)
            elif vendor == "Google Cloud Platform":
                self.fill_form_by_id("clientName", server_name)
                self.fill_form_by_id("projectId", project_id)
                self.fill_form_by_id("serviceAccountID", service_account_id)
                self.driver.find_element_by_xpath("//label[@for='p12KeyFileName']/input"
                                                 ).send_keys(p12_key)
                self.driver.find_element_by_xpath("//label[@for='privateKeysPassword']/input"
                                                 ).send_keys(private_key_password)
            elif vendor == "Azure - Classic":
                self.fill_form_by_id("serverName", server_name)
                self.fill_form_by_id("subscriptionId", subscription_id)
                self.driver.find_element_by_xpath("//label[@for='thumbprint']/input"
                                                 ).send_keys(thumbprint)
            elif vendor in ['Azure - Resource Manager', 'Azure - stack']:
                self.fill_form_by_id("serverName", server_name)
                self.fill_form_by_id("subscriptionId", subscription_id)
                self.fill_form_by_id("tenantId", tenant_id)
                self.fill_form_by_id("applicationId", application_id)
                self.driver.find_element_by_xpath("//label[@for='applicationPassword']/input"
                                                 ).send_keys(application_password)
            if self.check_if_entity_exists("id", "proxy"):
                if not self.check_if_entity_exists(
                        "xpath", "//li[@class='select2-search-choice']"):
                    for proxy in proxy_list:
                        self.log.info("Adding proxy %s to server", proxy)
                        self.select_value_from_dropdown("proxy", proxy)
            self.submit_form()
            if self.driver.find_element_by_link_text("Skip for now"):
                self.driver.find_element_by_link_text("Skip for now").click()
                self.wait_for_completion()
            if self.check_if_entity_exists(
                    "xpath", "//div/div[2]/p[@class='serverMessage error']"):
                raise Exception(self.driver.find_element_by_xpath(
                    "//div[2]/p[@class='serverMessage error']").text)
            else:
                if self.check_if_entity_exists("xpath",
                                               "//a[@data-ng-click = 'skipAddCollection()'"):
                    self.driver.find_element_by_xpath(
                        "//a[@data-ng-click = 'skipAddCollection()'").click()
                    self.wait_for_completion()
            self.log.info("Done with Add hypervisor")
        else:
            raise Exception("There is no Add hypervisor option")

    @adminconsole_handler()
    def hypervisor_restore(self, hypervisor_name, subclient_name=None):
        """
        Restores the given subclient in the hypervisor

        Args:
            hypervisor_name      (basestring):  the name of the hypervisor whose subclient is to
                                                be restored

            subclient_name       (basestring):  the name of the subclient to be restored

        Raises:
            Exception:
                if there is no subclient with the given name
                if restore could not be selected
                if the hypervisor was not backed up

        """
        self.log.info("Restoring subclient %s in hypervisor %s", subclient_name, hypervisor_name)
        self.search_for(hypervisor_name)
        url = self.current_url()
        self.select_by_col_name(hypervisor_name, "Restore", "Restore")
        if self.current_url() == url:
            if self.check_if_entity_exists("id", subclient_name):
                self.driver.find_element_by_id(subclient_name).click()
                self.wait_for_completion()
            else:
                raise Exception("There is no subclient named {0}".format(subclient_name))
            self.driver.find_element_by_xpath("//span[@text='Select content']").click()
            self.wait_for_completion()
        else:
            self.log.info("There was only backed up subclient in the hypervisor")
