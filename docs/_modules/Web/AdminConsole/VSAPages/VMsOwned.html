    @adminconsole_handler()
    def open_vm(self, vm_name):
        """
        Opens VM with the given name
        """
        self.search_for(vm_name)
        if self.check_if_entity_exists("link", vm_name):
            self.driver.find_element_by_link_text(vm_name).click()
            self.wait_for_completion()
        else:
            raise Exception("There is no VM named {0} owned by this user".format(vm_name))

    @adminconsole_handler()
    def open_hypervisor_of_vm(self, vm_name):
        """Opens the hypervisor corresponding to the VM name
            vm_name  :   a string, the name of the VM whose hypervisor has to be opened
        """
        self.search_for(vm_name)
        if self.check_if_entity_exists("link", vm_name):
            self.driver.find_element_by_xpath("//a[@title='{0}']/../../div[2]/a".format(
                vm_name
            )).click()
            self.wait_for_completion()
        else:
            raise Exception("There is no VM named {0} owned by this user".format(vm_name))

    @adminconsole_handler()
    def list_backup_vms(self):
        """
        Lists all the backup VM details
        """
        vms = {}
        while True:
            elements = self.driver.find_elements_by_xpath("//div[@class='ui-grid-canvas']/div")
            for elem in elements:
                vm_details = []
                key = elem.find_element_by_xpath("./div/div[1]/a").text
                for index in range(2, 9):
                    if index == 2:
                        vm_details.append(elem.find_element_by_xpath(
                            "./div/div["+str(index)+"]/a").text)
                        continue
                    elif index == 6:
                        vm_details.append(elem.find_element_by_xpath(
                            "./div/div["+str(index)+"]/div").text)
                        continue
                    vm_details.append(elem.find_element_by_xpath(
                        "./div/div["+str(index)+"]/span").text)
                vms[key] = vm_details
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        self.log.info("The list of backed up VMs owned by the user is "+str(vms))
        return vms

    @adminconsole_handler()
    def action_remove_vm(self, vm_name):
        """
        Removes the VM from the list of backed up VMs
        :param vm_name  (str)   --  the name of the VM to remove
        """
        self.search_for(vm_name)
        if self.check_if_entity_exists("link", vm_name):
            self.driver.find_element_by_xpath(
                "//a[@title='{0}']/../../div[9]/div/a/span".format(vm_name)).click()
            self.wait_for_completion()
            if self.check_if_entity_exists("link", "Remove"):
                self.driver.find_element_by_xpath(
                    "//a[@title='{0}']/../../div[9]/div/ul/li[4]/a".format(vm_name)).click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath("//div[3]/button[2]").click()
                self.wait_for_completion()
                if self.check_if_entity_exists(
                        "xpath",
                        "//div[@class='modal-body text-danger ng-binding ng-scope']"):
                    exp = self.driver.find_element_by_xpath("//div[@class='"
                                                            "modal-body text-danger ng-binding "
                                                            "ng-scope']").text
                    if self.check_if_entity_exists("xpath", "//button[@ng-click='close()']"):
                        self.driver.find_element_by_xpath("//button[@ng-click='close()']").click()
                        self.wait_for_completion()
                    raise Exception(exp)
            else:
                raise Exception("There is no option to remove the VM. Check if the logged "
                                "in user has sufficient permissions")
        else:
            raise Exception("The specified VM was not found. Check if the VM was backed up"
                            " and that the logged in user has sufficient permissions to view"
                            " that VM")

    @adminconsole_handler()
    def action_vm_backup(self, vm_name):
        """
        Backups the given VM
        :param vm_name  (str)   --  the name of the VM to backup
        :return job_id  (int)   --  the backup job ID
        """
        self.search_for(vm_name)
        if self.check_if_entity_exists("link", vm_name):
            self.driver.find_element_by_xpath(
                "//a[@title='{0}']/../../div[9]/div/a/span".format(vm_name)).click()
            self.wait_for_completion()
            if self.check_if_entity_exists("link", "Back up now"):
                self.driver.find_element_by_xpath(
                    "//a[@title='{0}']/../../div[9]/div/ul/li[1]/a".format(vm_name)).click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath(
                    "//div/div/div[2]/div/button[ \
                    @class='btn btn-primary cvBusyOnAjax']").click()
                self.wait_for_completion()
                job_id = None
                if self.driver.find_element_by_xpath(
                        "//div[@class='setup-content backup-options']/span[1]").is_displayed():
                    # Job started so the tag that displays the Job ID stays Valid
                    job_text = self.driver.find_element_by_xpath(
                        "//div[contains(text(),'Backup started for')]").text
                    # job_text=element1.text
                    # print job_text
                    split_job_id = job_text.split("Job")[-1].split(".")[0].strip()
                    job_id = int(split_job_id)
                    self.log.info("Backup job " + str(job_id) + " has started")
                    self.driver.find_element_by_xpath(
                        "//div[2]/div/button[contains(text(),'OK')]").click()
                    self.wait_for_completion()
                if not job_id:
                    raise Exception("The Job ID could not be obtained")
                return job_id
            else:
                raise Exception("There is no option to backup the given VM. Check if the"
                                " logged in user has sufficient permissions")
        else:
            raise Exception("The specified VM was not found. Check if the VM was backed up "
                            "and that the logged in user has sufficient permissions to "
                            "view that VM")
