"""
This module provides all the methods that can be done of the Inventory Manager page.


Classes:

    InventoryManager() ---> GovernanceApps() ---> AdminPage() ---> login_page --->
    AdminConsoleBase() ---> object()


InventoryManager  --  This class contains all the methods for action in Inventory
    Manager page and is inherited by other classes to perform GDPR related actions

    Functions:

    add_inventory()          --  adds an inventory
    search_for_inventory()    -- Searches for an inventory
    _click_inventory_action()    -- Clicks on an inventory action button
    navigate_to_inventory_details()    -- Navigates to inventory manager details page
    delete_inventory() -- Deletes a specific inventory
"""
from Web.AdminConsole.Components.table import Table
from Web.Common.page_object import WebAction
from Web.AdminConsole.GovernanceAppsPages.GovernanceApps import GovernanceApps



[docs]class InventoryManager(GovernanceApps):
    """
     This class contains all the methods for action in Inventory Manager page
    """

    def __init__(self, driver):
        super().__init__(driver)
        self.__table = Table(self)

    @WebAction()
    def add_inventory(self, inventory_name, index_server):
        """
        Adds an inventory

            :param inventory_name    (str)   --  Inventory name to be added
            :param index_server    (str)  -- Index Server to be selected

            Raise:
                Exception if inventory addition failed
        """
        self.log.info("Clicking on Add inventory")
        self.driver.find_element_by_xpath(
            "//*[text()='Add inventory']").click()
        self.wait_for_completion()
        self.log.info("Entering Inventory name")
        self.fill_form_by_id("name", inventory_name)
        self.log.info("Selecting Index Server")
        self.select_value_from_dropdown("analyticsEngine", index_server)
        self.log.info("Clicking on Save")
        self.driver.find_element_by_xpath(
            "//button[@class='btn btn-primary cvBusyOnAjax']").click()
        self.wait_for_completion()
        if self.check_if_entity_exists(
                "xpath", "//*[@class='serverMessage error']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//*[@class='serverMessage error']").text)

    @WebAction()
    def search_for_inventory(self, inventory_name):
        """
        Searches for an inventory

            Args:
                inventory_name (str)  - Inventory name to be searched for

            Returns True/False based on the presence of the Inventory
        """
        if self.check_if_entity_exists("id", "search-field"):
            self.log.info("Searching for Inventory: '%s'" % inventory_name)
            self.__table.search_for(inventory_name)
            if self.check_if_entity_exists(
                    "xpath", "//a[text()='%s']" %
                    inventory_name):
                self.log.info("Inventory found")
                return True
            self.log.info("Inventory not found")
            return False
        self.log.info("No Inventories found to search for")
        return False

    @WebAction()
    def _click_inventory_action(self, inventory_name):
        """
        Clicks on an inventory action button

            Args:
                inventory_name (str)  - Inventory name action to be clicked
        """
        self.log.info('Clicking the action button of %s' % inventory_name)
        self.driver.find_element_by_xpath(
            "//a[text()='%s']/../../div[5]/div" % inventory_name).click()
        self.wait_for_completion()

    @WebAction()
    def navigate_to_inventory_details(self, inventory_name):
        """
        Navigates to inventory manager details page

            Args:
                inventory_name (str)  - Inventory name details to be navigated

        """
        if self.search_for_inventory(inventory_name):
            self._click_inventory_action(inventory_name)
            self.log.info('Clicking on Details option')
            self.driver.find_element_by_xpath(
                "//a[text()='%s']/../..//a[text()='Details']" %
                inventory_name).click()
            self.wait_for_completion()
        else:
            raise Exception(
                "The inventory {0} is not present".format(inventory_name))

    @WebAction()
    def delete_inventory(self, inventory_name):
        """
        Adds an inventory

            Args:
                inventory_name (str)  - Inventory name to be deleted

            Raise:
                Exception if inventory deletion failed
        """
        if self.search_for_inventory(inventory_name):
            self._click_inventory_action(inventory_name)
            self.log.info('Clicking on Delete option')
            self.driver.find_element_by_xpath(
                "//a[text()='%s']/../..//a[text()='Delete']" %
                inventory_name).click()
            self.wait_for_completion()
            self.log.info('Clicking on Delete Confirmation')
            self.driver.find_element_by_xpath(
                "//div[@class='modal-content']//button[text()='Yes']").click()
            self.wait_for_completion()
            if self.check_if_entity_exists(
                    "xpath", "//div[4]/div/div[@class='growl-item alert ng-scope "
                             "alert-error alert-danger']"):
                raise Exception(
                    self.driver.find_element_by_xpath(
                        "//div[4]/div/div[@class='growl-item alert ng-scope alert-error "
                        "alert-danger']/div/div").text)
        else:
            raise Exception(
                "The inventory {0} is not present".format(inventory_name))
