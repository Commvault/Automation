    @adminconsole_handler()
    def get_data_source_name(self):
        """
        Returns data source name from the admin page
        """
        self.log.info("Obtaining data source name")
        data_source_name = str(self.driver.find_element_by_xpath(
            '//*[@id="customDatasourceReport"]/h1/span').text)
        self.log.info("data source name obtained is: %s" % data_source_name)
        return data_source_name

    @adminconsole_handler()
    def get_file_paths(self):
        """
        Returns the list of file paths shown on the current page

            Return:
                List of file paths
        """
        self.log.info("Obtaining all the file paths shown")
        file_paths_list = []
        rows = self.driver.find_elements_by_xpath(
            "//div[@class='ui-grid-canvas']/div")
        total_rows = int(len(rows) / 2)
        for row_id in range(1, total_rows + 1):
            file_paths_list.append(str(self.driver.find_element_by_xpath
                                       ("//div[@class='ui-grid-canvas']\
                                       /div[%d]//div[2]/div" % row_id).text))
        self.log.info('List of file paths obtained are: %s' % file_paths_list)
        return file_paths_list

    @adminconsole_handler()
    def get_file_path_row_id(self, file_path):
        """
        Returns the file path row id

            Args:
                file_path  - File Path of the row id

            Raise:
                Exception if File Path not found
        """
        self.log.info("Obtaining the row id for File Path: %s" % file_path)
        rows = self.get_file_paths()
        for row_id in range(1, len(rows) + 1):
            if file_path.lower() == str(self.driver.find_element_by_xpath
                                        ("//div[@class='ui-grid-canvas']\
                                        /div[%d]/div/div[2]/div"
                                         % row_id).text).lower():
                self.log.info("Row id obtained is: %d" % row_id)
                return row_id
        raise Exception("There is no file path: %s" % file_path)

    @adminconsole_handler()
    def file_path_checked_status(self, file_path):
        """
        Returns True/False based on the file path check box

            Return (Bool):
                True/False
        """
        row_id = self.get_file_path_row_id(file_path)
        try:
            self.log.info(
                "Checking if check box checked for File Path: %s" %
                file_path)
            self.driver.find_element_by_xpath(
                '//div[@class="ui-grid-canvas"]/div[%d]//div[contains\
                (@class, "ui-grid-row-selected")]' % row_id)
            self.log.info("File Path check box checked")
            return True
        except Exception:
            self.log.info("File Path check box not checked")
            return False

    @adminconsole_handler()
    def select_file(self, file_path):
        """
        Selects the file for a given file path
        """
        self.log.info("Selecting File: %s" % file_path)
        row_id = self.get_file_path_row_id(file_path)
        self.driver.find_element_by_xpath(
            '//div[@class="ui-grid-canvas"]/div[%d]//a' % row_id).click()
        self.wait_for_completion()

    @adminconsole_handler()
    def search_file(self, file_name, folder_name):
        """
        Search based on the file and folder name
        """
        search_string = '(FolderName_idx:"{0}") AND (FileName:"{1}")'.format(
            folder_name, file_name)
        self.driver.find_element_by_xpath(
            '//*[@id="reportViewerArea"]//cvsearchbar//input').clear()
        self.driver.find_element_by_xpath(
            '//*[@id="reportViewerArea"]//cvsearchbar//input').send_keys(search_string)
        self.driver.find_element_by_xpath(
            '//*[@id="reportViewerArea"]//cvsearchbar//button').click()
        self.wait_for_completion()

    @adminconsole_handler()
    def expand_entity(self, entity_name):
        """
        Expands the given entity

            Args:
                entity_name (str)  - Name of the entity
        """
        xpath = "//div[@class='modal-content']//span[text()='%s']\
        //following-sibling::i[contains(@class, 'down')]" % entity_name
        if not self.check_if_entity_exists('xpath', xpath):
            self.log.info("Expanding Entity: %s" % entity_name)
            self.driver.find_element_by_xpath(
                "//div[@class='modal-content']//span[text()='%s']" %
                entity_name) .click()
            self.wait_for_completion()
        else:
            self.log.info("Entity: %s already expanded" % entity_name)

    @adminconsole_handler()
    def get_entities(self):
        """
        Returns a Dictionary for all entities with its corresponding list of values

            Return (Dict):
                Dictionary with list of entity values
        """
        entities_list = []
        entities_dict = {}
        self.log.info("Obtaining all the Entity Names")
        elems = self.driver.find_elements_by_xpath(
            "//div[@class='modal-content']\
            //div[@role='tablist']//div//a/span/span[1]")
        for index in range(1, len(elems) + 1):
            entities_list.append(
                self.driver.find_element_by_xpath(
                    "//div[@class='modal-content']//div[@role='tablist']\
                    //div[%d]/div/h4/a/span/span[1]" %
                    index).text)
        self.log.info("Entity Names Obtained are: %s" % entities_list)
        for entity_name in entities_list:
            self.expand_entity(entity_name)
            self.log.info(
                "Obtaining Entity Values for Entity: %s" %
                entity_name)
            elems = self.driver.find_elements_by_xpath(
                "//div[@class='modal-content']//span[text()='%s']/ancestor::node()[5]\
                //div[contains(@class,'file-list')]/form/div" % entity_name)
            entity_values_list = []
            for index in range(1, len(elems) + 1):
                entity_values_list.append(
                    self.driver.find_element_by_xpath(
                        "//div[@class='modal-content']//span[text()='%s']\
                        /ancestor::node()[5]//div[contains(@class,'file-list')]\
                        /form/div[%d]/label" % (entity_name, index)).text)
            entity_values_list = sorted(entity_values_list, key=str.lower)
            entities_dict[entity_name] = entity_values_list
        self.log.info(
            "Entities Dictionary obtained is: '{0}'".format(entities_dict))
        return entities_dict

    @adminconsole_handler()
    def get_file_name(self):
        """
        Returns the file name

            Return (str):
                Name of the file
        """
        self.log.info("Obtaining File Name")
        file_name = self.driver.find_element_by_xpath(
            "//div[@class='modal-content']//div/span").text
        self.log.info("File Name Obtained is: %s" % file_name)
        return file_name

    @adminconsole_handler()
    def select_next_file(self):
        """
        Selects the next file to review by clicking on the right arrow if exists

            Return (Bool):
                True/False based on the status
        """
        xpath = "//div[@class='modal-content']//button[contains(@data-ng-click,'Next')]"
        try:
            self.log.info("Clicking on the next button")
            self.driver.find_element_by_xpath(xpath).click()
            self.wait_for_completion()
            return True
        except Exception:
            self.log.info("Next button not found")
            return False

    @adminconsole_handler()
    def close_file_preview(self):
        """
        Closes the file being previewed
        """
        self.log.info("Closing the file preview")
        self.driver.find_element_by_xpath(
            "//a[@class='modal__close-btn'][2]").click()
        self.wait_for_completion()
