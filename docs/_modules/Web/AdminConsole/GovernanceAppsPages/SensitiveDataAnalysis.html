    @adminconsole_handler()
    def add_project(self, project_name, plan_name, inventory_name):
        """
        Adds a project

            Args:
                project_name (str)  - Project name to be added
                plan_name (str)  - Plan name to be selected
                inventory_name (str)  - Inventory name to be selected

            Raise:
                Exception if project addition failed
        """
        self.log.info("Clicking on Add project")
        self.driver.find_element_by_xpath("//a[text()='Add project']").click()
        self.wait_for_completion()
        self.log.info("Entering Project name")
        self.fill_form_by_id("name", project_name)
        self.log.info("Selecting Plan: %s" % plan_name)
        self.select_value_from_dropdown("ediscoveryPlan", plan_name)
        self.log.info("Selecting Inventory: %s" % inventory_name)
        self.select_value_from_dropdown("ediscoveryInventory", inventory_name)
        self.log.info("Clicking on Save")
        self.driver.find_element_by_xpath(
            "//button[@class='btn btn-primary cvBusyOnAjax']").click()
        self.wait_for_completion()
        if self.check_if_entity_exists(
                "xpath", "//*[@class='serverMessage error']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//*[@class='serverMessage error']").text)

    @adminconsole_handler()
    def search_for_project(self, project_name):
        """
        Searches for a given project

            Args:
                project_name (str)  - Project name to be searched for

            Returns True/False based on the presence of the Project
        """
        self.log.info("Searching for Project: '%s'" % project_name)
        self.search_for(project_name)
        if self.check_if_entity_exists(
            "xpath", "//a[text()='%s']" %
                project_name):
            self.log.info("Project found")
            return True
        else:
            self.log.info("Project not found")
            return False

    @adminconsole_handler()
    def _click_project_action(self, project_name):
        """
        Clicks on project action button

            Args:
                project_name (str)  - project name action to be clicked
        """
        self.log.info('Clicking the action button of %s' % project_name)
        self.driver.find_element_by_xpath(
            "//a[text()='%s']/../../div[5]/div" % project_name).click()
        self.wait_for_completion()

    @adminconsole_handler()
    def navigate_to_project_details(self, project_name):
        """
        Navigates to project details page

            Args:
                project_name (str)  - project name details to be navigated

        """
        if self.search_for_project(project_name):
            self._click_project_action(project_name)
            self.log.info('Clicking on Details option')
            self.driver.find_element_by_xpath(
                "//a[text()='%s']/../..//a[text()='Details']" %
                project_name).click()
            self.wait_for_completion()
        else:
            raise Exception(
                "The Project {0} is not present".format(project_name))

    @adminconsole_handler()
    def delete_project(self, project_name):
        """
        Deletes a project

            Args:
                project_name (str)  - Project name to be deleted

            Raise:
                Exception if project deletion failed
        """
        if self.search_for_project(project_name):
            self._click_project_action(project_name)
            self.log.info('Clicking on Delete option')
            self.driver.find_element_by_xpath(
                "//a[text()='%s']/../..//a[text()='Delete']" %
                project_name).click()
            self.wait_for_completion()
            self.log.info('Clicking on Delete Confirmation')
            self.driver.find_element_by_xpath(
                "//div[@class='modal-content']//button[text()='Yes']").click()
            self.wait_for_completion()
            if self.check_if_entity_exists(
                    "xpath", "//div[4]/div/div[@class='growl-item alert ng-scope "
                             "alert-error alert-danger']"):
                raise Exception(
                    self.driver.find_element_by_xpath(
                        "//div[4]/div/div[@class='growl-item alert ng-scope alert-error "
                        "alert-danger']/div/div").text)
        else:
            raise Exception(
                "The project {0} is not present".format(project_name))
