    @PageService()
    def select_user(self, fileowners):
        """
        filter_ownerextension
            on entitlement management page, set fileowners under User filter
        Args:
                fileowner (list)  - list of file owners/users


        """

        self.log.info("find user from user lit")
        elemtexts = []
        if self.driver.find_elements_by_class_name("user-item"):
            self.log.info("found user from user list")

            elems = self.driver.find_elements_by_class_name("user-item")
            j = len(elems)
            for i in range(1, j + 1):
                csspath = ".user-item:nth-child(" + str(i) + ")"
                self.log.info(csspath)
                elemtext = self.driver.find_element_by_css_selector(
                    csspath).text
                self.log.info(elemtext)
                for fileowner in fileowners:
                    if elemtext.lower().find(fileowner.lower()) == -1:
                        continue
                    else:
                        elemtexts.append(elemtext)
                        self.driver.find_element_by_css_selector(
                            csspath).click()
                        self.wait_for_completion()
        else:
            raise Exception("failed to find user from user list")

        return elemtexts

    @PageService()
    def select_extension(self, extensions):
        """
        filter_ownerextension
            on entitlement management page, set extensinons under Extensions filter
        Args:
                externsion (list)  - list of file externsions


        """

        self.log.info("disable extension toggle")
        if self.driver.find_element_by_xpath(
                ".//*[@id='extensionsToggle']/div/div[2]"):
            self.driver.find_element_by_xpath(
                ".//*[@id='extensionsToggle']/div/div[2]").click()
            self.wait_for_completion()
        else:
            raise Exception(
                "failed to find extension toggle under filters -> externsions")

        if self.driver.find_elements_by_class_name("user-item"):
            self.log.info("found user from user list")
            elems = self.driver.find_elements_by_class_name("extension-item")
            # search through the extension items list and find the one for the
            # expected file and select it
            j = len(elems)
            for i in range(1, j + 1):
                csspath = ".extension-item:nth-child(" + \
                    str(i) + ") > .ng-binding"
                elemtext = self.driver.find_element_by_css_selector(
                    csspath).text
                self.log.info(elemtext)
                for extension in extensions:
                    self.checkbox_select(extension)
                    self.wait_for_completion()

        else:
            raise Exception("could not find extension from extension list")

    @PageService()
    def expand_tree_node(self, src_path, filename):
        """
        expand_tree_node

            Args:
                src_path (str)  - expand the folder path under file path tree node
                filename (str)  - filename

        """

        src = src_path.split('\\')
        src = [x for x in src if x != ""]
        src.append(filename)
        self.log.info(src[-1])

        for fpath in src:
            self.log.info("----------------")
            self.log.info(fpath)

            if self.check_if_entity_exists("css", "span.loadMoreLink"):
                by_css = By.CSS_SELECTOR
                csstr = "span.loadMoreLink"
                WebDriverWait(
                    self.driver, 30).until(
                        EC.visibility_of_element_located(
                            (by_css, csstr)))
                while True:
                    try:
                        WebDriverWait(
                            self.driver, 30).until(
                                EC.visibility_of_element_located(
                                    (by_css, csstr))).click()
                        print("Load more clicked")
                    except Exception:
                        break

            self.log.info("check folder/files records")
            if self.driver.find_elements_by_class_name("tree-node"):
                self.log.info("found the folder/files tree")
                elems = self.driver.find_elements_by_class_name(
                    "tree-node")[::-1]
            else:
                self.log.info("failed to find the record")
                raise Exception("failed to find the record")

            j = 1
            for elem in elems:
                if fpath.lower() == elem.text.lower():
                    logstr = "found file/folder " + elem.text
                    self.log.info(logstr)
                    self.driver.execute_script(
                        "arguments[0].scrollIntoView();", elem)
                    # elem.click()
                    self.driver.execute_script("arguments[0].click();", elem)
                    self.wait_for_completion()
                    break
                else:
                    j += 1
                    continue

            if j == len(elems) + 1:
                raise Exception("failed to find the file/folders " + fpath)

            if self.check_if_entity_exists("xpath", xpath):
                self.driver.find_element_by_xpath(xpath).click()
                self.wait_for_completion()
            else:
                raise Exception("could not find client from the list")
        else:
            raise Exception(
                "could not locate select client option on entitlement management page")

            self.driver.execute_script("arguments[0].click();", element)
            time.sleep(10)

            self.click_button("Review changes")

            # wait for Change permission job to finish. timeout (270seconds)
            i = 0

                if bodytext.find("Completed") == -1:

                    else:

                        continue
                else:
                    self.log.info("Change permission job completed")
                    break

            self.click_button("Ok")
        else:
            raise Exception("failed to find permission entries")

    @PageService()
    def get_available_perm(self, uname):
        """
        get_available_perm
        Args:
            uname (str)  - user name

        Raise:
            Exception if all permission for the user are not available for modify
        """

        # find a permission which are not gray out, then click it
        permlist = [
            '_allow_fullcontrol',
            '_allow_modify',
            '_allow_readandexecute',
            '_allow_read',
            '_allow_write',
            '_deny_fullcontrol',
            '_deny_modify',
            '_deny_readandexecute',
            '_deny_read',
            '_deny_write']
        prexpath = ".//*[@id='permissionsMain']/tbody/tr[2]/td/table/tbody/tr["
        i = 0
        while i < 10:
            checkboxid = uname + permlist[i]
            if i < 5:
                xpath = prexpath + str(i + 1) + "]/td[2]/input"
            else:
                xpath = prexpath + str(i - 4) + "]/td[3]/input"
            if self.check_if_entity_exists("id", checkboxid):
                element = self.driver.find_element_by_xpath(xpath)
                if element.is_enabled():
                    break
                else:
                    i += 1
            else:
                excep = "failed to find " + checkboxid
                raise Exception(excep)
        if i < 10:
            clickperm = "change permission " + checkboxid
            self.log.info(clickperm)
            return element

        raise Exception("all permission for selected user are gray out")
