    @WebAction()

        """Clicks ok button on select backup level modal to start backup"""
        ok_button = self.driver.find_element_by_xpath("//button[@data-ng-click='startBackup()']")

    @PageService()
    def select_instance(self, instance):
        """
        selects the given instance

        Args:
            instance(str) -- Instance to be select

        Returns:
            None

        Exception:
            if not able to select any instance

        """
        self.__table.access_link(instance)

    @PageService()
    def add_postgresql_database_server(
            self,
            os_type='Unix and Linux',
            dbserver_name=None,
            username=None,
            password=None,
            plan=None,
            unix_group=None):
        """
        Adds a new PostgreSQL database server

        Args:
            os_type             (str)       -- The type of the OS
                                                Default: Unix and Linux

            dbserver_name       (str)       -- Database server name

            username            (str)       -- Username for the DB server

            password            (str)       -- Password for the DB server

            plan                (str)       -- Plan for the DB server

            unix_group          (str)       -- Unix group for the DB server

        Returns:
            str         --  Job ID for the triggered job

        Exception:

            If user input is invalid

            If element not found

        """
        self.select_hyperlink("Add database server")
        self.select_hyperlink('PostgreSQL')

        if 'windows' in os_type.lower():
            self.select_radio('Windows')
        elif 'unix' in os_type.lower():
            self.select_radio('Unix and Linux')

        self.fill_form_by_id("serverName", dbserver_name)
        self.fill_form_by_id("userName", username)
        self.fill_form_by_id("password", password)
        self.cv_single_select("Plan", plan)

        if unix_group:
            self.fill_form_by_id("uGroup", unix_group)

        self.submit_form()

        return self.get_jobid_from_popup()

    @PageService(hide_args=True)
    def add_postgresql_database_instance(
            self,
            server_name=None,
            instance_name=None,
            db_username=None,
            password=None,
            maintenance_db=None,
            port=None,
            binary_directory=None,
            lib_directory=None,
            archive_log_directory=None,
            delete_archive_logs=False):
        """
        Adds a new PostgreSQL database server

        Args:
            server_name         (str)   --  Server to create a new PostgreSQL instance

            instance_name       (str)   --  Name of the instance to be created

            db_username         (str)   --  username for accessing the DB

            password            (str)   --  Password for accessing the DB

            maintenance_db      (str)   --  maintenance DB for the instance to be created

            port                (str)   --  port for the instance

            binary_directory    (str)   --  Full path for binary

            lib_directory       (str)   --  Full path for library

            archive_log_directory (str) --  Full path for archive logs

            delete_archive_logs (bool)  --  Deletes archive logs if set to True
                                                default: False

        Returns:
            None

        Exception:

            If user input is invalid

            If element not found

        """
        self.select_hyperlink("Add database instance")
        self.select_hyperlink('PostgreSQL')
        self.cvselect_from_dropdown('Server name', server_name)
        self.fill_form_by_id('databaseName', instance_name)
        self.fill_form_by_id('osUsername', db_username)
        self.fill_form_by_id('dbPassword', password)
        self.fill_form_by_id('maintenanceDB', maintenance_db)
        self.fill_form_by_id('port', port)
        self.fill_form_by_id('BinaryDirectory', binary_directory)
        self.fill_form_by_id('LibDirectory', lib_directory)
        self.fill_form_by_id('ArchiveLogDirectory', archive_log_directory)

        if delete_archive_logs:
            self.checkbox_select('archiveDelete')

        # Support for standby settings not added

        self.submit_form()

    @PageService()
    def instance_backup(
            self,
            instance=None,
            backupset_type=None,
            backupset_name='default',
            backup_level='Incremental'):
        """
        Performs backup operation on the selected instance

        Args:
            instance        (str)   -- instance to perform backup on

            backupset_type  (str)   -- Type of backupset (Dumpbased|Fsbased)

            backupset_name  (str)   -- Name of the backupset to run backup on
                                        default: 'default'

            backup_level    (str)   -- Backup level to be used for backup

        Returns:
            str     --  Job ID of the triggered backup job

        Exception:

            If user input is invalid

            If element not found

        """
        table = Table(self)
        table.access_link_by_column(instance, 'Backup')

        if 'dump' in backupset_type.lower():
            backupset_type = 'DumpBasedBackupSet'
        elif 'fs' in backupset_type.lower():
            backupset_type = 'FSBasedBackupSet'
        else:
            raise Exception(f'The Backupset Type {backupset_type} is Invalid')

        if self.check_if_entity_exists(
                'xpath',
                f"//label[contains(text(),'{backupset_type}')]/../..//label[contains(text(),'{backupset_name}')]"):
            self.driver.find_element_by_xpath(
                f"//label[contains(text(),'{backupset_type}')]/../..//label[contains(text(),'{backupset_name}')]"
                f"/../preceding-sibling::span[1]//input"
            ).click()
        else:
            raise Exception(f'Backupset {backupset_name} is Invalid')

        self.submit_form()
        self.select_radio(backup_level)

        return self.get_jobid_from_popup()

    @PageService()
    def __restore_options(self, destination_client=None, destination_instance=None):
        """
        Provides option fot Restores the backup

        Args:
            destination_client(str)   -- destination client to perform restore operation on

            destination_instance(str) -- instance to perform restore on

        Returns:
            str        --  Job ID for the triggered restore job

        Exception:

            If user input is invalid

            If restore failed

        """
        self.select_value_from_dropdown("destinationclient", destination_client)
        self.select_value_from_dropdown("destinationDatabase", destination_instance)

        # Point in time not included
        self.checkbox_select('startServer')
        self.submit_form()

        return self.get_jobid_from_popup()

    @PageService()
    def instance_restore(self, instance=None, destination_client=None, destination_instance=None):
        """
        Performs restore operation on selected instance

        Args:
            instance(str)             -- instance to be select

            destination_client(str)   -- destination client to be select

            destination_instance(str) -- destination instance to be select

        Returns:
            str     --  job ID for the triggered restore job

        Exception:

            If user input is invalid

            If restore failed

         """
        table = Table(self)
        table.access_link_by_column(instance, 'Restore')
        self.wait_for_completion()
        # To click on Restore
        if self.check_if_entity_exists(
                'xpath',
                "//div[@id='browseActions']/a[contains(text(), 'Restore')]"):
            self.driver.find_element_by_xpath("//div[@id='browseActions']/a[contains(text(), 'Restore')]").click()

        return self.__restore_options(destination_client, destination_instance)
