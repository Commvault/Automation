from Web.AdminConsole.Components.table import Table

    """class for subclients page"""

    def __init__(self, driver):

        super(Subclients, self).__init__(driver)
        self.__modal_dialog = ModalDialog(self)
        self.__table = Table(self)

    @PageService()
    def add_subclient(self, subclient_name, subclient_content, plan_name, backupset_name, storage_policy,
                      schedule_policies):
        """
        Method to create a subclient

        Args:
            subclient_name (str)    : Name of the subclient

            subclient_content (str) : Subclient content to be backed up

            plan_name (str)         : Name of the plan

            backupset_name (str)    : backup set name

            storage_policy (str)    : Name of the storage policy

            schedule_policies (list): list of schedule policies to be associated

        Returns:
            None

        Raises:

            Exception:

                --if failed to create a subclient

        """
        self.driver.find_element_by_xpath("//span[contains(text(),'Add subclient')]").click()
        self.cv_single_select("Backup set", backupset_name)
        self.fill_form_by_id("subclientName", subclient_name)

        if plan_name:
            self.toggle_enable("Use backup plan")
            self.cv_single_select("Plan", plan_name)
        else:
            self.toggle_disable("Use backup plan")
            self.cv_single_select("Storage policy", storage_policy)
            self.cvselect_from_dropdown("Schedule policy", schedule_policies)

        self.select_hyperlink("Content")
        path = self.driver.find_element_by_xpath("//input[@placeholder='Enter custom path']")
        path.clear()
        path.send_keys(subclient_content)
        self.wait_for_completion()
        self.driver.find_element_by_xpath("//i[@title='Add'][@data-ng-click='addCustomPath(0)']").click()
        self.wait_for_completion()
        self.click_button("Save")
        self.check_error_message()

    @PageService()
    def backup_subclient(self, client_name, subclient_name, backup_level):
        """
        Method to backup the given subclient

        Args:
             client_name (str)      : Name of the client

             subclient_name (str)   : subclient to be backed up

             backup_level   (str)   : type of backup

        Returns :
             job id  (int) : Job id of the backup job

        Raises:
            Exception :

             -- if fails to run the backup

        """
        self.__table.search_for(client_name)
        self.__table.access_action_item(subclient_name, 'Backup')
        self.select_radio(backup_level)
        self.driver.find_element_by_xpath('//button[@data-ng-click="startBackup()"]').click()
        return int(self.get_jobid_from_popup())

    @PageService()
    def restore_client(self, client_name, subclient_name, restore_file):
        """
        Method to Restore the given client data

        Args:

             client_name (str)      : Name of the client

             subclient_name (str)   : Name of the subclient

             restore_file   (str)   : Data to be restored

        Returns :

              job id  (int) : Job id of the restore job

        Raises:
            Exception :

             -- if fails to run the restore operation

        """
        self.__table.access_action_item(subclient_name, 'Restore')
        self.__table.search_for(restore_file)
        self.driver.find_element_by_xpath("//div[@ng-if='grid.options.enableSelectAll']").click()
        self.select_hyperlink("Restore")
        self.submit_form()
        return int(self.get_jobid_from_popup())

    @PageService()
    def delete_subclient(self, client_name, subclient_name):
        """
        Method to delete the given subclient

        Args:

             client_name (str)   : client name

             subclient_name(str) : subclient to be deleted

        Returns :
             None

        Raises:
            Exception :

             -- if fails to run the delete operation

        """
        self.__table.search_for(client_name)
        self.__table.access_action_item(subclient_name, 'Delete')
        self.__modal_dialog.type_text_and_save("DELETE")
        self.check_error_message()
