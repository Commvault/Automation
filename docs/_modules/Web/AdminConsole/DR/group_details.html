    def get_vm_details(self, vm_name):
        """Returns the replication VMs and their configuration from the table
            List with a dict for each row as {""}
        """
        table_data = self.__table.get_table_data()
        row_num = [idx for idx, name in enumerate(table_data['Source VM']) if name == vm_name]
        if not row_num:
            raise CVWebAutomationException(f"Cannot find the VM with the name {vm_name}")
        return {col: row[row_num[0]] for col, row in table_data.items()}

    @WebAction()
    def __is_general_settings_editable(self):
        """
        If parameters are editable then return true otherwise false
        Returns  (Boolean): True if the General settings is editable otherwise false
        """
        xp = "//div[contains(text(), 'General settings')]/..//div[contains(@data-ng-disabled, " \
             "'repTarget')]"
        parameters = self.__admin_console.driver.find_elements_by_xpath(xp)
        # status below will be listof 'true' if its disabled
        status = [each_param.get_attribute('disabled') for each_param in parameters]
        if [each_status for each_status in status if each_status != 'true']:
            # if disabled is not 'true' then its editable, so returning True
            return True
        return False

    @PageService()
    def is_general_settings_editable(self):
        """
        Returns  (Boolean): True if the General settings is editable otherwise false
        """
        return self.__is_general_settings_editable()

    @WebAction()
    def __get_selected_multi_select(self, multi_select_id: str) -> str:
        """Gets the current selected value from the multi-select"""
        return self.__driver.find_element_by_xpath(f"//*[@id='{multi_select_id}']//button").text

    @WebAction()
    def __get_selected_dropdown_option(self, select_id: str) -> str or None:
        """Gets the current value from the select"""
        options = self.__driver.find_elements_by_xpath(f"//*[@id='{select_id}']//option")
        for option in options:
            if option.get_attribute("selected") is not None:
                return option.text
        return None

    @WebAction()
    def __get_checkbox_value(self, checkbox_id: str) -> bool:
        """Gets the value of the given checkbox"""
        return self.__driver.find_element_by_xpath(f"//input[@type='checkbox' and @id='{checkbox_id}']").is_selected()

    @WebAction()
    def is_field_disabled(self, field_id: str) -> bool:
        """Checks if field is disabled and then waits for delay"""
        if self.__admin_console.check_if_entity_exists("xpath", f"//select[@name='{field_id}']//option"):
            return bool([element for element in
                         self.__driver.find_elements_by_xpath(f"//select[@name='{field_id}']//option")
                         if element.get_attribute("disabled") is not None])
        if self.__admin_console.check_if_entity_exists("xpath", f"//*[@id='{field_id}']//button"):
            return bool(self.__driver.find_element_by_xpath(f"//*[@id='{field_id}']//button")
                        .get_attribute('ng-disabled'))
        if self.__admin_console.check_if_entity_exists("xpath", f"//input[@name='{field_id}']"):
            disabled_attr = (self.__driver.find_element_by_xpath(f"//input[@name='{field_id}']")
                             .get_attribute('data-ng-disabled'))
            return disabled_attr is not None and "editMode" in disabled_attr
        raise CVWebAutomationException(f'In the edit VM, the field with id {field_id} does not exist')

class _AddVirtualMachine:
    """Class to introduce common functionality for adding cross-platform VMs"""
    def __init__(self, admin_console: AdminConsole, vm_type: str = SOURCE_HYPERVISOR_VMWARE):
        """Add content options from virtualization_replication"""
        self._admin_console = admin_console
        self._modal_panel = ModalPanel(admin_console)
        self._dialog = ModalDialog(admin_console)
        self.content = _Content(admin_console)  # Added for future purpose of removing select_vm_from_browse_tree
        self.vm_type = vm_type
        if vm_type == SOURCE_HYPERVISOR_VMWARE:
            self.override_options = _VMwareVMOptions(admin_console)
        elif vm_type == SOURCE_HYPERVISOR_AZURE:
            self.override_options = _AzureVMOptions(admin_console)

    def select_vm_from_browse_tree(self, virtual_machines: list):
        """Selects the given VMs from the browse tree"""
        for vm_name in virtual_machines:
            self._admin_console.search_vm(vm_name)

    @PageService()
    def __fields_edit_azure(self, override_args: dict):
        """Edit the fields for overriding Azure group"""
        self.override_options: _AzureVMOptions
        if override_args.get('vm_display_name'):
            self.override_options.set_vm_display_name(override_args.get('vm_display_name'))
        if override_args.get("resource_group"):
            self.override_options.select_resource_group(override_args.get("resource_group"))
        if override_args.get("region"):
            self.override_options.select_region(override_args.get("region"))
        if override_args.get("storage_account"):
            self.override_options.select_storage_account(override_args.get("storage_account"))
        if override_args.get("vm_size"):
            self.override_options.select_vm_size(override_args.get("vm_size"))
        if override_args.get("virtual_network"):
            self.override_options.virtual_network(override_args.get("virtual_network"))
        if override_args.get("security_group"):
            self.override_options.select_security_group(override_args.get("security_group"))
        if override_args.get("create_public_ip"):
            self.override_options.create_public_ip(enable=override_args.get("create_public_ip"))
        if override_args.get("restore_as_managed_vm"):
            self.override_options.restore_as_managed_vm(enable=override_args.get("restore_as_managed_vm"))

    @PageService()
    def __fields_edit_vmware(self, override_args: dict):
        """Edits the fields for VMWare Replication group"""
        self.override_options: _VMwareVMOptions
        if override_args.get('vm_display_name'):
            self.override_options.set_vm_display_name(override_args.get('vm_display_name'))
        if override_args.get('destination_host'):
            self.override_options.set_destination_host(override_args.get('destination_host'))
        if override_args.get('data_store'):
            self.override_options.select_datastore(override_args.get('data_store'))
        if override_args.get('resource_pool'):
            self.override_options.select_resource_pool(override_args.get('resource_pool'))
        if override_args.get('vm_folder'):
            self.override_options.set_vm_folder(override_args.get('vm_folder'))
        if override_args.get('network_settings'):
            self.override_options.expand_tab('Network settings')

    def add_vm(self, virtual_machines: list, override_args=None):
        """Add the VM with default options
        Args:
            virtual_machines (list): The virtual machines to add in the form
            {'Datastores and datastore clusters':[ds1,ds2],
                                 'Hosts and clusters':[h1,h2],
                                 'Tags and categories':[tag1,category1],
                                 'VMs and templates':[vm1,vm2],
                                 'Storage':[strg1, strg2],
                                 'By region': [region1, region2],
                                 'By zone': [zone1, zone2]
            }
            override_args   (dict): The dictionary values of all override options
                Sample:
                {"vm_display_name": "abcd",
                "destination_host": "172.16.5.4",
                "data_store": "ABCDEDFG05",
                "resource_pool": "/abcd",
                "vm_folder": "efg"
                }
        """
        self._admin_console.select_hyperlink('Add virtual machines')
        self.select_vm_from_browse_tree(virtual_machines)
        self._modal_panel.submit()
        if override_args:
            self._dialog.click_submit()
            if self.vm_type == SOURCE_HYPERVISOR_VMWARE:
                self.__fields_edit_vmware(override_args)
            elif self.vm_type == SOURCE_HYPERVISOR_AZURE:
                self.__fields_edit_azure(override_args)
            else:
                raise NotImplementedError(f"Override options are not implemented for {self.vm_type}")
            self._modal_panel.submit()
            self._modal_panel.submit()

        self._dialog.click_cancel()
        self._admin_console.wait_for_completion()
