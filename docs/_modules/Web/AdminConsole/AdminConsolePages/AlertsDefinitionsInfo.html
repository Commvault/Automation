    """
    This class provides the function or operations that can be performed on the alerts definitions
    tab on the Alerts Page
    """
    @adminconsole_handler()
    def add_user(self, name):
        """Adds users to the alert notification group

            Args:
                name  - username, group name, or email

            Raise:
                Exception if element is not present.
        """
        self.log.info("Adding user " + name + " to the alert's users or groups to notify")
        self.driver.find_element_by_xpath(
            "//a[@class='edit-users']").click()
        self.wait_for_completion()
        self.driver.find_element_by_xpath(
            "//div[@class='modal-content']/div[2]/div[2]/label/div/ul/li/input").send_keys(
                name)
        time.sleep(3)
        elements = self.driver.find_elements_by_xpath("//ul[@class='select2-results']/li")
        name = name + " ("+name+")"
        for elem in elements:
            if elem.find_element_by_xpath("./div/span").text == name:
                elem.find_element_by_xpath("./div/span").click()
                self.driver.find_element_by_xpath(
                    "//div[2]/div[2]/label/div/ul/li/input").send_keys(Keys.BACKSPACE)
        self.driver.find_element_by_id("addButton").click()
        self.driver.find_element_by_xpath("//div[4]/button[2]").click()
        self.wait_for_completion()
        if self.check_if_entity_exists(
                "xpath", "//span/div/div[@class='server-message error']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//span/div/div[@class='server-message error']").text)

    @adminconsole_handler()
    def remove_user(self, name):
        """Removes users from the alert notification group

            Args:
                name - username, group name or email of the user to be removed

            Raise:
                Exception if element is not present.
        """
        self.log.info("Removing user " + name + " from the alert's users or groups to notify")
        self.driver.find_element_by_xpath(
            "//a[contains(text(),'Edit')]").click()
        self.wait_for_completion()
        self.driver.find_element_by_xpath(
            "//label[contains(text(),'" +
            name +
            "')]/following-sibling::span[@class='delete-row']").click()
        self.wait_for_completion()
        self.driver.find_element_by_xpath("//div[4]/button[2]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def delete_alert(self):
        """
        Deletes the alert from the alert definitions page

            Raise:
                Exception if element is not present.
        """
        self.log.info("Deleting the alert")
        if self.check_if_entity_exists("link", "Delete"):
            self.driver.find_element_by_link_text("Delete").click()
            self.wait_for_completion()
            self.driver.find_element_by_link_text("Alerts definitions").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to delete the alert")
