[docs]class StorageCopy(AdminPage):
    """
    This class provides the function or operations that can be
    performed on the Storage Copy Page of the Admin Console
    """

    @adminconsole_handler()
    def edit_retention(self, data_aging=True, retention='30'):
        """
        To edit the retention value of the storage copy

        Args:
            data_aging          (bool)  -- To enable/disable data aging
                                            default: True

            retention           (str)   -- Retention period for the copy created (Infinite|Number of days)
                                            default: '30'

        """
        self.driver.find_element_by_xpath("""//a[@ng-click="onEdit('retention')"]""").click()
        if data_aging:
            self.toggle_enable('Data Aging')
            if "infinite" in retention.lower():
                self.checkbox_select('isInfiniteRetCheck')
            else:
                element = self.driver.find_element_by_xpath(
                    r"//input[@name='input'][@ng-model='policyRetentionValue']")
                element.clear()
                element.send_keys(retention)
        else:
            self.toggle_disable('Data Aging')

        # To click on save button
        self.driver.find_element_by_xpath(
            '//a[@ng-click="onSaveRetention()"][contains(text(),"Save")]').click()

        # To click on yes button
        self.driver.find_element_by_xpath(
            '//button[contains(text(), "Yes")][@ng-click="yes()"]').click()

        self.check_error_message()
