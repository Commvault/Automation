    @adminconsole_handler()
    def edit_user(self, full_name, username, email, groups, plans=None, enabled=True,
                  password=None):
        """Edits the user with modified details

            :param full_name    (str)   --  full name of the user
            :param username     (str)   --  the user name of the user
            :param email        (str)   --  the email ID of the user
            :param groups       (list)  --  list of user groups to be associated with the user
            :param plans        (list)  --  list of plans to be associated with the user
            :param enabled      (bool)  --  if the user should be enabled/ disabled
            :param password     (str)   --  the password of the user
        """
        self.select_user(username)
        self.driver.find_element_by_xpath("//a[@class='page-action-item ng-scope']").click()
        self.wait_for_completion()
        self.fill_form_by_id("fullName", full_name)
        self.fill_form_by_id("userName", username)
        self.fill_form_by_id("email", email)
        self.driver.find_element_by_xpath(
            "//isteven-multi-select[@id='userGroupType']/span/button").click()
        self.driver.find_element_by_xpath(
            "//isteven-multi-select[@id='userGroupType']/span/div/div[1]/div[1] \
            /button[2]").click()
        for group in groups:
            self.driver.find_element_by_xpath(
                "//div[@class='line-search']/button[@class='clearButton']").click()
            self.driver.find_element_by_xpath(
                "//div[@class='line-search']/input").send_keys(group)
            self.driver.find_element_by_xpath(
                "//div[@class='checkBoxContainer']//span[contains(text(),'"+group+"')]").click()
        self.driver.find_element_by_xpath(
            "//div[@class='line-searchOk']").click()

        if password:
            self.fill_form_by_id("password", password)

        if plans:
            self.driver.find_element_by_xpath(
                "//isteven-multi-select[@input-model='planList']/span/button").click()
            self.driver.find_element_by_xpath(
                "//isteven-multi-select[@input-model='planList']/span/div/div[1]/div[1] \
                /button[2]").click()
            for plan in plans:
                self.driver.find_element_by_xpath(
                    "//div[@class='line-search']/button[@class='clearButton']").click()
                self.driver.find_element_by_xpath(
                    "//div[@class='line-search']/input").send_keys(plan)
                self.driver.find_element_by_xpath(
                    "//div[@class='checkBoxContainer']//span[contains(text(),'" + plan +
                    "')]").click()
            self.driver.find_element_by_xpath(
                "//div[@class='line-searchOk']").click()

        if enabled:
            if not self.driver.find_element_by_id("enabled").is_selected():
                self.driver.find_element_by_xpath("//input[@id='enabled']/following-sibling::"
                                                  "label").click()
                self.wait_for_completion()

        self.driver.find_element_by_xpath(
            "//form[@id='addUserForm']/div[2]/button[2]").click()
        self.wait_for_completion()
        time.sleep(10)
        if self.check_if_entity_exists(
                "xpath", "//div[@class='setup-content smaller-inputs']/span[@class='error']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//div[@class='setup-content smaller-inputs']/span[@class='error']").text)

    @adminconsole_handler()
    def user_details(self):
        """Displays all the details about the user like name, email, user group and plans
        :return:
            user_details   (dict)  --  dict of users and their info
        """
        user_details = {}
        elements = self.driver.find_elements_by_xpath("//ul[@class='list-style-secondary']/li")
        for element in elements:
            key = element.find_element_by_xpath("./span[1]").text
            value = element.find_element_by_xpath("./span[2]").text
            user_details[key] = value
        self.log.info(user_details)
        return user_details

    @adminconsole_handler()
    def open_user_group(self, user_group):
        """
        Opens the user group associated with user
        :param user_group   -   the name of the user group to open
        """
        if self.check_if_entity_exists("link", user_group):
            self.driver.find_element_by_link_text(user_group).click()
            self.wait_for_completion()
        else:
            raise Exception("The user is not associated with the given user group {0}"
                            .format(user_group))
