    @adminconsole_handler()
    def add_storage(self, media_agent, path):
        """
        Adds storage
        """
        self.log.info("Inside Add Storage of StoragePoolDetails")
        if self.check_if_entity_exists("link", "Add storage"):
            self.driver.find_element_by_link_text("Add storage").click()
            self.wait_for_completion()
            self.log.info("Clicked on add storage")
            select = Select(self.driver.find_element_by_id("mediaAgent"))
            pattern = re.compile(media_agent, re.IGNORECASE)
            for option in select.options:
                value = option.get_attribute('label')
                if pattern.search(value):
                    option.click()
                    self.wait_for_completion()
                    break
            self.fill_form_by_id("mountPath", path)
            self.driver.find_element_by_xpath("//form/div[2]/button[2]/div").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("xpath", "//div[@class='setup-content']/p[1]"):
                raise Exception(self.driver.find_element_by_xpath(
                    "//div[@class='setup-content']/p[1]").text)
        else:
            raise Exception("There is no option to Add Storage Pool")

    @adminconsole_handler()
    def storage_pool_info(self):
        """
        Returns the details of storage pool
        :return:
            info    (dict)  -- details of storage pool
        """
        self.log.info("Fetching the details of the storage pool")
        elements = self.driver.find_elements_by_xpath("//li[@class = 'group']")
        key = ''
        info = {}
        for elem in elements:
            span_elements = elem.find_elements_by_xpath("./span")
            index = 1
            for span in span_elements:
                if index % 2 == 0:
                    value = span.text
                    info[key] = value
                else:
                    key = span.text
                index += 1
        self.log.info(info)
        return info

    @adminconsole_handler()
    def details(self):
        """
        Opens the storage pool dedupe details
        """
        self.log.info("Opening the storage pool dedupe details")
        if self.check_if_entity_exists(
                "xpath", "//span[@class='page-details-box-links ng-scope']/a"):
            self.driver.find_element_by_xpath(
                "//span[@class='page-details-box-links ng-scope']/a").click()
            self.wait_for_completion()

    @adminconsole_handler()
    def delete_storage_mount_path(self, mount_path):
        """
        Deletes the storage mount path
        :param mount_path   (str)   --  name of the mount path to delete
        """
        self.log.info("Deleting the storage mount path")
        storage_details_obj = StorageDetails(self.driver)
        storage_details_obj.action_delete_mount_path(mount_path)

    @adminconsole_handler()
    def share_storage_mount_path(
            self,
            mount_path,
            share):
        """
        Shares the storage mount path
        :param mount_path       (str)   --  mount path to share
        :param share            (dict)  --  info about MA and share access
        """
        storage_details_obj = StorageDetails(self.driver)
        storage_details_obj.share_mount_path(mount_path, share)
        self.driver.find_element_by_xpath(
            "//div[2]/div[3]/button[contains(text(),'Finish')]").click()
        self.wait_for_completion()


[docs]    def encrypt_storage(self, cipher=None, key_length=None, key_management_server=None):
        """
        To Encrypt the storage of the selected pool

        Args:
            cipher      (str)   -- Encryption method to be used

            key_length  (str)   -- Key length for the chosen cipher

            key_management_server   (str)   --  Key management server for the storage pool
        """
        self.select_hyperlink('Edit')
        if cipher and key_length:
            self.checkbox_select('encrypt')
            self.cv_single_select('Cipher', cipher)
            self.wait_for_completion()
            self.cv_single_select('Keylength', key_length)

        if key_management_server:
            self.cv_single_select('Key management server', key_management_server)
        self.submit_form()
        self.check_error_message()
