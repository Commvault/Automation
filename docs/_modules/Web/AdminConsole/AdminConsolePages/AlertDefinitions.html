    @adminconsole_handler()
    def add_alert_definition(self, alert_name, alert_type, clients, users, value=None,
                             groups=None):
        """Adds a new alert definition of the given type for the given entities and users

            Args:
                alert_name      :   name of alert to be created
                alert_type      :   type of alert to be created
                value           :   param value for x if the alert type consists of X
                    default is None
                clients         :   list of all clients to be associated with the alert
                users           :   list of users to be associated with the alert
                groups          :   list of user groups to be associated with the alert

            Raise:
                Exception if element is not present.
        """
        self.log.info("Adding a new alert definition")
        if self.check_if_entity_exists("link", "Add alert definition"):
            self.driver.find_element_by_link_text("Add alert definition").click()
            self.wait_for_completion()
            self.driver.find_element_by_id("alertName").send_keys(alert_name)
            Select(self.driver.find_element_by_name("alertType")).select_by_visible_text(
                alert_type)
            WebDriverWait(self.driver, 300).until(EC.presence_of_all_elements_located(
                (By.XPATH, "//div[@class='browse-tree alerts-browse']/div[3]/div[2]/div[1]/"
                           "div[1]/span")))
            self.wait_for_completion()
            if self.check_if_entity_exists("id", "alertParam"):
                if value:
                    self.driver.find_element_by_id("alertParam").clear()
                    self.driver.find_element_by_id("alertParam").send_keys(value)
            cgroup = self.driver.find_element_by_xpath("//div[@class="
                                                       "'browse-tree alerts-browse']/div[2]"
                                                       "/div[1]")
            if groups:
                cgroup.find_element_by_xpath("./button").click()
                self.wait_for_completion()
                elems = cgroup.find_elements_by_xpath("./../div[2]/div")
                for elem in elems:
                    name = elem.find_element_by_xpath("./div[1]/span")
                    if name.text.upper() in [g.upper() for g in groups]:
                        name.click()
                    else:
                        continue
            for client in clients:
                if self.check_if_entity_exists("xpath", "//div[@class="
                                                        "'browse-tree alerts-browse']/div[3]"
                                                        "//span[contains(text(),'" + client
                                               + "')]"):
                    self.driver.find_element_by_xpath("//div[@class="
                                                      "'browse-tree alerts-browse']/div[3]//"
                                                      "span[contains(text(),'" + client + "')]"
                                                     ).click()
                    self.wait_for_completion()
            for user in users:
                self.driver.find_element_by_xpath("//form/div[4]/label/div/ul/li/input"
                                                 ).send_keys(user)
                WebDriverWait(self.driver, 60).until(EC.presence_of_all_elements_located
                                                     ((By.XPATH,
                                                       "//li[@role='presentation']")))
                elements = self.driver.find_elements_by_xpath("//li[@role='presentation']")
                user = user + " ("+user+")"
                for elem in elements:
                    if elem.find_element_by_xpath("./div/span").text == user:
                        elem.find_element_by_xpath("./div/span").click()
                        self.driver.find_element_by_xpath("//form/div[4]/label/div/ul/li/input"
                                                         ).send_keys(Keys.BACKSPACE)
                    elif elem.find_element_by_xpath("./div/span").text == "Loading failed":
                        error = "The users could not be loaded in Add Alerts Definition"
                        raise Exception(error)
            self.driver.find_element_by_id("addButton").click()
            self.driver.find_element_by_xpath("//form/div[6]/button[2]").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("xpath", "//span[@class='server-message  error']"):
                raise Exception(self.driver.find_element_by_xpath("//span[@class="
                                                                  "'server-message  error']")
                                .text)
        else:
            raise Exception("There is no option to add a new alert definition. Check if the"
                            "user has required capabilities")

    @adminconsole_handler()
    def select_alert(self, alert_name):
        """
        Opens the alert with the given name alert_name - name of the alert to be opened

            Args:
                alert_name      : Name of the alert to be selected

            Raise:
                Exception if element is not present.
        """
        self.log.info("Selecting alert " + alert_name)
        self.driver.find_element_by_id("search-field").clear()
        self.driver.find_element_by_id("search-field").send_keys(alert_name)
        self.wait_for_completion()
        if self.check_if_entity_exists("link", alert_name):
            self.driver.find_element_by_link_text(alert_name).click()
            self.wait_for_completion()
        else:
            raise Exception("There is no alert " + alert_name)

    @adminconsole_handler()
    def delete_alert(self, alert_name):
        """Deletes the alert with the given name

            Args:
                alert_name - name of the alert to be deleted

            Raise:
                Exception if element is not present.
        """
        self.log.info("Deleting alert " + alert_name)
        self.driver.find_element_by_id("search-field").clear()
        self.driver.find_element_by_id("search-field").send_keys(alert_name)
        self.wait_for_completion()
        if self.check_if_entity_exists("link", alert_name):
            self.driver.find_element_by_xpath("//a[contains(text(),'" + alert_name +
                                              "')]/../../div[4]/a").click()
            self.wait_for_completion()
            self.driver.find_element_by_xpath("//button[contains(text(),'Yes')]").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no alert " + alert_name)


    @adminconsole_handler()
    def enable_alert(self, alert_name):
        """Enables a alert

            Args:
                alert_name - name of the alert to be enabled

            Raise:
                Exception if element is not present.
        """
        self.log.info("Enabling alert " + alert_name + " if not enabled")
        self.driver.find_element_by_id("search-field").clear()
        self.driver.find_element_by_id("search-field").send_keys(alert_name)
        self.wait_for_completion()
        if self.check_if_entity_exists("link", alert_name):
            if not self.driver.find_element_by_xpath(
                    "//a[contains(text(),'" + alert_name + "')]/../../div[3]/div/span/input")\
                    .is_selected():
                self.driver.find_element_by_xpath(
                    "//a[contains(text(),'" + alert_name + "')]/../../div[3]/div/span/label"
                    ).click()
                self.wait_for_completion()
        else:
            raise Exception("There is no alert " + alert_name)

    @adminconsole_handler()

            if self.driver.find_element_by_xpath("//a[contains(text(),'" + alert_name +
                                                 "')]/../../div[3]/div/span/input"
                                                ).is_selected():
                self.driver.find_element_by_xpath(
                    "//a[contains(text(),'" + alert_name + "')]/../../div[3]/div/span/label"
                    ).click()
                self.wait_for_completion()
        else:
            raise Exception("There is no alert " + alert_name)
