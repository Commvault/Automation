    @adminconsole_handler()
    def all_job_activity(self, activity_status='ON',
                         enable_after_delay=False,
                         delay_value=None,
                         custom_delay=None):
        """
        Enables / Disables the all job activity toggle.
        :param activity_status      (str)   --  the status of all job activity
            Possible values:    ON, OFF
        :param enable_after_delay   (bool)  --  if the job activity is off, then we can enable
                                                after a delay
            Default value:      False
        :param delay_value          (str)   --  the possible delay value to enable job activity
                                                after
            Default value:      None
        :param custom_delay:        (str)   --  the time to set if the delay value is custom
            Default value:      None
        """
        self.log.info("Enable / Disable AllJobActivity in CommServe")
        element = self.driver.find_element_by_xpath("//toggle-control[@title='Enable/Disable"
                                                    " All job activity']/span")
        status = element.get_attribute('class')
        if 'enabled' in status:
            if activity_status == 'OFF':
                element.click()
                self.wait_for_completion()
                if enable_after_delay:
                    self.driver.find_element_by_link_text("Enable after a delay").click()
                    self.driver.find_element_by_link_text(delay_value).click()
                    self.wait_for_completion()
                    if delay_value == 'Custom':
                        self.driver.find_element_by_id("enableTime").clear()
                        self.driver.find_element_by_id("enableTime").send_keys(custom_delay)
                        self.driver.find_element_by_xpath("//button[text()='Save']").click()
                        self.wait_for_completion()
        else:
            if activity_status == 'ON':
                element.click()
                self.wait_for_completion()

    @adminconsole_handler()
    def __add_security(self, user_role_dict):
        """Adds associations and owners for the commcell.
        :param user_role_dict   (dict)  -- key value pair of the user / user group and role to
                                            be added
                                           key -- user or user group name
                                           value -- role to be associated with user or user group
        """
        for user_group, role in user_role_dict.iteritems():
            self.driver.find_element_by_xpath("//div[@class='user-suggestion-single-select"
                                              "-wrapper']/div/a/span").click()
            self.driver.find_element_by_xpath("//div[@id='select2-drop']/div/input"
                                             ).send_keys(user_group)
            time.sleep(5)
            elements = self.driver.find_elements_by_xpath("//ul[@class='select2-results']/li")
            for elem in elements:
                pattern = re.compile(user_group + r'\b', re.IGNORECASE)
                value = elem.find_element_by_xpath("./div/span").text
                if pattern.search(value):
                    elem.find_element_by_xpath("./div/span").click()
            self.select_value_from_dropdown("adduserId", role)
            self.driver.find_element_by_xpath(
                "//div/div[1]/button[contains(text(),'Add')]").click()
        self.driver.find_element_by_xpath(
            "//div[2]/div[2]/button[contains(text(),'Save')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def __remove_security(self, user_role_dict):
        """Removes the security with the given usergroup and role for the commcell.

        :param user_role_dict   (dict)  -- key value pair of the user / user group and role
                                            to be removed
                                        key -- user or user group name
                                        value -- role to be associated with user or user group
        """
        elements = self.driver.find_elements_by_xpath(
            "//div/div[2]/div/div/ul[@class='users-list__user group ng-scope']")
        for elem in elements:
            for user_group, user_role in user_role_dict.iteritems():
                if elem.find_element_by_xpath(
                        "./li/span/label[1]").text == user_group:
                    if elem.find_element_by_xpath(
                            "./li/span/label[2]").text == user_role:
                        elem.find_element_by_xpath("./li/span/span/a").click()
        self.driver.find_element_by_xpath(
            "//div[2]/div[2]/button[contains(text(),'Save')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def security(self, action, entity_value):
        """
        Set the roles or custom permission for the users at the commcell level
        :param action           (str)   --  add or remove users from the list of associations
            Possible values :   add, remove
        :param entity_value     (dict)  --  the list of users or user groups to be associated
                                            with the role
        """
        self.log.info("Editing the security configuration")

        if self.check_if_entity_exists("xpath", "//cv-tile-component["
                                                "@data-title='Security']/div/div/div/div[3]/a"):
            self.driver.find_element_by_xpath("//cv-tile-component["
                                              "@data-title='Security']/div/div/div/div[3]/a")\
                .click()
            self.wait_for_completion()
            if action == 'add':
                self.__add_security(entity_value)
            elif action == 'remove':
                self.__remove_security(entity_value)
            else:
                error = "Please specify either 'add' or 'remove' as action" \
                    "if you want to add or remove security respectively"
                raise error

    def send_log(
            self,
            log_for,

            passive=False,

            output_path=None,
            folder=None,
            network_path=None,
            username=None,
            password=None,
            attach_to_email=True):
        """Send the log of the specific job on completion.

            logFor        : a string,
            jobID         : an int,     id of the job we want to send the logs of; default = None
            mediaAgents   : a list,     list of the media agents; default = None
            clients       : a list,     list of clients; default = None

            Upload        : a boolean,  ------; default = False
            passive       : a boolean,  ------; default = False

            folder        : a string,   ------; default = None
            nwPath        : a string,   ------; default = None
            username      : a string,   ------; default = None
            password      : a string,   ------; default = None
            attachToEmail : a boolean,  ------; default = True

        self.log.info("Sending logs of the specific criteria")
        if self.check_if_entity_exists("link", "Send logs"):
            self.driver.find_element_by_link_text("Send logs").click()
            self.wait_for_completion()

            if log_for == "Job ID":
                self.driver.find_element_by_id("useJobID").click()

                    "//div[1]/div/ng-include/div/label[2]/span[1]/label/span").click()

                        "s2id_autogen1").send_keys(media_agent_name)

                        "s2id_autogen1").send_keys(Keys.TAB)
                for client in clients:
                    self.driver.find_element_by_id(
                        "s2id_autogen2").send_keys(client)
                    self.driver.find_element_by_id(
                        "s2id_autogen2").send_keys(Keys.TAB)
                if search:
                    if self.driver.find_element_by_id(
                            "searchContent").is_enabled():

                            "//div[1]/div/ng-include/div/label[5]/span[1]/label").click()
                        self.driver.find_element_by_xpath(
                            "//div[1]/div/ng-include/div/label[6]/span[2]/input"
                            ).send_keys("error")

    @adminconsole_handler()
    def email_settings(self, smtp_server, smtp_port, sender_email, sender_name):
        """
        Sets the email configuration for the commcell
        :param smtp_server      (str)   --  the smtp server name
        :param smtp_port        (str)   --  the smtp port
        :param sender_email     (str)   --  the email address of the sender
        :param sender_name      (str)   --  the name of the sender
        """
        self.log.info("Editing the email configuration")
        if self.check_if_entity_exists("xpath", "//cv-tile-component["
                                                "@data-title='Email settings']/div/div/div/"
                                                "div/a"):
            self.driver.find_element_by_xpath("//cv-tile-component["
                                              "@data-title='Email settings']/div/div/div/"
                                              "div/a").click()
            self.wait_for_completion()
            self.fill_form_by_id("smtpServer", smtp_server)
            self.fill_form_by_id("smtpPort", smtp_port)
            self.fill_form_by_id("senderEmail", sender_email)
            self.fill_form_by_id("senderName", sender_name)
            self.driver.find_element_by_xpath(
                "//form/div[2]/button[2]").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to configure email settings. Please check "
                            "if the user has the required permission to perform this action")

    @adminconsole_handler()
    def default_plans(self, plan_dict):
        """
        Set the default plans to be set for all of the commcell
        :param plan_dict    (dict)  --  dict of plans to set as server, laptop and FS plan
        """
        try:
            self.log.info("Setting the default plan for the commcell")
            for key, value in plan_dict.iteritems():
                if key == 'Server plan':
                    self.select_value_from_dropdown("serverDefault", value)
                elif key == 'Laptop plan':
                    self.select_value_from_dropdown("laptopDefault", value)
                elif key == 'File system plan':
                    self.select_value_from_dropdown("fsDefault", value)
            self.driver.find_element_by_xpath("//form/div[2]/button[2]").click()
            self.wait_for_completion()
        except Exception as exp:
            self.error_out_screenshot()
            if self.check_if_entity_exists("xpath", "//form/div[2]/button[1]"):
                self.driver.find_element_by_xpath("//form/div[2]/button[1]").click()
                self.wait_for_completion()
            self.log.exception(str(exp))
            raise exp

    @adminconsole_handler()
    def auth_code_for_install(self, refresh=True):
        """
        Enable authcode generation for installation
        :param refresh      (bool)  --  refreshes the auth code that is present
            default: True
        :return: authcode
        """
        self.log.info("Enable and get the authcode for installation")
        element = self.driver.find_element_by_xpath("//cv-generate-auth-code//"
                                                    "toggle-control/span")
        status = element.get_attribute("class")
        if "enabled" in status:
            if refresh:
                self.driver.find_element_by_xpath("//cv-generate-auth-code/span[2]/span[2]"
                                                  "/span/label/a").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath("//button[contains(text(),'Yes')]").click()
                self.wait_for_completion()
        else:
            element.click()
            self.wait_for_completion()
        auth_code = self.driver.find_element_by_xpath("//cv-generate-auth-code/span[2]/"
                                                      "span[2]/span/label/span").text
        return auth_code
