    @adminconsole_handler()
    def add_role(self, role_name, roles):
        """
        Adds a new role with the given permissions

        Args:
            role_name (str) :  the name of the role
            roles (list)    : the list of all permissions associated with the role

        Returns:
            None
        """
        self.select_hyperlink("Add role")
        self.fill_form_by_id("name", role_name)
        for role in roles:
            role_element = self.driver.find_element_by_xpath("//span[contains(text(),'" +
                                                             role + "')]")
            parent_elem = role_element.find_element_by_xpath("./ancestor::ul/../../parent::li"
                                                            )
            if parent_elem.get_attribute("class") == "ivh-treeview-node ng-scope" \
                                                     " ng-isolate-scope" \
                                                     " ivh-treeview-node-collapsed":
                self.driver.execute_script("arguments[0].scrollIntoView();", parent_elem)
                parent_elem.find_element_by_xpath("./div/span[1]").click()
                self.wait_for_completion()
            role_element.find_element_by_xpath("./../span[2]/span/input").click()

        self.submit_form()
        self.check_error_message()

        # Verify that permissions are associated with the role properly
        self.search_for(role_name)
        self.select_hyperlink(role_name)
        self.checkbox_select("showSelected")
        for role in roles:
            if not self.check_if_entity_exists("xpath", "//span[contains(text(),'"
                                               + role + "')]"):
                raise Exception("The chosen permission was not associated with the role."
                                " Please check the logs.")
        self.cancel_form()

    @adminconsole_handler()
    def delete_role(self, role_name):
        """
        Deletes the role with the given name
        Args:
            role_name (str): the name of the role

        Returns:
            None
        """
        self.select_action_item(role_name, "Delete")
        self.click_yes_button()

    @adminconsole_handler()
    def list_roles(self):
        """
        Lists all the roles

        Returns:
            Roles List
        """
        elements = self.driver.find_elements_by_xpath("//div[@class='ui-grid-canvas']/div")
        roles_list = []
        for elem in elements:
            roles_list.append(elem.find_element_by_xpath("./div/div[1]/span/a").text)
        self.log.info(roles_list)
        return roles_list

    @adminconsole_handler()
    def edit_role(self, role_name, new_role_name, roles_info):
        """
        Edits a role with the given permissions

        Args:
            role_name (str)     : the name of the role
            new_role_name (str) :  the new name of the role
            roles_info (dic)    : it contains the roles to be added and removed as list

        Returns:
            None
        """
        self.search_for(role_name)
        self.select_hyperlink(role_name)
        self.fill_form_by_id("name", new_role_name)
        self.checkbox_deselect("showSelected")
        add_roles = roles_info['Add']
        for add_role in add_roles:
            if not self.driver.find_element_by_xpath("//span[contains(text(),'"
                                                     + add_role +
                                                     "')]/../span[2]/span/input"
                                                    ).is_selected():
                role_element = self.driver.find_element_by_xpath(
                    "//span[contains(text(),'" + add_role + "')]")
                parent_elem = role_element.find_element_by_xpath(
                    "./ancestor::ul/../../parent::li"
                    )
                if parent_elem.get_attribute("class") == "ivh-treeview-node ng-scope" \
                                                         " ng-isolate-scope" \

                    self.driver.execute_script("arguments[0].scrollIntoView();", parent_elem)
                    parent_elem.find_element_by_xpath("./div/span[1]").click()
                role_element.find_element_by_xpath("./../span[2]/span/input").click()
        remove_roles = roles_info['Remove']
        for rem_role in remove_roles:

        self.submit_form()
        self.wait_for_completion()
        self.check_error_message()
