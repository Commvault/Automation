    @adminconsole_handler()
    def add_disk_storage(
            self,
            storage_name,
            media,
            type_of_storage,
            folder_path,
            username=None,
            password=None):
        """Add new disk library for the MA specified
            The storage can either be created in the local path of the clients or on a network path

            storage_name    - name of the storage to be created
            media           - media agent to mount the storage on
            type_of_storage - type of path to mount the storage on, local or network
            folder_path     - directory path to mount the storage
            username        - the user name of the network machine where media is created
            password        - the password of the network machine where the media is created
        """
        self.__add_storage("Disk")
        self.fill_form_by_id("libraryName", storage_name)
        self.wait_for_completion()
        if self.check_if_entity_exists("id", "mediaAgent"):
            self.select_value_from_dropdown("mediaAgent", media)
            self.wait_for_completion()
        if type_of_storage == "Local path":
            type_of_storage = "localStorage"
        elif type_of_storage == "Network path":
            type_of_storage = "networkStorage"
        else:
            raise Exception("Please pass the correct value for type of storage")
        self.driver.find_element_by_xpath(
            "//input[@type='radio' and @value='" + type_of_storage + "']").click()
        if type_of_storage == 'networkStorage':
            self.fill_form_by_id("loginName", username)
            self.driver.find_element_by_xpath(
                "//form/div[2]/label[2]/input[@type='password']").send_keys(password)
        self.fill_form_by_id("mountPath", folder_path)
        if self.check_if_entity_exists("xpath", "//form/label[3]/span[@class='help-block']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//form/label[3]/span[class='help-block']").text)
        self.driver.find_element_by_xpath("//form/div[2]/button[2]/div").click()
        self.wait_for_completion()
        if self.check_if_entity_exists(
                "xpath", "//div/div[2]/p[1][@class='serverMessage error']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//div/div[2]/p[1][@class='serverMessage error']").text)
        else:
            self.log.info("Created new library " + storage_name)

    @adminconsole_handler()
    def add_cloud_storage(
            self,
            storage_name,
            media,
            type_of_storage,
            username,
            password,
            server_host,
            bucket,
            proxy_address=None,
            proxy_port=None,
            proxy_username=None,
            proxy_password=None):
        """Add new cloud library for the MA specified
            The storage can either be created in the local path of the clients or on a network path

            storage_name   - name of the storage to be created
            media         - media agent to mount the storage on
            type_of_storage - type of path to mount the storage on, local or network
            username      - username in case of network path
            password      - password in case of network path
            server_host    - host of the server in case of cloud library type
            bucket        - buckets to be associated with the storage
            proxy_addr     - proxy address for the cloud
            proxy_port     - proxy port for the cloud
            proxy_username - proxy username for the cloud
            proxy_password - proxy password for the cloud
        """
        self.__add_storage("Cloud")
        self.fill_form_by_id("libraryName", storage_name)
        self.select_value_from_dropdown("cloudType", type_of_storage)
        self.select_value_from_dropdown("mediaAgent", media)
        self.fill_form_by_id("ServerHost", server_host)
        self.fill_form_by_id("UserName", username)
        self.fill_form_by_id("Password", password)
        self.fill_form_by_id("Bucket", bucket)
        self.fill_form_by_id("libraryName", storage_name)
        self.driver.find_element_by_link_text("Proxy").click()
        self.wait_for_completion()
        if proxy_address:
            self.fill_form_by_id("ProxyAddress", proxy_address)
        if proxy_port:
            self.fill_form_by_id("ProxyPort", proxy_port)
        if proxy_username:
            self.fill_form_by_id("ProxyUserName", proxy_username)
        if proxy_password:
            self.fill_form_by_id("ProxyPassword", proxy_password)
        self.driver.find_element_by_xpath("//form/div[2]/button[2]/div").click()
        self.wait_for_completion()
        if self.check_if_entity_exists(
                "xpath", "//div/div[2]/span[@class='serverMessage error']"):
            raise Exception(self.driver.find_element_by_xpath(
                "//div/div[2]/span[@class='serverMessage error']").text)
        else:
            self.log.info("Created new library " + storage_name)

    @adminconsole_handler()
    def __add_storage(self, storage_type):
        """
        Selects the storage of given type to be added
        :param storage_type     (str)   --  type of storage target to add
        """
        if self.check_if_entity_exists("xpath", "//span[contains(text(),'Add storage')]"):
            self.driver.find_element_by_xpath("//span[contains(text(),'Add storage')]").click()
            self.driver.find_element_by_link_text(storage_type).click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to add storage")

    @adminconsole_handler()
    def storage_details(self):
        """Get the details of all the storage libraries in the form of a dictionary
            storage_name status Manufacturer Model

            Ex: {
                'storage1': 'Ready Disk Disk',
                'storage2': 'Ready Disk Disk'
            }
        """
        self.log.info("Printing all the library details as a dictionary")
        storage_dict = {}
        while True:
            elements = self.driver.find_elements_by_xpath(
                "//div[1]/div[2]/div[2]/div[@class='ui-grid-canvas']/div")
            for elem in elements:
                key = elem.find_element_by_xpath("./div/div[1]/a").text
                value = []
                for index in range(2, 4):
                    value.append(
                        elem.find_element_by_xpath(
                            "./div/div[" + str(index) + "]/div").text)
                value.append(elem.find_element_by_xpath("./div/div[4]/span").text)
                storage_dict[key] = value
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        self.log.info(storage_dict)

    @adminconsole_handler()
    def open_storage(self, storage):
        """Open the storage with the given name

            storage - name of the storage to be opened
        """
        self.log.info("Opening storage " + storage)
        self.__table.access_link(storage)

    @adminconsole_handler()
    def action_delete(self, storage_name):
        """Delete the storage with the given name
            storage_name - name of the storage to be removed
        """
        self.log.info("Going to delete the storage " + storage_name)
        self.__table.search_for(storage_name)
        if self.check_if_entity_exists("link", storage_name):
            self.driver.find_element_by_xpath(
                "//a[contains(text(),'" + storage_name + "')]/../../div[5]/div/a/span"
                ).click()
            if self.check_if_entity_exists("link", "Delete"):
                self.driver.find_element_by_xpath(
                    "//a[contains(text(),'" +storage_name + "')]/../../div[5]/div/ul//a["
                                                            "contains(text(),'Delete')]")\
                    .click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath(
                    "//div/div[3]/button[contains(text(),'Yes')]").click()
                self.wait_for_completion()
            else:
                raise Exception("There is no option to delete the library " + storage_name)
        else:
            raise Exception("There is no storage targt with name" + storage_name)
