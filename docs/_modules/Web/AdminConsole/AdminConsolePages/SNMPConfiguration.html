    @adminconsole_handler()
    def add_configuration(
            self,
            config_name,
            encryption_algorithm,
            username,
            password,
            privacy_algorithm,
            priv_password):
        """
        Method to create new SNMP Configuration

        Args:
            config_name (string): Name of the configuration to be created
            username (string): username for configuration
            password (string): password for configuration
            priv_password (string): privacy password for configuration
            new_encryption_algorithm (string): value of encryption algorithm to be selected for SNMP
            new_privacy_algorithm (string): value of privacy algorithm to be selected for SNMP

        Returns:
            None

        Raises:
            Exception:
                -- if configuration already exists
                -- if failed to create SNMP configuration

        """
        if self.check_if_entity_exists("link", "Add configuration"):
            self.driver.find_element_by_link_text(
                "Add configuration").click()
        else:
            exp = "User does not have permission to create SNMP configuration"
            self.log.info(exp)
            raise Exception(exp)

        self.fill_form_by_id("hostname", config_name)

        self.select_value_from_dropdown(
            "encryptionAlgorithm", encryption_algorithm)
        self.wait_for_completion()
        self.fill_form_by_id("username", username)
        self.fill_form_by_id("password", password)
        self.fill_form_by_id("confirmPassword", password)

        if privacy_algorithm:
            self.select_value_from_dropdown(
                "privacyAlgorithm", privacy_algorithm)
            self.fill_form_by_id("privacyPassword", priv_password)
            self.fill_form_by_id("confirmPrivacyPassword", priv_password)

        self.driver.find_element_by_xpath(
            "//button[contains(text(),'Save')]").click()

        if self.check_if_entity_exists(
                "xpath", "//span[@class='help-block']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='help-block']").text
            raise Exception(exp)
        if self.check_if_entity_exists(
                "xpath", "//span[@class='server-message error']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='server-message error']").text
            raise Exception(exp)
        self.wait_for_completion()

    @adminconsole_handler()
    def open_configuration(self, config_name):

        """Opens backup rule details
        Args:
            config_name (str): name for configuration name to be opened

        Returns:
            None

        Raises:
            Exception:
                None"""

        self.search_for(config_name)
        if self.check_if_entity_exists("link", config_name):
            self.driver.find_element_by_link_text(config_name).click()
            self.wait_for_completion()

    @adminconsole_handler()
    def edit_configuration(
            self,
            edit_config,
            old_config_name,
            new_config_name=None,
            new_encryption_algorithm=None,
            new_username=None,
            new_password=None,
            new_privacy_algorithm=None,
            new_priv_password=None):
        """
        Method to edit an existing SNMP Configuration

        Args:
            old_config_name (str): IP address of the configuration to be modified
            new_config_name (str): IP address of the configuration to be created
            new_username (str): username for new configuration
            new_password (str): password for new configuration
            new_priv_password (str): private password for new configuration
            new_encryption_algorithm (str): value of encryption algorithm to be updated for SNMP
            new_privacy_algorithm (str): value of privacy algorithm to be updated for SNMP

        Returns:
            None

        Raises:
            Exception:
                -- if failed to edit SNMP configuration

        """

        if edit_config == 1 or not new_config_name:
            self.open_configuration(old_config_name)
        else:
            self.open_configuration(new_config_name)

        if new_config_name:
            self.fill_form_by_id("hostname", new_config_name)

        if new_encryption_algorithm:
            self.select_value_from_dropdown(
                "encryptionAlgorithm", new_encryption_algorithm)
            self.wait_for_completion()
            if new_username:
                self.fill_form_by_id("username", new_username)
                self.fill_form_by_id("password", new_password)
                self.fill_form_by_id("confirmPassword", new_password)

            if new_privacy_algorithm:
                self.select_value_from_dropdown(
                    "privacyAlgorithm", new_privacy_algorithm)
                if new_priv_password:
                    self.fill_form_by_id("privacyPassword", new_priv_password)
                    self.fill_form_by_id("confirmPrivacyPassword", new_priv_password)

        self.driver.find_element_by_xpath(
            "//button[contains(text(),'Save')]").click()
        if self.check_if_entity_exists(
                "xpath", "//span[@class='help-block']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='help-block']").text
            self.log.info(exp)
            raise Exception(exp)
        if self.check_if_entity_exists("xpath", "//span[@class=\
        'server-message error']"):
            exp = self.driver.find_element_by_xpath("//span[@class=\
            'server-message error']").text
            self.log.info(exp)
            raise Exception(exp)
        self.wait_for_completion()

    @adminconsole_handler()
    def delete_configuration(self, config_name):
        """
        Method to delete SNMP Configuration
        Args:
            config_name (str): IPaddress/Name of the configuration to be deleted.
        Returns:
            None
        """
        self.search_for(config_name)
        self.driver.find_element_by_xpath(
            "//a[contains(text(),'" + config_name + "')]\
/ancestor::div[@class='ui-grid-row ng-scope']/div/div[5]/div/a").click()

        self.driver.find_element_by_link_text("Delete").click()
        self.wait_for_completion()
        self.driver.find_element_by_xpath(
            "//button[contains(text(),'Yes')]").click()

        if not self.driver.find_element_by_xpath("//div[@class='ng-binding ng-scope']")\
        .text == "SNMP configuration deleted successfully":
            exp = self.driver.find_element_by_xpath("//div[@class='ng-binding ng-scope']").text
            raise Exception(exp)
        else:
            self.log.info("Delete message displayed correctly")

        if self.check_if_entity_exists(
                "xpath", "//span[@class='help-block']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='help-block']").text
            self.log.info(exp)
            raise Exception(exp)
        if self.check_if_entity_exists("xpath", "//span[@class=\
        'server-message error']"):
            exp = self.driver.find_element_by_xpath("//span[@class=\
            'server-message error']").text
            self.log.info(exp)
            raise Exception(exp)
