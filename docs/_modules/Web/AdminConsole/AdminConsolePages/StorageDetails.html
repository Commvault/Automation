    @adminconsole_handler()
    def add_mount_path(
            self,
            media,
            storage_loc,
            drive,
            username=None,
            password=None):
        """Add a mount path to the storage

            media       - media agent you want to mount on
            storage_loc  - path or location of the directory you want to use
            drive       - the drive letter of the storage_loc
            username    - username to login with in case of network storage
            password    - password to login with in case of network storage
        """
        self.log.info("Adding mount path to the library")
        if self.check_if_entity_exists("link", "Add disk"):
            self.driver.find_element_by_link_text("Add disk").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("id", "mediaAgent"):
                Select(self.driver.find_element_by_name(
                    "mediaAgent")).select_by_visible_text(media)
            if storage_loc == "Local path":
                storage_loc = "localStorage"
            elif storage_loc == "Network path":
                storage_loc = "networkStorage"
            else:
                raise Exception("Please pass the correct value for type of storage")
            self.driver.find_element_by_xpath(
                "//input[@type='radio' and @value='" + storage_loc + "']").click()
            self.fill_form_by_id("mountPath", drive)
            if self.check_if_entity_exists(
                    "xpath", "//form/label[3]/span[class='help-block']"):
                exp = self.driver.find_element_by_xpath(
                    "//form/label[3]/span[class='help-block']").text
                raise Exception(exp)
            if storage_loc == 'networkStorage':
                self.fill_form_by_id("loginName", username)
                self.driver.find_element_by_xpath(
                    "//form/div[2]/label[2]/input[@type='password']").send_keys(password)
            self.driver.find_element_by_xpath(
                "//button[2]/div").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("xpath", "//div[@class='setup-content']/p[1]"):
                raise Exception(self.driver.find_element_by_xpath(
                    "//div[@class='setup-content']/p[1]").text)
            self.log.info("Added new path to the library")
        else:
            raise Exception("There is no link to Add mount path")

    @adminconsole_handler()
    def open_mount_path(self, mount_path):
        """Open the mount path of the storage with the given mount path
            mountPath - the path of the storage to be opened
        """
        self.log.info("Opening mount path " + mount_path)
        self.driver.find_element_by_id("search-field").clear()
        self.driver.find_element_by_id("search-field").send_keys(mount_path)
        self.wait_for_completion()
        if self.check_if_entity_exists("xpath", "//a[text()='" + mount_path + "']"):
            self.driver.find_element_by_partial_link_text(
                mount_path).click()
            self.wait_for_completion()
        else:
            raise Exception("There is no mount path " + mount_path)

    @adminconsole_handler()
    def share_mount_path(
            self,
            mount_path,
            share):
        """
        Share the mount path with other media agents
        :param mount_path   (str)   --  the mount path to share
        :param share        (dict)  --  info about media agents and share access
        """
        self.log.info("Sharing mountpath " + mount_path)
        self.driver.find_element_by_id("search-field").clear()
        self.driver.find_element_by_id("search-field").send_keys(mount_path)
        self.wait_for_completion()
        if self.check_if_entity_exists("xpath", "//a[text()='" + mount_path + "']"):
            self.driver.find_element_by_xpath(
                "//a[text()='" + mount_path + "']/../../div[5]/div/a/span").click()
            if self.check_if_entity_exists(
                    "xpath", "//a[text()='" + mount_path + "']/../../div[5]/div/ul/li[2] \
                    /a[contains(text(),'Share')]"):
                self.driver.find_element_by_xpath(
                    "//a[text()='" + mount_path + "']/../../div[5]/div/ul/li[2]/a[contains( \
                    text(),'Share')]").click()
                self.wait_for_completion()
                ma_agents = share.keys()
                for ma_g in ma_agents:
                    Select(self.driver.find_element_by_xpath(
                        "//div[contains(text(),'" + ma_g + "')]/../../div[3]/select"))\
                        .select_by_visible_text(share[ma_g])
                elements = self.driver.find_elements_by_xpath(
                    "//div[@class='setup-content wizard-scroll ng-scope']//div[@class="
                    "'ui-grid-contents-wrapper']/div[2]/div[2]/div[@class='ui-grid-canvas']"
                    "/div")
                index = 1
                flag = []
                for elem in elements:
                    f_ma = elem.find_element_by_xpath("./div/div[1]/div").text
                    for ma_g in ma_agents:
                        if f_ma == ma_g:
                            flag.append(index)
                            break
                        else:
                            continue
                    index += 1
                for flg in flag:
                    flg = str(flg)
                    self.driver.find_element_by_xpath(
                        "//div[1]/div/div[2]/div[@class='ui-grid-canvas']/div[" + flg + " \
                        ]/div/div/div/div/div").click()
            else:
                raise Exception("There is no option to share the mount path " + mount_path)
        else:
            raise Exception("There is no mount path by the name " + mount_path)

    @adminconsole_handler()
    def action_share_mount_path(
            self,
            mount_path,
            share,
            storage_loc,
            allowed_writers=None,
            writers=None,
            username=None,
            password=None,
            preferred=False,
            enable=False):
        """Share the mount path with specific media agents
            mountPath       - the mount path to be shared
            share           - type of access the ma can have
            storage_loc      - type of storage, local or network
            allowed_writers  - bool, to allocate the no. of writers manually
            writers         - total no. of writers allowed
            username        - username in case of network storage
            password        - password in case of network storage
            preferred       - choose whether this is the preferred mount path
            enable          - choose to enable this mountPath or
        """
        ret = self.share_mount_path(mount_path, share)
        if ret:
            self.driver.find_element_by_xpath(
                "//div[2]/div[3]/button[contains(text(), 'Next')]").click()
            self.wait_for_completion()
            if self.check_if_entity_exists(
                    "xpath", "//div[2]/div[3]/button[contains(text(), 'Next')]"):
                raise Exception(self.driver.find_element_by_xpath(
                    "//div[2]/div[1]/div/div/div[2]").text)
            else:
                self.driver.find_element_by_id("mountPath").send_keys(mount_path)
                if storage_loc == "Local path":
                    storage_loc = "localStorage"
                elif storage_loc == "Network path":
                    storage_loc = "networkStorage"
                else:
                    raise Exception("Please pass the correct value for type of storage")
                if storage_loc == 'networkStorage':
                    self.driver.find_element_by_name(
                        "loginName").send_keys(username)
                    self.driver.find_element_by_xpath(
                        "//form/div[2]/label[2]/input[@type='password']").send_keys(password)
                if allowed_writers == "Allocate number of writers":
                    self.driver.find_element_by_xpath(
                        "//form[@name='storageForm']/span/div[1]/label[3]/input").click()
                    self.driver.find_element_by_name(
                        "allowedWritersNumber").clear()
                    self.driver.find_element_by_name(
                        "allowedWritersNumber").send_keys(writers)
                if preferred:
                    if not self.driver.find_element_by_id(
                            "preferred").is_enabled():
                        self.driver.find_element_by_id(
                            "preferred").click()
                if enable:
                    if not self.driver.find_element_by_id(
                            "enable").is_enabled():
                        self.driver.find_element_by_id(
                            "enable").click()
                self.driver.find_element_by_xpath(
                    "//div[2]/div[3]/button[contains(text(), 'Finish')]").click()
                self.wait_for_completion()

    @adminconsole_handler()
    def action_delete_mount_path(self, mount_path):
        """Delete a mount path with the given name
            mountPath - the mount path to be removed
        """
        self.log.info("Deleting mountpath " + mount_path)
        if self.check_if_entity_exists("xpath", "//a[text()='" + mount_path + "']"):
            self.driver.find_element_by_xpath("//a[text()='" + mount_path + "']/../../div[5] \
                    /div/a/span").click()
            if self.check_if_entity_exists(
                    "xpath", "//a[text()='" + mount_path + "']/../../div[5]/div/ul/li[1]/a \
                    [contains(text(),'Delete')]"):
                self.driver.find_element_by_xpath(
                    "//a[text()='" + mount_path + "']/../../div[5]/div/ul/li[1]/a[contains( \
                    text(),'Delete')]").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath(
                    "//div/div[3]/button[contains(text(),'Yes')]").click()
                self.wait_for_completion()
                if self.check_if_entity_exists(
                        "xpath", "//div[@class='modal-body text-danger \
                         ng-binding ng-scope']"):
                    raise Exception(self.driver.find_element_by_xpath(
                        "//div[@class='modal-body text-danger ng-binding ng-scope']").text)
            else:
                raise Exception("There is no option to delete the mount path " + mount_path)
        else:
            raise Exception("There is no mount path named " + mount_path)

    @adminconsole_handler()
    def list_of_mount_paths(self):
        """List all the mount paths of the storage
        """
        self.log.info("Listing all the mount paths for the library")
        paths = self.driver.find_elements_by_xpath(
            "//div[@class='ui-grid-canvas']/div")
        mount_paths = {}
        for path in paths:
            key = path.find_element_by_xpath("./div/div[1]/a").text
            value = []
            for index in range(2, 4):
                value.append(
                    path.find_element_by_xpath(
                        "./div/div[" + str(index) + "]/div").text)
            mount_paths[key] = value
        self.log.info(mount_paths)

    @adminconsole_handler()
    def library_info(self):
        """Display the library information like total capacity,
            free space and throughput in the last 24 hours in the form of a dictionary
            :return:
                lib_info    (dict)  -- contains all the info about the library
        """
        self.log.info("Collecting information about the library")
        elements = self.driver.find_elements_by_xpath(
            "//div[@class='detailsPropsInner']")
        lib_info = {}
        for elem in elements:
            key = elem.find_element_by_xpath("./span[2]").text
            value = elem.find_element_by_xpath("./span[1]").text
            lib_info[key] = value
        self.log.info(lib_info)
        return lib_info
