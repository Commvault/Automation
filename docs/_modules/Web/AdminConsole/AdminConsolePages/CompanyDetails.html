    @adminconsole_handler()
    def edit_general_settings(self, general_settings):
        """
        Method to edit general setting associated o a company

        Args:
            general_settings (dict): dictionary containing values for general settings
                Eg. - general_settings = {'smtp':'comm.com', 'authcode': 'ON',
                                      'shared_laptop_usage': 'ON', 'UPN': 'ON'}
        Returns:
            None

        Raises:
            Exception:
                if fails to edit general settings
        """
        self.log.info("Editing general settings for the company")
        if self.check_if_entity_exists("xpath", "//div[@class='page-details group']\
        //span[contains(text(),'General')]"):
            general_tile = self.driver.find_element_by_xpath("//div[@class='page-details group']\
//span[contains(text(),'General')]/following-sibling::div")
            if general_settings['smtp']:
                self.log.info("Editing smtp associated to the company")
                general_tile.find_element_by_xpath(".//a[contains(text(),'Edit')]").click()
                self.fill_form_by_id("smtp", general_settings['smtp'])
                self.wait_for_completion()
                self.driver.find_element_by_xpath(
                    "//button[@class='btn btn-primary cvBusyOnAjax']").click()
                self.wait_for_completion()

            update_options = self.driver.find_elements_by_xpath("//toggle-control")

            if general_settings['authcode'] == "ON":
                self.log.info("Editing authcode requirement for installation for the company")
                if not update_options[0].find_element_by_xpath("./div").get_attribute("class") \
                    == "cv-material-toggle cv-toggle enabled":
                    update_options[0].find_element_by_xpath("./div").click()
            else:
                self.log.info("Editing authcode requirement for installation for the company")
                if not update_options[0].find_element_by_xpath("./div").get_attribute("class") \
                    == "cv-material-toggle cv-toggle":
                    update_options[0].find_element_by_xpath("./div").click()
                    self.wait_for_completion()
                    self.driver.find_element_by_xpath("//button[contains(text(),'Yes')]").click()
            self.wait_for_completion()

            if general_settings['UPN'] == "ON":
                self.log.info("Editing UPN usage for the company")
                if not update_options[1].find_element_by_xpath("./div").get_attribute("class") \
                    == "cv-material-toggle cv-toggle enabled":
                    update_options[1].find_element_by_xpath("./div").click()
            else:
                self.log.info("Editing UPN usage for the company")
                if not update_options[1].find_element_by_xpath("./div").get_attribute("class") \
                    == "cv-material-toggle cv-toggle":
                    update_options[1].find_element_by_xpath("./div").click()
            self.wait_for_completion()

            if self.check_if_entity_exists(
                    "xpath", "//span[@class='help-block']"):
                exp = self.driver.find_element_by_xpath(
                    "//span[@class='help-block']").text
                raise Exception(exp)

            if self.check_if_entity_exists(
                    "xpath", "//span[@class='serverMessage error']"):
                exp = self.driver.find_element_by_xpath(
                    "//span[@class='serverMessage error']").text
                raise Exception(exp)
        else:
            exp = "There is no option to edit the contacts of the company"
            raise Exception(exp)

    @adminconsole_handler()
    def edit_sites(self, primary_site, secondary_sites):
        """
        Edits the domain information of the company

        Args:
            primary_site(str) -- the name of the primary site to be added
            secondary_sites (list) -- list of all secondary sites to be added

        Returns:
            None

        Raises:
            Exception:
                -- if fails to set edit sites for the company
        """
        self.log.info("Editing sites associated to the company")
        self.log.info("Editing the site information of the company")
        if self.check_if_entity_exists(
                "xpath", "//cv-tile-component[@data-ac-id='label_domains-subscriptions']"):
            self.driver.find_element_by_xpath(
                "//cv-tile-component[@data-ac-id='label_domains-subscriptions']\
                //a[contains(text(),'Edit')]").click()
            self.wait_for_completion()
            if primary_site:
                self.fill_form_by_id("primaryDomainName", primary_site)

            if secondary_sites:
                old_secondary_site_entries = self.driver.find_elements_by_xpath(
                    "//a[@title='Remove']")
                for old in old_secondary_site_entries:
                    self.driver.find_element_by_xpath("//a[@title='Remove']").click()

                secondary_sites = [str(i) for i in secondary_sites.split(',')]
                for site in secondary_sites:
                    self.driver.find_element_by_link_text(
                        "Add an additional site").click()
                    self.wait_for_completion()
                index = len(secondary_sites)

                for elem in range(1, index+1):
                    self.driver.find_element_by_xpath(
                        "//form/div[2]/ul["+str(elem)+"]/li/label/input"
                    ).send_keys(secondary_sites[elem - 1])

            self.driver.find_element_by_xpath(
                "//form/div[3]/button[2]").click()
            self.wait_for_completion()

            if self.check_if_entity_exists(
                    "xpath", "//span[@class='help-block']"):
                exp = self.driver.find_element_by_xpath(
                    "//span[@class='help-block']").text
                raise Exception(exp)

            if self.check_if_entity_exists(
                    "xpath", "//span[@class='serverMessage error']"):
                exp = self.driver.find_element_by_xpath(
                    "//span[@class='serverMessage error']").text
                raise Exception(exp)
        else:
            exp = "There is no option to edit the domains of the company"
            raise Exception(exp)

    @adminconsole_handler()
    def extract_company_info(self):
        """
        Extracts all the information about the company

        Args:
            None

        Returns:
            company_info (dict) -- info about all the companies

        Raises:
            Exception:
                -- if fails to return company info
        """
        self.log.info("Extracting information about the company")
        company_info = {}
        elements = self.driver.find_elements_by_xpath(
            "//div[@class='page-details group']")
        for elem in elements:
            key = elem.find_element_by_xpath("./span").text
            values_list = []
            values_dict = {}
            div_elems = elem.find_elements_by_xpath("./div[@class='ng-scope']")

            if key == "Email settings":
                for div_elem in div_elems:
                    span_elems = div_elem.find_elements_by_xpath(
                        ".//span[@class='ng-scope']")
                    if not span_elems:
                        li_elems = div_elem.find_elements_by_xpath(".//li")
                        for li_elem in li_elems:
                            if not li_elem.find_element_by_xpath(".//span[1]").\
                                text == '':
                                if self.is_element_present(
                                        "./sapn", li_elem):
                                    val = li_elem.find_element_by_xpath(
                                        "./span").text
                                    values_dict[li_elem.find_element_by_xpath(
                                        "./sapn").text] = val
                                else:
                                    val = li_elem.find_element_by_xpath(
                                        "./span[2]").text
                                    if 'Edit' in val:
                                        val_new = val.rstrip('\nEdit')
                                        values_dict[li_elem.find_element_by_xpath(
                                            "./span[1]").text] = val_new
                                    else:
                                        values_dict[li_elem.find_element_by_xpath(
                                            "./span[1]").text] = val
                        company_info.update({key: values_dict})
                    else:
                        for span_elem in span_elems:
                            values_list.append(span_elem.text)
                        company_info.update({key: values_list})

            elif key == "Plans" or key == "Navigation preferences":
                for div_elem in div_elems:
                    li_elems = div_elem.find_elements_by_xpath(".//li")
                    for li_elem in li_elems:
                        tag = li_elem.find_element_by_xpath(".//a").text
                        values_list.append(tag)
                company_info.update({key: values_list})
            else:
                for div_elem in div_elems:
                    li_elems = div_elem.find_elements_by_xpath(".//li")
                    for li_elem in li_elems:
                        if self.is_element_present(
                                ".//toggle-control", li_elem):
                            if li_elem.find_element_by_xpath(
                                    ".//toggle-control/div").get_attribute("class") \
                                == "cv-material-toggle cv-toggle":
                                values_dict[li_elem.find_element_by_xpath(
                                    ".//span[@class='pageDetailColumn']").text] = "OFF"
                            else:
                                values_dict[li_elem.find_element_by_xpath(
                                    ".//span[@class='pageDetailColumn']").text] = "ON"
                        elif not li_elem.find_element_by_xpath(".//span[1]").\
                        text == '':
                            val = li_elem.find_element_by_xpath(
                                ".//span[2]").text
                            if 'Edit' in val:
                                val_new = val.rstrip('\nEdit')
                                values_dict[li_elem.find_element_by_xpath(
                                    ".//span[1]").text] = val_new
                            else:
                                values_dict[li_elem.find_element_by_xpath(
                                    ".//span[1]").text] = val
                company_info.update({key: values_dict})
        return company_info
