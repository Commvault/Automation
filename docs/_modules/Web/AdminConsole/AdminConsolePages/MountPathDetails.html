"""
This module provides the function or operations that can be performed on the
mount path selected on the AdminConsole

Class:

    MountPathDetails() -> StorageDetails() -> Storage() --
        -> AdminPage() -> login_page() -> AdminConsoleBase() -> object()

Functions:

edit_path()      -- edit the mount path details
mount_path_info() -- print the info of the mount path in the log file
open_storage()   -- navigate to the storage's main page

"""
from Web.Common.page_object import (
    WebAction,
    PageService
)
from Web.AdminConsole.Components.panel import PanelInfo
from Web.AdminConsole.Components.table import Table

    """
    Class for MountPathDetails
    """
    def __init__(self, admin_base):
        """
        Method to initiate Users class

        Args:
            admin_base   (Object) :   Admin base Class object
        """
        self.__admin_console = admin_base
        self.__panel = PanelInfo(self.__admin_console)
        self.__table = Table(self.__admin_console)

    @WebAction()
    def __fill_nw_cred(self, username, password):
        """
        fills network credentials
        Args:
            username: username
            password: password
        """
        self.__admin_console.driver.find_element_by_name("loginName").send_keys(username)
        self.__admin_console.driver.find_element_by_xpath(
            "//form/div[2]/label[2]/input[@type='password']").send_keys(password)

    @WebAction()
    def __fill_writers(self, writers):
        """
        fills writers count
        Args:
            writers: writer value to be filled
        """
        elem = self.__admin_console.driver.find_element_by_name("allowedWritersNumber")
        elem.clear()
        elem.send_keys(writers)

    @PageService()
    def edit_path(
            self,
            mount_path,
            access_type,
            storage_loc,
            username=None,
            password=None,
            allowed_writers=None,
            writers=None,
            preferred=False,
            enable=False):
        """Edit the mount path of the storage
            mount_path  - mount path to be edited
            accessType - type of access for this mount path
            storage_loc - type of storage, local or network
            username   - username in case of network storage
            password   - password in case of network storage
            writers    - total no. of writers allowed
            preferred  - choose whether this is the preferred mount path
            enable     - choose to enable this mountPath or not
        """
        self.__table.access_link(mount_path)
        self.__admin_console.select_value_from_dropdown("access_type", access_type)
        if storage_loc not in ['Local path', 'Network path']:
            raise Exception("Please pass the correct value for type of storage")
        self.__admin_console.select_radio(storage_loc)
        if storage_loc == 'Network path':
            self.__fill_nw_cred(username, password)
        if access_type == "Read/Write":
            if allowed_writers == "Allocate number of writers":
                self.__admin_console.select_radio('Allocate number of writers')
                self.__fill_writers(writers)
        if preferred:
            self.__admin_console.checkbox_select('preferred')
        if enable:
            self.__admin_console.checkbox_select('enable')
        self.__panel.submit()
        self.__admin_console.wait_for_completion()

    @PageService()
    def mount_path_info(self):
        """Gather info about the mount path
        """
        return self.__panel.get_details()
