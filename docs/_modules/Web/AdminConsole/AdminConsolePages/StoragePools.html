"""

    select_storage_pool()           -- opens a storage pool

    action_delete()                 -- removes a storage pool

from Web.AdminConsole.Components.table import Table
from Web.Common.handlers import adminconsole_handler
from Web.AdminConsole.AdminConsolePages.AdminPage import AdminPage
from Web.Common.page_object import PageService



[docs]class StoragePools(AdminPage):
    """
    This class provides the function or operations that can be
    performed on the Storage Pool Page of the Admin Console
    """

    def __init__(self, driver):
        super().__init__(driver)
        self.__table = Table(self)

    @PageService()
    def add_tape_storage_pool(self, pool_name=None, storage_target=None):
        """
        Add a new tape storage pool

        Args:
            pool_name       (str)   -- Name for the tape storage pool

            storage_target  (str)   -- storage target for creating the storage pool

        """
        self.select_hyperlink('Add storage pool')
        self.select_hyperlink('Tape')
        self.fill_form_by_id("storagePoolName", pool_name)
        self.select_value_from_dropdown("tapeStorage", storage_target)
        self.click_button('Finish')

    @adminconsole_handler()
    def add_disk_cloud_storage_pool(
            self,
            pool_name=None,
            media_agent=None,
            storage_target=None,
            username=None,
            password=None,
            path=None,
            partition_media_agent=None,
            partition_path=None):
        """
        To add a new disk/cloud storage pool

        Args:
            pool_name       (str)       -- Name of the storage pool to be created

            media_agent     (str)       -- Media gent to create storage pool on

            storage_target  (str)       -- Storage target to create storage pool

            username        (str)       -- username for the network path

            password        (str)       -- password for the network path

            path            (str)       -- local/network path for the storage pool

            partition_media_agent (str) -- Media agent for the partition settings

            partition_path  (list)      -- list of paths to be added to partition settings

        **Note** Media agent is required for creating a new storage,
                 to use an existing storage, storage target is required

        """
        self.select_hyperlink('Add storage pool')
        self.select_hyperlink('Disk/Cloud')
        self.fill_form_by_id("storagePoolName", pool_name)

        # To create a new storage
        if media_agent:
            self.driver.find_element_by_id("newStorage").click()
            self.select_value_from_dropdown("mediaAgent", media_agent)
            if username and password:
                self.driver.find_element_by_id('networkStorage').click()
                self.fill_form_by_id("userName", username)
                self.fill_form_by_id("password", password)
            else:
                self.driver.find_element_by_id('localStorage').click()

            self.fill_form_by_id("mountPath", path)

        # To use existing storage
        else:
            self.driver.find_element_by_id("existingStorage").click()
            self.select_value_from_dropdown("storage", storage_target)

        self.click_button('Next')
        self.check_error_message()
        self.select_value_from_dropdown('dedupeMediaAgent', partition_media_agent)

        if isinstance(partition_path, list):
            for path in partition_path:
                self.fill_form_by_id('partitionPath', path)
                self.select_hyperlink('Add partition')
        else:
            self.fill_form_by_id('partitionPath', partition_path)
            self.select_hyperlink('Add partition')

        self.click_button('Finish')
        self.check_error_message()

    @adminconsole_handler()
    def list_storage_pool(self):
        """Get the details of all the storage pools in the form of a dictionary
            storage_name status Manufacturer Model

            Ex: {
                'storage1': 'Ready Disk Disk',
                'storage2': 'Ready Disk Disk'
            }
            :return:
                storage_dict    (dict)  --  details of all storage pools
        """
        storage_dict = {}
        while True:
            elements = self.driver.find_elements_by_xpath(
                "//div[@class='ui-grid-contents-wrapper']/div \
                [2]/div[2]/div[@class='ui-grid-canvas']/div")
            for elem in elements:
                key = elem.find_element_by_xpath("./div/div[1]/div/a").text
                value = []
                for index in range(2, 7):
                    if index == 2:
                        value.append(elem.find_element_by_xpath("./div/div[" + str(index) +
                                                                "]/div/span").text)
                    elif index == 4:
                        value.append(elem.find_element_by_xpath("./div/div[" + str(index) +
                                                                "]/div").text)
                    else:
                        value.append(elem.find_element_by_xpath("./div/div[" + str(index) +
                                                                "]/span").text)
                storage_dict[key] = value
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        self.log.info(storage_dict)
        return storage_dict

    @PageService()
    def select_storage_pool(self, storage_pool):
        """
        selects the storage pool with the given name

        Args:
            storage_pool    (str)   -- Name of the storage pool to be opened

        """
        self.__table.access_link(storage_pool)

    @adminconsole_handler()
    def action_delete(self, storage_pool):
        """
        Deletes the storage pool with the given name

        Args:
            storage_pool - name of the storage to be removed

        """
        self.__table.access_action_item(storage_pool, 'Delete')
        self.driver.find_element_by_xpath("//button[text()='Yes'][@ng-click='yes()']").click()
        self.check_error_message()
