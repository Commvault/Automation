    @adminconsole_handler()
    def plan_info(self):
        """ Retrieves all plan information displayed from Plan Details Section"""

        self.log.info("Getting all plan information")
        details = {}
        title = self.driver.find_element_by_xpath('//h1[@class=\
        "page-title editable js-page-title ng-binding"]')
        details.update({"PlanName": title.text})
        elems = self.driver.find_elements_by_xpath(
            "//div[@class='page-details group']")
        for elem in elems:
            entity_name = elem.find_element_by_xpath("./span").text
            div_elements = elem.find_elements_by_xpath("./div")
            for div_elem in div_elements:
                keyvalue_list = []
                keyvalue_dict = {}
                key = ''
                rpo_count = 0
                if entity_name == "Storage":
                    div_elems_store = elem.find_elements_by_xpath(".//div[@class=\
                    'ng-isolate-scope']")
                    for div_elem_store in div_elems_store:
                        key = div_elem_store.find_element_by_xpath(
                            "./div[1]/span") .text
                        value = div_elem_store.find_element_by_xpath(
                            "./div[3]/span") .text
                        keyvalue_dict.update({key: value})
                    details.update({entity_name: keyvalue_dict})
                    break
                ul_elements = div_elem.find_elements_by_xpath(".//ul")
                if not ul_elements:
                    span_elements = div_elem.find_elements_by_xpath(".//span")
                    for span_elem in span_elements:
                        key = span_elem.text
                        keyvalue_list.append(key)
                    details.update({entity_name: keyvalue_list})
                    break
                for ul_elem in ul_elements:
                    li_elements = ul_elem.find_elements_by_xpath('.//li')
                    for li_elem in li_elements:
                        if rpo_count == 4:
                            break
                        inside_li_tags = li_elem.find_elements_by_xpath(
                            './/span')
                        if not inside_li_tags:
                            key = li_elem.text
                            keyvalue_list.append(key)
                            break
                        tag_iterate_count = 0
                        for tag in inside_li_tags:
                            if entity_name == "RPO":
                                rpo_count = rpo_count+1
                            tag_iterate_count += 1
                            if tag.text:
                                if tag_iterate_count == 1:
                                    key = tag.text
                                    if len(inside_li_tags) == 1:
                                        keyvalue_list.append(key)
                                        break
                                else:
                                    value = tag.text
                                    if 'Edit' in value:
                                        value = value.rstrip('\nEdit')
                                    keyvalue_dict.update({key: value})
                                    break
                            else:
                                if self.is_element_present(
                                        './Toggle-Control', tag):
                                    toggle_class = tag.find_element_by_xpath(
                                        './Toggle-Control').get_attribute('class')
                                    toggle_value = "OFF"
                                    if toggle_class == \
                                    "ng-isolate-scope enabled-activity":
                                        toggle_value = "ON"
                                    keyvalue_dict.update({key: toggle_value})
                                    break
                    if len(inside_li_tags) == 1:
                        details.update({entity_name: keyvalue_list})
                    elif not inside_li_tags:
                        details.update({entity_name: keyvalue_list})
                    else:
                        details.update({entity_name: keyvalue_dict})
        self.log.info(details)
        return details

    def edit_plan_storage_pool(self, additional_storage, edit_storage):
        """
        Method to edit storage pool associated with the plan

        Args:
            additional_storage (dictionary): Dictionary containing values for an additional
                storage, if it is to be added and the values of it
                Eg. - additional_storage = {'Add':False, 'storage_name':'Store',
                                                    'storage_pool':'Secondary',
                                                    'ret_period':'10'}
            edit_storage (dictionary): Dictionary containing values to edit existing storage
                Eg. - edit_plan_storage = {'Edit': True,
                                          'old_storage_name': self.tcinputs['primary_storage'],
                                          'new_storage_name':'Primary',
                                          'new_ret_period':'20'}

        Returns:
            None

        Raises:
            Exception:
                None
        """

        if self.check_if_entity_exists("xpath", "//div[@class='page-details group']\
        //span[contains(text(),'Storage')]"):

            if additional_storage['Add']:
                if self.check_if_entity_exists(
                        "xpath", "///cv-plan-storage-pool//\
a[@data-ng-click='addStorage(null, showSource)']"):
                    self.driver.find_element_by_xpath(
                        "//div[@class='page-details group']//span[contains(text()\
                        ,'Storage')]/following-sibling::div[@class ='ng-scope']\
                        //a[@data-ng-click='addStorage(null, showSource)']").click()

                    self.fill_form_by_id("storageName", additional_storage['storage_name'])
                    self.select_value_from_dropdown(
                        "storage", additional_storage['storage_pool'])
                    self.fill_form_by_id("retention", additional_storage['ret_period'])

                    self.driver.find_element_by_xpath(
                        "//button[@class='btn btn-primary cvBusyOnAjax ng-scope']").click()
                    self.wait_for_completion()
                else:
                    exp = "There is no option to add additional storage to the plan"
                    raise Exception(exp)

            if edit_storage['Edit']:
                if self.check_if_entity_exists(
                        "xpath", "//div[@class='page-details group']//span[contains(text()\
                    ,'Storage')]/..//div[@class='ui-grid-row ng-scope']\
                    //span[contains(text(),'"+edit_storage['old_storage_name']+"')]\
                    /../following-sibling::div/div[@class='btn-group ng-scope dropdown']/a"):
                    self.driver.find_element_by_xpath(
                        "//div[@class='page-details group']//span[contains(text()\
                    ,'Storage')]/..//div[@class='ui-grid-row ng-scope']\
                    //span[contains(text(),'"+edit_storage['old_storage_name']+"')]\
                    /../following-sibling::div/div[@class='btn-group ng-scope dropdown']/a")\
                    .click()
                    self.driver.find_element_by_xpath("//div[@class='page-details group']\
//span[contains(text(),'Storage')]/..//div[@class='ui-grid-row ng-scope']//span[contains(text()\
,'"+edit_storage['old_storage_name']+"')]/../following-sibling::div/div[@class=\
'btn-group ng-scope dropdown open']/ul/li/a").click()
                    self.wait_for_completion()
                    if edit_storage['new_storage_name']:
                        self.fill_form_by_id("storageName", edit_storage['new_storage_name'])
                    if edit_storage['new_ret_period']:
                        self.fill_form_by_id("retention", edit_storage['new_ret_period'])

                    self.driver.find_element_by_xpath(
                        "//button[@class='btn btn-primary cvBusyOnAjax ng-scope']").click()
                    self.wait_for_completion()
                else:
                    exp = "There is no option to edit the storage pool associated with this plan"
                    raise Exception(exp)
        else:
            self.log.info("There is no storage pool associated with this plan")

    @adminconsole_handler()
    def edit_plan_alerts(self, update_alerts):
        """
        Method to edit alerts associated with the plan

        Args:
            update_alerts (dict): dictionary containing alert details,
                    which alerts to enable/disable.
                Eg. - update_alerts = {"Backup" :None,
                                                "Jobfail": "Restore Job failed",
                                                "edge_drive_quota":None}

        Returns:
            None

        Raises:
            Exception:
                None
        """

        self.log.info("Editing plan alerts")
        if self.check_if_entity_exists("xpath", "//div[@class='page-details group']\
        //span[contains(text(),'Alerts')]"):
            self.driver.find_element_by_xpath(
                "//a[@data-ng-click='modifyAlertsList()']").click()
            self.wait_for_completion()

            alert_list = self.driver.find_elements_by_xpath("//ul[@data-ng-repeat='alert in globalAlerts']")

            if update_alerts["Backup"] == "No backup for last 4 days":
                if not alert_list[0].find_element_by_xpath("./li/input").is_selected():
                    alert_list[0].find_element_by_xpath("./li/label").click()
            else:
                if alert_list[0].find_element_by_xpath("./li/input").is_selected():
                    alert_list[0].find_element_by_xpath("./li/label").click()

            if update_alerts["Jobfail"] == "Restore Job failed":
                if not alert_list[1].find_element_by_xpath("./li/input").is_selected():
                    alert_list[1].find_element_by_xpath("./li/label").click()
            else:
                if alert_list[1].find_element_by_xpath("./li/input").is_selected():
                    alert_list[1].find_element_by_xpath("./li/label").click()

            if update_alerts["edge_drive_quota"] == "Edge drive quota alert":
                if not alert_list[2].find_element_by_xpath("./li/input").is_selected():
                    alert_list[2].find_element_by_xpath("./li/label").click()
            else:
                if alert_list[2].find_element_by_xpath("./li/input").is_selected():
                    alert_list[2].find_element_by_xpath("./li/label").click()

            self.driver.find_element_by_xpath(
                "//button[@class='btn btn-primary cvBusyOnAjax']").click()
            self.wait_for_completion()
        else:
            self.log.info("There is no option to edit the alerts associated with this plan")

            self.wait_for_completion()

                        if archiving_rules["file_create_time"]:
                            self.fill_form_by_id("fileCreatedTime", archiving_rules
                                                 ["file_create_time"])
                        if archiving_rules["archive_file_size"]:
                            self.fill_form_by_id("archiveFileSize", archiving_rules
                                                 ["archive_file_size"])
                        if archiving_rules["max_file_size"]:
                            self.fill_form_by_id("maxFileSize", archiving_rules
                                                 ["max_file_size"])
                        self.log.info(
                            "editing completed for values of Archiving rule........")

                        if archiving_rules["archive_read_only"]:
                            if not self.driver.find_element_by_id(
                                    "archiveReadOnlyFile").is_selected():
                                self.driver.find_element_by_xpath("//form[@id=\
                                'archivingRules-form']/div[3]/div[6]/label").click()
                        else:
                            if self.driver.find_element_by_id(
                                    "archiveReadOnlyFile").is_selected():
                                self.driver.find_element_by_xpath("//form[@id=\
                                'archivingRules-form']/div[3]/div[6]/label").click()

                        if archiving_rules["replace_file"]:
                            self.driver.find_element_by_id(
                                "replaceFile").click()
                        else:
                            self.driver.find_element_by_id(
                                "deleteFile").click()

                        self.driver.find_element_by_xpath(
                            "//button[@class='btn btn-primary cvBusyOnAjax']").click()

                    else:
                        self.driver.find_element_by_xpath(
                            "//button[@class='btn btn-primary cvBusyOnAjax']").click()
                        self.wait_for_completion()
            else:
                if self.check_if_entity_exists("xpath", "//span[contains(text(),\

                    self.driver.find_element_by_xpath(
                        "//button[@class='btn btn-primary cvBusyOnAjax ng-binding']").\
                        click()

    @adminconsole_handler()
    def remove_associated_users_and_groups(self, user_usergroup_de_association):
        """
        Method to remove users/user groups associated to a plan

        Args:
            user_usergroup (dictionary): Dictionary containing users or users groups to be
                de-associated from a plan (if All or just Users or just user groups)
                Eg. - "user_usergroup_de_association" : {"DeleteAll": true,
                                                        "DeleteAllUsers": false,
                                                        "DeleteAllUserGroups": false,
                                                        "Delete_Specific_user_usergroup": false}

        Returns:
            None

        Raises:
            Exception:
                None
        """

        if self.check_if_entity_exists("xpath", "//div[@class='page-details group']\
        /span[contains(text(),'Associated users and user groups')]"):

            self.driver.find_element_by_xpath("//div[@class='page-details group']\
            /span[contains(text(),'Associated users and user groups')]\
            /following-sibling::div//div[@class =\
            'page-details-box-links multiLinks ng-scope']/a").click()
            self.wait_for_completion()

            if user_usergroup_de_association['DeleteAll']:
                if self.check_if_entity_exists("xpath", "//div[@class='users-list']"):
                    user_list = self.driver.find_element_by_xpath("//div[@class='users-list']/ul")
                    users = user_list.find_elements_by_xpath("./li")
                    for user in users:
                        user.find_element_by_xpath(
                            ".//span[@class='delete-row']").click()
                else:
                    exp = 'There are no users or user groups associated with the plan'
                    raise Exception(exp)
            else:
                if user_usergroup_de_association['DeleteAllUsers']:
                    if self.check_if_entity_exists("xpath", "//div[@class='users-list']"):
                        users = self.driver.find_elements_by_xpath(
                            "//div[@class='users-list']//span[@class='user-type']")
                        for user in users:
                            user.find_element_by_xpath(".//span[@class='delete-row']").click()
                    else:
                        exp = 'There are no users associated with the plan'
                        raise Exception(exp)
                if user_usergroup_de_association['DeleteAllUserGroups']:
                    if self.check_if_entity_exists("xpath", "//div[@class='users-list']"):
                        users = self.driver.find_elements_by_xpath(
                            "//div[@class='users-list']//span[@class='group-type']")
                        for user in users:
                            user.find_element_by_xpath(".//span[@class='delete-row']").click()
                    else:
                        exp = 'There are no user groups associated with the plan'
                        raise Exception(exp)
                if user_usergroup_de_association['Delete_Specific_user_usergroup']:
                    if self.check_if_entity_exists("xpath", "//div[@class='users-list']"):
                        user_list = self.driver.find_element_by_xpath(
                            "//div[@class='users-list']/ul")
                        users = user_list.find_elements_by_xpath("./li")
                        for user in users:
                            for value in user_usergroup_de_association\
                                ['Delete_Specific_user_usergroup']:
                                if user.find_element_by_xpath(".//span[@class='ng-binding']")\
                                    .text == value:
                                    user.find_element_by_xpath(
                                        ".//span[@class='delete-row']").click()

            self.driver.find_element_by_xpath(
                "//button[@class='btn btn-primary cvBusyOnAjax ng-binding']").click()
            self.wait_for_completion()
            if self.check_if_entity_exists(
                    "xpath", "//h4[contains(text(),'Affected client list')]"):
                self.driver.find_element_by_xpath("//button[@class='btn btn-primary']").click()
            elif self.check_if_entity_exists(
                    "xpath", "//button[@class='btn btn-primary cvBusyOnAjax']"):
                self.driver.find_element_by_xpath(
                    "//button[@class='btn btn-primary cvBusyOnAjax']").click()
                self.wait_for_completion()
        else:
            self.log.info("No option to remove associations from this plan")
