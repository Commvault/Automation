    def __init__(self, driver):

    @adminconsole_handler()
    def add_arrays(self,
                   snap_vendor,
                   array_name,
                   user_name,
                   password,
                   control_host=None,
                   controllers=None,
                   retry_interval=300,
                   retry_count=5,
                   use_tunneling=False,
                   turn_off_snap_schedule=False,
                   turn_off_snap_delete=False,
                   update_host_alias=True,
                   array_host_aliases=None):
        """
        Creates an array if a snap plan already exists
        :param snap_vendor          (str)   --  the name of the vendor
        :param array_name           (str)   --  the name of the array to be added
        :param user_name            (str)   --  the username of the array
        :param password             (str)   --  the password of the array
        :param control_host         (str)   --  the control host of the array
        :param controllers          (list)  --  list of all controllers to the array
        :param retry_interval       (str)   --  the retry interval
        :param retry_count          (str)   --  the retry count
        :param use_tunneling        (bool)  --  to use host tunneling or not
        :param turn_off_snap_schedule(bool)  --  to turn off snapshot schedule
        :param turn_off_snap_delete (bool)  --  to auto delete snapshots or not
        :param update_host_alias    (bool)  --  auto update array host aliases
        :param array_host_aliases   (str)   --  array host aliases
        """
        self.log.info("Creating an array")
        if self.check_if_entity_exists("link", "Add array"):
            self.select_hyperlink("Add array")
            self.select_value_from_dropdown("vendorInput", snap_vendor)
            self.fill_form_by_id("arrayNameInput", array_name)
            if self.check_if_entity_exists("id", "arrayHostInput"):
                self.fill_form_by_id("arrayHostInput", control_host)
            self.fill_form_by_id("arrayUserName", user_name)
            self.fill_form_by_id("arrayPassword", password)
            self.button_next()

            if controllers:
                xpath = "//*[@id='ArrayControllers']"
                self.driver.find_element_by_xpath(xpath).click()
                self.wait_for_completion()
                select_option_xpath = xpath + "//div[@class='checkBoxContainer']"
                for cont in controllers:
                    if self.check_if_entity_exists("xpath", select_option_xpath):
                        span_xpath = select_option_xpath + "//span[contains(text(),'"\
                                     + cont + "')]"
                        self.driver.find_element_by_xpath(span_xpath).click()
                        self.wait_for_completion()
                    else:
                        raise Exception("The given controller was not present")
                self.scroll_into_view("ArrayControllers")
                self.driver.find_element_by_xpath(xpath).click()
            self.driver.find_elements_by_xpath("//button[contains(text(),'Next')]")[1].click()
            self.wait_for_completion()

            self.fill_form_by_id("0", retry_interval)
            self.fill_form_by_id("1", retry_count)
            if use_tunneling:
                self.toggle_enable("Use Tunneling")
            else:
                self.toggle_disable("Use Tunneling")

            if turn_off_snap_schedule:
                self.toggle_enable("Turn off Snapshot Schedule")
            else:
                self.toggle_disable("Turn off Snapshot Schedule")

            if turn_off_snap_delete:
                self.toggle_enable("Turn off Snapshot Auto Delete")
            else:
                self.toggle_disable("Turn off Snapshot Auto Delete")

            if update_host_alias:
                self.toggle_enable("Auto Update Array Host Aliases")
            else:
                self.toggle_disable("Auto Update Array Host Aliases")

            if array_host_aliases:
                for array in array_host_aliases:
                    self.fill_form_by_id("AliasInput", array)
                    self.click_button("Add")
            self.click_button("Save")
            self.wait_for_completion()
        else:
            raise Exception("There is no option to add an array. Please check if the logged "
                            "in user has sufficient permissions or a snap plan already exists")

    @PageService()
    def select_array(self, array_name):
        """
        Opens the array with the given name
        :param array_name   (str)   --  the name of the array to select
        """
        self.log.info("Opening the given array")
        self.__table.access_link(array_name)

    @PageService()
    def action_delete_array(self, array_name):
        """
        Deletes the array with the given name
        :param array_name   (str)   --  the name of the array to be deleted
        """
        self.__table.access_action_item(array_name, "Delete")
        self.click_button('Yes')

    @PageService()
    def action_list_snaps(self, array_name):
        """
        Lists the snaps on the array with the given name
        :param array_name   (str)   --  the name of the array whose snaps are to listed
        """
        self.__table.access_action_item(array_name, "List snaps")

    @adminconsole_handler()
    def array_list(self):
        """
        Lists all the arrays
        :return array_info  (list)  --  list of all arrays and their info
        """
        self.log.info("Listing all the arrays")
        arrays_info = []
        while True:
            array_list = self.driver.find_elements_by_xpath(
                "//div[@class='ui-grid-canvas']/div")
            for array in array_list:
                array_details = {}
                array_details['Name'] = array.find_element_by_xpath("./div/div[1]/a").text
                array_details['User name'] = array.find_element_by_xpath(
                    "./div/div[2]/div").text
                array_details['Snap vendor'] = array.find_element_by_xpath(
                    "./div/div[3]/div").text
                arrays_info.append(array_details)
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        arrays_info = [index for index in arrays_info if index != {}]
        self.log.info(arrays_info)
        return arrays_info

    @adminconsole_handler()
    def delete_all_snapshots(self):
        """
        Delete all  the snaps on the array
        """
        self.log.info("Deleting snaps for the given array")
        self.driver.find_element_by_xpath(
            "//div[@ng-model = 'grid.selection.selectAll']").click()
        if self.check_if_entity_exists("link", "Delete"):
            self.select_hyperlink("Delete")
            self.click_button_using_text('Yes')
            job_text = self.get_notification()
            if not job_text:
                raise Exception("Could not delete snaps")
            job_text = re.findall(r'\d+', job_text)[0]
            return job_text
        else:
            raise Exception("There is no option to delete snaps for the array")
