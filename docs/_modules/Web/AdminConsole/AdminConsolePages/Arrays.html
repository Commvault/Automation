    @adminconsole_handler()
    def add_arrays(self,
                   snap_vendor,
                   array_name,
                   user_name,
                   password,
                   retry_interval,
                   retry_count,
                   connect_host_group=False,
                   use_host=False,
                   diag_logging=False,
                   remote_snap_ma=None):
        """
        Creates an array if a snap plan already exists
        :param snap_vendor          (str)   --  the name of the vendor
        :param array_name           (str)   --  the name of the array to be added
        :param user_name            (str)   --  the username of the array
        :param password             (str)   --  the password of the array
        :param retry_interval       (str)   --  the retry interval
        :param retry_count          (str)   --  the retry count
        :param connect_host_group   (bool)  --  connect or not to the host group
        :param use_host             (bool)  --  to use or not to use the host
        :param diag_logging         (bool)  --  enable / disable diag logging
        :param remote_snap_ma       (str)   --  the remote snap MA to be added
        """
        self.log.info("Creating an array")
        if self.check_if_entity_exists("link", "Add array"):
            self.driver.find_element_by_link_text("Add array").click()
            self.wait_for_completion()
            self.select_value_from_dropdown("vendor", snap_vendor)
            self.fill_form_by_id("arrayName", array_name)
            self.fill_form_by_id("userName", user_name)
            self.fill_form_by_id("password", password)
            if self.check_if_entity_exists(
                    "xpath", "//form/div[1]/div[2]/uib-accordion/div/div/div[1]/h4/a"):
                self.driver.find_element_by_xpath(
                    "//form/div[1]/div[2]/uib-accordion/div/div/div[1]/h4/a").click()
                self.wait_for_completion()
                self.fill_form_by_id("mountRetryInterval", retry_interval)
                self.fill_form_by_id("mountRetryCount", retry_count)
                if connect_host_group:
                    if not self.driver.find_element_by_id("hostGroup").is_selected():
                        self.driver.find_element_by_xpath(
                            "//input[@id='hostGroup']/following-sibling::label").click()
                if use_host:
                    if not self.driver.find_element_by_id("useHost").is_selected():
                        self.driver.find_element_by_xpath(
                            "//input[@id='useHost']/following-sibling::label").click()
                if diag_logging:
                    if not self.driver.find_element_by_id("diagLogging").is_selected():
                        self.driver.find_element_by_xpath(
                            "//input[@id='diagLogging']/following-sibling::label").click()
                if remote_snap_ma is not None:
                    self.fill_form_by_id("remoteSnapMA", remote_snap_ma)
                self.driver.find_element_by_xpath("//button[contains(text(),'Save')]").click()
                self.wait_for_completion()
        else:
            raise Exception("There is no option to add an array. Please check if the logged "
                            "in user has sufficient permissions or a snap plan already exists")

    @adminconsole_handler()
    def select_array(self, array_name):
        """
        Opens the array with the given name
        :param array_name   (str)   --  the name of the array to select
        """
        self.log.info("Opening the given array")
        self.search_for(array_name)
        if self.check_if_entity_exists("link", array_name):
            self.driver.find_element_by_link_text(array_name).click()
            self.wait_for_completion()
        else:
            raise Exception("There is  no array with the given name")

    @adminconsole_handler()
    def action_delete_array(self, array_name):
        """
        Deletes the array with the given name
        :param array_name   (str)   --  the name of the array to be deleted
        """
        self.log.info("Deleting the given array")
        self.search_for(array_name)
        if self.check_if_entity_exists("link", array_name):
            self.driver.find_element_by_xpath(
                "//a[@text='"+array_name+"']/../../div[4]/div/a/span").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("link", "Delete"):
                self.driver.find_element_by_xpath(
                    "//a[@text='"+array_name+"']/../../div[4]/div/ul//a[contains(text(),"
                                             "'Delete')]").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath("//button[contains(text(),'Yes')]").click()
                self.wait_for_completion()
            else:
                raise Exception("There is no option to delete the array")
        else:
            raise Exception("There is  no array with the given name")

    @adminconsole_handler()
    def action_list_snaps(self, array_name):
        """
        Lists the snaps on the array with the given name
        :param array_name   (str)   --  the name of the array whose snaps are to listed
        """
        self.log.info("Listing snaps for the given array")
        self.search_for(array_name)
        if self.check_if_entity_exists("link", array_name):
            self.driver.find_element_by_xpath(
                "//a[text()='"+array_name+"']/../../div[4]/div/a/span").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("link", "List snaps"):
                self.driver.find_element_by_xpath(
                    "//a[text()='"+array_name+"']/../../div[4]/div/ul//a[contains \
                    (text(),'List snaps')]").click()
                self.wait_for_completion()
            else:
                raise Exception("There is no option to list the snaps for the array")
        else:
            raise Exception("There is  no array with the given name")

    @adminconsole_handler()
    def array_list(self):
        """
        Lists all the arrays
        :return array_info  (list)  --  list of all arrays and their info
        """
        self.log.info("Listing all the arrays")
        arrays_info = []
        while True:
            array_list = self.driver.find_elements_by_xpath(
                "//div[@class='ui-grid-canvas']/div")
            for array in array_list:
                array_details = {}
                array_details['Name'] = array.find_element_by_xpath("./div/div[1]/a").text
                array_details['User name'] = array.find_element_by_xpath(
                    "./div/div[2]/div").text
                array_details['Snap vendor'] = array.find_element_by_xpath(
                    "./div/div[3]/div").text
                arrays_info.append(array_details)
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        arrays_info = [index for index in arrays_info if index != {}]
        self.log.info(arrays_info)
        return arrays_info

    @adminconsole_handler()
    def delete_all_snapshots(self):
        """
        Delete all  the snaps on the array
        """
        self.log.info("Deleting snaps for the given array")
        self.driver.find_element_by_xpath(
            "//div[@ng-model = 'grid.selection.selectAll']").click()
        if self.check_if_entity_exists("link", "Delete"):
            self.driver.find_element_by_xpath(
                "//a[text() = 'Delete']").click()
            self.wait_for_completion()
            self.driver.find_element_by_xpath(
                "//button[text() = 'Yes']").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to delete snaps for the array")
