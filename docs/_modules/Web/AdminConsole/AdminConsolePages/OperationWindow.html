    @adminconsole_handler()
    def open_operation_rule(self, operation_rule_name):

        """Opens operation rule details
        Args:
            operation_rule_name (str): name for Backup rule to be opened

        Returns:
            None

        Raises:
            Exception:
                None"""

        self.search_for(operation_rule_name)
        self.driver.find_element_by_link_text(operation_rule_name).click()
        self.wait_for_completion()

    @adminconsole_handler()
    def add_operation_rule(
            self,
            operation_rule_name,
            start_date,
            end_date,
            operation_day,
            operation_time):
        """
        Method to create a new Backup Rule

        Args:
            operation_rule_name (str): name for operation rule to be created
            start_date (str): Start date of the operation rule
            end_date (str): End date of the operation rule
            operation_day (dict): dictionary with days for which the operation rule is to be
                    configured
                Eg.- operation_day = {'Monday':1, 'Wednesday':1, 'Friday':1, 'Sunday':1}
            operation_time (dict): dictionary with intervals for which the operation rule not
                     to be run
                Eg.- operation_time = {'0':1, '1':1, '2':1, '3':1, '4':1, '14':1, '15':1, '16':1,
                                                    '17':1, '18':1}

        Returns:
            None

        Raises:
            Exception:
                if failed to create a operation rule
        """

        if self.check_if_entity_exists("xpath", "//a[contains(text(),'Add rule')]"):
            self.driver.find_element_by_xpath(
                "//a[contains(text(),'Add rule')]").click()
            self.wait_for_completion()
        else:
            exp = "User does not have permissions to add operation rule"
            self.log.info(exp)
            raise exp

        self.log.info("Entering operation rule name")
        self.fill_form_by_id("operationRuleName", operation_rule_name)
        
        self.log.info("selecting operation rule days")
        for key, value in operation_day.items():
            if value == 1:
                if not self.driver.find_element_by_id(
                        "" + key + "").is_selected():
                    self.driver.find_element_by_xpath(
                        "//label[contains(text(),'" + key + "')]").click()
            else:
                if self.driver.find_element_by_id(
                        "" + key + "").is_selected():
                    self.driver.find_element_by_xpath(
                        "//label[contains(text(),'" + key + "')]").click()

        self.log.info("selecting operation rule times")
        for key, value in operation_time.items():
            if value == 1:
                if self.check_if_entity_exists(
                        "xpath", "//td[@id='"+key+"' and @class='ng-scope']"):
                    self.driver.find_element_by_xpath(
                        "//tr[@id='TIME']/td[@id='"+key+"']").click()
            else:
                if self.check_if_entity_exists(
                        "xpath", "//td[@id='"+key+"' and @class='ng-scope eng-selected-item']"):
                    self.driver.find_element_by_xpath(
                        "//tr[@id='TIME']/td[@id='"+key+"']").click()

        if start_date or end_date:
            self.log.info("Entering values for operation rule start date and end date")
            self.driver.find_element_by_xpath(
                "//label[contains(text(),'Enable date selection')]").click()
            if start_date == date.today().strftime("%m/%d/%Y"):
                self.driver.find_element_by_xpath("//label[@for='startDate']/..//button").click()
                self.driver.find_element_by_xpath("//button[@class=\
                    'btn btn-default btn-sm active']").click()
            else:
                self.fill_form_by_id("startDate", start_date)

            tomorrow = date.today()+datetime.timedelta(days=1)
            if end_date == tomorrow.strftime("%m/%d/%Y"):
                self.driver.find_element_by_xpath("//label[@for='endDate']/..//button").click()

                self.driver.find_element_by_xpath("//button[@class='btn btn-default btn-sm active']\
                    /../following-sibling::td/button").click()
            else:
                self.fill_form_by_id("endDate", end_date)

        self.log.info("All values are provided, submitting form for operation rule creation")
        self.driver.find_element_by_xpath(
            "//button[contains(text(),'Save')]").click()

        if self.check_if_entity_exists(
                "xpath", "//span[@class='cv-help-block']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='cv-help-block']").text
            self.log.exception(exp)
            raise Exception(exp)
        if self.check_if_entity_exists("xpath", "//span[@class=\
        'server-message error']"):
            exp = self.driver.find_element_by_xpath("//span[@class=\
            'server-message error']").text
            self.log.exception(exp)
            raise Exception(exp)
        self.wait_for_completion()

    @adminconsole_handler()
    def edit_operation_rule(
            self,
            edit_config,
            operation_rule_name,
            new_operation_rule_name=None,
            new_start_date=None,
            new_end_date=None,
            operation_day=None,
            operation_time=None):
        """
        Method to modify an existing Backup Rule

        Args:
            operation_rule_name (str): name of the existing operation rule
            new_operation_rule_name (str): new name for operation rule
            new_start_date (str): new start date of the operation rule
            new_end_date (str): new end date of the operation operation rule
            operation_day (dict): dictionary with days for which the operation rule is to be
                    modified
                Eg.- operation_day = {'Monday':1, 'Wednesday':1, 'Friday':1, 'Sunday':1}
            operation_time (dict): dictionary with intervals for which the operation rule 
                    not to be run
                Eg.- operation_time = {'0':1, '1':1, '2':1, '3':1, '4':1, '14':1, '15':1, '16':1,
                                                    '17':1, '18':1}
                    Time duration for which the back up is to be run
                    eg. 0 : 12:00 - 01:00 AM
                        1 : 01:00 - 02:00 AM etc.
        Returns:
            None

        Raises:
            Exception:
                if failed to modify a operation rule

        """
        if edit_config == 1 or not new_operation_rule_name:
            self.open_operation_rule(operation_rule_name)
        else:
            self.open_operation_rule(new_operation_rule_name)

        if new_operation_rule_name:
            self.log.info("Editing operation rule name")
            self.fill_form_by_id("operationRuleName", new_operation_rule_name)

        if operation_day:
            self.log.info("Editing operation rule days")
            for key, value in operation_day.items():
                if value == 1:
                    if not self.driver.find_element_by_id(
                            ""+key+"").is_selected():
                        self.driver.find_element_by_xpath(
                            "//label[contains(text(),'"+key+"')]").click()
                else:
                    if self.driver.find_element_by_id(
                            ""+key+"").is_selected():
                        self.driver.find_element_by_xpath(
                            "//label[contains(text(),'"+key+"')]").click()
            self.wait_for_completion()

        if operation_time:
            self.log.info("Editing operation rule times")
            for key, value in operation_time.items():
                if value == 1:
                    if self.check_if_entity_exists(
                            "xpath", "//td[@id='"+key+"' and @class='ng-scope']"):
                        self.driver.find_element_by_xpath(
                            "//tr[@id='TIME']/td[@id='"+key+"']").click()
                else:
                    if self.check_if_entity_exists(
                            "xpath", "//td[@id='"+key+"' and @class='ng-scope eng-selected-item']"):
                        self.driver.find_element_by_xpath(
                            "//tr[@id='TIME']/td[@id='"+key+"']").click()

        if (new_start_date is None and new_end_date is None):
            if self.driver.find_element_by_id("enableDate").is_selected():
                self.driver.find_element_by_xpath(
                    "//label[contains(text(),'Enable date selection')]").click()
        else:
            self.log.info("Editing start date and end date values")
            if not self.driver.find_element_by_id("enableDate").is_selected():
                self.driver.find_element_by_xpath(
                    "//label[contains(text(),'Enable date selection')]").click()
                if new_start_date:
                    self.fill_form_by_id("startDate", new_start_date)
                else:
                    self.driver.find_element_by_id("startDate").clear()
                if new_end_date:
                    self.fill_form_by_id("endDate", new_end_date)
                else:
                    self.driver.find_element_by_id("startDate").clear()
            else:
                if new_start_date:
                    self.fill_form_by_id("startDate", new_start_date)
                else:
                    self.driver.find_element_by_id("startDate").clear()
                if new_end_date:
                    self.fill_form_by_id("endDate", new_end_date)
                else:
                    self.driver.find_element_by_id("endDate").clear()

        self.driver.find_element_by_xpath(
            "//button[contains(text(),'Save')]").click()
        self.log.info("Editing completed")

        if self.check_if_entity_exists(
                "xpath", "//span[@class='cv-help-block']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='cv-help-block']").text
            self.log.exception(exp)
            raise Exception(exp)
        if self.check_if_entity_exists("xpath", "//span[@class=\
        'server-message error']"):
            exp = self.driver.find_element_by_xpath("//span[@class=\
            'server-message error']").text
            self.log.exception(exp)
            raise Exception(exp)
        self.wait_for_completion()

    @adminconsole_handler()
    def delete_operation_rule(self, operation_rule_name):
        """
        Method to delete a Backup Rule

        Args:
            operation_rule_name (str): name of the Backup rule to be deleted

        Returns:
            None

        Raises:
            Exception:
                if failed to delete a operation rule

        """

        self.search_for(operation_rule_name)
        self.driver.find_element_by_xpath(
            "//a[contains(text(),'" + operation_rule_name + "')]/../../div[5]/div/a").click()

        self.driver.find_element_by_link_text("Delete").click()
        self.wait_for_completion()
        self.driver.find_element_by_xpath(
            "//button[contains(text(),'Yes')]").click()
        self.wait_for_completion()

        if self.check_if_entity_exists(
                "xpath", "//span[@class='help-block']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='help-block']").text
            self.log.info(exp)
            raise Exception(exp)
        if self.check_if_entity_exists("xpath", "//span[@class=\
        'server-message error']"):
            exp = self.driver.find_element_by_xpath("//span[@class=\
            'server-message error']").text
            self.log.exception(exp)
            raise Exception(exp)
        self.wait_for_completion()
