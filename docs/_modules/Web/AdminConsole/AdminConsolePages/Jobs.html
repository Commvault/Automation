    @adminconsole_handler()
    def job_completion(self, job_id, retry=2):
        """
        Waits for job completion
        :param job_id   (str)   --  ID of the job to wait for
        :param retry    (int)   --  the number of times the job has to be resumed if it goes into
                                    pending
        """
        self.log.info("Getting the status of the job")
        job_id = str(job_id)
        # Navigating to jobs page
        self.navigate_to_jobs()

        if not self.select_job_by_id(job_id):
            self.log.info("Job is not present in the Active jobs page."
                          "Navigating to Job History")
            self.job_history()
            self.jobs_of_last_3_months()
            self.wait_for_completion()
            if not self.select_job_by_id(job_id):
                raise Exception("jobs is not present in Active jobs or job history")

        while True:
            if self.check_if_entity_exists("xpath", "//span[contains(text(), 'Status')]/"
                                                    "following-sibling::span"):
                status = self.driver.find_element_by_xpath(
                    "//span[contains(text(),'Status')]/following-sibling::span").text
                if status == "Running":
                    self.log.info("Waiting for Job " + str(job_id) + " to complete.")
                    time.sleep(10)
                    continue
                if status in ["Completed", "Failed", "Completed w/ one or more errors",
                              "Pending", "Killed", "Failed to Start"]:
                    if status in ["Completed", "Completed w/ one or more errors"]:
                        break
                    elif status in ["Failed", "Killed", "Failed to Start"]:
                        break
                    else:
                        if retry > 0:
                            self.resume_job(job_id)
                            retry = retry - 1
                            continue
                        else:
                            self.kill_job(job_id)
                            time.sleep(30)
                            continue
        return self.job_details()

    @adminconsole_handler()
    def __select_job_type(self, time_duration_type):
        """Displays jobs as of Last 24 hours, Last 7 days and All days
            timeDurationType - time duration to show job details of
        """
        self.driver.find_element_by_xpath("//cv-grid/div/div[1]/div/div/span[1]/a/span[1]"
                                         ).click()
        self.wait_for_completion()
        elements = self.driver.find_elements_by_xpath("//cv-grid/div/div[1]/div/div/span[1]/"
                                                      "ul/li")
        pattern = re.compile(time_duration_type, re.IGNORECASE)
        for li_elem in elements:
            if pattern.search(li_elem.find_element_by_xpath("./a/span").text):
                li_elem.find_element_by_xpath("./a/span").click()
                self.wait_for_completion()
                break
        if self.check_if_entity_exists("xpath", "//span[@class='server-message error']"):
            raise Exception(self.driver.find_element_by_xpath("//span[@class="
                                                              "'server-message error']"
                                                             ).text)

    @adminconsole_handler()
    def select_job_by_id(self, job_id):
        """
        Opens the job with given ID
        Args:
            job_id - id of the job to be selected

        Return:
            True    --  if the job opened successfully
        """
        job_id = str(job_id)
        self.log.info("Opening job " + job_id)
        if self.driver.find_element_by_xpath("//h2[contains(text(),'No jobs found')]"
                                            ).is_displayed():
            self.log.info("There are no active jobs")
            return False
        if self.check_if_entity_exists("xpath", "//div[@class = 'ui-grid-menu-button']"):
            self.driver.find_element_by_xpath("//div[@class = 'ui-grid-menu-button']").click()
            if self.check_if_entity_exists("xpath", "//div[@class = 'ui-grid-menu-inner']/"
                                                    "ul/li[6]"):
                self.driver.find_element_by_xpath("//div[@class = 'ui-grid-menu-inner']/ul/"
                                                  "li[6]/button").click()
                self.wait_for_completion()
                job_int = int(job_id)
                self.driver.find_element_by_xpath(
                    "//div[@class='ui-grid-filter-container ng-scope']/div/input")\
                    .send_keys(job_int)
                self.wait_for_completion()
                if self.check_if_entity_exists("xpath", "//a[text()='" + job_id + "']"):
                    self.driver.find_element_by_link_text(job_id).click()
                    self.wait_for_completion()
                    self.log.info("The Job" + job_id + " has been opened")
                    return True
                else:
                    return False
        else:
            raise Exception("There is no option to open toggle filter")

    @adminconsole_handler()
    def job_details(self):
        """Lists all the details of the job
            Return (True, 1) on successful completion
                    (False, func_name, error_msg) otherwise
            Ex: {
                'Status': 'Completed',
                'Data Written': '4.44 MB',
                'Job started by': 'Admin',
                'Backup type': 'Incremental',
                'Elapsed time': '2 min 8 sec',
                'iDataAgent': 'Virtual Server',
                'Software compression': 'Storage Policy',
                'Start time': 'Nov 18, 2015 00:23:57 AM',
                'Collection': 'NewCol',
                'Server': 'ADMINCONSOLECB',
                'Data transfered on network': '28.07 KB',
                'Instance': 'Hyper-V',
                'End time': 'Nov 18, 2015 00:26:14 AM',
                'Throughput': '3.67 MB/hr',
                'Last update time': 'Nov 18, 2015 00:26:14 AM',
                'Progress': '100%',
                'Type': 'Backup',
                'Source client computer': 'ADMINCONSOLECB',
                'Job started from': 'Interactive',
                'Size': '110.55 KB'
            }
        """
        self.log.info("Getting the job details")
        if not self.check_if_entity_exists(
                "xpath", "//ul[@class='list-style-secondary ng-scope']/li[3]/span[1]"):
            time.sleep(5)
        elements = []
        elements += self.driver.find_elements_by_xpath("//cv-tile-component"
                                                       "[@data-title='Job summary']//ul["
                                                       "@class='list-style-secondary ng-scope'"
                                                       "]/li")
        elements += self.driver.find_elements_by_xpath("//cv-tile-component"
                                                       "[@data-title='Job details']//ul["
                                                       "@class='list-style-secondary ng-scope'"
                                                       "]/li")
        count = len(elements)
        job_details = {}
        for i in range(count):
            k = elements[i].find_element_by_xpath("./span[1]").text
            if k in["Last update time", "Start time", "End time"]:
                value = elements[i].find_element_by_xpath("./span[2]/div").text
            else:
                value = elements[i].find_element_by_xpath("./span[2]").text
            job_details[k] = value
        if self.check_if_entity_exists(
                "xpath", "//cv-tile-component[@data-title='Job pending reason']"):
            k = "Job pending reason"
            value = self.driver.find_element_by_xpath(
                "//cv-tile-component[@data-title='Job pending reason']//li/span").text
            job_details[k] = value
        self.log.info(job_details)
        return job_details

    @adminconsole_handler()
    def __change_job_status(self, status):
        """Change the status of the job with id value *jobID*
            status - status to be selected of the job
        """
        if self.check_if_entity_exists("link", status):
            self.driver.find_element_by_link_text(status).click()
            self.wait_for_completion()
        else:
            raise Exception("The status of the job could not be changed")

    @adminconsole_handler()
    def kill_job(self, job_id):
        """Kills the job with the given ID
            jobID - id of the job to kill
        """
        self.log.info("Killing the job with id " + job_id)
        self.__change_job_status("Kill")

    @adminconsole_handler()
    def suspend_job(self, job_id):
        """Suspends the job with the given ID
            jobID - id of the job to suspend
        """
        self.log.info("Suspending the job with id " + job_id)
        self.__change_job_status("Suspend")

    @adminconsole_handler()
    def resume_job(self, job_id):
        """Resumes the job with the given ID
            jobID - id of the job to resume
        """
        self.log.info("Resuming the job with id " + job_id)
        self.__change_job_status(job_id, "Resume")
