    @adminconsole_handler()
    def open_event(self, event_id):
        """Opens a event with a given ID

            event_id - id of the event to be selected and opened
        """
        self.log.info("Opening event " + event_id)
        self.driver.find_element_by_id("search-field").clear()
        self.driver.find_element_by_id("search-field").send_keys(event_id)
        self.wait_for_completion()
        if self.check_if_entity_exists("link", event_id):
            self.driver.find_element_by_xpath(
                "//a[contains(text(),'" + event_id + "')]").click()
            self.wait_for_completion()
            elements = self.driver.find_elements_by_xpath(
                "//ul[@class='list-style-secondary ng-scope']/li")
            count = len(elements)-1
            event_details = {}
            for index in range(count):
                if index == 0:
                    key = elements[index].find_element_by_xpath(
                        "./span[1]").text
                    val = elements[index].find_element_by_xpath(
                        "./span[2]/div/div/span").text
                    event_details[key] = val
                else:
                    key = elements[index].find_element_by_xpath(
                        "./span[1]").text
                    val = elements[index].find_element_by_xpath(
                        "./span[2]").text
                    event_details[key] = val
            key = "Description"
            val = self.driver.find_element_by_xpath("//div[@class='col-md-12']//li/span").text
            event_details[key] = val
            self.log.info(event_details)
        else:
            raise Exception("Event " + event_id + " not found")
        if self.check_if_entity_exists("link", "Events"):
            self.driver.find_element_by_link_text("Events").click()
            self.wait_for_completion()

    @adminconsole_handler()
    def __select_event_type(self, event_type):
        """Displays events based on criticality
            event_type - criticality of the events
        """
        self.driver.find_element_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/a/span[1]").click()
        self.wait_for_completion()
        elements = self.driver.find_elements_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/ul/li")
        pattern = re.compile(event_type, re.IGNORECASE)
        for elem in elements:
            if pattern.search(elem.find_element_by_xpath("./a/span").text):
                elem.find_element_by_xpath("./a/span").click()
                self.wait_for_completion()
                break

    @adminconsole_handler()
    def critical_events(self, day=False):

    @adminconsole_handler()
    def list_of_events_for_job(self, job_id):
        """Lists all the events for a job
            job_id - id of the job to list the events for
        """
        self.log.info("List of events for the job {0} will be displayed as a dictionary"
                      .format(job_id))
        self.search_for(job_id)
        if self.check_if_entity_exists("xpath", "//div[5]/p/span[2]"):
            raise Exception("There are no events for the job")
        page = 0
        job_events = {job_id: []}
        while True:
            rows = self.driver.find_elements_by_xpath(
                "//div[@class='ui-grid-canvas']/div")
            for row in rows:
                jid = row.find_element_by_xpath("./div/div/div[3]/div")
                if jid.text == job_id:
                    cols = jid.find_elements_by_xpath("./../../div")
                    column_number = len(cols)
                    temp = []
                    for index in range(1, column_number + 1):
                        if index == 3:
                            continue
                        elif index == 1:
                            temp.append(
                                row.find_element_by_xpath(
                                    "./div/div/div[" + str(index) + "]/div/div/span").text)
                            continue
                        elif index == 6:
                            temp.append(
                                row.find_element_by_xpath(
                                    "./div/div/div[" + str(index) + "]/span").text)
                            continue
                        else:
                            temp.append(
                                row.find_element_by_xpath(
                                    "./div/div/div[" + str(index) + "]/div").text)
                            continue
                    job_events[job_id].append(temp)
            if self.cv_table_next_button_exists:
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    if not job_events[job_id] == []:
                        if page < 1:
                            self.cv_table_click_next_button()
                            self.wait_for_completion()
                            page += 1
                            continue
                        else:
                            break
                    else:
                        self.cv_table_click_next_button()
                        self.wait_for_completion()
                        continue
                else:
                    break
            else:
                break
        if not job_events[job_id] == []:
            self.log.info(job_events)
        else:
            raise Exception("There are no events for the job " + job_id)
