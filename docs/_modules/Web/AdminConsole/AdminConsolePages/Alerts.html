    @adminconsole_handler()
    def select_triggered_alerts(self):
        """
        Method to navigate to triggered alerts page

        Args:
            None

        Returns:
            None

        Raises:
            Exception:
                -- if fails to navigate to triggered alerts page
        """
        self.log.info("Navigating to Triggered Alerts page")
        if self.check_if_entity_exists("xpath", "//a[@href='#triggeredAlerts']"):
            self.driver.find_element_by_xpath(
                "//a[@href='#triggeredAlerts']").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no link Triggered alerts")

    @adminconsole_handler()
    def delete_current_triggered_alert(self, alert_name, client_name):
        """
        Method to delete a triggered alert

        Args:
           alert_name   (string) : Name of the alert to be deleted

           client_name  (string) : Name of client for which triggered alert to be deleted

        Returns:
            None

        Raises:
            Exception:
                -- if fails to delete the
        """
        count = 0
        self.log.info("Deleting triggered alert with name %s", alert_name)
        if self.check_if_entity_exists("xpath", "//div[@class='ui-grid-contents-wrapper']"):
            self.__table.search_for(alert_name)
            if self.check_if_entity_exists("link", alert_name):
                body_head = self.driver.find_element_by_xpath("//div[@class='ui-grid-render-\
                container ng-isolate-scope ui-grid-render-container-body']")
                rows = body_head.find_elements_by_xpath("./div[2]/div/div")
                for row in rows:
                    count += 1
                    if row.find_element_by_xpath(".//a[contains(text(),'"+alert_name+"')]"):
                        if row.find_element_by_xpath(".//a[contains(text(),'"+alert_name+"')]\
                        /../../../div[5]/div").text == client_name:
                            break

                checkbox_head = self.driver.find_element_by_xpath("//div[@class='left \
                ui-grid-render-container-left ui-grid-render-container ui-grid-render-container']")
                checkbox_head.find_element_by_xpath(
                    "./div[2]/div/div["+str(count)+"]/div/div/div/div/div/div").click()
                self.wait_for_completion()

                self.driver.find_element_by_xpath("//a[contains(text(),'Delete')]").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath("//button[@class='btn btn-primary']").click()
                time.sleep(1)

                if self.check_if_entity_exists(
                        "xpath", "//div[@class='growl-item alert ng-scope alert-success']"):
                    self.log.info("%s has been deleted successfully.", alert_name)
                elif self.check_if_entity_exists(
                        "xpath",
                        "//div[@class='growl-item alert ng-scope alert-error alert-danger']"):
                    exp = self.driver.find_element_by_xpath(
                        "//div[@class='growl-item alert ng-scope alert-error alert-danger']\
                        /div/div/div").text
                    raise Exception(exp)
                else:
                    exp = "Alert deletion message not displayed"
                    self.log.exception(exp)
                    raise Exception(exp)
            else:
                exp = "Alert %s is not displayed on Triggered alerts page", alert_name
                self.log.exception(exp)
                raise Exception(exp)
        else:
            self.log.info("No Triggered alerts displayed on the page")

    @adminconsole_handler()
    def delete_all_triggered_alerts(self):
        """
        Method to delete all triggered alerts

        Args:
           None

        Returns:
            None

        Raises:
            Exception:
                -- if fails to delete the subclient
        """
        self.log.info("Deleting all trigered alerts")
        self.driver.find_element_by_id("search-field").clear()
        if self.check_if_entity_exists("xpath", "//div[@class='ui-grid-contents-wrapper']"):
            self.driver.find_element_by_xpath(
                "//div[@class='ui-grid-contents-wrapper']/div[2]/div/div\
                //div[@class='ui-grid-cell-contents']/div").click()
            self.wait_for_completion()
            self.driver.find_element_by_xpath("//a[contains(text(),'Delete')]").click()
            self.wait_for_completion()
            self.driver.find_element_by_xpath("//button[@class='btn btn-primary']").click()
            time.sleep(1)
            if self.check_if_entity_exists(
                    "xpath", "//div[@class='growl-item alert ng-scope alert-success']"):
                exp = self.driver.find_element_by_xpath(
                    "//div[@class='growl-item alert ng-scope alert-success']/div/div/div")\
                    .text
                if exp == "Selected alerts deleted successfully":
                    self.log.info(exp)
                else:
                    self.log.exception(exp)
                    raise Exception(exp)
            else:
                exp = "Alert deletion message not displayed"
                self.log.exception(exp)
                raise Exception(exp)
        else:
            self.log.info("No Triggered alerts displayed on the page")

    @adminconsole_handler()
    def dump_alerts_info(self, computer_name, alert_info):
        """Dumps the alert info for the latest generated alert,
            of the specified computer name and alert type

            computer_name - name of the computer you want to dump alerts info
            alert_info    - alert you want to dump

            Ex: {
                'Previous Backup Size': '20.46 MB',
                'Storage Policies Used': 'Hawkeye_SP_Dedup',
                'CommCell': 'IRONMAN',
                'Detected Criteria': 'Decrease in Data size by 10 percentage',
                'Subclient': 'IndexBackup',
                'Failed Counts': '0',
                'Type': 'Job Management - Data Protection',
                'Status': 'Succeeded',
                'Current Backup Size': '0 Bytes',
                'Start Time': 'Sun Nov 15 00:00:13 2015',
                'End Time': 'Sun Nov 15 00:00:25 2015',
                'Backup Level': 'Full',
                'User': 'Administrator',
                'Scheduled Time': 'Sun Nov 15 00:00:03 2015',
                'Backup Set': 'defaultBackupSet',
                'Previous Job ID': '8639',
                'Protected Counts': '0',
                'Agent Type': 'Linux File System',
                'Client': 'Hawkeye',
                'Job ID': '8649',
                'Percentage Change': '-100',
                'Detected Time': 'Sun Nov 15 00:00:27 2015'
            }
        """
        self.log.info("Getting the information about alert " +
                      alert_info + " for computer " + computer_name)
        self.__table.search_for(alert_info)
        rows = self.driver.find_elements_by_xpath(
            "//cv-grid/div/div[2]/div/div[1]/div[3]/div[2]/div[@class='ui-grid-canvas']/div")
        elements = ""
        for row in rows:
            cname = row.find_element_by_xpath("./div/div/div[5]/div")
            if cname.text == computer_name:
                ainfo = cname.find_element_by_xpath(
                    "./../../div[6]/div/a")
                if ainfo.text == alert_info:
                    # coming out of the loop since by default,
                    # the alerts are displayed by the recently detected time.
                    # So as soon as the first matching pair is found the
                    # loop exits.
                    elements = ainfo
                    break
            else:
                continue
        if not elements:
            raise Exception("No such alert triggered for " + computer_name)
        else:
            elements.click()
            self.wait_for_completion()
            my_dict = {}
            arows = self.driver.find_elements_by_xpath(
                "//table[@id='contentTbl']/tbody/tr")
            index = 0  # for row numbering
            for arow in arows:
                if index == 0 or index == 3:
                    index += 1
                    continue
                else:
                    elems_uls = arow.find_elements_by_xpath("./td/ul")
                    for elem_ul in elems_uls:
                        data = elem_ul.text
                        lines = data.splitlines()
                        for line in lines:
                            line = line.strip()
                            key, val = line.split(": ", 1)
                            my_dict[key] = val
                    index += 1
            self.log.info(my_dict)
            self.driver.find_element_by_xpath(
                "/html/body/div[1]/div/div/a[@class='modal__close-btn']").click()
            self.wait_for_completion()

    @adminconsole_handler()
    def all_triggered_alerts_info(self):
        """Dumps all the alerts triggered for all the clients
            Eg.: {
                'Detected Criterion': 'Decrease in Data size by 10 percentage',
                'Detected time': 'Wed Nov 18 00:26:15 2015',
                'Type': 'Job Management - Data Protection',
                'Computer name': 'ADMINCONSOLECB',
                'Alert info': 'Decrease in data size'
            }
        """
        self.log.info("List of all triggered alerts")
        # for finding the rows
        rows = self.driver.find_elements_by_xpath(
            "//cv-grid/div/div[2]/div/div[1]/div[3]/div[2]/div[@class='ui-grid-canvas']/div")
        keys = [
            'Detected Criterion',
            'Type',
            'Detected time',
            'Computer name',
            'Alert info']
        for row in rows:
            cols = row.find_elements_by_xpath("./div/div/div")
            index = 0  # for skipping first column and last column
            assign_key = 0  # for assigning keys
            alert_info = {}
            for col in cols:
                if index == 0 or index == 6:
                    index += 1
                    continue
                elif index == 5:
                    val = col.find_element_by_xpath("./div/a").text
                    alert_info[keys[assign_key]] = val
                    index += 1
                    assign_key += 1
                else:
                    val = col.find_element_by_xpath("./div").text
                    alert_info[keys[assign_key]] = val
                    assign_key += 1
                    index += 1
            self.log.info(alert_info)

    @adminconsole_handler()
    def select_by_alert_type(self, alert_type):
        """
        Displays only the alerts of the given type
        :param alert_type       (str)   --  type of alert to select
        """
        self.log.info("Selecting the "+alert_type+" alerts")
        self.driver.find_element_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/a/span[1]").click()
        elements = self.driver.find_elements_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/ul/li")
        for elem_li in elements:
            if elem_li.find_element_by_xpath("./a/span").text == alert_type:
                elem_li.find_element_by_xpath("./a/span").click()
                self.wait_for_completion()
                break
