    @adminconsole_handler()
    def select_user(self, user):
        """
        Selects a user
        Params:user   (str)   -- the name of the user to select
        """
        self.search_for(user)
        if self.check_if_entity_exists("link", user):
            self.driver.find_element_by_xpath("//a[text()='"+user+"']").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no user with the name " + user)

    @adminconsole_handler(hide_args=True)
    def add_user(self, user_type, email, group, username, name, system_password=False,
                 password=None, external_provider=None, invite_user=False):
        """Adds new user with the specified details to the list of users
            name        - the full name of the user
            username    - the username that will be used
            email       - email address of the user
            password    - password of the user
            confirm_pass - password, needs to be same as the password argument
            group       - list of user groups the user needs to be added in
        """
        if self.check_if_entity_exists("link", "Add user"):
            self.driver.find_element_by_link_text("Add user").click()
            self.wait_for_completion()
            # Select user type
            self.driver.find_element_by_xpath("//input[@type='radio' and @value='" + user_type +
                                              "']").click()
            if user_type == "local":
                self.fill_form_by_id("fullName", name)
                self.fill_form_by_id("userName", username)
                if not system_password:
                    self.fill_form_by_id("password", password)
                    self.fill_form_by_id("confirmUserPassword", password)
                else:
                    self.driver.find_element_by_id("systemGeneratePassword").click()
            elif user_type == "external":
                self.driver.find_element_by_xpath(
                    "//*[@id='externalProvider']/span/button").click()
                self.driver.find_element_by_xpath(
                    "//*[@id='externalProvider']/span/div/div[1]/div/div[1]/input"
                    ).click()
                self.driver.find_element_by_xpath(
                    "//*[@id='externalProvider']/span/div/div[1]/div/div[1]/input"
                    ).send_keys(external_provider)
                self.driver.find_element_by_xpath(
                    "//*[@id='externalProvider']/span/div/div[2]/div/div/label/span"
                    ).click()
                self.driver.find_element_by_id("userName").send_keys(Keys.END + username)

            self.fill_form_by_id("email", email)

            self.driver.find_element_by_xpath(
                "//isteven-multi-select[@id='userGroupType']/span/button").click()
            for ugroup in group:
                self.driver.find_element_by_xpath(
                    "//div[@class='line-search']/button[@class='clearButton']").click()
                self.driver.find_element_by_xpath("//div[@class='line-search']/input").send_keys(
                    ugroup)
                self.driver.find_element_by_xpath(
                    "//div[@class='checkBoxContainer']//span[contains(text(),'" + ugroup + "')]"
                ).click()
            self.driver.find_element_by_xpath("//div[@class='line-searchOk ng-scope']").click()

            if invite_user:
                if not self.driver.find_element_by_id("inviteUser").is_selected():
                    self.driver.find_element_by_id("inviteUser").click()
                    self.wait_for_completion()

            self.driver.find_element_by_xpath("//*[@id='addUserForm']/div[2]/button[2]").click()
            self.wait_for_completion()

            if self.check_if_entity_exists(
                    "xpath", "//div/div[2]/span[@class='error']"):
                raise Exception(self.driver.find_element_by_xpath(
                    "//div/div[2]/span[@class='error']").text)
        else:
            raise Exception("There is no option to add user")

    @adminconsole_handler()
    def delete_user(self, user_name, owner_transfer):
        """Deletes the specified user
            user_name (str) - username of the user that needs to be deleted
			owner_transfer (str) - a user or user group to transfer ownership to before deleting
        """
        self.search_for(user_name)
        if self.check_if_entity_exists("link", user_name):
            self.driver.find_element_by_xpath(
                "//a[text()='"+user_name+"']/../../../div[6]/div/a/span").click()
            if self.check_if_entity_exists("link", "Delete"):
                self.driver.find_element_by_xpath(
                    "//a[contains(text(),'"+user_name+"')]/../../../div[6]/div/ul/li/a").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath(
                    "//div/div[3]/button[contains(text(), 'Yes')]").click()
                self.wait_for_completion()
                if self.check_if_entity_exists(
                        "xpath", "/html/body/div[1]/div/div[@class= 'modal-content']"):
                    self.log.info("Transferring Ownership")
                    self.driver.find_element_by_xpath(
                        "//div[@id='s2id_userList']/a/span[1]").click()
                    self.wait_for_completion()
                    self.driver.find_element_by_xpath(
                        "//div[@class='select2-search']/input").send_keys(owner_transfer)
                    self.wait_for_completion()
                    elements = self.driver.find_elements_by_xpath(
                        "//div[@id='select2-drop']/ul[@role='listbox']/li")
                    for elem in elements:
                        ugname = elem.find_element_by_xpath("./div/span").text
                        if ugname.split('(')[1].split(')')[0] == owner_transfer or \
                        ugname.split(' (')[0] == owner_transfer:
                            elem.find_element_by_xpath("./div/span").click()
                            self.wait_for_completion()
                            break
                    self.driver.find_element_by_xpath("//div[3]/button[2]").click()
                    self.wait_for_completion()
                if self.check_if_entity_exists(
                        "xpath", "//div[4]/div/div[@class='growl-item alert ng-scope  \
                        alert-error alert-danger']"):
                    raise Exception(self.driver.find_element_by_xpath(
                        "//div[4]/div/div[@class='growl-item alert ng-scope alert-error  \
                        alert-danger']/div/div").text)
            else:
                raise Exception("There is no option to delete the user group")
        else:
            raise Exception("There is no user with the name {0}".format(user_name))

    @adminconsole_handler()
    def user_info(self):
        """Displays all the users present in the CS and their corresponding info
        :return:
            user_list   (list)  --  list of users and their info
        """
        keys = ['UserName', 'Email', 'FullName', 'Plan']
        user_list = []
        while True:
            rows = self.driver.find_elements_by_xpath(
                "//div[@class='ui-grid-canvas']/div")
            for row in rows:
                user_info = {}
                columns = row.find_elements_by_xpath("./div/div")
                key = 0
                for index in range(1, len(columns)-1):
                    if index == 1 or index == 2 or index == 4:
                        val = row.find_element_by_xpath("./div/div["+str(index)+"]/span/a").text
                    else:
                        val = row.find_element_by_xpath("./div/div["+str(index)+"]/span").text
                    user_info[keys[key]] = val
                    key += 1
                user_list.append(user_info)
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        user_list = [index for index in user_list if index != {}]
        self.log.info(user_list)
        return user_list

    @adminconsole_handler()
    def __select_user_type(self, user_type):
        """
        Selects All users or laptop users
        :param user_type    (str)   --  type of user to select
        """
        self.driver.find_element_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/a/span[1]").click()
        self.wait_for_completion()
        elements = self.driver.find_elements_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/ul/li")
        pattern = re.compile(user_type, re.IGNORECASE)
        for li_elem in elements:
            if pattern.search(li_elem.find_element_by_xpath("./a/span").text):
                li_elem.find_element_by_xpath("./a/span").click()
                self.wait_for_completion()
                break

    def select_company(self, company_name):
        """
        Selects a company and displays its users
        :param company_name     (str)   --  the name of the company to select
        """
        if self.check_if_entity_exists(

    @adminconsole_handler()
    def import_bulk_users(self, user_type, file_path, user_groups, system_password=False,
                          external_provider=None):
        """
        Imports a large no of users into the commcell
        :return:
        """
        if not self.check_if_entity_exists("link", "Import bulk users"):
            raise Exception("There is no option to import bulk users. Please check the user"
                            "permissions")
        self.driver.find_element_by_link_text("Import bulk users").click()
        self.wait_for_completion()
        Select(self.driver.find_element_by_id("userType")).select_by_visible_text(user_type)
        element = self.driver.find_element_by_id("bulkImporter")
        self.driver.execute_script("arguments[0].style.visibility = 'visible'; "
                                   "arguments[0].style.height = '1px'; "
                                   "arguments[0].style.width = '1px'; "
                                   "arguments[0].style.opacity = 1", element)
        self.driver.find_element_by_id("bulkImporter").send_keys(file_path)

        if user_type == "External user":
            Select(self.driver.find_element_by_xpath("//select[@name='externalProvider']")
                  ).select_by_visible_text(external_provider)

        self.driver.find_element_by_xpath("//div[@data-ng-if='bulkUserCtrl.fileUploaded']//div["
                                          "@role='checkbox']").click()

        self.driver.find_element_by_xpath(
            "//isteven-multi-select[@id='userGroupType']/span/button").click()
        for ugroup in user_groups:
            self.driver.find_element_by_xpath(
                "//div[@class='line-search']/button[@class='clearButton']").click()
            self.driver.find_element_by_xpath("//div[@class='line-search']/input").send_keys(
                ugroup)
            self.driver.find_element_by_xpath(
                "//div[@class='checkBoxContainer']//span[contains(text(),'" + ugroup + "')]"
            ).click()
        self.driver.find_element_by_xpath("//div[@class='line-searchOk']").click()

        if user_type == 'Local user' and system_password:
            self.driver.find_element_by_id("systemGeneratePassword").click()
        self.driver.find_element_by_xpath("//form/div[2]/button[2]").click()
        self.wait_for_completion()
