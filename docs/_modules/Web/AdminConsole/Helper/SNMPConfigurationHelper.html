[docs]    def check_if_snmp_configuration_exists(self, config_name):
        """
        Method to verify existence of SNMP Configuration
        in SNMP Configuration page

        Args:
            config_name (str): IP address of the configuration to be verified

        Returns:
            None

        Raises:
            Exception:
                -- if failed to locate SNMP configuration
                    on SNMP configuration page

        """
        try:
            self.snmpconfig.navigate_to_snmp()
            if self.snmpconfig.check_if_entity_exists("id", "search-field"):
                self.snmpconfig.search_for(config_name)
                if self.snmpconfig.check_if_entity_exists("link", config_name):
                    return True
                return False
            return None
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)

        try:
            self.snmpconfig.navigate_to_snmp()
            self.snmpconfig.add_configuration(
                self.config_name,
                self.encryption_algorithm,
                self.username,
                self.password,
                self.privacy_algorithm,
                self.priv_password)
            self.encryption_algorithm = [self.encryption_algorithm]
            self.privacy_algorithm = [self.privacy_algorithm]
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def modify_configuration(self):
        """Calls the function to modify Configuration"""

        try:
            self.snmpconfig.navigate_to_snmp()
            for value_encrypt in self.encryption_algorithm:
                for value_private in self.privacy_algorithm:
                    self.edit_config += 1
                    self.snmpconfig.edit_configuration(
                        self.edit_config,
                        self._old_config_name,
                        self.config_name,
                        value_encrypt,
                        self.username,
                        self.password,
                        value_private,
                        self.priv_password)
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)

[docs]    def validate_snmp_configuration(self):
        """Validates SNMP configuration for the values, if retained correctly
        or not"""

        try:
            self.snmpconfig.navigate_to_snmp()
            if not self.check_if_snmp_configuration_exists(self.config_name):
                exp = "Config validation failed, configuration with name {0} \
                not displayed in SNMP page".format(self.config_name)
                self.log.exception(exp)
                raise Exception(exp)
            self.snmpconfig.open_configuration(self.config_name)

            if not self.driver.find_element_by_id(
                    "hostname").get_attribute('value') == self.config_name:
                exp = "Config validation failed, config name was not retained correctly"
                self.log.exception(exp)
                raise Exception(exp)
            self.log.info("Configuration Name is validated")

            if self.encryption_algorithm[0]:
                select = Select(self.driver.find_element_by_id("encryptionAlgorithm"))
                if not select.first_selected_option.text == self.encryption_algorithm[0]:
                    exp = "encryption algorithm validation failed, it was not retained \
correctly"
                    self.log.exception(exp)
                    raise Exception(exp)
                self.log.info("encryption algorithm is validated successfully")

                if not self.driver.find_element_by_id(
                        "username").get_attribute('value') == self.username:
                    exp = "Config validation failed, username was not retained correctly"
                    self.log.exception(exp)
                    raise Exception(exp)
                self.log.info("Username is validated")

                if self.privacy_algorithm[0]:
                    select1 = Select(self.driver.find_element_by_id("privacyAlgorithm"))
                    if not select1.first_selected_option.text == self.privacy_algorithm[0]:
                        exp = "privacy algorithm validation failed, it was not retained \
correctly"
                        self.log.exception(exp)
                        raise Exception(exp)
                    self.log.info("privacy algorithm validated successfully")

            self.log.info("Configuration validated successfully,\
all values are correctly retained")

            self.driver.find_element_by_xpath(
                "//button[contains(text(),'Cancel')]").click()
            self.snmpconfig.wait_for_completion()

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)
