    def __init__(self, driver=None, csdb=None):
        """
        Initializes the login helper module

        Args:
            driver (object) : the web driver object
            csdb (object)   : DBHelper object

        Raises:
             Exception:
                If the driver is not provided
        """
        self.driver = driver
        if not driver:
            raise Exception('Driver is not provided')
        self.login_obj = LoginPage(self.driver)
        self._log = logger.get_log()
        self._url = None
        self._username = None
        self._password = None
        self.csdb = csdb

    @property

    def _get_adminconsole_url(self, client_name=None):
        """
        Gets the client where adminconsole is installed

        Args:
            client_name (basestring):  the name of the client where adminconsole is installed

        Returns:
            adminconsole url
        """
        try:
            if client_name:
                hostname = client_name
            else:
                hostname = self.csdb._commcell.webconsole_hostname
            url = hostname + "/adminconsole/"
            return url

        except Exception as exp:
            self._log.exception(str(exp))
            raise exp


[docs]    def login(self,
              username=_CONSTANTS.ADMIN_USERNAME,
              password=_CONSTANTS.ADMIN_PASSWORD,
              client_name=None,
              remember_me=_CONSTANTS.WebConsole.STAY_LOGGED_IN,
              enable_ssl=False
              ):
        """
        Helper method for logging into admin console

        Args:
            username (basestring)   : the username to be logged in with, by default taken from
             the config file if not supplied
            password (basestring)   : the password for the above user, by default taken from the
             config file if not supplied
            client_name (basestring): the name of the client where adminconsole is installed
            remember_me (bool)      : select/deselect the keep me logged in checkbox
            enable_ssl (bool)       : if enabled logs in with https protocol

        Raises:
            Exception:
                If it does not admin console url
        """
        self._url = self._get_adminconsole_url(client_name)
        self._username = username
        self._password = password
        protocol = "https://" if enable_ssl else "http://"
        if "\\" not in self.username:
            url = f"{protocol}{self.username}:{self.password}@{self.url}"
        else:
            username = self.username.replace("\\", "%5C")
            url = f"{protocol}{username}:{self.password}@{self.url}"
        self.login_obj.navigate(url)

        self._log.info("*" * 10 + " Login to Admin console " + "*" * 10)
        if remember_me or _CONSTANTS.SECURITY_TEST:
            self.login_obj.login(self.username, self.password, True)
        else:
            self.login_obj.login(self.username, self.password)
