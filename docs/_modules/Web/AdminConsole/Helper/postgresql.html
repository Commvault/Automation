    def __init__(self, testcase):
        """
        Helper for PostgreSQL related files

        Args:
            testcase    (object)    -- object of TestCase class

        """
        self.driver = testcase.driver
        self.log = testcase.log
        self.csdb = testcase.csdb
        self.commcell = testcase.commcell
        self.jobs = load_module(
            'Jobs',
            constants.AUTOMATION_DIRECTORY +
            '\\Web\\AdminConsole\\AdminConsolePages'
        )
        self.job = self.jobs.Jobs(self.driver)
        self.tcinputs = testcase.tcinputs
        self.status = testcase.status
        self.db_instance = None
        self.admin_console_base = AdminConsoleBase(self.driver)

        """

[docs]    def check_job_status(self, jobid):
        """
        To check the status of the job in admin console jobs page

        Args:
            jobid   (str)   --  job id of the triggered job

        Raises:
            Exception

                if Job failed

        """
        self.job.job_completion(jobid)
        job_details = self.job.job_details()

        if "Completed" in job_details.get('Status'):
            self.status = constants.PASSED
            self.log.info("Job Completed successfully")
        else:
            raise Exception(job_details.get('Job pending reason'))



[docs]    def add_postgresql_database_server(
            self,
            os_type='Unix and Linux',
            dbserver_name=None,
            username=None,
            password=None,
            plan=None,
            unix_group=None):
        """
        Adds a new PostgreSQL database server

        Args:
            os_type             (str)       -- The type of the OS
                                                Default: Unix and Linux

            dbserver_name       (str)       -- Database server name

            username            (str)       -- Username for the DB server

            password            (str)       -- Password for the DB server

            plan                (str)       -- Plan for the DB server

            unix_group          (str)       -- Unix group for the DB server

        Returns:
            str         --  Job ID for the triggered job

        Exception:

            If user input is invalid

            If element not found

        """
        self.navigate_to_dbinstances_page()
        jobid = self.db_instance.add_postgresql_database_server(
            dbserver_name=dbserver_name,
            username=username,
            password=password,
            plan=plan,
            os_type=os_type,
            unix_group=unix_group
        )

        self.log.info("Job %s is started for adding database server", jobid)

        # To Check the Job status
        self.check_job_status(jobid)

        """
        Adds a new PostgreSQL database server

        Args:
            server_name         (str)   --  Server to create a new PostgreSQL instance

            instance_name       (str)   --  Name of the instance to be created

            db_username         (str)   --  username for accessing the DB

            password            (str)   --  Password for accessing the DB

            maintenance_db      (str)   --  maintenance DB for the instance to be created

            port                (str)   --  port for the instance

            binary_directory    (str)   --  Full path for binary

            lib_directory       (str)   --  Full path for library

            archive_log_directory (str) --  Full path for archive logs

            delete_archive_logs (bool)  --  Deletes archive logs if set to True
                                                default: False

        Returns:
            None

        Exception:

            If user input is invalid

            If element not found

        """
        self.navigate_to_dbinstances_page()
        self.db_instance.add_postgresql_database_instance(
            server_name=server_name,
            instance_name=instance_name,
            db_username=db_username,
            password=password,
            maintenance_db=maintenance_db,
            port=port,
            binary_directory=binary_directory,
            lib_directory=lib_directory,
            archive_log_directory=archive_log_directory,
            delete_archive_logs=delete_archive_logs
        )

        self.log.info('Successfully created database instance %s', instance_name)

            self,
            instance=None,
            backupset_type=None,
            backupset_name='default',
            backup_level='Incremental'):
        """
        Performs backup operation on the selected instance

        Args:
            instance        (str)   -- instance to perform backup on

            backupset_type  (str)   -- Type of backupset (Dumpbased|Fsbased)

            backupset_name  (str)   -- Name of the backupset to run backup on
                                        default: 'default'

            backup_level    (str)   -- Backup level to be used for backup

        Exception:

            If user input is invalid

            If element not found

        """

        jobid = self.db_instance.instance_backup(
            instance=instance,
            backupset_type=backupset_type,
            backupset_name=backupset_name,
            backup_level=backup_level
        )

        self.log.info("Backup job %s is started for instance %s", jobid, instance)

        # To Check the Job status
        self.check_job_status(jobid)

        return jobid

[docs]    def dumpbased_restore(self, job_id=None, destination_client=None, destination_instance=None, databases=None):
        """
        Performs DumpBasedBackupSet restore operation on selected instance

        Args:
            job_id  (str)               -- job id of the dumpbased backup job

            destination_client(str)     -- destination client to be select

            destination_instance(str)   -- destination instance to be select

            databases   (list)          -- Databases to restore

        Returns:
        str     --  job ID for the triggered restore job

        Exception:

            If user input is invalid

            If restore failed

        **Note** Run dumpbased_restore after Dumpbased Backup

        """
        self.job.initiate_job_based_subclient_restore(job_id)
        browse = Browse(self.admin_console_base)
        browse.select_for_restore(file_folders=databases)
        job_id = self.db_instance.__restore_options(destination_client, destination_instance)

        self.log.info("Dumpbased Restore job %s is started for instance %s", job_id, destination_instance)

        # To Check the Job status
        self.check_job_status(job_id)
