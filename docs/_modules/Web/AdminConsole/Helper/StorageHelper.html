    def _check_if_storage_exists(self, storage_obj, storage_name):
        """Check if the storage exists
        :param storage_obj      (obj)   --  object of either storage targets or storage pools
        :param storage_name     (str)   --  name of the storage target or storage pool
        :return:
            True    --  If the storage object is present
            False   --  If the storage object is not present
        """
        try:
            self.log.info("Checks if the storage exists")
            storage_obj.search_for(storage_name)
            if not storage_obj.check_if_entity_exists('link', storage_name):
                return False
            return True
        except Exception as exp:
            self.log.exception(str(exp))
            return False

[docs]    def create_storage_pool(self):
        """
        Creates a storage pool
        """
        self.storage_pools.navigate_to_storage_pools()
        ret = self._check_if_storage_exists(self.storage_pools, self._storage_pool)
        if ret:
            self.log.info("Storage Pool Exists")
        else:
            if self.pool_type == "Disk/Cloud":
                self.storage_pools.add_disk_cloud_storage_pool()
            elif self.pool_type == "Tape":
                self.storage_pools.add_tape_storage_pool()
            else:
                raise Exception("Please specify the correct type of storage pool to create")
