    def __init__(self, driver=None, csdb=None):

            'Restores',

        self.restores_obj = self.restores.Restores(self.driver)

            'Jobs',

        self._client_name = None
        self._backupset_name = 'defaultBackupSet'
        self._subclient_name = None
        self._subclient_content = None

        self._restore_path = 'C:\ACRestore'

[docs]    def navigate_to_fsbackupset(self):
        """Navigate to backupset"""
        self.restores_obj.navigate_to_servers()
        self.restores_obj.select_client(self._client_name)
        self.restores_obj.wait_for_completion()
        self.restores_obj.driver.find_element_by_link_text(
            "File System").click()
        self.restores_obj.wait_for_completion()
        self.restores_obj.driver.find_element_by_link_text(
            self._backupset_name).click()
        self.restores_obj.wait_for_completion()



[docs]    def create_fs_subclient(self):
        """Create FS Subclient"""
        try:

            self.navigate_to_fsbackupset()
            self.restores_obj.add_fs_subclient(
                self._subclient_name, self._plan_name, self._subclient_content)

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def fsbackup_now(self, bkp_level='FULL'):
        """Run Snap Backup"""
        try:
            self.navigate_to_fsbackupset()
            self.restores_obj.driver.find_element_by_link_text(
                self._subclient_name).click()
            self.restores_obj.wait_for_completion()
            self.snap_jobid = self.restores_obj.backup_now(bkp_level)
            job_details = self.job_obj.job_completion(self.snap_jobid[1])
            if job_details['Status'] not in [
                    "Completed", "Completed w/ one or more errors"]:
                raise Exception("Backup job failed. Please check the logs")
            return self.snap_jobid[1]

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def fs_restore(self):
        """Run restore operation"""
        try:
            self.navigate_to_fsbackupset()
            self.restores_obj.restore()
            job_details = self.restores_obj.submit_restore(
                [self._subclient_content], self._client_name, self._restore_path, True)
            job_id = job_details.split("\n")[0].strip()
            job_status = self.job_obj.job_completion(job_id)
            if job_status['Status'] not in [
                    "Completed", "Completed w/ one or more errors"]:
                raise Exception("Restore job failed. Please check the logs")

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def delete_subclient(self):
        """"Delete FS Subclient"""
        try:
            self.navigate_to_fsbackupset()
            self.restores_obj.delete_subclient(self._subclient_name)
            # Validation
            if self.restores_obj.check_if_entity_exists("link", self._subclient_name):
                raise Exception(
                    "Validation : Subclient deletion failed, still exists in Admin Console Page")
            else:
                self.log.info("Validation : Subclient deletion successful from Admin Console Page")

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)
