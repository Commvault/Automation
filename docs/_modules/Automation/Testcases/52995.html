"""This testcase will verify you can install a client with an authcode and it gets assigned to the correct
default plan for non MSP Commcell.

TestCase:   Class for executing this test case

TestCase:

    __init__()      --  initialize TestCase class

    setup()         --  setup the requirements for the test case

    run()           --  run function of this test case

"""

from AutomationUtils.cvtestcase import CVTestCase
from Install.InstallUtils.client_installation import Installation
from Laptop.laptophelper import LaptopHelper
from Server.Security.securityhelper import OrganizationHelper


[docs]class TestCase(CVTestCase):
    """Test case class for validating Laptop client onboarding through custom package
        with auth code"""

    def __init__(self):
        """Initializes test case class object"""
        super(TestCase, self).__init__()
        self.name = """[Laptop] - Non MSP - Install with Auth Code and validate client association to default plan"""
        self.product = self.products_list.LAPTOP
        self.feature = self.features_list.NOTAPPLICABLE
        self.show_to_user = True
        self.tcinputs = {
            "Default_Plan": None, # Default Laptop plan for commcell [ AdminConsole->Commcell->Administration ]
            "Windows_machine_host_name": None,
            "Windows_machine_user_name": None, # Non admin user with enough privileges to install the client
            "Windows_machine_password": None,
            "Mac_machine_host_name": None,
            "Mac_machine_user_name": None,
            "Mac_machine_password": None,
            "SP revision": None
        }

        # In case no revision needs to be passed as input, set it to 0 in inputs
        # Need it for the current release.

        # FUTURE ENHANCEMENTS
        # Remove plan details from input and get from the Commcell properties

        # PRE-REQUISITES OF THE TESTCASE
        # - MSP should not be enabled ever on the commcell
        # - A laptop Plan should be created on commcell which should be associated to the commcell
        # Also follow the steps here:

        # Login with a non-administrator user on the laptop, for example: testlab\testuser_01
        # This is because for non MSP scenario we do not want the laptop to be registered and activated by an admin
        # as he might have installed the laptop..
        # Laptop user should ideally be the one who should see the backup running in the backup monitor once he first
        # logs in.
        # In case the testlab\testuser_01 is not logged in during registration or after registration,
        # the client will not be associated to the Default Plan. Also the client will not be a part of the Plan's
        # client group after registration.
        # And hence the OSC auto backup will not be triggered too.

        # User defined
        self._os_list = ['Windows', 'Mac']

        self.laptop = None


[docs]    def run(self):
        """ Main function for test case execution."""
        try:

            self.laptop = LaptopHelper(self)

            self.tcinputs['commserver_version'] = self.commcell.commserv_version
            self.tcinputs["commserv_host_name"] = self.commcell.commserv_hostname
            self.tcinputs["commserv_client_name"] = self.commcell.commserv_name

            for _os in self._os_list:
                machine_object = self.laptop.utility.get_machine_object(
                    self.tcinputs[_os+'_machine_host_name'],
                    self.tcinputs[_os+'_machine_user_name'],
                    self.tcinputs[_os+'_machine_password']
                )
                self.tcinputs[_os+'_machine_object'] = machine_object
                self.tcinputs[_os+'_machine_client_name'] = machine_object.machine_name

            #-------------------------------------------------------------------------------------
            self.laptop.tc.log_step("""
                - Make sure the Commserver is a non MSP Commcell
                    Turn on authcode from the commcell level from AdminConsole->Commcell->Administration
                    Set Default Laptop plan for the Commcell from Adminconsole->Commcell->Administration

                - Create a custom package from cloud workflow that uses "Do not require end user interaction"
                    *without the authcode.
                - Download the package from cloud and copy it on the client machine.
                - Install relevant platform packages on client
                    <Package executable> /wait /silent /install /silent /authcode <AUTH CODE>
                    Do *not run SimCalwrapper.
                    It's not needed when we are registering through auth code in installation command.
                - Once the client is registered a backup job should get triggered automatically from OSC schedule.
                    Wait for job to complete.
                - Modify subclient content and add new content. Automatic incremental backup should be triggered for
                    new content added.
                - Execute out of place restore for the content backed by backup job.

                Validation:
                        - Verify nchatterflag is off in registry for clients post registration.
                        - Verify Plan client group associations for clients
                        - The client should be a part of the default plan that was set for commcell
                        - Client appears in a plan in devices for Commcell
            """, 200)
            package_names = {'Windows':'WinX64_Client.exe', 'Mac':'Mac_Client.pkg'}

            #-------------------------------------------------------------------------------------
            self.laptop.tc.log_step("""
                1. Make sure the Commserver is a non MSP Commcell
                    Turn on authcode from the commcell level and set Default Laptop plan for the Commcell
            """)
            orghelper = OrganizationHelper(self.commcell)
            _ = orghelper.disable_msp()
            self.tcinputs["install_authcode"] = Installation(self.tcinputs, self.commcell).commcell_install_authcode()
            orghelper.commcell_default_plan(self.tcinputs['Default_Plan'])

            #-------------------------------------------------------------------------------------
            self.laptop.tc.log_step("""
                2. Create a custom package from cloud workflow that uses "Do not require end user interaction"
                    *without the authcode, and download the package from cloud.
            """)
            self.tcinputs["proxy_list"] = self.laptop.custom_package.proxy_list(self.commcell)
            _ = self.laptop.custom_package.create(self.tcinputs)
            _ = self.laptop.custom_package.download(self.tcinputs["proxy_list"])

            for _os in self._os_list:
                self.laptop.tc.log_step("Custom package Installation for [{0}]".format(_os))

                #---------------------------------------------------------------------------------
                self.laptop.tc.log_step("""
                    Install relevant platform packages on client
                    <Package executable> /wait /silent /install /silent /authcode <AUTH CODE>
                    Do *not run SimCalwrapper.
                """)
                self.tcinputs["client_user_name"] = self.tcinputs[_os+'_machine_user_name']
                self.tcinputs["client_password"] = self.tcinputs[_os+'_machine_password']
                self.tcinputs["client_host_name"] = self.tcinputs[_os+'_machine_host_name']
                self.tcinputs["client_name"] = self.tcinputs[_os+'_machine_client_name']
                self.tcinputs['executable_name'] = package_names[_os]

                client_name = self.tcinputs["client_name"]
                installer = Installation.create_installer_object(self.tcinputs, self.commcell)
                installer.install_client()
                self.commcell.refresh()

                #---------------------------------------------------------------------------------
                self.laptop.tc.log_step("""
                    Once the client is registered a backup job should get triggered automatically from OSC schedule.
                    Wait for job to complete.
                    Modify subclient content and add new content. Automatic incremental backup should be triggered for
                    new content added.
                    Execute out of place restore for the content backed by backup job.
                """)
                self.laptop.utils.osc_backup_and_restore(self.tcinputs[_os+'_machine_object'], False)

                #---------------------------------------------------------------------------------
                self.laptop.tc.log_step("""
                    c.
                        Validation:
                        -Verify nchatterflag is off in registry for clients post registration.
                        -Verify Plan client group associations for clients
                        -The client should be a part of the default plan that was set for commcell
                        -Client appears in a plan in devices.
                """)
                orghelper.validate_client(clients=client_name,
                                          chatter_flag=0,
                                          client_groups=[self.tcinputs['Default_Plan'] + ' clients'])

        except Exception as excp:
            self.laptop.tc.fail(excp)
        finally:
            self.laptop.utils.cleanup_jobs()
