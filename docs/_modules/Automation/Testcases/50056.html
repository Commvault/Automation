[docs]class TestCase(CVTestCase):
    """Class for executing

        File System Commit - Basic Functionality Scenarios
        This test case will verify the basic functionality of FS commit.
        The following scenarios will be covered as part of this test case.

        1. Full (Commit) -> Incremental
        2. Full -> Incremental (Commit) -> Incremental
        3. Full -> Incremental (Commit) -> Synthetic Full -> Incremental
        4. Full (Commit) -> Incremental (Commit) -> Incremental (Commit) -> Differential

    """

    def __init__(self):
        """Initializes test case class object"""
        super(TestCase, self).__init__()
        self.name = "File System Commit - Basic Functionality Scenarios"
        self.show_to_user = True
        self.tcinputs = {
            "TestPath": None,
            "StoragePolicyName": None
        }
        self.helper = None
        self.storage_policy = None
        self.num_of_dirs = 1
        self.slash_format = None
        self.test_path = None
        self.runid = None
        self.id = None
        self.client_machine = None



[docs]    def setup(self):
        """Initializes pre-requisites for this test case"""
        self.helper = FSHelper(self)
        self.helper.populate_tc_inputs(self)
        self.num_of_files = int(self.tcinputs.get('NumOfFiles', 10))
        self.file_size_in_kb = int(self.tcinputs.get('FileSizeInKb', 102400))
        self.threshold = int(self.tcinputs.get('Threshold', 3))

        """Main function for test case execution"""
        try:
            self.log.info(self.__doc__)

            # IMPROVING READABILITY BY REMOVING REPEATED USAGE OF self.
            os_sep = self.slash_format
            test_path = self.test_path
            runid = str(self.runid)
            id = str(self.id)
            storage_policy = self.storage_policy

            backupset_name = '_'.join(("backupset", id))
            self.log.info("Create a backupset for the scenarios if not already present.")
            self.helper.create_backupset(backupset_name)

            # actual_commit_files IS THE COMMIT FILES
            # PREPARED BY SCAN FOR THE REFERENCE COMMITTED JOB PARAMETERS.

            # expected_commit_files IS THE COMMIT FILES
            # WE EXPECT SCAN TO HAVE PREPARED BASED ON WHAT WE THINK THE COMMIT PARAMETERS MUST BE.

            for scan_type in ScanType:

                # Skip change journal scan for UNIX
                if self.client_machine.os_info != 'WINDOWS' and scan_type.value == ScanType.CHANGEJOURNAL.value:
                    continue

                # SCENARIO 1 BEGINS HERE
                scenario_num = "SCENARIO_1"
                scenario = "Full (Commit) -> Incremental"
                self.log.info("**********")
                self.log.info("{} : {}".format(scenario_num, scenario))

                subclient_name = '_'.join(("subclient", id, scan_type.name.lower(), scenario_num))
                content = list()
                content.append(os_sep.join((test_path, subclient_name)))
                run_path = os_sep.join((content[0], runid))
                full_data_path = os_sep.join((run_path, "full"))
                incr_1_data_path = os_sep.join((run_path, "incr_1"))

                self.log.info("Step 1.01 : Create a subclient.")
                self.helper.create_subclient(name=subclient_name,
                                             storage_policy=storage_policy,
                                             content=content, scan_type=scan_type,
                                             allow_multiple_readers=True)

                self.log.info("Step 1.02 : Add data for the FULL backup.")
                self.client_machine.generate_test_data(full_data_path,
                                                       self.num_of_dirs,
                                                       self.num_of_files,
                                                       self.file_size_in_kb)

                self.log.info("Step 1.03 : Run a FULL backup and COMMIT it.")
                job_1 = self.helper.run_backup(backup_level="Full", wait_to_complete=False)
                self.helper.commit_job(job_1[0], self.threshold, CommitCondition.FILES)

                self.log.info("Step 1.04 : Create the expected commit files for FULL job {}.".format(job_1[0].job_id))
                expected_commit_files = self.helper.create_expected_commit_files(job_1[0])

                self.log.info("Step 1.05 : Add new data for the INCREMENTAL.")
                self.client_machine.generate_test_data(incr_1_data_path,
                                                       self.num_of_dirs,
                                                       self.num_of_files,
                                                       self.file_size_in_kb)

                self.log.info("Step 1.06 : Run an INCREMENTAL backup and let it COMPLETE.")
                job_2 = self.helper.run_backup(backup_level="Incremental", wait_to_complete=True)

                self.log.info("Step 1.07 : Retrieve the commit files prepared by scan.")
                actual_commit_files = self.helper.get_actual_commit_files(job_1[0])

                self.log.info("Step 1.08 : Validating commit files for "
                              "committed job {}.".format(job_1[0].job_id))
                self.helper.validate_commit_files(expected_commit_files, actual_commit_files)

                self.log.info("END OF {}".format(scenario_num))
                self.log.info("**********")
