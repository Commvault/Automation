    def __init__(self):
        """Initializes test case class object"""
        super(TestCase, self).__init__()

        self.name = "Basic Acceptance Test of PostgreSQL backup and restore"
        self.product = self.products_list.POSTGRESQL
        self.feature = self.features_list.DATAPROTECTION
        self.show_to_user = True

    def _run_backup(self, subclient, backup_type):
        """Initiates the backup job for the specified subclient

        Args:
            subclient            (obj)       -- Subclient object for which backup needs to be run

            backup_type          (str)       -- Type of backup (FULL/INCREMENTAL)

        Returns:
            job                              -- JobID of the backup job

        Raises:
            BaseException:
            if unable to start the backup job

        """
        log = logger.get_log()
        job = subclient.backup(backup_type)
        log.info("Started %s backup with Job ID: %s",
                 backup_type, str(job.job_id))
        if not job.wait_for_completion():
            raise Exception(
                "Failed to run %s backup job with error: %s",
                str(backup_type),
                str(job.delay_reason)
            )
        log.info("Successfully finished %s backup job", (backup_type))

        return job

    def _run_restore(self, subclient, db_list, client_name, instance_name):
        """Initiates the restore job for the specified subclient

        Args:
            subclient            (Obj)       -- Subclient object for which restore needs to be run

            backup_type          (str)       -- Type of backup (FULL/INCREMENTAL)

            client_name          (str)       -- Name of the client which subclient belongs to

            instance_name        (str)       -- Name of the instance which subclient belongs to

        Returns:
            job                              -- JobID of the restore job

        Raises:
            BaseException:
            if unable to start the backup job

        """
        log = logger.get_log()
        log.debug("db_list = %s", str(db_list))
        log.debug("client_name = %s", str(client_name))
        log.debug("instance_name = %s", str(instance_name))
        backupset_name = subclient._backupset_object.backupset_name
        job = subclient.restore_postgres_server(db_list, client_name, instance_name)
        log.info("Started %s Restore with Job ID: %s",
                 backupset_name, str(job.job_id))
        if not job.wait_for_completion():
            raise Exception(
                "Failed to run restore job with error: %s",
                str(job.delay_reason)
            )

        log.info("Successfully finished %s restore job", str(backupset_name))

        return job

        """Main function for test case execution"""
        log = logger.get_log()

        try:
            log.info("Started executing %s testcase", str(self.id))
            pgsql_helper_object = pgsqlhelper.PostgresHelper(
                self.commcell, self.client, self.instance)

            subclient_list = list()
            for i in self.instance.subclients.all_subclients:
                if i.startswith('dumpbased'):
                    subclient_list.append(i.split("\\")[1])
            log.debug("List of subclients configured for dumpbased"
                      "backupset: %s", str(subclient_list))
            self.postgres_data_population_size = self.tcinputs['TestDataSize']
            num_of_databases = self.postgres_data_population_size[0]
            num_of_tables = self.postgres_data_population_size[1]
            num_of_rows = self.postgres_data_population_size[2]
            # Specifying Maintanance Database and Template0 -- hard coded right now
            ignoredb_list = ["postgres", "template0"]
            pgsql_server_user_name = self.instance._properties[
                'postGreSQLInstance']['SAUser']['userName']
            pgsql_server_password = pgsql_helper_object._postgres_db_password
            pgsql_server_port = self.instance._properties['postGreSQLInstance']['port']
            pgsql_server_hostname = self.client.client_hostname
            pgsql_db_object = database_helper.PostgreSQL(
                pgsql_server_hostname,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password,
                "postgres")

            pgsql_bin_dir = self.instance._properties['postGreSQLInstance']['BinaryDirectory']
            log.info("Bin Directory: %s", str(pgsql_bin_dir))
            log.info("Postgres server Port: %s", str(pgsql_server_port))
            log.info("postgres server password: %s", str(pgsql_server_password))

            pgsql_data_dir = pgsql_helper_object.get_postgres_data_dir(
                pgsql_bin_dir, pgsql_server_password, pgsql_server_port)
            log.info(pgsql_data_dir)

            ################# DumpBased Backup/Restore Operations ########################
            log.info(
                "#" * (10) + "  DumpBased Backup/Restore Operations  " + "#" * (10))
            backup_set = self.instance.backupsets.get("dumpbasedbackupset")
            subclient_dict = (backup_set.subclients._subclients)
            subclient = backup_set.subclients.get("default")
            self.subclient_list = list(subclient_dict.keys())
            log.debug(subclient_list)

            db_prefix = "auto_full_dmp"
            log.info("Generating Test Data")
            pgsql_helper_object.generate_test_data(
                pgsql_server_hostname,
                num_of_databases,
                num_of_tables,
                num_of_rows,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password,
                True,
                db_prefix)
            log.info("Test Data Generated successfully")

            db_list_before_backup = None
            # Get Subclient content
            # check if subclient exists or not if not default subclient
            log.info("Collecting DB List")
            if subclient.subclient_name.lower() == "default":
                db_list_before_backup = pgsql_db_object.get_db_list()
                if db_list_before_backup is None:
                    log.error(
                        "Unable to get the database list.Cleaning up test data.")
                    return 1
                # Get list of all the subclients content and exclude them from total list
                # of Databases
                log.debug(db_list_before_backup)
                all_other_sub_clients_contents = list()
                for sbclnt in self.subclient_list:
                    if sbclnt.lower() != backup_set.subclients.default_subclient.lower():
                        log.info("Subclient name is not default subclient")
                        log.debug(sbclnt)
                        sub_client_new = backup_set.subclients.get(sbclnt)
                        subc_content_db_list = None
                        log.debug(sub_client_new.content)
                        subc_content_db_list = sub_client_new.content
                        log.debug(subc_content_db_list)
                        sub_client_db_list = []
                        for contents in subc_content_db_list:
                            sub_client_db_list.append(
                                contents['postgreSQLContent']['databaseName'])
                        log.info(
                            "Database list of %s Subclient is %s", sbclnt, sub_client_db_list)
                        for database in sub_client_db_list:
                            all_other_sub_clients_contents.append(
                                database.lstrip("/"))
                for db_name in all_other_sub_clients_contents:
                    if db_name in db_list_before_backup:
                        db_list_before_backup.remove(db_name)
            else:
                if self.subclient.subclient_name not in self.subclient_list:
                    raise Exception(
                        "{0} SubClient Does't exist".format(
                            self.subclient.subclient_name))

                sub_client_content = self.subclient.content
                db_list = []
                for i in sub_client_content:
                    db_list.append(
                        sub_client_content[i]['postgreSQLContent']['databaseName'])

                db_list_before_backup = db_list

                log.info(subclient.content)
                log.info("Subclient Contents:")
                log.info(db_list)
                db_list_before_backup = pgsql_helper_object.strip_slash_char(db_list_before_backup)

            # Get the subclient content Info before backup
            log.info("Collect information of the subclient content before backup")
            for i in ignoredb_list:
                if i in db_list_before_backup:
                    db_list_before_backup.remove(i)

            # Colecting Meta data
            before_full_backup_db_list = pgsql_helper_object.generate_db_info(
                db_list_before_backup,
                pgsql_server_hostname,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password)

            ###################### Running Full Backup ##############################
            log.info(
                "#" * (10) + "  Running Dumpbased Full Backup  " + "#" * (10))
            self._run_backup(subclient, "FULL")

            # appending "/" to dbnames for dumpbased restore
            db_list = ["/" + ele for ele in db_list_before_backup]
            time.sleep(float(10))
            log.info("Sleeping for 10 seconds")

            log.info("Deleting Automation Created databases")
            tc_name = "auto"
            pgsql_helper_object.cleanup_tc_db(
                pgsql_server_hostname,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password,
                tc_name)

            # ###################### Running restore ###################################
            log.info(
                "#" * (10) + "  Running Dumpbased Restore  " + "#" * (10))
            self._run_restore(
                subclient, db_list, self.client.client_name, self.instance.instance_name)
            after_backup_db_list = pgsql_db_object.get_db_list()
            if after_backup_db_list is None:
                log.error(
                    "Unable to get the database list")
                raise Exception("Unable to get the database list")

            # Get subclient content info after restore
            log.info("Collect information of the subclient content after restore")
            for i in ignoredb_list:
                if i in after_backup_db_list:
                    after_backup_db_list.remove(i)
            after_restore_db_info = pgsql_helper_object.generate_db_info(
                after_backup_db_list,
                pgsql_server_hostname,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password)

            # validation
            log.info(
                "Validating the database information collected before Full Backup \

            # validate subclient content information collected before backup and after restore
            ret_code = pgsql_helper_object.validate_db_info(

            if ret_code != True:

                    "Database information validation failed")

                log.info("Database information validation passed successfully")

            # ########################## FS Backup/Restore Operation #################

            log.info(

            backup_set = self.instance.backupsets.get("fsbasedbackupset")
            subclient_dict = (backup_set.subclients._get_subclients())
            subclient = backup_set.subclients.get("default")
            self.subclient_list = list(subclient_dict.keys())
            log.debug(subclient_list)

            self._run_backup(subclient, "FULL")

            # Modifications to DB before incremental
            db_prefix = "auto_incr_fs"
            pgsql_helper_object.generate_test_data(
                pgsql_server_hostname,
                num_of_databases,
                num_of_tables,
                num_of_rows,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password,
                True,
                db_prefix)
            log.info("Test Data Generated successfully")

            # Colecting Meta data
            db_list_before_backup = pgsql_db_object.get_db_list()
            if db_list_before_backup is None:
                log.error(
                    "Unable to get the database list")
                raise Exception("Data validation failure")
            # Get the subclient content Info before backup
            log.info("Collect information of the subclient content before backup")
            for i in ignoredb_list:
                if i in db_list_before_backup:
                    db_list_before_backup.remove(i)

            before_full_backup_db_list = pgsql_helper_object.generate_db_info(
                db_list_before_backup,
                pgsql_server_hostname,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password)

            log.info(
                "#" * (10) + "  Running FSBased Incremental Backup  " + "#" * (10))
            self._run_backup(subclient, "INCREMENTAL")

            time.sleep(float(10))
            log.info("Sleeping for 10 seconds")

            # stopping postgres server before the restore
            ret_code = pgsql_helper_object.stop_postgres_server(

            if ret_code == 0:

            elif ret_code == -2:
                log.info("Server was not running, Continuing to restore phase")
            else:
                log.info("Server stop failure, error")

            # Running FS Restore
            log.info(
                "#" * (10) + "  Running FSBased Restore  " + "#" * (10))
            db_list = ["/data"]
            self._run_restore(

            db_list = ["/" + ele for ele in db_list_before_backup]

                    "Unable to get the database list")
                raise Exception("Unable to get the database list")
            log.info("Collect information of the subclient content after restore")
            for i in ignoredb_list:

                    after_backup_db_list.remove(i)

            after_restore_db_info = pgsql_helper_object.generate_db_info(
                after_backup_db_list,
                pgsql_server_hostname,
                pgsql_server_port,

            # validation

                Full Backup and after Inplace Restore for DumpBasedBackupset")
            # validate subclient content information collected before backup and after restore
            ret_code = pgsql_helper_object.validate_db_info(
                before_full_backup_db_list, after_restore_db_info)
            if ret_code != True:
                log.error(
                    "Database information validation failed")
                raise Exception("Data validation failure")
            else:
                log.info("Database information validation passed successfully")

            time.sleep(float(10))
            log.info("sleeping for 10 seconds")

            log.info("Deleting Automation Created databases")
            tc_name = "auto"
            pgsql_helper_object.cleanup_tc_db(
                pgsql_server_hostname,
                pgsql_server_port,
                pgsql_server_user_name,
                pgsql_server_password,
                tc_name)

        except Exception as excp:
            log.error('Failed with error: %s', str(excp))
            self.result_string = str(excp)
            self.status = constants.FAILED
