[docs]    def init_tc(self):
        """ Initial configuration for the test case. """
        try:
            self.browser = BrowserFactory().create_browser_object()
            self.browser.open()
            self.webconsole = WebConsole(self.browser, self.commcell.webconsole_hostname)
            self.utils.webconsole = self.webconsole
            self.webconsole.login()
            self.webconsole.goto_reports()

            # Deletes any previous report if any with the same name.
            self.utils.cre_api.delete_custom_report_by_name(self.name, suppress=True)

            # Navigate to Custom Report Builder Page.
            navigator = Navigator(self.webconsole)
            navigator.goto_report_builder()

            self.report_builder = builder.ReportBuilder(self.webconsole)
            self.report_builder.set_report_name(self.name)

        except Exception as exception:
            raise CVTestCaseInitFailure(exception) from exception


    @test_step
    def add_inputs(self):
        """ Adds inputs to the builder."""

        self.daterange = DateRange(self.input_name[0])
        self.report_builder.add_input(self.daterange)
        self.daterange.enable_options(
            custom_range=True,
            time=True,
            minutes=True,
            hours=True,
        )
        self.daterange_controller = DateRangeController(self.input_name[0])
        self.daterange.add_html_controller(self.daterange_controller)
        self.daterange.save()

        self.time = Date(self.input_name[1])
        self.report_builder.add_input(self.time)
        self.time_picker_controller = TimePickerController(self.input_name[1])
        self.time.add_html_controller(self.time_picker_controller)
        self.time.save()

        self.date = Date(self.input_name[2])
        self.report_builder.add_input(self.date)
        self.date_picker_controller = DatePickerController(self.input_name[2])
        self.date.add_html_controller(self.date_picker_controller)
        self.date.save()

        self.date_time = Date(self.input_name[3])
        self.report_builder.add_input(self.date_time)
        self.date_time_picker_controller = DateTimePickerController(self.input_name[3])
        self.date.add_html_controller(self.date_time_picker_controller)
        self.date.save()

        self.string = String(self.input_name[4])
        self.report_builder.add_input(self.string)
        self.list_box_controller = ListBoxController(self.input_name[4])
        self.string.add_html_controller(self.list_box_controller)
        self.list_box_controller.set_labels_and_values(["A", "B", "C"], ["1", "2", "3"])
        self.string.save()

        titles = self.report_builder.get_all_input_titles()
        for loop_var in range(len(self.input_name)):
            if self.input_name[loop_var] not in titles:
                raise CVTestStepFailure("Added input {0} is not displayed in the report builder"
                                        " page".format(self.input_name[loop_var]))

    @test_step
    def associate_inputs(self):
        """Add HTML input and parameter to Dataset and preview"""
        dataset = builder.Datasets.DatabaseDataset()
        self.report_builder.add_dataset(dataset)
        dataset.set_dataset_name("Test Dataset")
        dataset.set_sql_query("SELECT @daterange_from [DateRangeFrom], "
                              "@daterange_to [DateRangeTo], "
                              "@time_t [Time], "
                              "@datetime_t [DateTime], "
                              "@listbox [ListBox], "
                              "@date_date[Date]")

        dataset.add_parameter("daterange", self.daterange)
        dataset.add_parameter("time_t", self.time)
        dataset.add_parameter("datetime_t", self.date_time)
        dataset.add_parameter("date_date", self.date)
        dataset.add_parameter("listbox", self.string)

        dataset.save()

        self.table = builder.DataTable("Automation Table 1")
        self.report_builder.add_component(self.table, dataset)
        self.table.add_column_from_dataset()

    @test_step
    def validate_data(self):
        """Validate data on both on builder and viewer"""
        self.populate_input_and_validate()

        self.report_builder.save(deploy=True)
        self.report_builder.open_report()

        self.report_viewer = viewer.CustomReportViewer(self.webconsole)
        self.table = viewer.DataTable("Automation Table 1")
        self.report_viewer.associate_component(self.table)

        self.report_viewer.associate_input(self.daterange_controller)
        self.report_viewer.associate_input(self.time_picker_controller)
        self.report_viewer.associate_input(self.date_picker_controller)
        self.report_viewer.associate_input(self.date_time_picker_controller)
        self.report_viewer.associate_input(self.list_box_controller)

        self.populate_input_and_validate()

    @test_step
    def bookmark_report_url(self):
        """Bookmark the report URL with input and validate data"""
        self.browser.driver.refresh()
        self.table = viewer.DataTable("Automation Table 1")
        self.report_viewer.associate_component(self.table)
        self.populate_input_and_validate(populate=False)


[docs]    def populate_input_and_validate(self, populate=True):
        """Populates input and validates if set to true"""
        if populate:
            self.daterange_controller.set_starting_date("10", "07", "2018")
            self.daterange_controller.set_ending_date("22", "12", "2018")
            self.daterange_controller.set_starting_time("10", "30")
            self.daterange_controller.set_ending_time("06", "30", am=False)
            self.time_picker_controller.set_time_controller("12", "30", "AM")
            self.date_picker_controller.set_date_controller("14", "04", "2018")
            self.date_time_picker_controller.set_date_time_controller("14", "01", "2018",
                                                                      "05", "30", am=False)
            self.list_box_controller.select_value("A")
            self.list_box_controller.apply()

        data = self.table.get_table_data()
        expected_result = {'DateRangeFrom': ['Jul 10, 2018, 10:30:00 AM'],
                           'DateRangeTo': ['Jul 22, 2018, 6:30:00 PM'], 'Time': ['12:30:00 AM'],
                           'DateTime': ['Jan 14, 2018, 4:00:00 PM'], 'ListBox': ['1'],
                           'Date': ['Apr 14, 2018']}

        if data != expected_result:
            raise CVTestStepFailure(
                "Unexpected data [%s], expected [%s]" % (
                    str(data), str(expected_result)))
