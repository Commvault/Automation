[docs]    def run(self):
        """Main function for test case execution"""
        table_name = "T52808"
        database_name = "DB52808"
        res_db_list = []
        try:
            self.log.info("Started executing %s testcase", self.id)
            #  Test data Population
            self.sybase_cv_helper.sybase_populate_data(
                database_name, table_name)
            # Full backup on default subclient
            self.log.info("Full Backup")
            full_job = self.sybase_cv_helper.backup_and_validation(
                self.subclient, backup_type='full')
            self.log.info("Full job : %s completed", str(full_job.job_id))

            # Add test table before next transaction log backup to custom user
            # database
            table_name_tl = "{0}_TLOG1".format(table_name)
            self.sybase_cv_helper.single_table_populate(
                database_name, table_name_tl)

            # Transaction Log Backup
            self.log.info("Transaction Log  Backup")
            tl_job = self.sybase_cv_helper.backup_and_validation(
                self.subclient, backup_type='incremental')
            self.log.info("TL job : %s completed", str(tl_job.job_id))

            # Storing custom database table list and table content details
            # before restore
            self.log.info(
                "Storing custom database table list"
                "and table content details before restore ")
            status, table_list_before_restore = self.sybase_helper.get_table_list(
                database_name)
            status, full_job_table_content_before_restore = self.sybase_helper.get_table_content(
                database_name, table_name)
            status, tl_job_table_content_before_restore = self.sybase_helper.get_table_content(
                database_name, table_name_tl)

            # Drop the user database
            drop_status = self.sybase_helper.drop_sybase_database(
                database_name)
            self.log.info("Database Drop Status : %s", str(drop_status))

            # Perform restore of dropped database
            self.log.info("Partial Restore")
            res_db_list.append(database_name)
            partial_restore_job = self.instance.restore_database(
                database_list=res_db_list)
            if not partial_restore_job.wait_for_completion():
                raise Exception(
                    "Failed to run Restore job with error: {0}".format(
                        partial_restore_job.delay_reason))

            # Data Validation after restore
            status, table_list_after_restore = self.sybase_helper.get_table_list(
                database_name)
            status, full_job_table_content_after_restore = self.sybase_helper.get_table_content(
                database_name, table_name)
            status, tl_job_table_content_after_restore = self.sybase_helper.get_table_content(
                database_name, table_name_tl)

            table_status = False
            table_content_status = False
            if table_list_before_restore == table_list_after_restore:
                table_status = True
            if (full_job_table_content_before_restore == full_job_table_content_after_restore
               ) and (tl_job_table_content_before_restore == tl_job_table_content_after_restore):
                table_content_status = True

            # TC status setting based on data validation results
            if table_status and table_content_status:
                self.log.info("Data validation succeeded")
                self.log.info("Test Case Passed")
                self.status = constants.PASSED
            else:
                self.log.info("Data validation failed")
                self.log.info("Test Case  Failed")
                self.status = constants.FAILED
        except BaseException as exp:
            self.log.error("Testcase failed with exception : %s", str(exp))
            self.log.exception("Detailed Exception : %s", str(sys.exc_info()))
            self.result_string = str(exp)
            self.status = constants.FAILED
        finally:
            self.log.info("Cleanup Function")
            self.sybase_cv_helper.sybase_cleanup_test_data(database_name)
            self.sybase_cv_helper.sybase_delete_database_from_subclient(
                self.subclient, database_name)
