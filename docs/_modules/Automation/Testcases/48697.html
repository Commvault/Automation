[docs]    def run(self):
        """Main function for test case execution"""
        try:
            # Initialize test case inputs
            self.log.info(self.__doc__)
            self.log.info(
                "\nStep 1 : Create backupset as default backupset for this "
                "testcase if it doesn't already exist."
            )
            backupset_name = "backupset_" + self.id
            self.ma_helper.create_default_backupset(backupset_name)

            self.log.info(
                "\nStep 2 : Define the following monikers as "
                "content for the default subclient of the default backupset."
                r"\%Desktop%"
                r"\%Documents%"
                r"\%Music%"
                r"\%Pictures%"
                r"\%Videos%"
            )
            subclients_object = self.backupset.subclients
            self.subclient = subclients_object.get('default')

            self.ma_helper.update_subclient(
                self.storage_policy,
                ["\\%Desktop%", "\\%Music%", "\\%Pictures%", "\\%Videos%", "\\%Documents%"]
            )

            self.log.info("\nStep 3 : Create some test data under the well-known folders.")
            suffix = "full_incr1"
            self.ma_helper.generate_test_data_for_ma(self.src_machine, suffix)

            self.log.info("\nStep 4 : Run a Full or first Incremental Backup.")
            job_full_inc1 = self.ma_helper.run_backup_verify()

            self.log.info(
                "\nStep 5 : Perform both regular FS and MA content validation if applicable."
            )
            if (job_full_inc1[0].backup_level).lower() == "full":
                self.ma_helper.ma_restore_verify(
                    job_full_inc1[0],
                    suffix,
                    ma_validation_flag=True
                )
            else:
                self.ma_helper.ma_restore_verify(
                    job_full_inc1[0],
                    suffix,
                    ma_validation_flag=False
                )

            self.log.info("\nStep 6 : Perform addition and modification"
                          " of test data under well-known folders.")
            suffix = "incr2"
            self.ma_helper.generate_test_data_for_ma(self.src_machine, suffix, add_incr=True)

            self.log.info("\nStep 7 : Run an Incremental Backup before Synthetic Full.")
            job_inc2 = self.ma_helper.run_backup_verify(
                backup_level="Synthetic_full",
                incremental_backup=True,
                incremental_level="BEFORE_SYNTH")

            self.log.info("\nStep 8 : Perform regular FS backup validation.")
            self.ma_helper.ma_restore_verify(job_inc2[0], suffix, ma_validation_flag=False)

            self.log.info("\nStep 9 : Perform addition and modification "
                          "of test data under well-known folders.")
            suffix = "incr3"
            self.ma_helper.generate_test_data_for_ma(self.src_machine, suffix, add_incr=True)

            self.log.info("\nStep 10 : Run an Incremental Backup.")
            job_inc3 = self.ma_helper.run_backup_verify(backup_level="Incremental")

            self.log.info("\nStep 11 : Perform both regular FS backup validation "
                          "and MA validation.")
            self.ma_helper.ma_restore_verify(job_inc3[0], suffix, ma_validation_flag=True)

            self.log.info(
                "\nStep 12 : Perform a Sync. Restore of "
                "the backed up data to the destination client."
            )
            job_sync_restore1 = self.ma_helper.run_sync_restore(self.dst_client)

            if job_sync_restore1.summary['status'].lower() == "completed":
                self.log.info("***TEST CASE COMPLETED SUCCESSFULLY AND PASSED***")
                self.status = constants.PASSED
            else:
                self.log.info("***TEST CASE FAILED***")
                self.status = constants.FAILED

        except Exception as excp:
            error_message = "Failed with error: {}".format(str(excp))
            self.log.error(error_message)
            self.result_string = str(excp)
            self.status = constants.FAILED
