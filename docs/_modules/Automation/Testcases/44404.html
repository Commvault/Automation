    """Testcase to verify Health report tile Subclients without storage policy association"""
    def __init__(self):
        super(TestCase, self).__init__()
        self.name = "Metrics: Health - Subclients without storage policy association"

        self.tile = None

[docs]    def setup(self):
        """Intializes Private metrics object required for this testcase"""
        self.private_metrics = PrivateMetrics(self.commcell)

        self.private_metrics.save_config()

            self.log.info(

            )
            self._input_client.enable_backup()



[docs]    def init_webconsole(self):
        """Initialize the application to the state required by the testcase"""
        try:
            self.browser = BrowserFactory().create_browser_object()
            self.browser.open()
            self.webconsole = WebConsole(self.browser, self.commcell.webconsole_hostname)
            self.webconsole.login()
            self.webconsole.goto_reports()
            self.navigator = Navigator(self.webconsole)
            self.navigator.goto_commcell_dashboard(self.commcell.commserv_name)
            self.navigator.goto_health_report()
            self.tile = Subclients_SP_Assoc(self.webconsole)
        except Exception as msg:
            raise CVTestCaseInitFailure(msg) from msg


    @test_step_handler()
    def validate_status(self):
        """Validate Warning status of health Tile"""
        visible_status = self.tile.get_health_status()
        if visible_status != HealthConstants.STATUS_WARNING:
            raise CVTestStepFailure(
                "Self Service not in [{0}] Status, visible status is [{1}]"
                .format(HealthConstants.STATUS_WARNING, visible_status)
            )

    @test_step_handler()
    def validate_outcome(self):
        """Verify outcome contains client and subclient counts"""
        return self.tile.get_entity_counts()

    @test_step_handler()
    def validate_details(self, entities, include_client=True):
        """Validate details of the health Tile"""
        self.tile.access_view_details()
        table = MetricsTable(self.webconsole)
        clients = list(set(table.get_data_from_column('Client')))  # get distinct clients
        subclients = table.get_data_from_column('Subclient')
        if len(clients) != entities['client']:
            raise CVTestStepFailure(
                "Client Count in summary {0} but it is different in details page"
                " visible clients : {1}".format(
                    str(entities['client']),
                    str(clients)
                )
            )

        if len(subclients) != entities['Subclient']:
            raise CVTestStepFailure(
                "Subclient Count in summary {0} but it is different in details page"
                " visible clients : {1}".format(
                    str(entities['subclients']),
                    str(subclients)
                )
            )
        if include_client:
            if self.tcinputs['Client_without_SP'] not in clients:
                raise CVTestStepFailure(
                    "Expected client [{0}] not found in details page {1}".format(
                        self.tcinputs['Client_without_SP'],
                        str(clients)
                    )
                )
        else:
            if self.tcinputs['Client_without_SP'] in clients:
                raise CVTestStepFailure(
                    "Backup disabled client [{0}] found in details page {1}".format(
                        self.tcinputs['Client_without_SP'],
                        str(clients)
                    )
                )


[docs]    def disable_bkp_upload(self):
        """Disable backup activity on the client and perform upload now"""
        self.log.info(f"Disabling Backup activity for client {self.tcinputs['Client_without_SP']}")
        self._input_client.disable_backup()
        self.utils.private_metrics_upload()

[docs]    def run(self):
        try:
            self.utils.private_metrics_upload()
            self.init_webconsole()
            self.validate_status()
            entities = self.validate_outcome()
            self.validate_details(entities)
            self.disable_bkp_upload()
            self.navigator.goto_health_report()
            entities = self.validate_outcome()
            self.validate_details(entities, include_client=False)

        except Exception as error:
            self.utils.handle_testcase_exception(error)
        finally:
            self.log.info(
                f"Enabling Backup activity for client {self.tcinputs['Client_without_SP']}"
            )
            self._input_client.enable_backup()
            WebConsole.logout_silently(self.webconsole)
            Browser.close_silently(self.browser)
