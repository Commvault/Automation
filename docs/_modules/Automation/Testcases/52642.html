'''
Test case to check the basic acceptance of operation window in Admin console.

It verifies
1. Creation of operation rule based on different criteria's passed as arguments
2. Validates if the operation rule is created successfully.
3. Editing of operation rule created in above steps.
4. Deletion of operation rule created, edited & verified in above steps.
'''

from AutomationUtils.cvtestcase import CVTestCase
from Web.Common.cvbrowser import BrowserFactory
from Web.AdminConsole.Helper.LoginHelper import LoginMain
from Web.AdminConsole.Helper.OperationWindowHelper import OperationWindowMain
from Reports.utils import TestCaseUtils


[docs]class TestCase(CVTestCase):

    ''' Basic Acceptance test for Backup rule configuration '''

    def __init__(self):
        '''
       Initializing the Test case file
        '''
        super(TestCase, self).__init__()
        self.name = "Basic Acceptance Test Backup window in AdminConsole"
        self.product = self.products_list.COMMSERVER
        self.feature = self.features_list.ADMINCONSOLE
        self.show_to_user = False
        self.utils = TestCaseUtils(self)
        self.tcinputs = {
            "operation_rule_name": None,
            "new_operation_rule_name": None,
            "start_date_combo_list": None,
            "end_date_combo_list": None
            }


[docs]    def run(self):

        try:
            self.log.info("Started executing {0} testcase".format(self.id))
            self.log.info("*" * 10 + " Initialize browser objects " + "*" * 10)

            factory = BrowserFactory()
            browser = factory.create_browser_object()
            browser.open()
            driver = browser.driver

            self.log.info("Creating the login object")
            login_obj = LoginMain(driver, self.csdb)
            login_obj.login(self.inputJSONnode['commcell']['commcellUsername'],
                            self.inputJSONnode['commcell']['commcellPassword'])
            self.log.info("login successful")

            operation_window_obj = OperationWindowMain(driver)

            start_date_list = self.tcinputs['start_date_combo_list']
            end_date_list = self.tcinputs['end_date_combo_list']

            operation_window_obj.operation_rule_name = self.tcinputs['operation_rule_name']
            operation_window_obj.start_date = start_date_list[0]
            operation_window_obj.end_date = end_date_list[0]

            operation_window_obj.create_operation_rule()
            self.log.info("Backup rule created successfully")

            operation_window_obj.validate_operation_rule()
            self.log.info("initial validation for operation rule is successfully completed")

            operation_window_obj.operation_rule_name = self.tcinputs['new_operation_rule_name']
            operation_window_obj.start_date = [start_date_list[1]]
            operation_window_obj.end_date = [end_date_list[1]]

            operation_window_obj.operation_day = dict.fromkeys(['Tuesday', 'Thursday',
                                                                     'Saturday'], 1)
            operation_window_obj.operation_day.update(dict.fromkeys(['Monday', 'Wednesday',
                                                                     'Friday', 'Sunday'], 0))
            operation_window_obj.operation_time = dict.fromkeys(['6', '8', '10', '11', '12'
                                                                      , '13', '22', '23'], 1)
            operation_window_obj.operation_time.update(dict.fromkeys(['0', '1', '2', '3', '4',
                                                                      '14', '15', '16', '17',
                                                                      '18'], 0))

            operation_window_obj.modify_operation_rule()
            self.log.info("Backup rule edited successfully")

            operation_window_obj.validate_operation_rule()
            self.log.info("Validation for operation rule after editing is successfully \
completed")

            operation_window_obj.start_date = start_date_list
            operation_window_obj.end_date = end_date_list

            operation_window_obj.modify_operation_rule()
            self.log.info("Multiple Editing for Backup verified successfully,\
now proceeding to delete backup rule")

        except Exception as exp:
            self.utils.handle_testcase_exception(exp)

        finally:

            operation_window_obj.del_operation_rule()
            self.log.info("Backup rule deleted successfully. \
            Test case completed. Exiting module.....Thanks")
            browser.close()
