            "Tenant_username": None,
            "Tenant_password": None,

            "Mac_machine_user_name": None,
            "Mac_machine_password": None,
            "Windows_machine_host_name": None,
            "Windows_machine_user_name": None,
            "Windows_machine_password": None,

[docs]    def setup(self):
        """Initializes pre-requisites for test case"""
        try:
            self.laptop = LaptopHelper(self, company=self.tcinputs['Tenant_company'])

            self.tcinputs['organization_object'] = self.laptop.organization.company
            self.tcinputs['commserver_version'] = self.commcell.commserv_version
            self.tcinputs["commserv_host_name"] = self.commcell.commserv_hostname
            self.tcinputs["commserv_client_name"] = self.commcell.commserv_name
            self.tcinputs["commcell_username"] = self.tcinputs["Tenant_username"]
            self.tcinputs["commcell_password"] = self.tcinputs["Tenant_password"]

            for _os in self._os_list:
                machine_object = self.laptop.utility.get_machine_object(
                    self.tcinputs[_os+'_machine_host_name'],
                    self.tcinputs[_os+'_machine_user_name'],
                    self.tcinputs[_os+'_machine_password']
                )
                self.tcinputs[_os+'_machine_object'] = machine_object
                self.tcinputs[_os+'_machine_client_name'] = machine_object.machine_name

        except Exception as excp:
            raise Exception("Failed in setup module with exception [{0}]".format(excp))



[docs]    def run(self):
        """ Main function for test case execution."""
        try:


            #-------------------------------------------------------------------------------------
            self.laptop.tc.log_step("""
                1. Turn on authcode from the commcell level
                    Set default plan for Tenant Company

                2. Create a custom package that uses "Do not require end user interaction", with
                    auth code, and download the package from cloud.

                    a. Install custom package *without authcode with options:
                        <InstallationPackage>.exe /silent /install /silent
                        Run Simcalwrapper
                    b.
                        Wait for laptop full backup job to start from osc schedule.
                        Change subclient content, wait for incremental backup to trigger.
                        Execute out of place restore.

                    c. Validation
                        - Verify chatterflag is off in registry for clients
                        - Verify Plan client group associations for clients
                        - The client should be a part of the default plan that was set
                            for the commcell.
                        - Client appears in a plan in devices admin console page
            """, 200)

            package_names = {'Windows':'WinX64_Client.exe', 'Mac':'Mac_Client.pkg'}


            #-------------------------------------------------------------------------------------
            self.laptop.tc.log_step("""
                1. Turn off/on authcode from the commcell level, and validate if the auth code
                    can be disabled/enabled.
                   Set default plan for Tenant Company, and validate that the default
                    plan can be set.
            """)
            _ = self.laptop.organization.modify_auth_code('disable')
            self.tcinputs["authcode"] = self.laptop.organization.modify_auth_code('enable')
            org_default_plan = self.laptop.organization.default_plan
            self.laptop.organization.default_plan = self.tcinputs['Default_Plan']


            #-------------------------------------------------------------------------------------
            self.laptop.tc.log_step("""
                2. Create a custom package that uses "Do not require end user interaction", with
                    auth code, and download the package.
            """)
            self.tcinputs["proxy_list"] = self.laptop.custom_package.proxy_list(self.commcell)
            _ = self.laptop.custom_package.create(self.tcinputs)
            _ = self.laptop.custom_package.download(self.tcinputs["proxy_list"])

            for _os in self._os_list:
                self.laptop.tc.log_step("Custom package Installation for [{0}]".format(_os))

                #---------------------------------------------------------------------------------
                self.laptop.tc.log_step("""
                    a.
                        Install custom package and register *without authcode arguments:
                        <InstallationPackage>.exe /silent /install /silent
                        Run Simcallwrapper
                """)
                self.tcinputs["client_user_name"] = self.tcinputs[_os+'_machine_user_name']
                self.tcinputs["client_password"] = self.tcinputs[_os+'_machine_password']
                self.tcinputs["client_host_name"] = self.tcinputs[_os+'_machine_host_name']
                self.tcinputs["client_name"] = self.tcinputs[_os+'_machine_client_name']
                self.tcinputs['executable_name'] = package_names[_os]

                client_name = self.tcinputs["client_name"]
                installer = Installation.create_installer_object(self.tcinputs, self.commcell)
                installer.install_client()
                if _os == 'Windows':
                    installer.execute_register_me_command()
                self.commcell.refresh()


                #---------------------------------------------------------------------------------
                self.laptop.tc.log_step("""
                    b.
                        Wait for laptop full backup job to start from osc schedule.
                        Change subclient content, wait for incremental backup to trigger.
                        Execute out of place restore.
                """)
                self.laptop.utils.osc_backup_and_restore(
                    self.tcinputs[_os+'_machine_object'], False
                )

                #---------------------------------------------------------------------------------
                self.laptop.tc.log_step("""
                    c.
                        Validation:
                        -Verify nchatterflag is off in registry for clients post registration.
                        -Verify Plan client group associations for clients
                        -The client should be a part of the default plan that was set for commcell
                        -Client appears in a plan in devices admin console page.
                """)
                self.laptop.organization.validate_client(clients=client_name, chatter_flag=0)

        except Exception as excp:
            self.laptop.tc.fail(excp)
        finally:
            self.laptop.utils.cleanup_jobs()
            self.laptop.organization.default_plan = org_default_plan
