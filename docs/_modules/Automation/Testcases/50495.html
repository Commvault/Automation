    def __init__(self):
        """Initializes TestCase object"""
        super(TestCase, self).__init__()
        self.name = "Data Server IP Acceptance test case"
        self.product = self.products_list.MEDIAAGENT
        self.feature = self.features_list.MEDIAMANAGEMENT

            "ClientName": None,

            "MALibraryMountPath": None,

            "ShareMAName": None,
            "SharedMALibraryMountPath": None,
            "DedupPath": None,
            "RestoreLocation": None
        }

    def _run_backup(self, backup_type):
        """Initiates backup job and waits for completion"""

        self.log.info("*" * 10 + " Starting Subclient {0} Backup ".format(backup_type) + "*" * 10)
        job = self.subclient.backup(backup_type)
        self.log.info("Started {0} backup with Job ID: {1}".format(backup_type, str(job.job_id)))
        if not job.wait_for_completion():
            raise Exception(
                "Failed to run {0} backup job with error: {1}".format(
                    backup_type, job.delay_reason
                )
            )

        self.log.info("Successfully finished {0} backup job".format(backup_type))

        return job

[docs]    def run(self):
        """Execution method for this test case"""
        try:
            self._log.info("Started executing {0} testcase".format(self.id))
            randomint = str(random.randint(100000, 9999999))
            full_library_name = "Share_DiskLibrary_{}".format(randomint)
            non_dedupsp = "non_dedupsp_{}".format(randomint)
            dedup_sp = "DedupSP_{}".format(randomint)
            mediaagent = storage.MediaAgent(self.commcell, self.client.client_name)
            backupset_name = "DiskShare_backupset_{}".format(randomint)
            subclient_name = "DiskShare_subclient_{}".format(randomint)

            if "MALibraryMountPath" in self.tcinputs:
                malibrarypath = self.tcinputs["MALibraryMountPath"]
            else:
                malibrarypath = os.path.join(self.client.install_directory, "Automation", "MAPath")
            if "subClientContent" in self.tcinputs:
                subclient_content = self.tcinputs["subClientContent"]
            else:
                subclient_content = os.path.join(
                    self.client.install_directory, "Automation", "share_maname")
            if "ShareMAName" in self.tcinputs:
                share_maname = self.tcinputs["ShareMAName"]
            else:
                share_maname = Client(self.commcell, self.commcell._commserv_name)
            if "SharedMALibraryMountPath" in self.tcinputs:
                sharelibrarypath = self.tcinputs["SharedMALibraryMountPath"]
            else:
                sharelibrarypath = os.path.join(
                    share_maname.install_directory, "Automation", "shareMAPath")
            if "DedupPath" in self.tcinputs:
                dedup_path = self.tcinputs["DedupPath"]
            else:
                dedup_path = os.path.join(
                    share_maname.install_directory,
                    "Automation",
                    "DedupPath" + randomint)

            if "RestoreLocation" in self.tcinputs:
                restore_location = self.tcinputs["RestoreLocation"]
            else:
                restore_location = os.path.join(
                    self.client.install_directory, "Automation", "share_maname")
            self._log.info("Creating Library and Storage Policy for full DR backup")
            self._log.info("Creating Library with name {}".format(full_library_name))
            disklibraries = storage.DiskLibraries(self.commcell)
            if not disklibraries.has_library(full_library_name):
                fulldisklibrary = disklibraries.add(full_library_name, mediaagent, malibrarypath)
            else:
                fulldisklibrary = disklibraries.get(full_library_name)
            # details=DiskLibraries.get(full_library_name)
            sharedoutput = fulldisklibrary.shareMountpath(
                share_maname.client_name, sharelibrarypath)
            if 'mountPath' in sharedoutput:
                self._log.info("share operation is successful")
            else:
                self._log.error('Failed with error: ' + str(sharedoutput))
                self.result_string = str(sharedoutput)
                self.status = constants.FAILED
                return
            storage_obj = storage.StoragePolicies(self.commcell)
            if not storage_obj.has_policy(non_dedupsp):
                non_dedupsp = storage_obj.add(
                    non_dedupsp,
                    fulldisklibrary,
                    mediaagent,
                    retention_period=0,
                    retention_cycles=1)
                self._log.info(
                    "Created Non dedup Storage Policy with name {}".format(non_dedupsp))
            else:
                self._log.info(
                    "SP already exists, so creating SP object {}".format(non_dedupsp))
                non_dedupsp = storage.StoragePolicy(self.commcell, non_dedupsp)

            if not storage_obj.has_policy(dedup_sp):
                storage_obj.add(
                    dedup_sp,
                    fulldisklibrary,
                    mediaagent, dedup_path,
                    retention_period=0,
                    retention_cycles=1)
                self._log.info(
                    "Created dedup Storage Policy with name {}".format(dedup_sp))
            else:
                self._log.info(
                    "SP already exists, so creating SP object {}".format(dedup_sp))
            dedup_sp = storage.StoragePolicy(self.commcell, dedup_sp)
            self._log.info("Creating BackupSet with name {}".format(backupset_name))
            backupset = Backupsets(self.agent)
            backupset_name = backupset.add(backupset_name)
            subclient = Subclients(backupset_name)
            subclient_name = subclient.add(subclient_name, non_dedupsp.storage_policy_name)
            subclient_name.content = [subclient_content]

            self._log.info("Create Machine class object")
            client_machine = Machine(self.client.client_name, self.commcell)

            self._log.info("Read subclient content")
            self._log.info("Subclient Content: {0}".format(subclient_name.content))

            self._log.info("Generating test data at: {0}".format(subclient_content))
            client_machine.generate_test_data(subclient_content)
            self.subclient = subclient_name

            for iteration in range(2):
                self._log.info("iteration {0}".format(str(iteration)))
                self._run_backup('FULL')
                self._log.info("Generating test data at: {0}".format(subclient_content))
                client_machine.generate_test_data(subclient_content)
                self._run_backup('INCREMENTAL')

            job = self.subclient.restore_out_of_place(
                mediaagent.media_agent_name,
                restore_location,
                subclient_name.content,
                fs_options={
                    "media_agent": share_maname.client_name})
            self._log.info("started restore job {0}".format(job))
            if not job.wait_for_completion():
                raise Exception(
                    "Failed to run {0} restore job with error: {1}".format(
                        job, job.delay_reason
                    )
                )
            dataging = self.commcell.run_data_aging(
                storage_policy_name=dedup_sp.storage_policy_name)

            if not job.wait_for_completion():
                raise Exception(
                    "Failed to run {0} restore job with error: {1}".format(
                        dataging, dataging.delay_reason
                    )
                )

            self._log.info("Testcase execution completed")
        except Exception as exp:
            self._log.error('Failed with error: ' + str(exp))
            self.result_string = str(exp)
            self.status = constants.FAILED
