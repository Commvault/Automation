'''
Test case to check the basic acceptance of SNMP in Admin console.

It verifies
1. Creation of SNMP configuration based on different criteria's passed as arguments
2. Validates if the SNMP configuration is created successfully.
3. Editing of SNMP configuration created in above steps.
4. Deletion of SNMP configuration created, edited & verified in above steps.
'''

from AutomationUtils.cvtestcase import CVTestCase
from Web.Common.cvbrowser import BrowserFactory
from Web.AdminConsole.Helper.LoginHelper import LoginMain
from Web.AdminConsole.Helper.SNMPConfigurationHelper import SNMPConfigurationMain
from Reports.utils import TestCaseUtils


[docs]class TestCase(CVTestCase):

    ''' Basic Acceptance test for SNMP Configuration '''

    def __init__(self):
        '''
       Initializing the Test case file
        '''
        super(TestCase, self).__init__()
        self.name = "Basic Acceptance Test for SNMP configuration in AdminConsole"
        self.product = self.products_list.COMMSERVER
        self.feature = self.features_list.ADMINCONSOLE
        self.show_to_user = False
        self.utils = TestCaseUtils(self)
        self.tcinputs = {
            "config_name": None,
            "new_config_name": None,
            "encryption_value_combo_list": None,
            "privacy_value_combo_list": None
        }


[docs]    def run(self):

        try:
            self.log.info("Started executing {0} testcase".format(self.id))
            self.log.info("*" * 10 + " Initialize browser objects " + "*" * 10)

            factory = BrowserFactory()
            browser = factory.create_browser_object()
            browser.open()
            driver = browser.driver

            self.log.info("Creating the login object")
            login_obj = LoginMain(driver, self.csdb)

            login_obj.login(self.inputJSONnode['commcell']['commcellUsername'],
                            self.inputJSONnode['commcell']['commcellPassword'])
            self.log.info("login successful")

            snmp_configuration_obj = SNMPConfigurationMain(driver)

            encryption_algo_list = self.tcinputs['encryption_value_combo_list']
            privacy_algo_list = self.tcinputs['privacy_value_combo_list']

            snmp_configuration_obj.config_name = self.tcinputs['config_name']
            snmp_configuration_obj.encryption_algorithm = encryption_algo_list[0]
            snmp_configuration_obj.privacy_algorithm = privacy_algo_list[0]

            snmp_configuration_obj.create_configuration()
            self.log.info("SNMP configuration created successfully")

            snmp_configuration_obj.validate_snmp_configuration()
            self.log.info("Initial validation completed successfuly")

            snmp_configuration_obj.config_name = self.tcinputs['new_config_name']
            snmp_configuration_obj.username = "Tester B"
            snmp_configuration_obj.password = "Builder!123"
            snmp_configuration_obj.priv_password = "Builder!1234"
            snmp_configuration_obj.encryption_algorithm = [encryption_algo_list[1]]
            snmp_configuration_obj.privacy_algorithm = [privacy_algo_list[1]]

            snmp_configuration_obj.modify_configuration()
            self.log.info("SNMP configuration edited successfully")

            snmp_configuration_obj.validate_snmp_configuration()
            self.log.info("Validation after editing completed successfully")

            snmp_configuration_obj.encryption_algorithm = encryption_algo_list
            snmp_configuration_obj.privacy_algorithm = privacy_algo_list

            snmp_configuration_obj.modify_configuration()
            self.log.info("Multiple editing verified successfully")

        except Exception as exp:
            self.utils.handle_testcase_exception(exp)

        finally:

            snmp_configuration_obj.del_configuration()
            self.log.info("SNMP configuration deleted successfully.\
                    Test case completed. Exiting module.....Thanks")
            browser.close()
