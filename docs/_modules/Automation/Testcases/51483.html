""""Test Case to check basic acceptance of IntelliSnap operations

steps:
Creates a plan, add-edit arrays, associate server and creates subclient
run snapbackup and restore operations
disassociate array, delete array, delete subclient and delete plan

Pre-requisite:
Storage pool should be configured

TestCase: Class for executing this test case

TestCase:
    __init__()      --  initialize TestCase class

    run()           --  run function of this test case
"""

from AutomationUtils.cvtestcase import CVTestCase
from AutomationUtils import logger
from AutomationUtils import constants
from Web.AdminConsole.Helper.fs_helper import FSHelper
from Web.AdminConsole.Helper.array_helper import ArrayHelper
from Web.AdminConsole.Helper.PlanHelper import PlanMain
from Web.Common.cvbrowser import BrowserFactory
from Web.AdminConsole.Helper.LoginHelper import LoginMain

    """Class for executing Basic acceptance Test of IntelliSnap FS backup and Restore test case"""

    def __init__(self):
        """Initializes test case class object"""
        super(TestCase, self).__init__()
        self.name = "Basic Acceptance Test of Admin Console"
        self.product = self.products_list.CTE
        self.feature = self.features_list.SNAPBACKUP



[docs]    def run(self):
        """Main function for test case execution"""
        log = logger.get_log()

        try:
            log.info("Started executing {0} testcase".format(self.id))

            log.info("*" * 10 + " Initialize browser objects " + "*" * 10)
            factory = BrowserFactory()
            browser = factory.create_browser_object()
            browser.open()
            driver = browser.driver

            log.info("Creating the login object")
            login_obj = LoginMain(driver, self.csdb)

            ##Login
            login_obj.login(self.inputJSONnode['commcell']['commcellUsername'],
                            self.inputJSONnode['commcell']['commcellPassword']
                            )
            log.info("Login completed successfully. Will continue with snap plan creation")

            ##Create FileSystem Object and Array Object, Assign variables
            fs_obj = FSHelper(driver, self.csdb)
            array_obj = ArrayHelper(driver, self.csdb)
            plan_obj = PlanMain(driver)
            plan_obj.plan_name = self.tcinputs['plan_name']
            plan_obj.storage_poolname = self.tcinputs['storagepool_name']
            fs_obj.client_name = self.tcinputs['client_name']
            fs_obj.subclient_name = self.tcinputs['subclient_name']
            fs_obj.subclient_content = self.tcinputs['subclient_content']
            fs_obj.plan_name = self.tcinputs['plan_name']
            array_obj.array_vendor = self.tcinputs['array_vendor']
            array_obj.array_name = self.tcinputs['array_name']
            array_obj.array_new_user = self.tcinputs['array_user']
            array_obj.array_new_password = self.tcinputs['array_password']
            array_obj.client_name = self.tcinputs['client_name']

            #Create a Snap Plan
            plan_obj.add_plan()

            #Add Array
            array_obj.add_array()
            #Edit username-password
            array_obj.edit_array()

            #Associate Server
            array_obj.associate_array()

            fs_obj.create_fs_subclient()
            #Run Backup And Restore
            fs_obj.fsbackup_now()
            fs_obj.fs_restore()

            array_obj.list_snap()

            #Cleanup and Delete Test
            array_obj.delete_all_snap()
            fs_obj.delete_subclient()
            array_obj.disassociate_array()
            array_obj.delete_array()
            plan_obj.delete_plans()

            browser.close()

        except Exception as exp:
            browser.close()
            log.error('Failed with error: ' + str(exp))
            self.result_string = str(exp)
            self.status = constants.FAILED
