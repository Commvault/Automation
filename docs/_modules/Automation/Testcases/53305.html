[docs]    def run(self):
        """Execution method for this test case"""
        try:
            self.log.info("Started executing {0} testcase".format(self.id))
            logfile_to_check = ["ApplicationMgr.log"]
            loglines_with_required_calls = []
            client_machine = Machine(self.client)
            validatelog = os.path.join(self.client.log_directory, logfile_to_check[0])
            idautil = idautils.CommonUtils(self)
            entities = CVEntities(self)
            server = serverhelper.ServerTestCases(self)
            response_map = entities.create(
                ["disklibrary", "storagepolicy", "backupset", "subclient"])
            subclient_content = response_map["subclient"]["content"][0]
            subclient = response_map['subclient']['object']
            all_jobs_statistics = {}

            def laptop_operations(backup_type="Full"):
                '''
                performs laptop backup and generates statistics
                @args
                backup_type  (str) -- backup type
                '''
                self.log.info("Generating test data at: {0}".format(subclient_content))
                client_machine.generate_test_data(subclient_content)
                server.rename_remove_logfile(client_machine, validatelog)
                job = idautil.subclient_backup(subclient, backup_type)
                all_jobs_statistics[job.job_id] = server.validatelogs(
                    client_machine, loglines_with_required_calls,
                    validatelog, serverconstants.statics_constant())
                self.log.info("Job {} Statastics {}".format(
                    job.job_id, all_jobs_statistics[job.job_id]))
                server.rename_remove_logfile(client_machine, validatelog)

            for iteration in range(2):
                self.log.info("started iteration: {0}".format(str(iteration)))
                for iter in range(2):
                    if iter == 0:
                        backup_type = "Full"
                    else:
                        backup_type = "Incremental"
                    laptop_operations(backup_type)

            self.log.info("All jobs statistics are {}".format(all_jobs_statistics))
            headers = ["Request Type:"]
            for key in all_jobs_statistics.keys():
                headers.append("JOB ID: " + key)
            string_to_check = {}
            for key, value in all_jobs_statistics.items():
                for item in value.keys():
                    string_to_check[item] = []
                break
            for key, value in all_jobs_statistics.items():
                for key in string_to_check:
                    string_to_check[key].append(value[key])
            self.log.info("All jobs statistics formatted data {}".format(string_to_check))
            data = server.generate_email_body(headers, string_to_check)
            mail = Mailer(commcell_object=self.commcell)
            mail.mail("Laptop backup call statistics information", data)
            if client_machine.check_file_exists(validatelog.replace(".log", "_1.log")):
                client_machine.remove_directory(validatelog.replace(".log", "_1.log"))
            install_path = self.client.install_directory.split(os.path.sep)[0]
            if not install_path.endswith(os.path.sep):
                install_path += os.path.sep
            fileobj = open(os.path.join(install_path, logfile_to_check[0]), "w")
            for line in loglines_with_required_calls:
                fileobj.write(line)
                fileobj.write("\n")
            fileobj.close()
            self.log.info("Testcase execution completed")
        except Exception as exp:
            self.log.error('Failed with error: ' + str(exp))
            self.result_string = str(exp)
            self.status = constants.FAILED
        finally:
            entities.cleanup()
