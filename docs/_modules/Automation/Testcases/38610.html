[docs]    def run(self):
        """Run function for test case execution"""
        log = logger.get_log()

        try:
            log.info("Started executing %s testcase", self.id)

            # Getting the client hostname
            client_hostname = self.client.client_hostname

            # Getting the SAUsername for MySQL
            mysql_user = self.instance.mysql_username

            # Getting the MySQL Server port number
            mysql_port = int(self.port)

            # Creating MYSQLHelper Object
            log.info("Create MYSQLHelper class object")
            self.mysql_helper = MYSQLHelper(
                self.commcell, self.subclient, self.instance, client_hostname, mysql_user,
                mysql_port)

            # Initializing counter variables
            count_full = 0
            count_inc1 = 0
            count_inc2 = 0

            full_tables_dict = {}
            inc_tables_dict1 = {}
            inc_tables_dict2 = {}

            # Checking the basic settings required for Automation
            log.info(
                "Check Basic Setting of mysql server before stating the test cases")
            self.mysql_helper.basic_setup_on_mysql_server()

            # Checking whether Binary Logging is enabled or not in MySQL Server
            # (Required to run Incremental Backup)
            self.mysql_helper.log_bin_on_mysql_server()

            log.info("Read subclient content")
            log.info("Subclient Content: %s", self.subclient.content)

            if self.subclient.content == [] or self.subclient.content == ['/']:
                raise Exception(
                    "Subclient Content is empty please add subclient content from Commcell Console"
                )

            # Timestamp For Tablenames (Full Backup)
            timestamp_full = str(int(time.time()))

            # Populating Databases For Full Backup
            for each_db in self.subclient.content:
                table_list = self.mysql_helper.create_table_with_text_db(
                    each_db,
                    table_name="full_38610_" +
                    timestamp_full,
                    no_of_tables=10,
                    column_in_each_table=5,
                    drop_table_before_create=False)
                full_tables_dict[each_db] = table_list

            # Counter variable value incremented after tables created for Full Backup
            count_full += 1

            # Getting Database Size and table sizes (Full)
            db_size_full_bk, table_size_full_bk = self.mysql_helper.get_database_and_table_sizes(
                self.subclient.content)

            # Running Full Backup
            self.mysql_helper.run_backup('FULL')

            log.info("sleeping for 2 Min")
            time.sleep(120)
            log.info("Starting the test case")

            # Timestamp For Table names (Incremental Backup 1)
            timestamp_inc1 = str(int(time.time()))

            # Populating Databases For Incremental Backup 1
            for each_db in self.subclient.content:
                table_list = self.mysql_helper.create_table_with_text_db(
                    each_db,
                    table_name="inc1_38610_" +
                    timestamp_inc1,
                    no_of_tables=10,
                    column_in_each_table=5,
                    drop_table_before_create=False)
                inc_tables_dict1[each_db] = table_list

            # Counter variable value incremented after tables created for Incremental Backup 1
            count_inc1 += 1

            # Getting Database Size and table sizes (Incremental 1)
            db_size_inc1_bk, table_size_inc1_bk = self.mysql_helper.get_database_and_table_sizes(
                self.subclient.content)

            # Running Incremental Backup
            self.mysql_helper.run_backup('INCREMENTAL')

            log.info("sleeping for 2 Min")
            time.sleep(120)
            log.info("Starting the test case")

            # Timestamp For Table names (Incremental Backup 2)
            timestamp_inc2 = str(int(time.time()))

            # Populating Databases For Incremental Backup 2
            for each_db in self.subclient.content:
                table_list = self.mysql_helper.create_table_with_text_db(
                    each_db,
                    table_name="inc2_38610_" +
                    timestamp_inc2,
                    no_of_tables=10,
                    column_in_each_table=5,
                    drop_table_before_create=False)
                inc_tables_dict2[each_db] = table_list

            # Counter variable value incremented after tables created for Incremental Backup 2
            count_inc2 += 1

            # Getting Database Size and table sizes (Incremental 2)
            db_size_inc2_bk, table_size_inc2_bk = self.mysql_helper.get_database_and_table_sizes(
                self.subclient.content)

            # Running Incremental Backup
            self.mysql_helper.run_backup('INCREMENTAL')

            log.info("sleeping for 2 Min")
            time.sleep(120)
            log.info("Starting the test case")

            # Dropping the Automation created tables for Full Backup
            if count_full != 0:
                if full_tables_dict != {}:
                    self.mysql_helper.clean_up_tables(full_tables_dict)
            count_full -= 1

            # Dropping the Automation created tables for Incremental Backup 1
            if count_inc1 != 0:
                if inc_tables_dict1 != {}:
                    self.mysql_helper.clean_up_tables(inc_tables_dict1)
            count_inc1 -= 1

            # Dropping the Automation created tables for Incremental Backup 2
            if count_inc2 != 0:
                if inc_tables_dict2 != {}:
                    self.mysql_helper.clean_up_tables(inc_tables_dict2)
            count_inc2 -= 1

            # Removing mysql and sys system db's from restore list
            paths = self.subclient.content.copy()

            if '\\mysql' in paths:
                paths.remove('\\mysql')

            if '\\sys' in paths:
                paths.remove('\\sys')

            # Running In Place Data Only Restore
            log.info("Running In Place Restore - Data Only")
            job_r1 = self.instance.restore_in_place(
                paths,
                self.staging,
                self.client.client_name,
                self.instance.instance_name,
                data_restore=True,
                log_restore=False)
            log.info("Started restore with Job ID: %s", str(job_r1.job_id))
            if not job_r1.wait_for_completion():
                raise Exception(
                    "Failed to run data only restore job with error: {1}".format(
                        job_r1.delay_reason
                    )
                )

            log.info("Successfully finished data only restore job")
            count_full += 1

            log.info("sleeping for 5 Min")
            time.sleep(300)
            log.info("Starting the test case")

            # Getting Database Size and Table sizes after Data Only Restore
            db_size_data_res, tb_size_data_res = self.mysql_helper.get_database_and_table_sizes(
                self.subclient.content)

            # Comparing Database and Table Sizes (Size after Full Backup and Data Only Restore)
            self.mysql_helper.compare_db_and_table_sizes(
                db_size_full_bk,
                table_size_full_bk,
                db_size_data_res,
                tb_size_data_res)

            # Running In Place Log Only Restore
            log.info("Running In Place Restore - Log Only")
            job_r2 = self.instance.restore_in_place(
                paths,
                self.staging,
                self.client.client_name,
                self.instance.instance_name,
                data_restore=False,
                log_restore=True)
            log.info("Started restore with Job ID: %s", str(job_r2.job_id))
            if not job_r2.wait_for_completion():
                raise Exception(
                    "Failed to run data only restore job with error: {1}".format(
                        job_r2.delay_reason
                    )
                )

            log.info("Successfully finished log only restore job")
            count_inc1 += 1
            count_inc2 += 1

            log.info("sleeping for 5 Min")
            time.sleep(300)
            log.info("Starting the test case")

            # Getting Database Size and Table Sizes after Log Only Restore
            db_size_log_res, tb_size_log_res = self.mysql_helper.get_database_and_table_sizes(
                self.subclient.content)

            # Comparting Database and Table Sizes (Size after incremental backup and
            # Log Only Restore)
            self.mysql_helper.compare_db_and_table_sizes(
                db_size_inc2_bk,
                table_size_inc2_bk,
                db_size_log_res,
                tb_size_log_res)

            # Dropping the Automation reated tables for Full Backup
            if count_full != 0:
                if full_tables_dict != {}:
                    self.mysql_helper.clean_up_tables(full_tables_dict)
            count_full -= 1

            # Dropping the Automation reated tables for Incremental Backup 1
            if count_inc1 != 0:
                if inc_tables_dict1 != {}:
                    self.mysql_helper.clean_up_tables(inc_tables_dict1)
            count_inc1 -= 1

            # Dropping the Automation reated tables for Incremental Backup 2
            if count_inc2 != 0:
                if inc_tables_dict2 != {}:
                    self.mysql_helper.clean_up_tables(inc_tables_dict2)
            count_inc2 -= 1

            # Running In Place Data + Log Restore
            log.info("Running In Place Restore - Data + Log")
            job_r3 = self.instance.restore_in_place(
                paths,
                self.staging,
                self.client.client_name,
                self.instance.instance_name,
                data_restore=True,
                log_restore=True)
            log.info("Started restore with Job ID: %s", str(job_r3.job_id))
            if not job_r3.wait_for_completion():
                raise Exception(
                    "Failed to run data only restore job with error: {1}".format(
                        job_r3.delay_reason
                    )
                )

            log.info("Successfully finished data only restore job")
            count_full += 1
            count_inc1 += 1
            count_inc2 += 1

            log.info("sleeping for 5 Min")
            time.sleep(300)
            log.info("Starting the test case")

            # Getting Database Size and Table sizes after Data + Log Restore
            db_sz_datalog_res, tb_sz_datalog_res = self.mysql_helper.get_database_and_table_sizes(
                self.subclient.content)

            # Comparing Database and Table Sizes (Size after Full Backup and Data Only Restore)
            self.mysql_helper.compare_db_and_table_sizes(
                db_size_inc2_bk,
                table_size_inc2_bk,
                db_sz_datalog_res,
                tb_sz_datalog_res)

        except Exception as excp:
            log.error('Failed with error: ' + str(excp))
            self.result_string = str(excp)
            self.status = constants.FAILED

        finally:
            # Dropping the Automation reated tables for Full Backup
            if count_full != 0:
                if full_tables_dict != {}:
                    self.mysql_helper.clean_up_tables(full_tables_dict)

            # Dropping the Automation reated tables for Incremental Backup 1
            if count_inc1 != 0:
                if inc_tables_dict1 != {}:
                    self.mysql_helper.clean_up_tables(inc_tables_dict1)

            # Dropping the Automation reated tables for Incremental Backup 2
            if count_inc2 != 0:
                if inc_tables_dict2 != {}:
                    self.mysql_helper.clean_up_tables(inc_tables_dict2)
