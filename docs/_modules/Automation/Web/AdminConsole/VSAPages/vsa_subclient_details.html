    @adminconsole_handler()
    def subclient_storage_pool_info(self):
        """
        Displays the storage pool info of the subclient

        Returns:
            pool_info   (list):  lists the storage pool info

        """
        self.select_configuration_tab()
        elements = self.driver.find_elements_by_xpath("//cv-tile-component[@data-title='"
                                                      "Storage pool']/div/div[3]/div/div[2]/ul/li")
        pool_info = {}
        for element in elements:
            pool_info[element.find_element_by_xpath(
                "./span[1]").text] = element.find_element_by_xpath("./span[2]").text
        self.log.info("The storage pool associated with plan of the subclient is %s",
                      str(pool_info))
        return pool_info

    @adminconsole_handler()
    def schedule_info(self):
        """
        Displays all the schedules associated with the subclient

        Returns:
            schedules  (list):  list of all schedules associated to the subclient

        """
        self.select_configuration_tab()
        elements = self.driver.find_elements_by_xpath("//cv-tile-component[@data-title="
                                                      "'Schedules']/div/div[3]/div/div[2]/ul")
        schedules = []
        for element in elements:
            schedules.append(element.find_element_by_xpath("./li").text)
        self.log.info("The schedules associated with the subclient are %s", str(schedules))
        return schedules

    @adminconsole_handler()
    def restore(self, recovery_time=None):
        """
        Opens the select restore page

        Args:
            recovery_time   (basestring):   the backup date in 01-September-1960 format

        Raises:
            Exception:
                if there is no option to restore

        """
        self.select_overview_tab()
        self.recovery_point_restore(recovery_time)

    @adminconsole_handler()
    def backup_now(self, bkp_type):
        """
        Starts a backup job for the collection

        Args:

            bkp_type (str): the backup level

        Returns:
            Job ID of backup job

        """
        self.select_overview_tab()
        self.driver.execute_script("window.scrollTo(0,0)")
        self.select_hyperlink("Backup")
        self.submit_backup(bkp_type)
        return self.backup_job_id()

    @adminconsole_handler()
    def manage_content(self):
        """
        Manages the collection content by adding or removing VMs and rules

        Raises:
            Exception:
                if there is no link to Manage the subclient content

        """
        self.select_overview_tab()
        self.select_hyperlink("Manage")

    @adminconsole_handler()
    def subclient_settings(self, reader=2, collect_file_details=False, bkp_type=None, cbt=True,
                           auto_detect_owner=False, transport_mode=None,
                           free_space=None, ds_free_space_check=True):
        """
        Edits the no of readers assigned to the collection and the backup type
        of the virtual machine

        Args:
            reader                   (int):         the number of readers

            collect_file_details     (bool):        to enable or disable metadata collection

            bkp_type                 (basestring):  app / crash consistent backup

            cbt                      (bool):        enable / disable CBT

            auto_detect_owner        (bool):        to auto detect the owner of the VM

            transport_mode           (basestring):  the transport mode to be used for backup

            free_space               (int):         the amount of free space to be set

            ds_free_space_check      (bool):        to enable / disable datastore freespace check

        Raises:
            Exception:
                when the subclient setting could not be edited or
                when there is an error while updating the settings

        """
        self.select_configuration_tab()
        if cbt:
            self.toggle_enable("Use changed block tracking")
        else:
            self.toggle_disable("Use changed block tracking")

        if auto_detect_owner:
            self.toggle_enable("Auto detect VM Owner")
        else:
            self.toggle_disable("Auto detect VM Owner")

        if collect_file_details:
            self.toggle_enable("Collect file details")
        else:
            self.toggle_disable("Collect file details")

        if self.check_if_entity_exists("xpath", "//cv-tile-component[@data-title='Settings']/"
                                                "div/div[3]/div/div[2]/div/a["
                                                "contains(text(),'Edit')]"):
            self.tile_select_hyperlink("Settings", "Edit")
            self.fill_form_by_id("noOfReaders", reader)

            if bkp_type:
                self.driver.find_element_by_xpath(
                    "//div[@class='frequency']//span[contains("
                    "text(),'" + bkp_type + "')]").click()

            if transport_mode:
                self.select_value_from_dropdown("transportMode", transport_mode)

            if self.check_if_entity_exists("xpath", "//input[@id='datastoreFreespaceCheckCB']"):
                if not ds_free_space_check:
                    self.checkbox_deselect("datastoreFreespaceCheckCB")
                else:
                    self.checkbox_select("datastoreFreespaceCheckCB")
                    if free_space:
                        self.fill_form_by_id("datastoreFreespaceRequired", free_space)

            self.submit_form()
            self.check_error_message()
        else:
            raise Exception("There is no option to edit the subclient properties")

    @adminconsole_handler()
    def subclient_summary(self):
        """
        Provides the summary info of the subclient like last backup time, next backup time

        Returns:
            summary_info    (dict): info about the subclient

        """
        self.select_overview_tab()
        elements = self.driver.find_elements_by_xpath("//cv-tile-component[@data-title='Summary']"
                                                      "/div/div[3]/div/div[2]/ul/li")
        summary_info = {}
        for element in elements:
            key = element.find_element_by_xpath("./span[1]").text
            if key == "Data management":
                class_value = element.find_element_by_xpath("./span[2]/toggle-control/div"
                                                           ).get_attribute("class")
                if "enabled" in class_value:
                    value = "Enabled"
                else:
                    value = "Disabled"
            else:
                try:
                    value = element.find_element_by_xpath("./span[2]").text
                except Exception:
                    value = None
            summary_info[key] = value
        self.log.info("Summary of the subclient is %s", str(summary_info))
        return summary_info

    @adminconsole_handler()
    def snap_mount_esx_host(self, esx_host):
        """
        Sets the given ESX host as the snap mount ESX

        Args:
            esx_host (basestring):      the esx host to set as snap mount esx

        """
        self.select_configuration_tab()
        self.tile_select_hyperlink("Snap mount esx", "Edit")
        self.select_destination_host(esx_host)
        self.submit_form()
