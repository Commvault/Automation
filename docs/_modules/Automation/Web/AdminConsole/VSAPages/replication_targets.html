    @adminconsole_handler()
    def add_replication_target(
            self,
            vendor_type,
            rep_target_name,
            destination_hypervisor,
            proxy,
            vm_display_name,
            destination_host,
            resource_pool,
            datastore,
            destination_network,
            prefix=False,
            test_failover_options=False,
            expiration_hours=True,
            expiration_time_hours=1,
            expiration_days=False,
            expiration_time_days=2,
            configure_isolated_network=False,
            gateway_template=None,
            migrate_vms=False):
        """
        Adds a new replication target with the specified inputs and proxy.

        Args:
            vendor_type             (basestring)   --  name of the vendor (vmware, amazon etc.)

            rep_target_name         (basestring)   --  name the user gives to the
                                                        replication target

            destination_hypervisor  (basestring)   --  client with the hypervisor.

            proxy                   (basestring)   --  proxy given by the user.

            vm_display_name         (basestring)   --  allows the user to give the prefix or
                                                        suffix in the name

            destination_host        (basestring)   --  name of esx host

            datastore               (basestring)   --  specify a datastore from the dropdown

            resource_pool           (basestring)   --  specify the resource pool from the
                                                        dropdown

            destination_network     (basestring)   --  specify the destination network from
                                                    the dropdown

            prefix                  (bool)  --  if prefix has to be added in vm display name

            test_failover_options   (bool)  --  to test failover options after replication

            expiration_hours        (bool)  --  if the replicated VM has to be up for hours

            expiration_time_hours   (int)   --  the no of hours the replicated VMs need to be up

            expiration_days         (bool)  --  if the replicated VM has to be up for days

            expiration_time_days    (int)   --  the no of days the replicated VMs need to be up

            configure_isolated_network  (bool)  --  if the replicated VMs need to be placed in
                                                    a isolated network

            gateway_template        (basestring)   --  the name of the gateway template to use for
                                                        isolated network

            migrate_vms             (bool)  --  if the replicated VMs need to be migrated

        Raises:
            Exception:
                if there is no option to add a replication target because the logged in user did
                    not have sufficient permissions or
                if there an error while creating replication targets

        """
        self.driver.execute_script("window.scrollTo(0,0)")

        self.select_hyperlink("Add replication target")

        self.select_value_from_dropdown("vendorTypes", vendor_type)
        self.fill_form_by_id('replicationTargetName', rep_target_name)

        if prefix:
            self.select_radio("azureDisplayNamePrefix")

        # self.fill_form_by_id("displayNamePrefixSuffix", vm_display_name)
        self.driver.find_element_by_xpath("//div/input[@name='displayNamePrefixSuffix']"
                                         ).send_keys(vm_display_name)

        if vendor_type == 'VMware':
            self.driver.find_element_by_xpath("//cv-select-hypervisors[@vendor='VMW']").click()
            self.driver.find_element_by_xpath("//cv-select-hypervisors//div[@class="
                                              "'line-search']/input"
                                             ).send_keys(destination_hypervisor)
            elements = self.driver.find_elements_by_xpath("//cv-select-hypervisors//div["
                                                          "@class='checkBoxContainer']/div")
            for element in elements:
                if element.find_element_by_xpath("./div/label/span").text == " "\
                        + destination_hypervisor:
                    element.find_element_by_xpath("./div/label/span").click()
            self.driver.find_element_by_xpath("//cv-select-proxy[@vendor='VMW']").click()
            self.driver.find_element_by_xpath("//cv-select-proxy//div[@class='line-search']"
                                              "/input").send_keys(proxy)
            elements = self.driver.find_elements_by_xpath("//cv-select-proxy//div[@class="
                                                          "'checkBoxContainer']/div")
            for element in elements:
                if element.find_element_by_xpath("./div/label/span").text == " " + proxy:
                    element.find_element_by_xpath("./div/label/span").click()
            self.wait_for_completion()

            # Browsing and selecting the destination host
            self.driver.find_element_by_xpath('//span[@class="input-group-btn"]/'
                                              'button').click()
            self.wait_for_completion()

            self.select_destination_host(destination_host)
            self.submit_form()

            self.select_value_from_dropdown("dataStore", datastore, True, False)
            self.select_value_from_dropdown("resourcePool", resource_pool, False, False)
            self.select_value_from_dropdown("networkSettingsDestination", destination_network)

            if test_failover_options:
                self.checkbox_select("showVirtualLab")
                if expiration_hours:
                    self.fill_form_by_id("expirationTimeInHours", expiration_time_hours)
                if expiration_days:
                    self.select_radio("Day(s)")
                    self.fill_form_by_id("expirationTimeInDays", expiration_time_days)
                if configure_isolated_network:
                    self.checkbox_select("createIsolatedNetwork")
                    self.select_destination_host(gateway_template)
                    self.submit_form()
                if migrate_vms:
                    self.checkbox_select("migrateVMs")
            self.submit_form()

    @adminconsole_handler()
    def select_replication_target(self, rep_target_name):
        """
        Opens the replication target with the given name

        Args:
            rep_target_name     (basestring)   --  name of replication target

        Raises:
            Exception:
                if there is no replication target with the given name

        """
        self.search_for(rep_target_name)
        self.select_hyperlink(rep_target_name)

    @adminconsole_handler()
    def action_delete(self, rep_target_name):
        """
        delete a replication target with the specified name

        Args:
            rep_target_name     (basestring)   --  name of replication target

        Raises:
            Exception:
                if there an error while deleting the replication target or
                if there is no replication target with the given name

        """
        self.search_for(rep_target_name)

        if self.check_if_entity_exists("link", rep_target_name):
            self.driver.find_element_by_xpath("//a[text()='" + rep_target_name
                                              + "']/../../div[5]/div/a/span").click()
            self.wait_for_completion()
            if self.check_if_entity_exists("link", "Delete"):
                self.driver.find_element_by_xpath("//a[text()= '"
                                                  + rep_target_name
                                                  + "']/ancestor::div[@class='ng-isolate-scope']"
                                                    "/div[5]//a[contains"
                                                    "(text(),'Delete')]").click()
                self.wait_for_completion()
                self.driver.find_element_by_xpath("//div[3]/button[2]").click()
                self.wait_for_completion()

            if self.check_if_entity_exists("xpath",
                                           "//div/div/div[@class='growl-item alert ng-scope "
                                           "alert-error alert-danger']"):
                raise Exception(self.driver.find_element_by_xpath(
                    "//div/div/div[@class='growl-item alert ng-scope alert-error alert-danger']"
                    "/div/div").text)
        else:
            raise Exception("There is no replication target named {0}".format(rep_target_name))
