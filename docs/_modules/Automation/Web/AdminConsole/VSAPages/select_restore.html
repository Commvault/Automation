from Web.Common.handlers import adminconsole_handler
from Web.AdminConsole.AdminConsolePages.AdminPage import AdminPage



[docs]class SelectRestore(AdminPage):
    """
    This class contains all the methods to select any type of restore to be done.
    """
    @adminconsole_handler()
    def select_guest_files(self):
        """
        Opens the guest files restore page
        """
        self.log.debug("Guest Files Restore")
        self.driver.find_element_by_xpath(
            "//div[contains(text(), 'Guest files')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def select_vm_files(self):
        """
        Opens the virtual machine files restore page
        """
        self.log.info("Virtual Machine Files Restore")
        self.driver.find_element_by_xpath(
            "//div[contains(text(),'Virtual machine files')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def select_full_vm_restore(self):
        """
        Opens the full VM restore page
        """
        self.log.info("Full Virtual Machine Restore")
        if self.check_if_entity_exists("xpath", "//div[contains(text(),"
                                                "'Full virtual machine')]"):
            self.driver.find_element_by_xpath(
                "//div[contains(text(),'Full virtual machine')]").click()
        elif self.check_if_entity_exists("xpath", "//div[contains(text(),'Full instance')]"):
            self.driver.find_element_by_xpath(
                "//div[contains(text(),'Full instance')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def select_disk_restore(self):
        """
        Opens the disk level restore page
        """
        self.log.info("Virtual Machine Disks Restore")
        if self.check_if_entity_exists("xpath", "//div[contains(text(),'Disk level')]"):
            self.driver.find_element_by_xpath(
                "//div[contains(text(),'Disk level')]").click()
        elif self.check_if_entity_exists("xpath", "//div[contains(text(),'Attach volume')]"):
            self.driver.find_element_by_xpath(
                "//div[contains(text(),'Attach volume')]").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def select_source_copy(self, copy_name):
        """
        Selects the copy from which browse and restore has to happen

        Args:
            copy_name   (basestring):   the name of the copy to browse and restore from

        Raises:
            Exception:
                if the copy could not be selected

        """
        self.select_hyperlink("Select source")
        elements = self.driver.find_elements_by_xpath("//a[.='Select source']/../ul/li")
        for element in elements:
            copy_elem = element.find_element_by_xpath("./a")
            if copy_elem.text == copy_name:
                copy_elem.click()
                self.wait_for_completion()
                break
        if not self.check_if_entity_exists("xpath", "//a[.='" + copy_name + "']"):
            raise Exception("The copy could not be selected.")
