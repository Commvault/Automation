    @adminconsole_handler()
    def open_global_filter(self, global_filter_path):
        """Method to open global exceptions"""

        self.select_hyperlink(global_filter_path)

    @adminconsole_handler()
    def add_global_filter(self, global_filter_path):
        """
        Method to create a global filter
        Args:
            global_filter_path(dict): dictionary with the global filter path to be added
              Eg.- global_filter_path = {'windows_global_filter_path' : "c\\test", 'unix_global_filter_path' :
                                                         "/root/test.txt" }
        Returns:
            None
        Raises:
            Exception:
                if failed to create a global filter
         """

        if global_filter_path['windows_global_filter_path']:
            if self.check_if_entity_exists("xpath", "//span[contains(text(),'Add windows exception')]"):
                self.driver.find_element_by_xpath("//span[contains(text(),'Add windows exception')]").click()
                self.wait_for_completion()
            elif self.check_if_entity_exists("xpath", "//h1[contains(text(),'Windows global exceptions')]"
                                                      "/../../span/span/span/a"):
                self.driver.find_element_by_xpath(
                    "//h1[contains(text(),'Windows global exceptions')]/../../span/span/span/a").click()
                self.wait_for_completion()
            else:
                exp = "Add windows global filter operation is failed"
                self.log.exception(exp)
                raise Exception(exp)

            self.log.info("Entering windows global filter path")
            self.fill_form_by_id("filterAddData", global_filter_path['windows_global_filter_path'])
            self.log.info("submitting the form")
            self.submit_form()
            self.log.info("windows global filter was created successfully.")
            self.check_error_message()

        if global_filter_path['unix_global_filter_path']:
            if self.check_if_entity_exists("xpath", "//p[contains(text(),'Add unix Exception')]"):
                self.driver.find_element_by_xpath("//p[contains(text(),'Add unix Exception')]").click()
                self.wait_for_completion()

            elif self.check_if_entity_exists("xpath", "//h1[contains(text(),'Unix global exceptions')]"
                                                      "/../../span/span/span/a"):
                self.driver.find_element_by_xpath("//h1[contains(text(),'Unix global exceptions')]"
                                                  "/../../span/span/span/a").click()
                self.wait_for_completion()
            else:
                exp = "Add unix global filter operation is failed"
                self.log.exception(exp)
                raise Exception(exp)

            self.log.info("Entering unix global filter path")
            self.fill_form_by_id("filterAddData", global_filter_path['unix_global_filter_path'])
            self.log.info("submitting the form")
            self.submit_form()
            self.log.info("Unix global filter was created successfully")

            self.check_error_message()

    @adminconsole_handler()
    def edit_global_filter(self, global_filter_path, new_global_filter_path=None):
        """
        Method to modify an existing  global filter
        Args:
            global_filter_path(dict): dictionary for path of the existing  global filter
            new_global_filter_path(dict):dictionary for  new path of the  global filter
             Eg:- new_global_filter_path = {'new_windows_global_filter_path': "*.bat" ,
                'new_unix_global_filter_path': "**temp*"}
        Returns:
            None
        Raises :
            Exception:
                 if failed to modify the  global filter
        """

        if new_global_filter_path['windows_global_filter_path']:
            self.log.info("Editing windows global filter")
            self.open_global_filter(global_filter_path['windows_global_filter_path'])
            self.fill_form_by_id("filterEditData", new_global_filter_path['windows_global_filter_path'])
            self.submit_form()
            self.log.info("Editing completed for Windows global filter path")

        if new_global_filter_path['unix_global_filter_path']:
            self.log.info("Editing unix global filter")
            self.open_global_filter(global_filter_path['unix_global_filter_path'])
            self.fill_form_by_id("filterEditData", new_global_filter_path['unix_global_filter_path'])
            self.submit_form()
            self.log.info("Editing completed for Unix global filter path")

        self.check_error_message()

    @adminconsole_handler()
    def delete_global_filter(self, global_filter_path):
        """
        Method to delete a global filter

        Args:
            global_filter_path(dict) :dictionary for global filter path to be deleted
        Returns:
            None
        Raises:
            Exception :
                if failed delete the global filter path
        """
        if global_filter_path['windows_global_filter_path']:
            if self.check_if_entity_exists("xpath", "//a[contains(text(),"
                                                    "'" + global_filter_path[
                                                        'windows_global_filter_path'] + "')]/../../div[2]/a"):
                self.driver.find_element_by_xpath("//a[contains(text(),"
                                                  "'" + global_filter_path[
                                                      'windows_global_filter_path'] + "')]/../../div[2]/a").click()
                self.wait_for_completion()
                self.click_yes_button()

            else:
                exp = "Failed to delete the windows global exception"
                self.log.exception(exp)
                raise Exception(exp)
            self.wait_for_completion()

        if global_filter_path['unix_global_filter_path']:

            if self.check_if_entity_exists("xpath", "//a[contains(text(),"
                                                    "'" + global_filter_path[
                                                        'unix_global_filter_path'] + "')]/../../div[2]/a"):
                self.driver.find_element_by_xpath("//a[contains(text(),"
                                                  "'" + global_filter_path[
                                                      'unix_global_filter_path'] + "')]/../../div[2]/a").click()
                self.wait_for_completion()
                self.click_yes_button()

            else:
                exp = "Failed to delete the unix global exception"
                self.log.exception(exp)
                raise Exception(exp)
            self.wait_for_completion()
