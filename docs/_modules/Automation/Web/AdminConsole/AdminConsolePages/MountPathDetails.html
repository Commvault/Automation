"""
This module provides the function or operations that can be performed on the
mount path selected on the AdminConsole

    MountPathDetails() -> StorageDetails() -> Storage() --
        -> AdminPage() -> login_page() -> AdminConsoleBase() -> object()

edit_path()      -- edit the mount path details
mount_path_info() -- print the info of the mount path in the log file
open_storage()   -- navigate to the storage's main page

"""
from Web.AdminConsole.Helper.Imports import *
from Web.AdminConsole.AdminConsolePages.StorageDetails import StorageDetails

    @adminconsole_handler()
    def edit_path(
            self,
            mount_path,
            access_type,
            storage_loc,
            username=None,
            password=None,
            allowed_writers=None,
            writers=None,
            preferred=False,
            enable=False):
        """Edit the mount path of the storage
            mount_path  - mount path to be edited
            accessType - type of access for this mount path
            storage_loc - type of storage, local or network
            username   - username in case of network storage
            password   - password in case of network storage
            writers    - total no. of writers allowed
            preferred  - choose whether this is the preferred mount path
            enable     - choose to enable this mountPath or not
        """
        self.log.info("Editing mount path info")
        if self.check_if_entity_exists("link", mount_path):
            self.driver.find_element_by_link_text(mount_path).click()
            self.wait_for_completion()
            self.select_value_from_dropdown("access_type", access_type)

            if storage_loc == "Local path":
                storage_loc = "localStorage"
            elif storage_loc == "Network path":
                storage_loc = "networkStorage"
            else:
                exp = "Please pass the correct value for type of storage"
                raise Exception(exp)
            if storage_loc == 'networkStorage':
                self.driver.find_element_by_name("loginName").send_keys(username)
                self.driver.find_element_by_xpath(
                    "//form/div[2]/label[2]/input[@type='password']").send_keys(password)
            if access_type == "Read/Write":
                if allowed_writers == "Allocate number of writers":
                    self.driver.find_element_by_xpath(
                        "//form[@name='storageForm']/span/div[1]/label[3]/input").click()
                    self.driver.find_element_by_name(
                        "allowedWritersNumber").clear()
                    self.driver.find_element_by_name(
                        "allowedWritersNumber").send_keys(writers)
            if preferred:
                if not self.driver.find_element_by_id(
                        "preferred").is_enabled():
                    self.driver.find_element_by_id("preferred").click()
            if enable:
                if not self.driver.find_element_by_id(
                        "enable").is_enabled():
                    self.driver.find_element_by_id("enable").click()
            self.driver.find_element_by_xpath(
                "//form/div[2]/button[2]/div[contains(text(), 'Save')]") .click()
            self.wait_for_completion()
        else:
            raise Exception("There is no link to edit the mount path")

    @adminconsole_handler()
    def mount_path_info(self):
        """Gather info about the mount path
        """
        self.log.info("Listing the details of the mount path")
        elements = self.driver.find_elements_by_xpath(
            "//div[2]/div/div/span/div[@class='detailsBar cols-5 ng-scope']/div")
        mount_path_info = {}
        for elem in elements:
            key = elem.find_element_by_xpath("./div/span[2]").text
            value = elem.find_element_by_xpath("./div/span[1]").text
            mount_path_info[key] = value
        self.log.info(mount_path_info)

    @adminconsole_handler()
    def open_storage(self):
        """Opens the storage's main page
        """
        self.log.info("Opens the main storage page")
        self.driver.find_element_by_xpath("//div/span/ul[1]/li/a").click()
        self.wait_for_completion()
