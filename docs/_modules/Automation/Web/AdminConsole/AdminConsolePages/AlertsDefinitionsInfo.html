                      "User":[]},

            Exception:

    @adminconsole_handler()
    def edit_alert_entities(self, alert_entities):

        ''' Method to alert entities tile for given alert

        Args:
            alert_entities   (list)      : server groups or clients for which alert is defined

                Eg. alert_entities = ["Server groups", "Clients"]

        Returns:
            None

        Raises:
            Exception:
                -- if fails to edit alert entities
        '''

        self.log.info("Editing Servers groups and clients")
        self.driver.find_element_by_xpath(
            "//a[@ng-click='openEntitiesEditModal()']").click()
        self.wait_for_completion()

        while self.check_if_entity_exists("xpath", "//div[@class='ng-scope selected']"):
            self.driver.find_element_by_xpath(
                "//div[@class='ng-scope selected']/span/span").click()

        if self.check_if_entity_exists(
                "xpath", "//button[@class='collapsed']"):
            collapsed_elems = self.driver.find_elements_by_xpath(
                "//button[@class='collapsed']")
            for elem in collapsed_elems:
                elem.click()
            entities = self.driver.find_element_by_xpath(
                "//div[@class='browse-tree alerts-browse']")
            for value in alert_entities['server_group_list']:
                entities.find_element_by_xpath(
                    "//span[./text()='"+str(value)+"']").click()
            for value in alert_entities['server_list']:
                entities.find_element_by_xpath(
                    "//span[./text()='"+str(value)+"']").click()

        self.click_button("Save")

    @adminconsole_handler()
    def edit_alert_summary(self, ind_notification):

        ''' Method to edit alert summary tile for given alert

        Args:
            ind_notification  (str)    :  individual notification to be enabled/disabled
                    Eg. - ind_notification = 'ON'

        Returns:
            None

        Raises:
            Exception:
                -- if fails to edit alert summary
        '''

        self.log.info("Editing individual notification")
        if ind_notification == 'ON':
            self.toggle_enable('Send individual notification')
        else:
            self.toggle_disable('Send individual notification')

    @adminconsole_handler()
    def delete_alert_from_details(self):
        """
        Deletes the alert from the alert details page

        Returns:
            None

        Raises:
            Exception:
                -- if fails to delete alert
        """
        self.log.info("Deleting the alert")
        self.select_hyperlink('Delete')
        if self.check_if_entity_exists(
                "xpath", "//div[@class='growl-item alert ng-scope alert-success']"):
            exp = self.driver.find_element_by_xpath(
                "//div[@class='growl-item alert ng-scope alert-success']/div/div/div")\
                    .text
            if  exp == "Alert definition deleted.":
                self.log.info(exp)
            else:
                self.log.exception(exp)
                raise Exception(exp)
        else:
            exp = "Alert deletion message not displayed"
            self.log.exception(exp)
            raise Exception(exp)

    @adminconsole_handler()
    def alert_info(self, alert_target):
        '''Method to extract information about an alert

        Returns:
            None

        Raises:
            Exception:
                -- if fails to extract alert info from details page
        '''
        try:
            details = {}
            count = 0
            title = self.driver.find_element_by_xpath(
                "//h1[@class='page-title editable js-page-title ng-binding']")
            details.update({"AlertName":title.text})
            elems = self.driver.find_elements_by_xpath("//div[@class='page-details group']")
            for elem in elems:
                entity_name = elem.find_element_by_xpath("./span").text
                if entity_name == 'Alert target':
                    keyvalue_list = []
                    keyvalue_dict = {}
                    key = elem.find_element_by_xpath("./div/div/ul/li/span").text
                    value = elem.find_element_by_xpath("./div/div/ul/li/span[2]").text
                    keyvalue_dict.update({key: value})
                    if alert_target['Email'] or alert_target['Console']:
                        recipients = elem.find_elements_by_xpath(
                            "./div/div/div[2]/cv-tabset-component/div/div")
                        for recipient in recipients:
                            count += 1
                            if count > 1 and not alert_target['Email']:
                                break
                            else:
                                recipient.find_element_by_xpath(
                                    "./../../ul/li["+str(count)+"]/a").click()
                                key = recipient.find_element_by_xpath(
                                    "./../../ul/li["+str(count)+"]/a").text
                                li_elems = recipient.find_elements_by_xpath(".//li")
                                temp_dict = {}
                                for li_elem in li_elems:
                                    k = li_elem.find_element_by_xpath("./span[1]/label").text
                                    v = li_elem.find_element_by_xpath("./span[2]").text
                                    temp_dict.update({k: v})
                                keyvalue_dict.update({key:temp_dict})
                        details.update({entity_name: keyvalue_dict})
                else:
                    div_elements = elem.find_elements_by_xpath("./div")
                    for div_elem in div_elements:
                        keyvalue_list = []
                        keyvalue_dict = {}
                        key = ''
                        ul_elements = div_elem.find_elements_by_xpath(".//ul")
                        if not ul_elements:
                            inner_div_elems = div_elem.find_elements_by_xpath(
                                ".//div[@class='ng-scope']")
                            for inner_div_elem in inner_div_elems:
                                key = inner_div_elem.find_element_by_xpath("./span").text
                                keyvalue_list.append(key)
                            details.update({entity_name: keyvalue_list})
                        for ul_elem in ul_elements:
                            li_elements = ul_elem.find_elements_by_xpath('.//li')
                            for li_elem in li_elements:
                                if self.is_element_present(
                                        ".//toggle-control", li_elem):
                                    if li_elem.find_element_by_xpath(
                                            ".//toggle-control/div").get_attribute("class") \
                                        == "cv-material-toggle cv-toggle":
                                        keyvalue_dict[li_elem.find_element_by_xpath(
                                            ".//span[@class='pageDetailColumn']").text] = "OFF"
                                    else:
                                        keyvalue_dict[li_elem.find_element_by_xpath(
                                            ".//span[@class='pageDetailColumn']").text] = "ON"
                                inside_li_tags = li_elem.find_elements_by_xpath('.//span')
                                tag_iterate_count = 0
                                for tag in inside_li_tags:
                                    tag_iterate_count += 1
                                    if tag.text:
                                        if tag_iterate_count == 1:
                                            key = tag.text
                                            if len(inside_li_tags) == 1:
                                                keyvalue_list.append(key)
                                                break
                                        else:
                                            value = tag.text
                                            if 'Edit' in value:
                                                value = value.rstrip('\nEdit')
                                            keyvalue_dict.update({key: value})

                            if len(inside_li_tags) == 1:
                                details.update({entity_name: keyvalue_list})
                            else:
                                details.update({entity_name: keyvalue_dict})
            self.log.info(details)
            return details
        except Exception as exp:
            raise Exception(exp)
