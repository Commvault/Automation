    @adminconsole_handler()
    def __set_plans(self, plans):
        """
        Fill the plans field on company creation page with plans in the arguments

        Args:
            plans (list): list of plans to be set for the company to be created

        Returns:
            None

        Raises:
            Exception:
                -- if fails to set plans for the company
        """
        self.log.info("Setting plans for the company")
        plans_id = self.driver.find_element_by_xpath(
            "//isteven-multi-select[@id='plans']")
        plans_id.find_element_by_xpath("//span/button").click()
        plans_id.find_element_by_xpath(
            "//span/div/div[1]/div[1]/button[2]").click()

        for plan in plans:
            self.driver.find_element_by_xpath(
                "//div/button[@class='clearButton']").click()
            self.driver.find_element_by_xpath(
                "//div[@class='line-search']/input").send_keys(plan)
            self.wait_for_completion()
            listed_span_elems = self.driver.find_elements_by_xpath(
                "//div[@class='acol']/label/span")
            for span_elem in listed_span_elems:
                if span_elem.text.strip() == plan:
                    span_elem.click()
                    break
        time.sleep(5)
        self.driver.find_element_by_xpath(
            "//div[@class='line-searchOk ng-scope']").click()

    @adminconsole_handler()
    def __enable_custom_domain(self, enable_custom_domain):
        """
        Method to fill the Custom domain on company creation page

        Args:
            enable_custom_domain (boolean): value if custom domain to be enabled/disabled for
                the company to be created

        Returns:
            None

        Raises:
            Exception:
                -- if fails to set enable_custom_domain for the company
        """
        self.log.info("Setting value, if custom domain to be defined or not")
        if enable_custom_domain:
            if not self.driver.find_element_by_xpath(
                    "//input[@id='enableCustomDomain']").is_selected():
                self.driver.find_element_by_xpath("//input[@id=\
                'enableCustomDomain']/following-sibling::label[contains(text()\
                ,'Enable custom domain')]").click()
                self.wait_for_completion()
            else:
                self.log.info("Custom domain already enabled")
        else:
            self.log.info("Custom domain need not be enabled")

    @adminconsole_handler()
    def __submit_company(self):
        """
        Saves the company created by clicking the submit button

        Args:
            None

        Returns:
            None

        Raises:
            Exception:
                -- if fails to click on Submit button
        """
        self.log.info("Clicking on Submit button")
        buttons = self.driver.find_elements_by_xpath(
            "//form[@id='addSubscriptionForm']/div[2]/button")
        for button in buttons:
            if button.text == 'Save':
                button.click()
                self.wait_for_completion()
                break

    @adminconsole_handler()
    def add_company(
            self,
            company_name,
            email,
            contact_name,
            plans,
            company_alias,
            smtp_alias,
            mail_template,
            enable_custom_domain,
            primary_domain):
        """ Adds the company with the specified inputs in the arguments.

            Args:
                company_name (string) : Name of the cmpany to be created
                email (string) : email of Tenant admin for the company to be created
                contact_name (string) : Contact name for the company
                plans (list): plans to be associated to the company
                company_alias (string): company alias for the company
                smtp_alias (string): smtp alias for the company
                enable_custom_domain (boolean): enable_custom_domain to be set for the company
                    to be created
                primary_domain (string): primary_domain to be set for the company to be created

            Returns:
                None

            Raises:
                Exception:
                    -- if fails to set primary_domain for the company
        """
        self.log.info("Creating new company")
        self.__click_add_company()
        self.wait_for_completion()
        self.__set_company_name(company_name)
        self.__set_email(email)
        self.__set_contact_name(contact_name)
        if plans:
            self.__set_plans(plans)
        time.sleep(5)
        self.__set_company_alias(company_alias)
        if smtp_alias:
            self.__set_smtp(smtp_alias)
        if mail_template:
            self.__set_mailtemplate(mail_template)
        if enable_custom_domain:
            self.__enable_custom_domain(enable_custom_domain)
            self.__set_primary_domain(primary_domain)
        self.__submit_company()
        self.wait_for_completion()

        if self.check_if_entity_exists(
                "xpath", "//span[@class='help-block']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='help-block']").text
            self.log.exception(exp)
            raise Exception(exp)

        if self.check_if_entity_exists(
                "xpath", "//span[@class='serverMessage error']"):
            exp = self.driver.find_element_by_xpath(
                "//span[@class='serverMessage error']").text
            self.log.exception(exp)
            raise Exception(exp)

    @adminconsole_handler()
    def delete_company(self, company_name):
        """checks if the given company exists and deletes the same

        Args:
            Company_name (string) - name of the company to be deleted

        Returns:
            None

        Raises:
            Exception:
                -- if fails delete company with given name
        """
        self.log.info("Deleting the company {0}".format(company_name))
        self.search_for(company_name)
        if self.check_if_entity_exists("link", company_name):
            company_id = self.driver.find_element_by_xpath(
                "//a[text()='" + company_name + "']")
            company_id.find_element_by_xpath(
                "//../../div[3]/div/a/span").click()

            if self.check_if_entity_exists("link", "Delete company"):
                delete_link = company_id.find_element_by_xpath(
                    "//../../div[3]/div/ul//a")
                if delete_link.text == 'Delete company':
                    delete_link.click()
                    self.wait_for_completion()
                    self.driver.find_element_by_xpath(
                        "//div[3]/button[2]").click()
                    self.wait_for_completion()
                    alert_path = "//div[3]/div/div[@class='growl-item\
                     alert ng-scope alert-error alert-danger']"

                if self.check_if_entity_exists("xpath", alert_path):
                    exp = self.driver.find_element_by_xpath(
                        alert_path).text
                    self.wait_for_completion()
                    raise Exception(exp)
            else:
                exp = "There is no option to delete the company"
                self.log.exception(exp)
                raise Exception(exp)
        else:
            exp = "There is no company with the given name"
            self.log.exception(exp)
            raise Exception(exp)
