    @adminconsole_handler()
    def enable_http_proxy(
            self,
            proxy_server,
            proxy_port,
            proxy_user_name=None,
            proxy_password=None):
        """
        set up http proxy on commcell.

        Args:
            proxy_server    (str)   --  Host name or IP address of the proxy server.

            proxy_port      (int)   --  port no on which proxy server listens to.

            proxy_user_name (str)   --  User name to authenticate against proxy.

            proxy_password   (str)  --  Password to authenticate against proxy.

        """

        self.toggle_enable("Proxy server")
        self.toggle_edit("Proxy server")  # in case already enabled click edit.
        self.fill_form_by_id("proxyServer", proxy_server)
        self.fill_form_by_id("proxyPort", proxy_port)
        self.submit_form()
        if proxy_user_name and proxy_password:
            self.toggle_enable("User authentication")
            self.fill_form_by_id("userName", proxy_user_name)
            self.fill_form_by_id("password", proxy_password)
            self.fill_form_by_id("confirmPassword", proxy_password)
            self.submit_form()

    @adminconsole_handler()
    def disable_http_proxy(self):
        """
        Disables http proxy
        :return:
        """
        self.toggle_disable("Proxy server")

    @adminconsole_handler()
    def set_internet_options(self, internet_option, gateway_client=None):
        """
        Configures internet gateway from maintenence page.

        Args:
            internet_option (Enum) -- InternetOptions defined in adminconsoleconstants.
            Eg:InternetOptions.USE_CLIENT

            gateway_client  (str) -- name of the client , if InternetOptions.USE_CLIENT
            option is selected


        Returns:

        """
        self.tile_select_edit_icon("Internet options")
        self.select_radio(internet_option.value)
        if acdefines.InternetOptions.USE_CLIENT.value == internet_option.value:
            if isinstance(gateway_client, str):
                self.select_value_from_dropdown("gatewayClient", gateway_client)
            else:
                raise Exception("Invalid Input parameter :gateway_client")
        self.submit_form()

    @adminconsole_handler()
    def _download_software(self, download_option=None, sp_version=None, os_to_download=None):
        """
        Fills download software page with configured settings.

        Args:
            download_option (str) -- download option to be chosen

            sp_version  (str)   -- sp version to download
                                    Example: 'SP12'

            os_to_download  (list) --  List of os to be downloaded

        Returns:
            None

        Raises:
            Exception

                if inputs are invalid

        """
        if download_option:
            # To select the download option
            self.select_radio(download_option)

        if acdefines.DownloadOptions.GIVEN_SP_AND_HF.value == download_option:
            if isinstance(sp_version, str):
                self.select_value_from_dropdown("servicePackValue", sp_version.upper())
            else:
                raise Exception("Invalid Input parameter :sp_version")

        if os_to_download:
            self.select_hyperlink("Operating system options")
            win_os = []
            unix_os = []

            # To separate windows and unix os from the given lsit
            for os in os_to_download:
                if 'windows' in os:
                    win_os.append(os)
                else:
                    unix_os.append(os)

            if win_os:
                # To deselect default value
                if 'windows64' not in win_os:
                    self.checkbox_deselect('windows64')

                # To select windows OS
                self.checkbox_select(win_os, "Windows")

            if unix_os:
                # To select Unix OS
                self.checkbox_select(unix_os, "Unix")

                # To deselect default value
                if 'linux64' not in unix_os:
                    self.checkbox_deselect('linux64')

    @adminconsole_handler()
    def run_download_software(self, download_option=None, sp_version=None, os_to_download=None):
        """
        Runs download software job with configured settings.

        Args:
            download_option (str) -- download option to be chosen

            sp_version  (str)   -- sp version to download
                                    Example: 'SP12'

            os_to_download  (list) --  List of os to be downloaded

        Returns:
            (str)       --  Job id of the download job triggered

        Raises:
            Exception

                if inputs are invalid

                if job fails

                if job failed to start

        Usage:

            * Enum DownloadOptions defined in adminconsoleconstants can be used for providing
              input to download_option

                >>> DownloadOptions.LATEST_HF_FOR_INSTALLED_SP.value

            * Enum DownloadOSID defined in adminconsoleconstants can be used for providing input
              to os_to_download

                >>> [DownloadOSID.WINDOWS_32.value, acd.DownloadOSID.UNIX_AIX32.value]

            * If no arguments are given, by default Latest hotfixes for the installed service pack
              is chosen as the
              download_option and WindowsX64 is chosen as the os_to_download

        """
        # To load Download software
        self.fetch_download_software()

        # To select the run button
        self.tile_select_run_icon("Download software")

        # To fill the Download software page
        self._download_software(download_option=download_option,
                                sp_version=sp_version,
                                os_to_download=os_to_download)

        # To click the Download button
        self.submit_form()

        if self.check_if_entity_exists(
                "xpath",
                "//h1[contains(text(), 'Download software')]//..//..//a[contains(@href, 'jobs')]"):

            # To get the job id
            jobid = self.driver.find_element_by_xpath(
                "//h1[contains(text(), 'Download software')]//..//..//a[contains(@href, 'jobs')]"
            ).text

        else:
            raise Exception("Job not started, please check the logs")

        # To click OK button
        self.click_button("OK")

        # return the job id
        return jobid
