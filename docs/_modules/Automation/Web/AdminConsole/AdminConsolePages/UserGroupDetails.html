    @adminconsole_handler()
    def edit_user_group(self,
                        group_name=None,
                        description=None,
                        plan=None,
                        quota=False,
                        group_enabled=True,
                        shared_laptop_usage=False,
                        quota_limit=None):
        """
        edit the user group details
        :param quota_limit          (str)   --  the size in GB of the group quota
        :param group_name           (str)   --  the new name of the group
        :param plans                (list)  --  list of plans to be associated with user group
        :param group_enabled        (bool)  --  enable / disable the group
        :param description          (str)   --  description of the user group
        :param quota                (bool)  --  enable / disable quota
        :param shared_laptop_usage  (bool)  --  enable / disable device activation
        """

        self.log.info('CLicking on Edit Hyperlink to open edit window')
        self.select_hyperlink("Edit")
        if group_name:
            self.log.info('Editing user group name')
            self.fill_form_by_id("userGroupName", group_name)
        if description:
            self.log.info('Editing user group description')
            self.fill_form_by_id("groupDesc", description)
        if quota:
            self.log.info('Enabling user group quota')
            self.checkbox_select("quotaEnabled")
            if quota_limit:
                self.log.info('Setting user group quota limit')
                self.fill_form_by_id("quotaLimit", quota_limit)
        else:
            self.log.info('Disabling user group quota')
            self.checkbox_deselect("quotaEnabled")

        if group_enabled:
            self.log.info('Enabling user group')
            self.checkbox_select("groupEnabled")
        else:
            self.log.info('Disabling user group')
            self.checkbox_deselect("groupEnabled")

        if shared_laptop_usage:
            self.log.info('Enabling shared laptop usage')
            self.checkbox_select("userCentric")
        else:
            self.log.info('Disabling shared laptop usage')
            self.checkbox_deselect("userCentric")

        if plan:
            self.log.info('Setting up plan for user group')
            self.cv_single_select('Plan', plan)

        self.log.info('Editing completed, submitting form to save new values')
        self.submit_form()

    @adminconsole_handler()
    def add_users(self, users):
        """Adds new users to the group
            users   :   a list, the list of users to be added
        """
        if self.check_if_entity_exists("link", "Add users"):
            self.driver.find_element_by_link_text("Add users").click()
            self.wait_for_completion()
            x_list = users
            if self.driver.find_element_by_id("showSelected").is_selected():
                self.driver.find_element_by_xpath(
                    "//input[@id='showSelected']/following-sibling::label").click()
            selected = []
            for user in users:
                self.log.info("adding user with username : %s", user)
                self.search_for(user)
                users_list = self.driver.find_elements_by_xpath(
                    "//cv-grid/div/div[2]/div/div[1]/div[2]/div[2]/div[ \
                    @class='ui-grid-canvas']/div")
                index = 0
                for username in users_list:
                    index += 1
                    temp_user_name = username.find_element_by_xpath("./div/div[1]/span").text
                    if temp_user_name == user:
                        self.driver.find_element_by_xpath(
                            "//cv-grid/div/div[2]/div/div[1]/div[1]/div/div[2]/div[ \
                            @class='ui-grid-canvas']/div["+str(index)+"] \
                            /div/div/div/div/div").click()
                        selected.append(temp_user_name)
            x_list = list(set(users) - set(selected))
            self.driver.find_element_by_xpath("//*[@id='editUsersInGroupForm']/div[3]/"
                                              "button[2]").click()
            self.wait_for_completion()
            if x_list:
                raise Exception("There are no users with the name or the user is already added "
                                +str(x_list))
        else:
            raise Exception("There is no option to add users to the group. Check if the user"
                            " has permissions")

    @adminconsole_handler()
    def remove_users(self, users):
        """Removes users from the group
            users   :   a list, the list of users to be removed
        """
        self.log.info("Removing users from the group")
        items = []
        for user in users:
            self.log.info("removing user with username : %s", user)
            self.search_for(user)
            while True:
                if self.check_if_entity_exists("xpath", "//a[@title='"+user+"']"):
                    self.driver.find_element_by_xpath(
                        "//a[@title='" + user + "']/../../div[4]/span/a[contains(text(),"
                                                "'Remove')]").click()
                    self.wait_for_completion()
                    self.driver.find_element_by_xpath(
                        "//div[@class='modal-content']/div[3]/button[2]").click()
                    self.wait_for_completion()
                    break
                else:
                    if self.cv_table_next_button_exists():
                        if self.driver.find_element_by_xpath(
                                "//button[@ng-disabled='cantPageForward()']").is_enabled():
                            self.driver.find_element_by_xpath(
                                "//button[@ng-disabled='cantPageForward()']/div").click()
                            self.wait_for_completion()
                            continue
                        else:
                            items.append(user)
                            break
                    else:
                        items.append(user)
                        break
        if items:
            raise Exception("There are no users with the name "+str(items))

    @adminconsole_handler()
    def list_users(self):
        """Lists all the users associated with the group
        :return:
            users   (list)  --  list of all users associated with the group
        """
        elements = self.driver.find_elements_by_xpath("//div[3]/div[2]/div[@class="
                                                      "'ui-grid-canvas']/div")
        users = []
        for elem in elements:
            user = {'User name': elem.find_element_by_xpath("./div/div[3]/a").text,
                    'Full name': elem.find_element_by_xpath("./div/div[2]/div").text,
                    'Email': elem.find_element_by_xpath("./div/div[1]/div").text}
            users.append(user)
        self.log.info(users)
        return users


[docs]    def get_user_group_details(self):
        '''
        Fetches and displays all the details about the user group like name, description,
        enabled, plans and Username of users etc.

        Returns:
            group_details   (dict):  dict of user group details
        '''
        group_details = {}
        elements = self.driver.find_elements_by_xpath(
            "//ul[@class='list-style-secondary ng-scope']/li")
        for element in elements:
            if self.is_element_present('./cv-plan-migration', element):
                key = element.find_element_by_xpath("./cv-plan-migration/div/label").text
                if self.is_element_present('./cv-plan-migration/div/span/div/a', element):
                    value = element.find_element_by_xpath(
                        "./cv-plan-migration/div/span/div/a").text
                else:
                    value = element.find_element_by_xpath(
                        "./cv-plan-migration/div/span/div/span").text
                group_details[key] = value
            else:
                key = element.find_element_by_xpath("./span[1]").text
                if self.is_element_present("./span[2]", element):
                    value = element.find_element_by_xpath("./span[2]").text
                else:
                    value = 'No'
                if value == '✓':
                    value = 'Yes'
                elif value == '✘':
                    value = 'No'
                group_details[key] = value

        self.log.info(group_details)
        return group_details
