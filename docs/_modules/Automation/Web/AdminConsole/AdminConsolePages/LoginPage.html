"""
This module provides the function or operations that can be performed on the
Login page on the AdminConsole

Class:

    LoginPage() -> AdminConsoleBase() -> object()

Functions:

_init_()            -- initialize the class object
set_username()      -- fill the username field with the specified username
set_password()      -- fill the password field with the specified password
submit_login()      -- login to admin console
login()             -- attempt login to AdminConsole using the username and password provided
forgot_password()   -- attempt to recover the password for the given username
"""

from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.by import By
from Web.AdminConsole.Helper.AdminConsoleBase import AdminConsoleBase
from Web.Common.handlers import adminconsole_handler



[docs]class LoginPage(AdminConsoleBase):
    """
    This class provides the operations that can be perfomed on the Login Page of adminconsole
    """
    def __init__(self, browser):
        super(LoginPage, self).__init__(browser)
        self.login_flag = 1

    @adminconsole_handler()
    def submit_login(self):
        """
        Login to AdminConsole using the username and password provided
        """
        self.driver.find_element_by_id("loginbtn").click()
        self.wait_for_completion()
        if self.check_if_entity_exists("xpath",
                                       "//*[@id='loginform']/div/div/form/div[1]/div[1]"):
            raise Exception(self.driver.find_element_by_xpath(
                "//*[@id='loginform']/div/div/form/div[1]/div[1]").text)
        try:
            WebDriverWait(self.driver, 300).until(EC.presence_of_element_located((
                By.XPATH, "//div[@class='loadingNav loaded']")))
            WebDriverWait(self.driver, 30).until(EC.presence_of_element_located((
                By.ID, "user-account-name")))
            self.wait_for_completion()
        except Exception as exp:
            raise Exception("Admin Console dashboard did not load in 5 minutes "+str(exp))

        ret_code = self.check_if_entity_exists("xpath",
                                               "div[@class='center-container']/div/div/div[3]")
        if ret_code:
            raise Exception(self.driver.find_element_by_xpath("//div[@class='error']").text)
        else:
            if self.check_if_entity_exists(
                    "xpath",
                    "//div[@class='loginMsg vw-login-message error alert vw-alert-danger']"):
                raise Exception(self.driver.find_element_by_xpath(
                    "//div[@class='loginMsg vw-login-message error alert vw-alert-danger']").text)
            if self.check_if_entity_exists("xpath", "//div[@class='button-container']/button"):
                self.log.info("Product is in Evaluation. Clicking on Okay")
                self.driver.find_element_by_xpath("//div[@class='button-container']/button").click()
                self.log.info("Clicked on Okay")
                self.wait_for_completion()
        self.log.info("Login successful")
        self.login_flag = 0

    @adminconsole_handler(hide_args=True)
    def login(self, user_name, password, remember_me=True):
        """
        Login to AdminConsole by using the username and password specified

        Args:
            user_name (basestring) : username to be used to login
            password (basestring)  : password to be used to login
            remember_me (bool)      : select/deselect the keep me logged in checkbox
        """
        self.fill_form_by_id("username", user_name)
        self.driver.find_element_by_id("continuebtn").click()
        self.wait_for_completion()
        self.fill_form_by_id("password", password)
        if remember_me:
            self.checkbox_select("stayactivebox")
        else:
            self.checkbox_deselect("stayactivebox")
        self.submit_login()

    @adminconsole_handler()
    def forgot_password(self, user_name):
        """
        To reset the password

        Args:
            user_name (basestring): username to be used
        """
        self.log.info("Selecting Forgot password")
        if self.check_if_entity_exists("xpath", "//a[@href='forgotPassword.jsp']"):
            self.driver.find_element_by_xpath("//a[@href='forgotPassword.jsp']").click()
            self.wait_for_completion()
            self.fill_form_by_id("username", user_name)
            self.driver.find_element_by_xpath(".//*[@id='loginbtn']").click()
            self.wait_for_completion()
            WebDriverWait(self.driver, 60).until(EC.presence_of_element_located(
                (By.ID, "forgotpass-title")))
            if self.check_if_entity_exists("xpath", "//a[@href='index.jsp']"):
                self.driver.find_element_by_xpath("//a[@href='index.jsp']").click()
                self.wait_for_completion()
