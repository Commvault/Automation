    @adminconsole_handler()
    def add_tape_storage_pool(self, pool_name, storage_media):
        """add a new tape storage library
        :param pool_name        (str)   --  the name of the storage pool to be added
        :param storage_media    (str)   --  the name of the media agent
        """
        self.__add_storage_pool("Tape")
        self.fill_form_by_id("storagePoolName", pool_name)
        self.select_value_from_dropdown("tapeStorage", storage_media)
        self.wait_for_completion()
        self.driver.find_element_by_xpath(
            "//button[@class='btn btn-primary cvBusyOnAjax setup-btn--hidden']").click()
        self.wait_for_completion()
        if self.check_if_entity_exists(
                "xpath", "//div[@class='setup-content ng-scope']/div[1] "):
            raise Exception(self.driver.find_element_by_xpath(
                "//div[@class='setup-content ng-scope']/div[1]").text)

    @adminconsole_handler()
    def add_disk_cloud_storage_pool(
            self, pool_name, storage, partition_settings, media=None, library=None, path=None,
            path_type=None, network_username=None, network_password=None):
        """Add new library for the MA specified
            The storage can either be created in the local path of the clients or on a network path

            poolName            - name of the storage pool to be created
            storage             - new or existing storage to be used
            partitionSettings   -  dedupe partition information
            media               - media agent to be used to create a new storage
            library             - name of the existing library to be used
            path                - path for the new library to be created
            type                - Tape or Disk Storage pool
        """
        self.__add_storage_pool("Disk/Cloud")
        if self.cb_flag:
            cb_time_stamp = datetime.datetime.fromtimestamp(
                datetime.datetime.now().timestamp()).strftime('%d_%m_%Y_%H%M%S')
        self.fill_form_by_id("storagePoolName", pool_name)
        if storage == "New storage":
            self.driver.find_element_by_id("newStorage").click()
            self.scroll_into_view("mediaAgent")
            self.select_value_from_dropdown("mediaAgent", media)
            if path_type == 'Local path':
                self.fill_form_by_id("mountPath", path)
            else:
                self.fill_form_by_id("userName", network_username)
                self.fill_form_by_id("password", network_password)
                self.fill_form_by_id("mountPath", path)
                if self.check_if_entity_exists(
                        "xpath", "//div[2]/div[1]/div[@class='serverMessage error']"):
                    raise Exception(self.driver.find_element_by_xpath(
                        "div[2]/div[1]/div[@class='serverMessage error']").text)
        else:
            self.driver.find_element_by_id("existingStorage").click()
            self.wait_for_completion()
            self.select_value_from_dropdown("storage", library)
        self.driver.find_element_by_xpath(
            "//div/button[@data-ng-click='saveGeneralTab()']").click()
        self.wait_for_completion()
        select = Select(self.driver.find_element_by_id("dedupeMediaAgent"))
        if not isinstance(partition_settings, dict):
            partition_settings = ast.literal_eval(partition_settings)
        for key in partition_settings.keys():
            partition_paths = partition_settings[key]
            for partition_path in partition_paths:
                if self.cb_flag:
                    partition_path = partition_path+"_"+cb_time_stamp
                    self.log.info("Partition_path"+partition_path)
                pattern = re.compile(key, re.IGNORECASE)
                for option in select.options:
                    value = option.get_attribute('label')
                    if pattern.search(value):
                        option.click()
                        self.fill_form_by_id("partitionPath", partition_path)
                        self.driver.find_element_by_id("addDDBInfo").click()
                        if self.check_if_entity_exists("xpath", "//div \
                                [@class='serverMessage error']"):
                            exp = self.driver.find_element_by_xpath(
                                "//div[@class='serverMessage error \
                                ']").text
                            raise Exception(exp)
                        self.wait_for_completion()
        self.driver.find_element_by_id("submitForCreatePool").click()
        self.wait_for_completion()
        if self.check_if_entity_exists("xpath",
                                       "//div[@class='set-up-page']/div[2]/ng-include/div[2]"):
            raise Exception(self.driver.find_element_by_xpath(
                "//div[@class='serverMessage error']").text)

    @adminconsole_handler()
    def __add_storage_pool(self, pool_type):
        """
        Adding a new storage pool
        """
        if self.check_if_entity_exists("link", "Add storage pool"):
            self.driver.find_element_by_link_text("Add storage pool").click()
            self.driver.find_element_by_link_text(pool_type).click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to Add Storage Pool")

    @adminconsole_handler()
    def list_storage_pool(self):
        """Get the details of all the storage pools in the form of a dictionary
            storage_name status Manufacturer Model

            Ex: {
                'storage1': 'Ready Disk Disk',
                'storage2': 'Ready Disk Disk'
            }
            :return:
                storage_dict    (dict)  --  details of all storage pools
        """
        storage_dict = {}
        while True:
            elements = self.driver.find_elements_by_xpath(
                "//div[@class='ui-grid-contents-wrapper']/div \
                [2]/div[2]/div[@class='ui-grid-canvas']/div")
            for elem in elements:
                key = elem.find_element_by_xpath("./div/div[1]/div/a").text
                value = []
                for index in range(2, 7):
                    if index == 2:
                        value.append(elem.find_element_by_xpath("./div/div[" + str(index) +
                                                                "]/div/span").text)
                    elif index == 4:
                        value.append(elem.find_element_by_xpath("./div/div[" + str(index) +
                                                                "]/div").text)
                    else:
                        value.append(elem.find_element_by_xpath("./div/div[" + str(index) +
                                                                "]/span").text)
                storage_dict[key] = value
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        self.log.info(storage_dict)
        return storage_dict

    @adminconsole_handler()
    def open_storage_pool(self, storage_pool):
        """Open the storage pool with the given name
            storage_pool - name of the storage to be opened
        """
        self.search_for(storage_pool)
        if self.check_if_entity_exists("link", storage_pool):
            self.driver.find_element_by_xpath("//a[text()='"+ storage_pool +"']").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no storage with the name " + storage_pool)

    @adminconsole_handler()
    def action_delete(self, storage_pool):
        """Delete the storage_pool with the given name

            storage_pool - name of the storage to be removed
        """
        self.search_for(storage_pool)
        if self.check_if_entity_exists("link", storage_pool):
            elements = self.driver.find_elements_by_xpath("//div[@class='ui-grid-canvas']/div")

        for element in elements:
            if element.find_element_by_xpath("./div/div[1]/div/a").text == storage_pool:
                element.find_element_by_xpath("./div/div[7]/div/a/span").click()
                if self.check_if_entity_exists("link", "Delete"):
                    element.find_element_by_xpath(
                        "./div/div[7]/div/ul//a[contains(text(),'Delete')]").click()
                    self.wait_for_completion()
                    self.driver.find_element_by_xpath(
                        "//div/div[3]/button[contains(text(),'Yes')]").click()
                    self.wait_for_completion()
                    if self.check_if_entity_exists("xpath",
                                                   "//div[@class='modal-body text-danger"
                                                   " ng-binding ng-scope']"):
                        raise Exception(self.driver.find_element_by_xpath(
                            "//div[@class='modal-body text-danger ng-binding ng-scope']").text)
                else:
                    raise Exception("There is no option to delete the storagePool " + storage_pool)
            else:
                raise Exception("There is no storage pool with name "+ storage_pool)
