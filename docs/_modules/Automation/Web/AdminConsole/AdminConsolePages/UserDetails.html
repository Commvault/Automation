    @adminconsole_handler()
    def edit_user(self,
                  email,
                  full_name=None,
                  user_name=None,
                  groups=None,
                  plan=None,
                  enabled=True,
                  password=None,
                  admin_password=None):
        """Modifies the user with given details

        Args:
            full_name    (basestring):  full name of the user

            user_name    (basestring):  the user name of the user

            email        (basestring):  the email ID of the user

            groups       (basestring/list): one group name or list of user groups to be
                                            associated with the user

            plan         (list):        list of plans to be associated with the user

            enabled      (bool):        if the user should be enabled/ disabled

            password     (basestring):  the password of the user

        """

        self.log.info('Opening edit dialog for the user')
        self.select_hyperlink("Edit")

        if full_name:
            self.log.info('Entering fullname for the user')
            self.fill_form_by_id("fullName", full_name)

        if user_name:
            self.log.info('Entering username for the user')
            self.fill_form_by_id("userName", user_name)

        self.log.info('Entering email for the user')
        self.fill_form_by_id("email", email)

        if groups:
            self.cvselect_from_dropdown(self.props['label.userGroup'], groups)

        if password:
            self.log.info('Entering password for the user')
            self.fill_form_by_id("password", password)

            self.log.info('Confirming password for the user')
            self.fill_form_by_id("confirmUserPassword", password)

            self.log.info('Entering admin password for confirmation')
            self.fill_form_by_id("currentUserPassword", admin_password)

        if plan:
            self.log.info('Setting up plan association for the user')
            self.cvselect_from_dropdown(self.props['tableHeader.plan'], plan)

        if enabled:
            self.log.info('Enabling user')
            self.checkbox_select("enabled")
        else:
            self.log.info('Disabling user')
            self.checkbox_deselect("enabled")

        self.log.info('Submitting form to complete edit operation.')
        self.submit_form()
        self.check_error_message()

    @adminconsole_handler()
    def get_user_details(self):
        """Displays all the details about the user like name, email, user group and plans

        Returns:
            user_details   (dict):  dict of users and their info
        """
        user_details = {}
        elements = self.driver.find_elements_by_xpath("//ul[@class='list-style-secondary']/li")
        for element in elements:
            if self.is_element_present('./cv-plan-migration', element):
                key = element.find_element_by_xpath("./cv-plan-migration/div/label").text
                if self.is_element_present('./cv-plan-migration/div/span/div/a', element):
                    value = element.find_element_by_xpath(
                        "./cv-plan-migration/div/span/div/a").text
                else:
                    value = element.find_element_by_xpath(
                        "./cv-plan-migration/div/span/div/span").text
                user_details[key] = value
            else:
                key = element.find_element_by_xpath("./span[1]").text
                value = element.find_element_by_xpath("./span[2]").text
                if value == '✓':
                    value = 'Yes'
                elif value == '✘':
                    value = 'No'
                user_details[key] = value
        self.log.info(user_details)
        return user_details

    @adminconsole_handler()
    def open_user_group(self, user_group):
        """
        Opens the user group associated with user

        Args:
            user_group  (basestring):   the name of the user group to open

        """
        if self.check_if_entity_exists("link", user_group):
            self.select_hyperlink(user_group)
        else:
            raise Exception("The user is not associated with the given user group {0}"
                            .format(user_group))
