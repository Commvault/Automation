    @adminconsole_handler()
    def navigate_to_home(self):
        """
        Navigates to the main page(servers)
        """
        if self.check_if_entity_exists("css", "span.logo-text"):
            self.driver.find_element_by_css_selector(
                "span.logo-text").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no option to Navigate to Home page using the Admin"
                            "Console text")

    @adminconsole_handler()
    def user_settings(self):
        """
        Navigates to the user settings page
        """
        self.driver.find_element_by_id("user-account-name").click()
        self.driver.find_element_by_xpath("//a[@href='#changePassword']").click()
        self.wait_for_completion()

    @adminconsole_handler()
    def logout(self, url=None):
        """
        Logs out from the current user
        :param url      (str)   --  adminconsole URL
        """
        self.driver.find_element_by_id("user-account-name").click()
        elements = self.driver.find_elements_by_xpath(
            "//header/div[2]/div[4]/ul/li")
        pattern = re.compile(self.props['label.logout'], re.IGNORECASE)
        for element in elements:
            logout = element.find_element_by_xpath("./a").text
            if pattern.search(logout):
                self.driver.find_element_by_link_text(logout).click()
                self.wait_for_completion()
                # clicks on log in again option

                if url:
                    self.navigate(url)
                else:
                    self.driver.find_element_by_xpath(
                        "//div[@class='links']/a[1]").click()
                    self.wait_for_completion()
