    @adminconsole_handler()
    def __select_user_group_type(self, user_group_type):
        """
        Selects All users groups or AD user groups or Local groups
        """
        self.driver.find_element_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/a/span[1]").click()
        self.wait_for_completion()
        elements = self.driver.find_elements_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/ul/li")
        pattern = re.compile(user_group_type, re.IGNORECASE)
        for elem_li in elements:
            if pattern.search(elem_li.find_element_by_xpath("./a/span").text):
                elem_li.find_element_by_xpath("./a/span").click()
                self.wait_for_completion()
                break

    @adminconsole_handler()
    def add_user_group(self, group_name, description=None, quota=False, group_enabled=True,
                       shared_laptop_usage=False, quota_limit=None):
        """Adds a new user group with the given name
            group_name          :   a string, name of the user group to be added
            description         :   a string, the description of the user group
            quota               :   a boolean, by default the quot is not enabled
            group_enabled       :   a boolean, to activate or deactivate device
            shared_laptop_usage :   a boolean, to enable shared laptop usage
            quote_limit         :   a string, the size in GB of the group quota limit
        """
        if self.check_if_entity_exists("link", "Add user group"):
            self.log.info('Clicking on Add Usr Group Link')
            self.select_hyperlink("Add user group")

            self.log.info('Entering user group name')
            self.fill_form_by_id("userGroupName", group_name)
            if description:
                self.log.info('Entering user group description')
                self.fill_form_by_id("groupDesc", description)
            if quota:
                self.log.info('Enabling quota for User Group')
                self.checkbox_select("quotaEnabled")
                if quota_limit:
                    self.log.info('Setting up quota limit')
                    self.fill_form_by_id("quotaLimit", quota_limit)
            else:
                self.log.info('Disabling quota for User Group')
                self.checkbox_deselect("quotaEnabled")

            if group_enabled:
                self.log.info('Enabling User Group')
                self.checkbox_select("groupEnabled")
            else:
                self.log.info('Disabling User Group')
                self.checkbox_deselect("groupEnabled")

            if shared_laptop_usage:
                self.log.info('Enabling shared laptop usage for user group')
                self.checkbox_select("userCentric")
            else:
                self.log.info('Disabling shared laptop usage for user group')
                self.checkbox_deselect("userCentric")

            self.log.info('Submitting form for user group creation')
            self.submit_form()
            self.check_error_message()
        else:
            raise Exception("There is no option to add a new user group. Check if the user has"
                            " sufficient permissions")

    @adminconsole_handler()
    def action_delete_user_group(self, group_name, owner_transfer):
        """Deletes the user group with the given name
            group_name     :   a string, name of the user group to be deleted
            owner_transfer :   a user or user group to transfer ownership to before deleting
        """
        self.search_for(group_name)
        if self.check_if_entity_exists("link", group_name):
            self.select_action_item(group_name, 'Delete')
            self.click_yes_button()
            self.transfer_ownership(owner_transfer)
            self.check_error_message()
        else:
            raise Exception("There is no user group with the name " + group_name)

    @adminconsole_handler()
    def open_user_group(self, group_name):
        """Opens user group with the given name
            group_name   :   a string, name of the user group to be opened
        """
        self.search_for(group_name)
        if self.check_if_entity_exists("link", group_name):
            self.driver.find_element_by_link_text(group_name).click()
            self.wait_for_completion()
        else:
            raise Exception("There is no user group with the name " + group_name)

    @adminconsole_handler()
    def list_user_group(self):
        """Lists all the user groups
        :return:
            user_group_list     (list)  -- list of user groups
        """
        user_group_list = {}
        if self.cv_table_next_button_exists():
            if self.driver.find_element_by_xpath(
                    "//button[@ng-click='paginationApi.seek(1)']").is_enabled():
                self.cv_table_click_first_button()
                self.wait_for_completion()
        while True:
            elems = self.driver.find_elements_by_xpath(
                "//div[3]/div[2]/div[@class='ui-grid-canvas']/div")
            for elem in elems:
                key = elem.find_element_by_xpath("./div/div[1]/a").text
                value = elem.find_element_by_xpath("./div/div[2]/div").text
                user_group_list[key] = value
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        self.log.info(user_group_list)
        return user_group_list

        """
        Adds users to the given group
        :param group_name   :   a string, the name of the group to which the users must be added
        :param user_list    :   a list, the list of users to be added to the group
        :return:
        """
        self.search_for(group_name)
        if self.check_if_entity_exists("link", group_name):
            self.driver.find_element_by_xpath("//a[contains(text(),'" + group_name +
                                              "')]/../../div[5]/div/a/span").click()
            if self.check_if_entity_exists("link", "Add users"):
                self.driver.find_element_by_xpath(
                    "//a[contains(text(),'" + group_name +
                    "')]/../../div[5]/div/ul/li[1]/a").click()
                self.wait_for_completion()

                if self.driver.find_element_by_id("showSelected").is_selected():
                    self.driver.find_element_by_xpath("//input[@id='showSelected']/"
                                                      "following-sibling::label").click()
                    self.wait_for_completion()
                for user in user_list:
                    self.search_for(user)
                    elements = self.driver.find_elements_by_xpath("//div[@class='"
                                                                  "ui-grid-contents-wrapper']/"
                                                                  "div[2]/div[2]/div/div"
                                                                 )
                    i = 1
                    for element in elements:
                        if element.find_element_by_xpath("./div/div[1]/span").text == user:
                            break
                        i += 1
                    self.driver.find_element_by_xpath("//div[@class='ui-grid-contents-wrapper']/"
                                                      "div[1]/div/div[2]/div/div[{0}]//div[@role='"
                                                      "checkbox']".format(str(i))).click()
                self.driver.find_element_by_xpath("//*[@id='editUsersInGroupForm']/div[3]/"
                                                  "button[2]").click()
                self.wait_for_completion()
            else:
                raise Exception("There is no option to add users to the user group")
