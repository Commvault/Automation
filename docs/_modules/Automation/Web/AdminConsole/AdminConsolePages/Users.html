    @adminconsole_handler()
    def select_user(self, user):
        """
        Selects a user

        Args:
            user    (basestring):   the name of the user to select

        """
        self.search_for(user)
        if self.check_if_entity_exists("link", user):
            self.driver.find_element_by_xpath("//a[text()='"+user+"']").click()
            self.wait_for_completion()
        else:
            raise Exception("There is no user with the name " + user)

    @adminconsole_handler(hide_args=True)
    def add_user(self,
                 user_type,
                 email,
                 group=None,
                 username=None,
                 name=None,
                 system_password=False,
                 password=None,
                 external_provider=None,
                 invite_user=False):
        """Adds new user with the specified details to the list of users

        Args:
            user_type           (basestring):   type of user to add [local / external]

            email               (basestring):   the email of the user

            group               (basestring/List) :  one group name or list of user
                                                    groups to attach to

            username            (basestring):   the username of the user

            name                (basestring):   the display name of the user

            system_password     (bool):         if the system password needs to be used

            password            (basestring):   the password of the user

            external_provider   (basestring):   if external user, then the provider name

            invite_user         (bool):         if the users should be sent an email invite

        """
        if self.check_if_entity_exists("link", self.props['pageHeader.addUser']):
            self.select_hyperlink(self.props['pageHeader.addUser'])

            self.log.info('Selecting User type')
            self.select_radio(user_type)

            if user_type == self.props['label.localUser']:
                if name:
                    self.log.info('Entering fullname for the user')
                    self.fill_form_by_id("fullName", name)

                if username:
                    self.log.info('Entering username for the user')
                    self.fill_form_by_id("userName", username)

                self.log.info('Entering email for the user')
                self.fill_form_by_id("email", email)

                if not system_password:
                    self.log.info('Entering password for the user')
                    self.fill_form_by_id("password", password)

                    self.log.info('Confirming password for the user')
                    self.fill_form_by_id("confirmUserPassword", password)
                else:
                    self.log.info('Selecting checkbox to enable system password for the user')
                    self.checkbox_select("systemGeneratePassword")

            elif user_type == self.props['label.externalUser']:

                self.log.info('Selecting external provider for the user')
                self.cvselect_from_dropdown(self.props['label.externalProvider'],
                                            external_provider)

                self.log.info('Entering username for the user')
                self.driver.find_element_by_id("userName").send_keys(Keys.END + username)

                self.log.info('Entering provider mail for the user')
                self.fill_form_by_id("providerEmail", email)

            if group:
                self.log.info('Selecting User group associations for the User')
                self.cvselect_from_dropdown(self.props['label.userGroup'], group)

            if invite_user:
                self.log.info('Selecting checkbox for inviting User')
                self.checkbox_select("inviteUser")
            else:
                self.log.info('De-selecting checkbox for inviting User')
                self.checkbox_deselect("inviteUser")

            self.log.info('Submitting form to complete new user creation')
            self.submit_form()
            self.check_error_message()
        else:
            raise Exception("There is no option to add user")

    @adminconsole_handler()
    def delete_user(self, user_name, owner_transfer):
        """Deletes the specified user

        Args:
            user_name       (basestring):   username of the user that needs to be deleted

            owner_transfer  (basestring):   a user or user group to transfer ownership to

        """
        self.log.info('Selecting Delete from action item menu')
        self.select_action_item(user_name, self.props['action.delete'],
                                self.props['label.actions'])
        self.log.info('Confirming user deletion')
        self.click_yes_button()
        self.transfer_ownership(owner_transfer)
        self.log.info('User : %s is deleted successfully', user_name)

    @adminconsole_handler()
    def user_info(self):
        """Displays all the users present in the CS and their corresponding info

        Returns:
            user_list   (list):     list of users and their info

        """
        keys = ['UserName', 'Email', 'FullName', 'Plan']
        user_list = []
        while True:
            rows = self.driver.find_elements_by_xpath(
                "//div[@class='ui-grid-canvas']/div")
            for row in rows:
                user_info = {}
                columns = row.find_elements_by_xpath("./div/div")
                key = 0
                for index in range(1, len(columns)-1):
                    if index == 1 or index == 2 or index == 4:
                        val = row.find_element_by_xpath("./div/div["+str(index)+"]/span/a").text
                    else:
                        val = row.find_element_by_xpath("./div/div["+str(index)+"]/span").text
                    user_info[keys[key]] = val
                    key += 1
                user_list.append(user_info)
            if self.cv_table_next_button_exists():
                if self.driver.find_element_by_xpath(
                        "//button[@ng-disabled='cantPageForward()']").is_enabled():
                    self.cv_table_click_next_button()
                    self.wait_for_completion()
                    continue
                else:
                    break
            else:
                break
        user_list = [index for index in user_list if index != {}]
        self.log.info(user_list)
        return user_list

    @adminconsole_handler()
    def __select_user_type(self, user_type):
        """
        Selects All users or laptop users

        Args:
            user_type   (basestring):   type of user to select

        """
        self.driver.find_element_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/a/span[1]").click()
        self.wait_for_completion()
        elements = self.driver.find_elements_by_xpath(
            "//cv-grid/div/div[1]/div/div/span[1]/ul/li")
        pattern = re.compile(user_type, re.IGNORECASE)
        for li_elem in elements:
            if pattern.search(li_elem.find_element_by_xpath("./a/span").text):
                li_elem.find_element_by_xpath("./a/span").click()
                self.wait_for_completion()
                break

    @adminconsole_handler()
    def select_company(self, company_name):
        """
        Selects a company and displays its users

        Args:
            company_name    (basestring):   the name of the company to select

        """
        if self.check_if_entity_exists(
                "xpath", "//cv-company-dropdown/div/label/select"):
            Select(self.driver.find_element_by_xpath(
                "//cv-company-dropdown/div/label/select")).select_by_visible_text(company_name)
            self.wait_for_completion()

    @adminconsole_handler()
    def import_bulk_users(self, user_type, file_path, user_groups, system_password=False,
                          external_provider=None):
        """
        Imports a large no of users into the commcell
        """
        if not self.check_if_entity_exists("link", self.props['pageHeader.importBulkUsers']):
            raise Exception("There is no option to import bulk users. Please check the user"
                            "permissions")
        self.select_hyperlink(self.props['pageHeader.importBulkUsers'])
        self.select_value_from_dropdown("userType", user_type)
        element = self.driver.find_element_by_id("bulkImporter")
        self.driver.execute_script("arguments[0].style.visibility = 'visible'; "
                                   "arguments[0].style.height = '1px'; "
                                   "arguments[0].style.width = '1px'; "
                                   "arguments[0].style.opacity = 1", element)
        self.driver.find_element_by_id("bulkImporter").send_keys(file_path)

        if user_type == self.props['label.externalUser']:
            self.select_value_from_dropdown('externalProvider', external_provider)

        # Selecting all the imported users
        self.driver.find_element_by_xpath("//div[@data-ng-if='bulkUserCtrl.fileUploaded']//div["
                                          "@role='checkbox']").click()
        self.cvselect_from_dropdown(self.props['label.userGroup'], user_groups)

        if user_type == self.props['label.localUser'] and system_password:
            self.driver.find_element_by_id("systemGeneratePassword").click()
        self.submit_form()
        self.check_error_message()
