[docs]    def validate_plan_details(self):
        """validates if the plan details are displayed correctly"""

        try:
            for key, value in self.plan_name.items():
                self.plan_details.navigate_to_plan()
                if not self.__check_if_plan_exists(value):
                    exp = "Validation failed. Plan with Given name {0} does not exist".\
                    format(value)
                    raise Exception(exp)
                self.plan_details.select_plan(value)
                displayed_val = self.plan_details.plan_info()
                if not self.validatekeyedit:
                    displayed_val, validatekeydict = self.__set_default_values_if_none(
                        displayed_val, key, value, get_validatekeydict=True)
                elif self.validatekeyedit:
                    displayed_val, validatekeydict = self.__set_default_values_if_none(
                        displayed_val, key, value, get_validatekeydict="Post_Edit_Validation")

                validatekeydict = ast.literal_eval(validatekeydict)
                for key_dict, val_dict in validatekeydict.items():
                    if isinstance(val_dict, list):
                        self.log.info('Entity given_val "{0}"'.format(val_dict))
                        if set(displayed_val[key_dict]) == set(validatekeydict[key_dict]):
                            self.log.info(
                                "{0} displayed for {1} matches with {2} given" .format(
                                    displayed_val[key_dict], key, validatekeydict[key_dict]))
                        else:
                            exp = "{0} displayed for {1} does not match with {2} given " .format(
                                displayed_val[key_dict], key, validatekeydict[key_dict])
                            raise Exception(exp)
                    elif isinstance(val_dict, str):
                        self.log.info('Entity given_val "{0}"'.format(val_dict))
                        if displayed_val[key_dict] == validatekeydict[key_dict]:
                            self.log.info(
                                "{0} displayed for {1} matches with {2} given" .format(
                                    displayed_val[key_dict], key_dict, validatekeydict[key_dict]))
                        else:
                            exp = "{0} displayed for {1} does not match with {2} given " \
                                .format(displayed_val[key_dict], key_dict, validatekeydict\
                                        [key_dict])
                            raise Exception(exp)
                    else:
                        self.log.info('Entity given_val :{0}'.format(val_dict))
                        for item, valuedict in val_dict.items():
                            d_val = displayed_val[key_dict][item]
                            key_val = validatekeydict[key_dict][item]
                            if d_val == key_val:
                                self.log.info("{0} values match" .format(item))
                            else:
                                exp = "{0} displayed for {1} does not match with {2} given " \
                                    .format(d_val, item, key_val)
                                raise Exception(exp)
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)

    def __get_no_of_users(self):

        list_usergroups = []
        no_of_users = 0

        self.plan_details.navigate_to_users()
        for value in self.user_usergroup_association:
            self.plan_details.search_for(value)
            if not self.plan_details.check_if_entity_exists("link", value):
                list_usergroups.append(value)
            elif self.plan_details.check_if_entity_exists("link", value):
                no_of_users += 1

        if list_usergroups != []:
            for value in list_usergroups:
                self.plan_details.navigate_to_user_groups()
                self.plan_details.search_for(value)
                self.driver.find_element_by_link_text(value).click()
                user_grid = self.driver.find_element_by_xpath("//div[@class='ui-grid-canvas']")
                users_usergroup = user_grid.find_elements_by_xpath(
                    ".//div[@class='ui-grid-row ng-scope']")
                for user in users_usergroup:
                    no_of_users += 1
        return no_of_users
