    def __init__(self, driver=None, csdb=None):
        """
        Initializes the login helper module

        Args:
            driver (object) : the web driver object
            csdb (object)   : DBHelper object

        Raises:
             Exception:
                If the driver is not provided
        """
        self.driver = driver
        if not driver:
            raise Exception('Driver is not provided')

        self.login_page = load_module(
            'LoginPage',
            constants.AUTOMATION_DIRECTORY +
            '\\Web\\AdminConsole\\AdminConsolePages')

        self.login_obj = self.login_page.LoginPage(self.driver)
        self._log = logger.get_log()
        self._url = None
        self._username = None
        self._password = None
        self.csdb = csdb

    def _get_adminconsole_url(self, client_name=None):
        """
        Gets the client where adminconsole is installed

        Args:
            client_name (basestring):  the name of the client where adminconsole is installed

        Returns:
            adminconsole url
        """
        try:
            if client_name:
                query = "select net_hostname, name from app_client where id in (select ClientId " \
                        "from simInstalledPackages where simPackageID = 726)"
                self.csdb.execute(query)
                name_hostname = self.csdb.fetch_all_rows()
                for name in name_hostname:
                    if name[1].lower() == client_name.lower():
                        hostname = name
                        break

            elif client_name is None:
                query = "select top 1 ClientId from simInstalledPackages where simPackageID=726"
                self.csdb.execute(query)
                client_id = self.csdb.fetch_one_row()[0].strip()

                query = "select net_hostname from APP_Client where id = {0}".format(str(client_id))
                self.csdb.execute(query)
                hostname = self.csdb.fetch_one_row()
            url = "https://" + hostname[0] + "/adminconsole"

            return url
        except Exception as exp:
            self._log.exception(str(exp))
            raise exp


[docs]    def login(self, username, password, client_name=None, remember_me=False):
        """
        Helper method for logging into admin console

        Args:
            username (basestring)   : the username to be logged in with
            password (basestring)   : the password for the above user
            client_name (basestring): the name of the client where adminconsole is installed
			remember_me (bool)      : select/deselect the keep me logged in checkbox

        Raises:
            Exception:
                If it does not admin console url
        """
        try:
            self.url = self._get_adminconsole_url(client_name)
            self.username = username
            self.password = password
            self.login_obj.navigate(self.url)

            self._log.info("*" * 10 + " Login to Admin console " + "*" * 10)
            if remember_me:
                self.login_obj.login(self.username, self.password, True)
            else:
                self.login_obj.login(self.username, self.password)
        except Exception as exp:
            self._log.exception(str(exp))
            raise exp



[docs]    def re_login(self, username, password):
        """
        Re login to admin console after log out

        Raises:
            Exception:
                If it unable to login
        """
        try:
            self._log.info("Logging back in")
            self.login_obj.fill_form_by_id("username", username)
            self.login_obj.fill_form_by_id("password", password)
            self.login_obj.submit_login()
        except Exception as exp:
            self._log.exception(str(exp))
            raise exp
