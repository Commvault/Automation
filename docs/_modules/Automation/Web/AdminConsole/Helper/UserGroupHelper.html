            'UserGroups',

        self.log = logger.get_log()

        self._group_name = 'AutoTestGroup'
        self._description = 'AutoTestGroup'

        self._quota_limit = 50
        self._plan = None

        self._owner_transfer = 'master'

    def __check_if_user_group_exists(self, group_name):
        """
        checks if the given user already exists

        Args:
            user (str): Name of the user

        Returns:
            None

        Raises:
            Exception:
                None
        """
        try:

            self.user_groups.search_for(group_name)
            return self.user_groups.check_if_entity_exists("link", group_name)

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)

[docs]    def validate_user_group(self):
        """validates if the user group details are displayed correctly"""

        try:
            self.user_groups.navigate_to_user_groups()
            if not self.__check_if_user_group_exists(self.group_name):
                exp = "Validation failed. User with Given name {0} does not exist".\
                format(self.group_name)
                raise Exception(exp)
            self.user_groups.open_user_group(self.group_name)
            displayed_val = self.user_group_details.get_user_group_details()
            user_list = []
            user_details = self.user_group_details.list_users()
            for user in user_details:
                user_list.append(user['User name'])
            displayed_val.update({'Users':user_list})

            if not self._edit_flag:
                validatekeydict = self.__set_default_values_if_none(
                    get_validatekeydict='Pre-edit')
            else:
                validatekeydict = self.__set_default_values_if_none(
                    get_validatekeydict='Post-edit')

            validatekeydict = ast.literal_eval(validatekeydict)
            for key_dict, val_dict in validatekeydict.items():
                if isinstance(val_dict, list):
                    self.log.info('Entity given_val "{0}"'.format(val_dict))
                    if set(displayed_val[key_dict]) == set(validatekeydict[key_dict]):
                        self.log.info(
                            "{0} displayed for {1} matches with {2} given" .format(
                                displayed_val[key_dict], key_dict, validatekeydict[key_dict]))
                    else:
                        exp = "{0} displayed for {1} does not match with {2} given " .format(
                            displayed_val[key_dict], key_dict, validatekeydict[key_dict])
                        raise Exception(exp)
                elif isinstance(val_dict, str):
                    self.log.info('Entity given_val "{0}"'.format(val_dict))
                    if displayed_val[key_dict] == validatekeydict[key_dict]:
                        self.log.info(
                            "{0} displayed for {1} matches with {2} given" .format(
                                displayed_val[key_dict], key_dict, validatekeydict[key_dict]))
                    else:
                        exp = "{0} displayed for {1} does not match with {2} given " \
                            .format(displayed_val[key_dict], key_dict, validatekeydict\
                                    [key_dict])
                        raise Exception(exp)
                else:
                    self.log.info('Entity given_val :{0}'.format(val_dict))
                    for item in val_dict.items():
                        d_val = displayed_val[key_dict][item]
                        key_val = validatekeydict[key_dict][item]
                        if d_val == key_val:
                            self.log.info("{0} values match" .format(item))
                        else:
                            exp = "{0} displayed for {1} does not match with {2} given " \
                                .format(d_val, item, key_val)
                            raise Exception(exp)
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)


    def __set_default_values_if_none(self, get_validatekeydict):
        """this function sets default values to the parameters provided for user group creation
            for comparison against displayed values"""
        try:

            if self.group_enabled:
                enabled = 'Yes'
            else:
                enabled = 'No'

            if self.shared_laptop_usage:
                shared_laptop_usage = 'Yes'
            else:
                shared_laptop_usage = 'No'

            if get_validatekeydict == 'Pre-edit':
                validatekeydict = """{'Group name':'""" + str(self.group_name) + """',
                'Group description': '""" + str(self.description) + """',
                'Group enabled': '""" + str(enabled) + """',
                'Shared laptop usage': '"""+str(shared_laptop_usage)+"""',
                'Quota limit': '"""+str(self.quota_limit)+" GB"+"""',
                'Users':"""+str(self.user_list)+"""}"""

            elif get_validatekeydict == 'Post-edit':
                validatekeydict = """{'Group name':'""" + str(self.group_name) + """',
                'Group description': '""" + str(self.description) + """',
                'Group enabled': '""" + str(enabled) + """',
                'Shared laptop usage': '"""+str(shared_laptop_usage)+"""',
                'Plan': '"""+str(self.plan)+"""',
                'Users':"""+str(self.user_list)+"""}"""

            return validatekeydict

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)
