"""
This module provides the function or operations that can be used to run
basic operations on Users page.

To begin, create an instance of UserMain for users test case.

definition of AdminConsoleInfo

    def __check_if_user_exists(self, user):
        """
        checks if the given user already exists

        Args:
            user (str): Name of the user

        Returns:
            None

        Raises:
            Exception:
                None
        """
        try:

            self.users.search_for(user)
            return self.users.check_if_entity_exists("link", user)

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)


[docs]    def add_new_user(self):
        """calls add user method from Users page"""

        try:

            self.users.navigate_to_users()
            self.users.add_user(self.user_type,
                                self.email,
                                self.user_groups,
                                self.user_name,
                                self.full_name,
                                self.system_password,
                                self.password,
                                self.external_provider,
                                self.invite_user)
            if not self.user_type == 'External user':
                self._user_list.append(self.user_name)
                self._old_user_name = self.user_name

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def delete_user(self):
        """Calls method to delete the specified user"""
        try:

            if self._user_list:
                self.users.navigate_to_users()
                for user in self._user_list:
                    self.users.delete_user(user, self._owner_transfer)

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def edit_local_user_details(self):
        '''
        Calls method to edit user details
        '''
        try:

            self.users.navigate_to_users()
            self.users.select_user(self._old_user_name)
            self.user_details.edit_user(self.email,
                                        self.full_name,
                                        self.user_name,
                                        self.user_groups,
                                        self.plan,
                                        self.user_enabled,
                                        self.password,
                                        self.admin_password)
            if self.user_name != self._old_user_name:
                self._user_list.append(self.user_name)
                self._user_list.remove(self._old_user_name)

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def validate_user(self):
        """validates if the user details are displayed correctly"""

        try:
            self.users.navigate_to_users()
            if not self.__check_if_user_exists(self.user_name):
                exp = "Validation failed. User with Given name {0} does not exist".\
                format(self.user_name)
                raise Exception(exp)
            self.users.select_user(self.user_name)
            displayed_val = self.user_details.get_user_details()

            validatekeydict = self.__set_default_values_if_none(
                get_validatekeydict=True)

            validatekeydict = ast.literal_eval(validatekeydict)
            for key_dict, val_dict in validatekeydict.items():
                if isinstance(val_dict, list):
                    self.log.info('Entity given_val "{0}"'.format(val_dict))
                    if set(displayed_val[key_dict]) == set(validatekeydict[key_dict]):
                        self.log.info(
                            "{0} displayed for {1} matches with {2} given" .format(
                                displayed_val[key_dict], key_dict, validatekeydict[key_dict]))
                    else:
                        exp = "{0} displayed for {1} does not match with {2} given " .format(
                            displayed_val[key_dict], key_dict, validatekeydict[key_dict])
                        raise Exception(exp)
                elif isinstance(val_dict, str):
                    self.log.info('Entity given_val "{0}"'.format(val_dict))
                    if displayed_val[key_dict] == validatekeydict[key_dict]:
                        self.log.info(
                            "{0} displayed for {1} matches with {2} given" .format(
                                displayed_val[key_dict], key_dict, validatekeydict[key_dict]))
                    else:
                        exp = "{0} displayed for {1} does not match with {2} given " \
                            .format(displayed_val[key_dict], key_dict, validatekeydict\
                                    [key_dict])
                        raise Exception(exp)
                else:
                    self.log.info('Entity given_val :{0}'.format(val_dict))
                    for item in val_dict.items():
                        d_val = displayed_val[key_dict][item]
                        key_val = validatekeydict[key_dict][item]
                        if d_val == key_val:
                            self.log.info("{0} values match" .format(item))
                        else:
                            exp = "{0} displayed for {1} does not match with {2} given " \
                                .format(d_val, item, key_val)
                            raise Exception(exp)
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)


    def __set_default_values_if_none(self, get_validatekeydict=False):
        """this function sets default values to the parameters provided for user creation
            for comparison against displayed values"""
        try:

            if self.user_enabled:
                enabled = 'Yes'
            else:
                enabled = 'No'

            if self.full_name:
                full_name = self.full_name
            else:
                full_name = ''

            group_list = ''
            list_count = 0
            if self.user_groups:
                for group in self.user_groups:
                    list_count += 1
                    if list_count == 1:
                        group_list = group_list + group
                    else:
                        group_list = group_list +", "+group
            else:
                group_list = 'No group is found'

            if not self.plan:
                self.plan = 'No plan'

            if get_validatekeydict is True:
                validatekeydict = """{'Name':'""" + str(full_name) + """',
                'Email': '""" + str(self.email) + """',
                'Enabled': '""" + str(enabled) + """',
                'Group': '"""+str(group_list)+"""',
                'Plan': '"""+str(self.plan)+"""'}"""
            return validatekeydict

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)
