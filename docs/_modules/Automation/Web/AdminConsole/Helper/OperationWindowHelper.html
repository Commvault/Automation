"""
This module provides the function or operations that can be used to run
basic operations on Backup Window page.

To begin, create an instance of BackupRuleMain for Backup Rule test cases.

To initialize the class variables, pass the instance object to the appropriate
definition of AdminConsoleInfo

Call the required definition using the instance object.

navigate_to_operation   -- Navigates to Backup Window page
create_operation_rule   -- Add a configuration to the Backup Window page
modify_operation_rule   -- Modifies/Edits the Added Backup Rule
del_operation_rule      -- Deletes the Back up rule added by create_operation_rule method
                           after validation
validate_operation_rule -- Validates the Backup rule added by create_operation_rule method
"""

    def __init__(self, driver):
        '''
        Initialize method for BackupRuleMain
        '''
        self.driver = driver

        if not driver:
            raise Exception('Driver is not provided')

        self.operation_window_module = load_module(
            'OperationWindow',
            constants.AUTOMATION_DIRECTORY +
            '\\Web\\AdminConsole\\AdminConsolePages')
        self.operation_window = self.operation_window_module.OperationWindow(self.driver)

        self.log = logger.get_log()
        self._operation_rule_name = None
        self._old_operation_rule_name = None
        self._start_date = None
        self._end_date = None
        self._operation_day = dict.fromkeys(['Monday', 'Wednesday', 'Friday',
                                                'Sunday'], 1)
        self._operation_time = dict.fromkeys(['0', '1', '2', '3', '4', '14',
                                                '15', '16', '17', '18'], 1)
        self.edit_config = 0

    @property
    def operation_rule_name(self):
        """ Get operation rule name"""
        return self._operation_rule_name

    @operation_rule_name.setter
    def operation_rule_name(self, value):
        """ Set operation rule name"""
        self._operation_rule_name = value
        if self._old_operation_rule_name is None:
            self._old_operation_rule_name = value

[docs]    def check_if_operation_rule_exists(self, operation_rule_name):
        """Checks if operation rule exists

        Args:
            operation_rule (str): name for Backup rule to be opened

        Returns:
            True, False or None based on if conditions

        Raises:
            Exception:
                None
        """
        try:
            if self.operation_window.check_if_entity_exists("id", "search-field"):
                self.operation_window.search_for(operation_rule_name)
                if self.operation_window.check_if_entity_exists("link", operation_rule_name):
                    return True
                return False
            return None
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def create_operation_rule(self):
        """Calls the function to create Backup Rule"""

        try:
            self.operation_window.navigate_to_operation_window()
            self.operation_window.add_operation_rule(
                self.operation_rule_name,
                self.start_date,
                self.end_date,
                self.operation_day,
                self.operation_time)
            self.start_date = [self.start_date]
            self.end_date = [self.end_date]
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def modify_operation_rule(self):
        """Calls the function to modify/edit Backup Rule"""

        try:
            self.operation_window.navigate_to_operation_window()
            for start_date_val in self.start_date:
                for end_date_val in self.end_date:
                    self.edit_config += 1
                    self.operation_window.edit_operation_rule(
                        self.edit_config,
                        self._old_operation_rule_name,
                        self.operation_rule_name,
                        start_date_val,
                        end_date_val,
                        self.operation_day,
                        self.operation_time)
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)

[docs]    def validate_operation_rule(self):
        """Validates operation Rule, if values are retained correctly or not"""

        try:
            if not self.check_if_operation_rule_exists(self.operation_rule_name):
                exp = "Operation rule validation failed, operation rule with name {0} \
not displayed in Operation Window page".format(self.operation_rule_name)
                raise Exception(exp)
            self.operation_window.open_operation_rule(self.operation_rule_name)

            if not self.driver.find_element_by_id(
                    "operationRuleName").get_attribute('value') == self.operation_rule_name:
                exp = "Operation Rule validation failed,\
operation Rule name was not retained correctly"
                raise Exception(exp)
            self.log.info("Operation rule name is retained correctly")

            for key, value in self.operation_day.items():
                if value == 1:
                    if not self.driver.find_element_by_id(
                            "" + key + "").is_selected():
                        exp = "Operation Rule validation failed,\
operation day values were not retained correctly"
                        raise Exception(exp)
            self.log.info("Operation day values is retained correctly")

            for key, value in self.operation_time.items():
                if value == 1:
                    if not self.operation_window.check_if_entity_exists(
                            "xpath",
                            "//td[@id='"+key+"' and @class='ng-scope eng-selected-item']"):
                        exp = "Operation Rule validation failed,\
operation time values were not retained correctly"
                        raise Exception(exp)
            self.log.info("Operation time values is retained correctly")

            if ((self.start_date[0]) or (self.end_date[0])):
                if self.driver.find_element_by_id("enableDate").is_selected:
                    if self.start_date[0]:
                        if not self.driver.find_element_by_id("startDate").\
                        get_attribute('value') == self.start_date[0]:
                            exp = "Operation Rule validation failed,\
start date was not retained correctly"
                            raise Exception(exp)
                        self.log.info("Start date is retained correctly")
                    if self.end_date[0]:
                        if not self.driver.find_element_by_id("endDate").\
                        get_attribute('value') == self.end_date[0]:
                            exp = "Operation Rule validation failed,\
end date was not retained correctly"
                            raise Exception(exp)
                        self.log.info("End date is retained correctly")
                else:
                    exp = "Operation Rule validation failed,\
date values were not retained correctly"
                    raise Exception(exp)

            self.log.info("Operation rule validated successfully, all values were correctly\
retained")

            self.driver.find_element_by_xpath(
                "//button[contains(text(),'Cancel')]").click()
            self.operation_window.wait_for_completion()

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)
