"""
This module provides the function or operations that can be used to run
basic operations on companies or subscriptions page.

To begin, create an instance of CompanyMain for companies test case.

To initialize the class variables, pass the instance object to the appropriate
definition of AdminConsoleInfo

Call the required definition using the instance object, with no arguments to
be passed, only the flags needs to be passed into the method call while calling
the member function.


Class:

    CompanyMain() -> Comapanies() -> object()

check_if_company_exists()     -- Checks if the given company already exists or not
addcompany()                  -- Calls method add_company from companies class
deletecompany()               -- Calls methods to de-associate plans and delete company
                                 from CompanyDetails & Company class
validatecompany()             -- Validates the values displayed for company against
                                 values provided as input
__set_default_values_if_none()-- provides defaults for values provided to compare
                                 with values displayed
edit_company_details()        -- Edits company tiles from company details page
"""

[docs]    def check_if_company_exists(self):
        """
            Method to check if the given company exists
        """
        try:
            company_exists = False
            self.companies.search_for(self.company_name)
            if self.companies.check_if_entity_exists("link", self.company_name):
                company_exists = True
            return company_exists
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)



[docs]    def select_company(self):
        """
            Method to search for a given company and open company details page
        """
        self.log.info("Selecting the company")

        if self.check_if_company_exists():
            self.driver.find_element_by_xpath(
                "//a[text()='" + self.company_name + "']").click()
            self.companies.wait_for_completion()
        else:
            exp = "There is no company with the given name"
            self.log.exception(exp)
            raise Exception(exp)

[docs]    def validate_company(self):
        """
            Method to validate displayed values against given values
        """
        try:
            self.companies.navigate_to_companies()
            if not self.check_if_company_exists():
                exp = "Company with name {0} does not exist, validation failed".\
                format(self.company_name)
                self.log.exception(exp)
                raise Exception(exp)
            self.select_company()
            displayed_val = self.companies.company_info()

            if self._validatekeydict:
                displayed_val, validatekeydict = self.__set_default_values_if_none\
                (displayed_val, get_validatekeydict="Post-edit")
            else:
                displayed_val, validatekeydict = self.__set_default_values_if_none\
                (displayed_val, get_validatekeydict="Pre-edit")
            validatekeydict = ast.literal_eval(validatekeydict)

            for key, value in validatekeydict.items():
                if isinstance(value, list):
                    count = 0
                    self.log.info('Entity given val "{0}"'.format(value))
                    if (key == 'Plans' and
                            displayed_val[key] != "None" and
                            validatekeydict[key] != "None"):
                        planname = sep = rest = ""
                        validatekeydict[key] = sorted(validatekeydict[key])
                        maxval = max(len(displayed_val[key]), len(validatekeydict[key]))
                        for plan in sorted(displayed_val[key]):
                            if count < maxval:
                                planname, sep, rest = plan.partition('(Laptop Plan )')
                                if sep == '':
                                    planname, sep, rest = plan.partition(\
                                                                '(FSServer Plan )')
                                if sep == '':
                                    planname, sep, rest = plan.partition(\
                                                                '(Server Plan )')
                                if sep == '':
                                    planname, sep, rest = plan.partition(\
                                                                '(Database Plan )')
                                if str(planname).strip() == validatekeydict[key][count]:
                                    self.log.info("{0} displayed for {1} matches"
                                                  .format(plan, key))
                                else:
                                    exp = "{0} displayed for {1} does not match \
                                    with {2}".format(key, plan, validatekeydict[key][count])
                                    self.log.exception(exp)
                                    raise Exception(exp)
                            else:
                                break
                            count += 1
                    else:
                        if set(displayed_val[key]) == set(validatekeydict[key]):
                            self.log.info("{0} displayed for {1} matches with {2} given"
                                          .format(displayed_val[key], key, validatekeydict[key]))
                        else:
                            exp = "{0} displayed for {1} does not match with {2} given".\
                                    format(displayed_val[key], key, validatekeydict[key])
                            self.log.exception(exp)
                            raise Exception(exp)
                elif isinstance(value, str):
                    if displayed_val[key] == validatekeydict[key]:
                        self.log.info("{0} displayed for {1} matches with {2} given"
                                      .format(displayed_val[key], key, validatekeydict[key]))
                    else:
                        exp = "{0} displayed for {1} does not match with {2} given ".\
                        format(displayed_val[key], key, validatekeydict[key])
                        self.log.exception(exp)
                        raise Exception(exp)
                else:
                    self.log.info('Entity displayed_val :{0}'.format(value))
                    for item, valuedict in value.items():
                        d_val = displayed_val[key][item]
                        key_val = validatekeydict[key][item]
                        if d_val == key_val:
                            self.log.info("{0} values match" .format(item))
                        else:
                            exp = "{0} displayed for {1} does not match with {2} given".\
                            format(d_val, item, key_val)
                            self.log.exception(exp)
                            raise Exception(exp)
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)


    def __set_default_values_if_none(self, displayed_val, get_validatekeydict=None):
        """Method to set default values for the company details"""
        try:
            default_comp_values = {"Requires authcode for installation":"OFF",
                                   "Secondary domain name":"Not set",
                                   "Navigation preferences":"Navigation customization",
                                   "Email settings":"Not configured",
                                   "Plans":"None",
                                   "Shared Laptop Usage":"OFF",
                                   "Use UPN instead of e-mail":"OFF"}
            for key, value in displayed_val.items():
                if isinstance(value, list):
                    if displayed_val[key] == [None]:
                        displayed_val[key] = default_comp_values[key]
                elif isinstance(value, str):
                    if displayed_val[key] == [None]:
                        displayed_val[key] = default_comp_values[key]
                else:
                    for item, valuedict in value.items():
                        if displayed_val[key][item] is None:
                            displayed_val[key][item] = default_comp_values[key][item]

            _query = "select attrVal from App_CompanyProp where componentNameId=\
                (select id from UMDSProviders where serviceType=5 and domainName=\
                '{0}') and attrName='Creation Time'".format(self.company_alias)
            self.csdb.execute(_query)
            unix_company_time = self.csdb.fetch_one_row()
            creation_time = datetime.datetime.fromtimestamp(int(unix_company_time[0]))
            comp_creation_time = creation_time.strftime('%b %#d, %#I:%M:%S %p')

            if get_validatekeydict == "Pre-edit":
                validatekeydict = """{'CompanyName': '"""+str(self.company_name)+"""',
                'Contacts': {'Contact name': '"""+str(self.contact_name)+"""'},
                'Plans':"""+str(self.plans)+""",
                'Email settings':['"""+str(default_comp_values["Email settings"])+"""'],
                'General': {'Company created on': '"""+str(comp_creation_time)+"""',
                'Company alias':'"""+str(self.company_alias)+"""',
                'Associated SMTP':'"""+str(self.smtp)+"""',
                'Requires authcode for installation': '"""+str(default_comp_values[\
                                        "Requires authcode for installation"])+"""'},
                'Navigation preferences': ['"""+str(default_comp_values[\
                                                "Navigation preferences"])+"""'],
                'Sites': {'Primary site name': '"""+str(self.primary_site)+"""',
                'Secondary site name':'"""+str(default_comp_values[\
                                                "Secondary domain name"])+"""'}}"""
                return displayed_val, validatekeydict

            elif get_validatekeydict == "Post-edit":
                validatekeydict = """{'CompanyName': '"""+str(self.company_name)+"""',
                'Contacts': {'Contact name': '"""+str(self.contact_name)+"""'},
                'Plans':"""+str(self.plans)+""",
                'Email settings':{'Sender name':'"""+str(self.sender_name)+"""',
                'Sender email':'"""+str(self.sender_email)+"""'},
                'General': {'Company created on': '"""+str(comp_creation_time)+"""',
                'Company alias':'"""+str(self.company_alias)+"""',
                'Associated SMTP':'"""+str(self.general_settings['smtp'])+"""',
                'Requires authcode for installation': '"""+str(self.general_settings\
                ['authcode'])+"""',
                'Use UPN instead of e-mail':'"""+str(self.general_settings['UPN'])+"""'},
                'Navigation preferences': ['"""+str(default_comp_values[\
                                                "Navigation preferences"])+"""'],
                'Sites': {'Primary site name': '"""+str(self.primary_site)+"""',
                'Secondary site name':'"""+str(self.secondary_sites)+"""'}}"""
                return displayed_val, validatekeydict

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)
