"""
This module provides the function or operations that can be used to run
basic operations on Global Exceptions page.

To begin, create an instance of GlobalExceptionsMain for Global exceptions test cases.

To initialize the class variables, pass the instance object to the appropriate
definition of AdminConsoleInfo

Call the required definition using the instance object.

navigate_to_global_exceptions  	-- Navigates to Global exceptions page
create_global_filter   	-- Add a new windows or unix global filter
modify_global_filter   	-- Modifies/Edits the Added windows or unix  global filter
del_global_filter      	-- Deletes the global filter added by
                                   create_global_filter method after validation
validate_global_filter 	-- Validates the global filter added by create_global_filter
"""

from Web.AdminConsole.Helper.LoadModule import load_module

[docs]class GlobalExceptionsMain(object):
    """Helper file to provide arguments and handle function call to main file"""

    def __init__(self, driver=None):
        """Initialize method for GlobalExceptionsMainMain """
        self.driver = driver

        if not driver:
            raise Exception('Driver is not provided')

        self.global_exceptions_module = load_module(
            'global_exceptions',
            constants.AUTOMATION_DIRECTORY +
            '\\Web\\AdminConsole\\AdminConsolePages')
        self.global_exceptions = self.global_exceptions_module.GlobalExceptions(self.driver)

        self.log = logger.get_log()

        self._global_filter_path = {'windows_global_filter_path':'C:\test1.txt',
                                    'unix_global_filter_path':'root/test1.bat'}
        self._old_global_filter_path = None

    @property
    def global_filter_path(self):
        """ Get global filter path"""
        return self._global_filter_path

    @global_filter_path.setter
    def global_filter_path(self, value):
        """ Set windows global filter path"""
        self._global_filter_path = value
        if self._old_global_filter_path is None:
            self._old_global_filter_path = value


[docs]    def check_if_global_filter_exists(self, global_filter_path):
        """" checks if  global filter exists

        Args :

            global_filter_path(dict) : dictionary with path of the global filter to be opened

        Returns :
        True, False or None based on if conditions

        Raises:
            Exception:
                None
        """

        try:
            if self.global_exceptions.check_if_entity_exists("link", global_filter_path):
                return True
            return False
        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)

    def _get_filter_value(self, value):
        """to get the attribute value"""
        self.global_exceptions.open_global_filter(value)
        filter_value = self.driver.find_element_by_id("filterEditData").get_attribute('value')
        return filter_value


[docs]    def validate_global_filter(self):
        """Validates  global path , if path is retained correctly or not"""

        try:
            if self.global_filter_path['windows_global_filter_path']:
                if not self._get_filter_value(
                        self.global_filter_path['windows_global_filter_path']) == \
                        self.global_filter_path['windows_global_filter_path']:
                    exp = "windows global exception validation failed, windows global filter path " \
                          "was not retained correctly"
                    raise Exception(exp)
                self.log.info("Windows Global filter path is retained correctly")
                self.global_exceptions.cancel_form()

            if self.global_filter_path['unix_global_filter_path']:
                if not self._get_filter_value(
                        self.global_filter_path['unix_global_filter_path']) == \
                        self.global_filter_path['unix_global_filter_path']:
                    exp = "validation failed, unix global filter path was not retained correctly"
                    raise Exception(exp)
                self.log.info("Unix Global filter path is retained correctly")
                self.global_exceptions.cancel_form()

        except Exception as exp:
            error = pprint_handler_errors(exp)
            raise Exception(error)
