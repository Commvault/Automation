    @adminconsole_handler()
    def add_nas_client(
            self,
            nas_server_name,
            nas_host_name,
            ndmp_ida_=None,
            cifs_ida=None,
            nfs_ida=None,
            ndmp_username=None,
            ndmp_password=None,
            ndmp_port=10000,
            accessnode_ma=None):
        """
        Adds a new NAS File Server with the Chosen iDAs and Access Node
        Server name is the name the user gives the NAS File Server
        Hostname is the actual name of the File server.
        The hostname can either be IP, FQDN or short name if the File Server is in the
        same domain as the CS
        Access Node is one of the MAs which is designated for Communication with File Server
        NAS Client can have 3 iDAs -- NDMP, CIFS and NFS.
        When NDMP is selected (it consumes NAS Client License) which asks for NDMP username & pwd


        Args:
            nas_server_name  (basestring):  the client name of the NAS server to be opened
            nas_host_name  (basestring):  the host name of the NAS server to be opened
            ndmp_ida_       (integer):  0 or 1 to select or deselect NDMP iDA
            cifs_ida        (integer):  0 or 1 to select ordeselect CIFS iDA
            nfs_ida _       (integer):  0 or 1 to select or deselect NFS iDA
            ndmp_username   (basestring): the NDMP user name
            ndmp_password   (basestring): the NDMP password
            ndmp_port       (integer): NDMP Port number on the File Server -- default : 10000
            accessnode_ma   (basestring): Detect MA for the File Server(Optional)
        """
        ma_cnt = 0
        opt = 0
        self.log.info("Adding NAS File Server " + nas_host_name + " as " + nas_server_name)
        self.select_hyperlink("Add NAS server")
        self.fill_form_by_id("hostName", nas_host_name)
        self.fill_form_by_id("clientName", nas_server_name)
        if ndmp_ida_:
            self.log.info("Adding NDMP iDA")
            self.checkbox_select("nasIda")
            self.wait_for_completion()
            self.fill_form_by_id("ndmpLogin", ndmp_username)
            self.fill_form_by_id("ndmpPassword", ndmp_password)
            if ndmp_port != 10000:
                self.fill_form_by_id("ndmpPort", ndmp_port)
            self.log.info("Checking list of MAs")
            element = self.driver.find_element_by_id('detectMa')
            for option in element.find_elements_by_tag_name('option'):
                ma_cnt = ma_cnt + 1
                cmptxt = option.text
                cmptxt = cmptxt[17 : len(cmptxt) - 14]
                if not accessnode_ma and str(cmptxt) == str(accessnode_ma):
                    opt = ma_cnt
            self.log.info("No. of MAs shown in Detect MA " + str(ma_cnt-1))
            self.log.info("Detect MA in sequence.." + str(opt))
            if ma_cnt == 1:
                self.log.info("Error in populating list of MAs")
                raise Exception("Error in populating list of MAs")
            elif ma_cnt > 1:
                self.log.info("No. of MAs shown in Detect MA" + str(ma_cnt - 1))
                self.log.info("No MA has been passed. Thus CS has been selected as Detect MA")
            elif ma_cnt > 1 and opt > 1:
                self.log.info("No. of MAs shown in Detect MA" + str(ma_cnt - 1))
                self.log.info("Explicit MA has been selected as Detect MA -- " + accessnode_ma)
                self.select_value_from_dropdown("detectMa", str(opt), True)
            else:
                self.log.info("No MAs listed for Detect MA.....")
                raise Exception("No MAs listed for Detect MA.....")
        if cifs_ida:
            self.log.info("Adding CIFS iDA")
            self.checkbox_select("windowsIda")
        if nfs_ida:
            self.log.info("Adding NFS iDA")
            self.checkbox_select("unixIda")
        self.click_button("Save")

    @adminconsole_handler()
    def action_release_license(self, server_name):
        """"
        Release CIFS and NFS Licenses

        Args:
            server_name  (basestring):  the name of the server to be opened
        """
        self.select_action_item(server_name, "Release license")
        if ("Server File System - Linux File System" in
                self.driver.find_element_by_tag_name("html").text):
            self.checkbox_select("Server File System - Linux File System")
        if ("Server File System - Windows File System" in
                self.driver.find_element_by_tag_name("html").text):
            self.checkbox_select("Server File System - Windows File System")
        if "NDMP - NDMP" in self.driver.find_element_by_tag_name("html").text:
            self.checkbox_select("NDMP - NDMP")
        self.click_button("OK")
        self.log.info("Release license successful on Client:" + server_name)


    @adminconsole_handler()
    def action_backup(self, server_name, idataagent, subclient_name, backup_type):
        """"
        Running backup of subClients under defaultbackupset

        Args:
            server_name  (basestring):  the name of the server to be opened
            idataagent  (basestring):   the name of the iDA (NDMP or CIFS or NFS)
            subclient_name (basestring):    name of the subclient to be backed up
            backup_type (basestring):   the backup type to be run
        """

        self.search_for(server_name)
        self.driver.find_element_by_link_text("Backup").click()
        self.wait_for_completion()
        if idataagent == 'NDMP':
            self.log.info("Finding for SubClient under NDMP Agent")
            self.driver.find_element_by_xpath \
                ("//label[text()='" + subclient_name + "']/../..//input").click()
            self.wait_for_completion()
        elif idataagent == 'CIFS':
            self.log.info("Finding for subClient under CIFS Agent")
            self.driver.find_element_by_xpath \
                ("//label[text()='" + subclient_name + "']/../..//input").click()
            self.wait_for_completion()
        elif idataagent == 'NFS':
            self.log.info("Finding for subClient under NFS Agent")
            self.driver.find_element_by_xpath \
                ("//label[text()='" + subclient_name + "']/../..//input").click()
            self.wait_for_completion()
        else:
            raise Exception("Invalid iDA name passed")
        self.driver.find_element_by_xpath("//button[@type='submit']").click()
        self.wait_for_completion()
        self.submit_backup(backup_type)
        job_id = self.backup_job_id()
        return job_id


    @adminconsole_handler()
    def action_restore(self, server_name, idataagent, subclient_name):
        """"
            Running Restore of Data backed up by SubClient under defaultbackupSet of NAS Client

            Args:
                server_name  (basestring):  the name of the server to be opened
                idataagent  (basestring):   the name of the iDA (NDMP or CIFS or NFS)
                subclient_name (basestring):    name of the subclient to be backed up
        """

        self.log.info("Checking Action Item.....")
        self.search_for(server_name)
        #self.select_hyperlink("Restore")
        self.driver.find_element_by_xpath("*//a[.='Restore']/..").click()
        self.wait_for_completion()
        self.log.info("Clicked on Restore")
        if "Backup content" in self.driver.find_element_by_tag_name("html").text:
            self.log.info("Looks like there is only one backed up subclient")
            self.wait_for_completion()
        elif "Please select a subclient or instance to restore" \
                in self.driver.find_element_by_tag_name("html").text:
            self.log.info("Multiple subClients present")
            if idataagent == 'NDMP':
                self.log.info("SubClient  under NDMP Agent")
                self.driver.find_element_by_xpath\
                    ("//label[text()='" + subclient_name + "']/../..//input").click()
                self.wait_for_completion()
            elif idataagent == 'Windows File System':
                self.log.info("Finding for subClient under CIFS Agent")
                self.driver.find_element_by_xpath\
                    ("//label[text()='" + subclient_name + "']/../..//input").click()
                self.wait_for_completion()
            elif idataagent == 'Linux File System':
                self.driver.find_element_by_xpath("//label[text()='" + subclient_name +
                                                  "']/../..//input").click()
                self.wait_for_completion()
            else:
                self.log.info("Invalid iDA passed")
                raise Exception("Invalid Agent Passed")
            self.click_button("Select")
        else:
            self.log.info("Looks like there is nothing to Restore")
            return
