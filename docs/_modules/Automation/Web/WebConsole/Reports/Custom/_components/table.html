class TableViewer(CRComponentViewer):
    """Actions common to Pivot and Table Viewer go here"""

    @WebAction()
    def __click_export_to_csv(self):
        """Click the export to CSV on the Table"""
        button = self._driver.find_element_by_xpath(
            self._x + "//*[@title='Export to CSV']"
        )
        button.click()

    @WebAction()
    def __click_additional_columns(self):
        """Click Additional columns dropdown"""
        dropdown = self._driver.find_element_by_xpath(
            self._x + "//button[@title='Add or remove columns']")
        dropdown.click()

    @WebAction()
    def __click_column_on_additional_col_menu(self, column_name):
        """Click column on hidden column dropdown"""
        checkbox = self._driver.find_element_by_xpath(
            self._x + "//*[@id='previewTable_Colums']//*[.='%s']" %
            column_name
        )
        checkbox.click()

    @WebAction()
    def __is_additional_column_enabled(self, column_name):
        """Return true if additional column is enabled"""
        checkbox = self._driver.find_element_by_xpath(
            self._x + "//*[@id='previewTable_Colums']//*[.='%s']/../i" %
            column_name
        )
        return "ui-grid-icon-ok" in checkbox.get_attribute("class")

    @WebAction()
    def __read_additional_columns(self):
        """Read available hidden columns"""
        columns_names = self._driver.find_element_by_xpath(
            self._x + "//*[@id='previewTable_Colums']//i"
        )
        return [columns_name.text for columns_name in columns_names]

    @WebAction(log=False)
    def __read_table_row_data(self, row_index, include_hidden=False):
        """Read all the previewed data from table"""
        hidden_xp = "" if include_hidden else "[not(contains(@class, 'ng-hide'))]"
        rows = self._driver.find_elements_by_xpath(
            "(" + self._x +
            f"//tbody/tr[not(@id='filterRow_previewTable')])[{row_index}]"
            "//td" + hidden_xp
        )
        return [
            row.text.strip()
            if row.text.strip() != ""
            else row.get_attribute("title")
            for row in rows
        ]

    @WebAction()
    def __get_row_count(self):
        """Get number of rows (including table header)"""
        return len(self._driver.find_elements_by_xpath(
            self._x + "//table//tr[not(@id='filterRow_previewTable')]"
        ))

    @WebAction()
    def __get_column_names(self, include_hidden=False):
        """Read all column names from table"""
        col_xp = "//th" + (
            "" if include_hidden else "[not(contains(@class, 'ng-hide'))]"
        ) + "//*[@title]"
        columns = self._driver.find_elements_by_xpath(self._x + col_xp)
        return [
            column.text.strip()
            if column.text.strip() != ""
            else column.get_attribute("title")
            for column in columns
        ]

    @WebAction()
    def __click_row_select(self):
        """click on number of rows  """
        tbl_length_xp = self._x + "//select[contains(@class, 'page-size')]"
        select_obj = self._driver.find_element_by_xpath(tbl_length_xp)
        select_obj.click()

    @WebAction()
    def __select_rows(self, number_of_results):
        """Select number of rows to be displayed on table"""
        tbl_length_xp = self._x + "//select[contains(@class, 'page-size')]"
        Select(
            self._driver.find_element_by_xpath(tbl_length_xp)
        ).select_by_visible_text(
            str(number_of_results)
        )

    @WebAction()
    def __show_table_menu(self):
        """Show table menu"""
        menu_xp = self._x + "//*[@title='Show Table Menu']"
        menu = self._driver.find_element_by_xpath(menu_xp)
        menu.click()
        sleep(2)

    @WebAction()
    def __click_configure_alert(self):
        """Click on configure alert of table"""
        button = self._driver.find_element_by_xpath(
            self._x + "//*[@title='Configure Alert']"
        )
        button.click()

    @WebAction()
    def __is_filter_enabled(self):
        """Check if filter is enabled"""
        filter_xp = self._x + "//tr[contains(@id,'filterRow_previewTable')]"
        filter_inp = self._driver.find_elements_by_xpath(filter_xp)
        if filter_inp:
            return True
        return False

    @WebAction()
    def __apply_filter(self, column_name, value):
        """Applies the filter to the corresponding column """
        filter_xp = "//*[@id='{0}']//tr[@id='filterRow_previewTable']" \
                    "/td[count(//*[@title='{1}']/../preceding-sibling::th)+1]//input"\
            .format(self.id, column_name)
        filter_ = self._driver.find_element_by_xpath(filter_xp)
        filter_.clear()
        filter_.send_keys(value)
        filter_.send_keys(Keys.RETURN)
        sleep(2)

    @WebAction()
    def __get_cell_attributes(self, columns):
        """Fetches cell value and its attributes."""
        dict_ = dict()
        for column in columns:
            list_ = list()
            col_xpath = "//*[@id='{0}']//td[count(//*[@id='{0}']//*[@title='{1}']" \
                        "/../preceding-sibling::th)+1]" \
                .format(self.id, column)
            cells = self._driver.find_elements_by_xpath(col_xpath)
            for cell in cells:
                attributes = dict()
                attributes['bg_color'] = cell.value_of_css_property('background-color')
                attributes['font_color'] = cell.value_of_css_property('color')
                attributes['font_size'] = cell.value_of_css_property('font-size')
                attributes['data'] = cell.text.strip()
                list_.append(attributes)

            dict_[column] = list_
        return dict_

    @WebAction()
    def __enable_filter(self):
        """Enables filter on the table"""
        filter_icon = self._x +\
            "//*[@class='compButtons  icon-filter hideOnExportFriendly ng-scope']"
        filter_ = self._driver.find_element_by_xpath(filter_icon)
        filter_.click()

    @WebAction()
    def __fetch_table_title(self):
        """Fetches the table title"""
        title = self._driver.find_element_by_xpath(
            self._x + "//div[@class='reportstabletitle panel-table-title']/span")
        return title.text

    @PageService()
    def get_table_rows(self):
        """Get all the table data"""
        return [
            self.__read_table_row_data(row_index)
            for row_index in range(1, self.__get_row_count())
        ]

    @PageService()
    def get_table_columns(self):
        """Get all visible column names"""
        return self.__get_column_names()

    @PageService()
    def set_number_of_rows(self, number_of_results=50):
        """
        selects the number of rows to be shown in table
        Args:
            number_of_results: value to select in number of rows
        """
        self.__click_row_select()
        self.__select_rows(number_of_results)
        self.__click_row_select()  # to close the panel
        self._webconsole.wait_till_load_complete()

    @PageService()
    def export_to_csv(self):
        """Export to CSV using the Table Level CSV export option"""
        self.__show_table_menu()
        self.__click_export_to_csv()
        self._webconsole.wait_till_load_complete()

    @PageService()
    def configure_alert(self):
        """Opens configure alert window"""
        self.__show_table_menu()
        self.__click_configure_alert()

    @PageService()
    def enable_hidden_column(self, column_name):
        """Enable hidden column"""
        self.__click_additional_columns()
        if column_name not in self.__read_additional_columns():
            raise CVWebAutomationException(
                "Column [%s] not found inside Table [%s]" % (
                    column_name, self.id
                )
            )
        if self.__is_additional_column_enabled(column_name) is False:
            self.__click_column_on_additional_col_menu(column_name)
        self.__click_additional_columns()

    @PageService()
    def get_table_data(self, show_hidden_data=False):
        """Dump all data in the table as json"""
        cols = self.__get_column_names(include_hidden=show_hidden_data)
        rows = [
            self.__read_table_row_data(row_index, include_hidden=show_hidden_data)
            for row_index in range(1, self.__get_row_count())
        ]
        if rows:
            return dict(zip(cols, map(list, zip(*rows))))

        return dict(zip(cols, [[]] * len(cols)))

    @PageService()
    def set_filter(self, column_name, value):
        """Send string in filter. this will enable filter if not enabled

        Args:
            column_name: name of the column to set filter

            value: value to be sent in filter

        """
        if self.__is_filter_enabled() is False:
            self.__enable_filter()
        self.__apply_filter(column_name, value)

    @PageService()
    def get_attributed_table_data(self):
        """Fetches the table data along with style information."""
        columns = self.__get_column_names(include_hidden=False)
        return self.__get_cell_attributes(columns)

    @PageService()
    def get_table_title(self):
        """Returns the table title"""
        return self.__fetch_table_title()


class TableProperties(CRComponentProperties):
    """Actions common to Pivot and Table Properties panel go here"""

    @WebAction()
    def __enter_custom_code(self, title, script):
        """Types in the passed script of the table component."""
        table = self._driver.find_element_by_xpath("//label[@title='%s']/..//textarea" % title)
        table.send_keys(script)

    @PageService()
    def set_cell_expression(self, script):
        """Sets cell expression

        Args:
            script (str)    : script which is to be entered

        """
        self._select_current_component()
        self._click_scripts_tab()
        self.__enter_custom_code("Cell Expression (evalauted for all columns)", script)

    @PageService()
    def set_row_style(self, script):
        """Sets row style

        Args:
            script (str)    : script which is to be entered

        """
        self._select_current_component()
        self._click_scripts_tab()
        self.__enter_custom_code("Row Style", script)


class DataTableBuilder(TableBuilder):
    """This class contains all the builder actions specific to Data Table"""

    @property
    def category(self):
        """
        Returns:Category type as 'Table'
        """
        return "Table"

    @property
    def name(self):
        """
        Returns:Name as 'Data Table'
        """
        return "Data Table"

    @WebAction()
    def __drag_dataset_to_table(self):
        """Drag dataset to table"""
        self._drag_dataset_to_component(self._x + "//div[@id='table_%s']" % self.id)

    @WebAction()
    def __drag_column_to_table(self, column_name):
        """Drag column to table"""
        self._drag_column_from_dataset(column_name, self._x + "//div[@id='table_%s']" % self.id)

    @WebAction()
    def __click_add_button(self):
        """Clicks add button on the table"""
        button = self._driver.find_element_by_xpath(self._x + "//button[@title='Add New Button']")
        button.click()

    @WebAction()
    def __get_button_id(self):
        """Retrieve component ID"""
        elements = self._driver.find_elements_by_xpath(self._x + "//div[@class='buttonPanel ng-scope']//ul/li")
        button = elements[-2].find_element_by_xpath(".//button")
        return button.get_attribute('id').strip()

    @PageService()
    def add_column_from_dataset(self, column=None):
        """Add column from associated dataset to table

        Args:
            column (iterable): Any iterable of column name
        """
        if column:
            self._validate_dataset_column(self.dataset_name, column)
            self.__drag_column_to_table(column)
        else:
            self.__drag_dataset_to_table()
        self._webconsole.wait_till_line_load()

    @PageService()
    def add_column(self, column):
        """Adds Column to the table

        Args:
            column (obj): Object of type column which is to be added.

        """
        if not isinstance(column, Column):
            raise TypeError("Invalid component type")
        self._select_current_component()
        self.__drag_column_to_table(column.title)
        self._webconsole.wait_till_line_load()
        xpath = self._x + "//th/span[@title='%s']" % column.title
        column.configure_builder_component(self._webconsole, self.dataset_name,
                                           self.page_name, column.title, xpath)

    @PageService()
    def add_button(self, button):
        """Adds Button to the table

        Args:
            button (obj): Object of type button which is to be added.

        """

        if not isinstance(button, Button):
            raise TypeError("Invalid component type")
        self.__click_add_button()
        id_ = self.__get_button_id()
        xpath = self._x + "//button[@id='%s']" % id_
        button.configure_builder_component(self._webconsole, self.dataset_name,
                                           self.page_name, id_, xpath)
        button.set_component_title(button.title)

    @PageService()
    def associate_column(self, column):
        """Configures column object"""
        if not isinstance(column, Column):
            raise TypeError("Invalid component type")
        xpath = self._x + "//th/span[@title='%s']" % column.title
        column.configure_builder_component(self._webconsole, self.dataset_name,
                                           self.page_name, column.title, xpath)


class DataTableViewer(TableViewer):
    """This class contains all the viewer actions specific to Data Table"""

    @WebAction()
    def __select_row(self, row_no):
        """Selects the given row in the table"""
        row = self._driver.find_element_by_xpath(self._x + f"//tr[{row_no}]//td[1]//input")
        row.click()

    @PageService()
    def select_rows(self, rows):
        """Selects the given list of rows"

        Args:
            rows (list) : List of row numbers for which checkbox have to be selected.

        """
        list(map(self.__select_row, rows))

    def associate_button(self, button):
        if not isinstance(button, ButtonViewer):
            raise ValueError("invalid component type")
        button.configure_viewer_component(self._webconsole, page="Page0")
        button._set_x(None, self._x + f"//button[@id='{button.id}']")

class PreviewTable(DataTableViewer):

    def __init__(self):
        super().__init__("")

    def configure_viewer_component(self, webconsole, page=""):  # page is empty string for method
        self._webconsole = webconsole                           # signature compatibility with
        self._browser = webconsole.browser                      # base class
        self._driver = webconsole.browser.driver
        self.page_name = ""
        self._set_x(None, xp="//*[@id='PreviewTable']")
