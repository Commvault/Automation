Source code for Automation.Web.Common.handlers

"""
This module defines all the decorators necessary for the browser and functionality
handlers.

These decorators are defined for the following reasons,

 * Automatic logging: The first line of the doc string is logged along with argument to avoid
        calling the logger every time. Please note that the logging format of the handler is
        slightly different from the default log, handler logs have the following format, the
        difference in formatting is because of the addition of function arguments and class name

        `time line_number module_name class_name handler_code message [function_name(arg_name=arg_val)]`


 * Automatic Error Handling: Handlers automatically catch the exception generated by
        the function it decorates and handles it according to the handler/exception type.

        Handlers would introspect the function's class for WebConsole, Browser and WebDriver
        objects, and if it has any of those objects it would capture the screenshot and
        the URL when an exception is raised

        Example code::
            try:
                webconsole.login() # if username XPath was changed,
                                   # and if login failed, you would
                                   # error
            except Exception as e:
                err = get_errors(exception_object)
                print(err)

        Output::
            OrderedDict(
            [
                ('SeleniumError',     'no such element: Unable to locate element:
                                      {"method":"xpath","selector":"//input[@id='username']'),
                ('BrowserOperation', 'Set username')
                ('Functionality',    'Login using store login popup; '),
                ('TestCase',         'Check if login is working')
                ('Screenshot',       'C:\\Automation\\Log Files\\Screenshot_50167 1507871141.png'),
                ('URL',              'http://machine/webconsole/login.do?disableSSO')
            ]

        Having said this, the handlers DO NOT MODIFY the exception object's normal functionality,
        you can continue to use the exception generated like you do, exception type/args remain
        unchanged and are propagated up the stack just just like they would when used without
        handlers
"""

_BROWSER_NAME_LOGGING = _CONSTANTS.LOG_BROWSER_NAME
__SCREENSHOTS = {}


def _lookup_attrib(class_obj, attrib_type):
    return [attrib for attrib in class_obj.__dict__.values()
            if isinstance(attrib, attrib_type)]


def _lookup_webconsole(class_obj):
    from Web.WebConsole.webconsole import WebConsole
    return _lookup_attrib(class_obj, WebConsole)


def _lookup_browser(class_obj):
    from Web.Common.cvbrowser import Browser
    return _lookup_attrib(class_obj, Browser)


def _lookup_webdriver(class_obj):
    from selenium.webdriver.remote.webdriver import WebDriver
    return _lookup_attrib(class_obj, WebDriver)


def _get_driver(class_obj):
    webconsole = _lookup_webconsole(class_obj)
    if webconsole:
        return webconsole[0].browser.driver
    browser = _lookup_browser(class_obj)
    if browser:
        return browser[0].driver
    driver = _lookup_webdriver(class_obj)
    if driver:
        return driver[0]
    return None


def _get_url(class_obj):
    driver = _get_driver(class_obj)
    if driver is not None:
        return driver.current_url
    else:
        return ""


def _get_screenshot(driver, excp):
    global __SCREENSHOTS
    _LOG = get_log()
    if id(excp) in __SCREENSHOTS.keys():
        return __SCREENSHOTS[id(excp)]
    try:
        ret = False
        file_path = ""
        if driver is not None:
            file_name = (
                "Screenshot TC"
                + threading.current_thread().getName() + " "
                + str(id(excp)) + " "
                + str(time.time()).split(".")[0]
                + ".png"
            )
            file_path = os.path.join(LOG_DIR, file_name)
            ret = driver.save_screenshot(file_path)
        if ret is True:
            __SCREENSHOTS[id(excp)] = file_path
            return file_path
        _LOG.error("Unable to capture screenshot at [%s]" % file_path)
        return ""
    except Exception as e:
        _LOG.exception("Error while taking screenshot [%s]" % str(e))
        return ""


def _get_non_handler_errors(err):
    err_dict = {}
    if len(err.args) > 0:
        if isinstance(err, WebDriverException):
            err_dict["1_SeleniumError"] = type(err).__name__ + ": " + \
                                          err.args[0].splitlines()[0]
        else:
            err_dict["1_" + type(err).__name__] = err.args
    else:
        err_dict["1_" + type(err).__name__] = ""
    return err_dict


def _get_all_function_args(log, hide_args, func, args, kwargs):
    arg_dict = {}
    if log is True and hide_args is False:
        arg_specs = inspect.getfullargspec(func)
        func_arg_names = arg_specs.args
        func_defaults = arg_specs.defaults if arg_specs.defaults is not None else ()
        func_arg_values = list(args) + list(func_defaults)
        arg_dict = dict(zip(func_arg_names, func_arg_values))
        arg_dict.update(kwargs)
    return arg_dict


def _get_func_doc(func):
    try:
        return func.__doc__.strip().splitlines()[0]
    except Exception:
        raise ValueError("Unable to log %s without doc string" % func.__name__)


def _get_handler_errors(err):
    err_dict = {}
    handler_error_attribs = [
        e for e in dir(err) if e.startswith("_cv_attribute_")
    ]
    err_dict.update({
        (attrib, err.__getattribute__(attrib))
        for attrib in handler_error_attribs
    })
    return err_dict


def _get_log_record(func, class_obj, level_name, level_num, msg):
    # copy the level code from python's lib/logging/__init__.py file
    return logging.makeLogRecord({
        'module': func.__module__.split(".")[-1],
        'levelname': level_name,
        'levelno': level_num,
        'lineno': func.__code__.co_firstlineno + 1,
        'funcName': str(class_obj.__class__.__name__),
        'msg': msg
    })


def _format_function_args(args_dict):
    args_str = ""
    for arg_name, arg_value in args_dict.items():
        if arg_name != 'self':
            args_str += """%s="%s",""" % (arg_name, arg_value)
    if args_str != "":
        args_str = args_str[:-1]
    return args_str


def _format_handler_errors(err):
    temp_errs = OrderedDict()
    ret_errs = OrderedDict()
    temp_errs.update(
        sorted(
            {(k.replace("_cv_attribute_", ""), v)
                for k, v in err.items()}
        )
    )
    for temp_err in temp_errs.items():
        key = ""
        key_splits = temp_err[0].split("_")
        if len(key_splits) > 1:
            if key_splits[0].isdigit():
                key_splits = key_splits[1:]
                key = " ".join(key_splits)
        else:
            key = temp_err[0]
        ret_errs[key] = temp_err[1]
    return ret_errs


def _handle_logging(log, hide_args, func, prefix, args, kwargs):
    if log is False:
        return

    func_log_str = func.__name__ + "(" + _format_function_args(
        _get_all_function_args(log, hide_args, func, args, kwargs)
    ) + ")"
    level = prefix.strip().lower()

    _LOG = get_log()
    browser_name = ""
    if _BROWSER_NAME_LOGGING:
        _browser = _lookup_browser(args[0])
        if _browser:
            browser_name = "[" + _browser[0].name + "]"
    if level == "bh":
        log_msg = (
            "   . %s %s [%s]" % (
                browser_name, _get_func_doc(func), func_log_str
            )
        ).strip()
        _LOG.handle(_get_log_record(func, args[0], "BH", 10, log_msg))

    elif level == "fh":
        log_msg = (
            "%s %s [%s]" % (
                browser_name, _get_func_doc(func), func_log_str
            )
        ).strip()
        _LOG.handle(_get_log_record(func, args[0], "FH", 20, log_msg))

    elif level == "ts":
        log_msgs = ["-" * 30 + " Starting Test Step " + "-" * 30]
        log_msgs += ["   " + _get_func_doc(func)]
        log_msgs += ["-" * 80]
        for log_msg in log_msgs:
            _LOG.handle(
                _get_log_record(func, args[0], "TS", 25, log_msg)
            )

    else:
        log_msg = (
            "%s %s [%s]" % (
                browser_name, _get_func_doc(func), func_log_str
            )
        ).strip()
        _LOG.handle(_get_log_record(func, args[0], "", 25, log_msg))

[docs]def add_error(error: Exception, level: str, msg: str):
    """
    Add custom errors to the handler generated error

    Args:
        error (Exception): Any subclass of exception object
        level (str): custom level that has to be added
        msg (str): custom message that has to be set to the level
    """
    if not issubclass(error.__class__, Exception):
        raise ValueError("Argument [error] is not an exception type")
    if type(level) is not str:
        raise ValueError("Argument [level] received is not of type str")
    if type(msg) is not str:
        raise ValueError("Argument [msg] received is not of type str")
    if not level.startswith("_cv_attribute_"):
        level = "_cv_attribute_5_" + level
    if not hasattr(error, level):
        error.__setattr__(level, msg)




[docs]def get_errors(err):
    """Extract all the errors generated by the handlers"""
    errors = {}
    errors.update(_get_handler_errors(err))
    errors.update(_get_non_handler_errors(err))
    return _format_handler_errors(errors)




[docs]def pprint_handler_errors(err, border=True):
    """Get the errors recorded by handlers as a formatted string

    To get the raw dictionary of errors use ``get_errors(err)``

    Args:
        err (Exception): Exception object
        border (bool): Enable or disable border string
    """
    summary = (
        "\n".join(
            ["%-30s  : %s" % (level, msg)
             for level, msg in get_errors(err).items()]
        )
    )
    if border:
        return "\n$" + ("*" * 50) + "$\n" + \
               "Error Summary" + "(" + err.__class__.__name__ + "):\n" + summary + \
               "\n$" + ("*" * 50) + "$\n"
    return summary




[docs]def browser_handler(log: bool=True, hide_args: bool=False, delay=0):
    """Use browser handler to decorate methods that have XPaths in them
    handling the browser based operations.

    Args:
        log (bool): Set this to false to disable the logging done by
            browser handler, this takes higher precedence than the
            configurations inside config.json
        hide_args (bool): Don't print the arguments passed to the
            function on logs
        delay (bool): Delay the function execution (in seconds)

    Example::

        @browser_handler()
        def some_function(self):
            self._browser.driver.find_element_by_xpath("//doc").click()
    """
    def _browser_handler(func):
        def _browser_handler_executor(*args, **kwargs):
            try:

                _handle_logging(log, hide_args, func, "BH", args, kwargs)
                time.sleep(delay)
                return func(*args, **kwargs)

            except WebDriverException as e:

                add_error(
                    e,
                    "_cv_attribute_2_BrowserOperation",
                    _get_func_doc(func) + "; %s" % _format_function_args(
                        _get_all_function_args(
                            log, hide_args, func, args, kwargs
                        )
                    )
                )
                add_error(
                    e,
                    "_cv_attribute_8_Screenshot",
                    _get_screenshot(_get_driver(args[0]), e)
                )
                add_error(e, "_cv_attribute_9_URL", _get_url(args[0]))
                raise

        return _browser_handler_executor
    return _browser_handler

    """Use functionality handler to decorate any functions which rely on
    browser handlers and expose the functionality to the TestCases

    Args:
        log (bool): Set this to false to disable the logging done by
            functionality handler, this takes higher precedence than
            the configurations inside config.json

        hide_args (bool): Don't print the arguments passed to the
            function on logs

    Example::

        @functionality_handler(log=False)
        def some_function(self):
            # function code
    """
    def _functionality_handler(func):
        def _functionality_handler_executor(*args, **kwargs):
            try:

                _handle_logging(log, hide_args, func, "FH", args, kwargs)
                return func(*args, **kwargs)

            except CVException as e:

                add_error(
                    e,
                    "_cv_attribute_3_Functionality",
                    _get_func_doc(func) + "; %s" % _format_function_args(
                        _get_all_function_args(
                            log, hide_args, func, args, kwargs
                        )
                    )
                )
                add_error(
                    e,
                    "_cv_attribute_8_Screenshot",
                    _get_screenshot(_get_driver(args[0]), e)
                )
                add_error(e, "_cv_attribute_9_URL", _get_url(args[0]))
                raise

        return _functionality_handler_executor
    return _functionality_handler




[docs]def test_step_handler():
    """
    This handler is used to decorate test steps inside the testcase

    We additionally add the test step as doc string to the
    CVTestStepFailure exception
    """
    def _test_step_handler(func):
        def _test_step_handler_executor(*args, **kwargs):
            try:

                _handle_logging(True, False, func, "TS", args, kwargs)
                return func(*args, **kwargs)

            except Exception as e:

                e.test_step = _get_func_doc(func)
                add_error(e, "_cv_attribute_6_TestStep", _get_func_doc(func))
                if isinstance(e, CVException):
                    add_error(
                        e,
                        "_cv_attribute_8_Screenshot",
                        _get_screenshot(_get_driver(args[0]), e)
                    )
                    add_error(e, "_cv_attribute_9_URL", _get_url(args[0]))
                raise

        return _test_step_handler_executor
    return _test_step_handler




[docs]def adminconsole_handler(log: bool=True, hide_args: bool=False):
    """
    Use this handler to decorate any AdminConsole function to automate
    the process of taking screenshot and managing exception.
    """
    def _adminconsole_handler(func):
        def _adminconsole_handler_executor(*args, **kwargs):
            try:

                _handle_logging(log, hide_args, func, "", args, kwargs)
                return func(*args, **kwargs)

            except Exception as e:

                e.test_step = _get_func_doc(func)
                add_error(e, "_cv_attribute_6_TestStep", _get_func_doc(func))
                add_error(
                    e,
                    "_cv_attribute_8_Screenshot",
                    _get_screenshot(_get_driver(args[0]), e)
                )
                add_error(e, "_cv_attribute_9_URL", _get_url(args[0]))
                args[0].close_popup()
                raise

        return _adminconsole_handler_executor
    return _adminconsole_handler
