        """creates a session with Azure Blob Storage
        Args :
            account_name       (str)      --       name of the Azure blob account
            account_key        (str)      --       key of the Azure blob account

        try:

[docs]    def delete_contents_azure(self, session, content):
        """deletes the specified contents from Azure cloud
        Args :
            session        (obj)     --       Azure blob session object
            content        (list)    --       Part of the subclient content which has to be
            deleted from the cloud
        Returns :
            None

        """
        for item in content:
            windows_path = Path(item)
            container_name = Path(item).parts[1]
            if (len(windows_path.parts)) == 2:
                session.delete_container(container_name)
                continue
            path_to_file = ("/".join(item.strip("/").split('/')[1:]))
            generator = session.list_blobs(container_name)
            blob_list = []
            for i in generator:
                blob_list.append(i.name)
            if path_to_file in blob_list:
                try:
                    session.delete_blob(container_name, path_to_file)
                except azure.common.AzureMissingResourceHttpError:
                    self.log.error("the specified container or blobs do not exist")
            else:
                generator = session.list_blobs(container_name, path_to_file+'/', delimiter='/')
                for blob in generator:
                    try:
                        session.delete_blob(container_name, blob.name)
                    except azure.common.AzureMissingResourceHttpError:
                        self.log.error("the specified container or blobs do not exist")

[docs]    def download_container_azure(self, session, container_name):
        """downloads a container from Azure cloud
        Args :
            session        (obj)   --   Azure blob session object
            container_name (str)   --   Name of the azure blob container which has to be downloaded

        Returns :
            None

        """
        generator = session.list_blobs(container_name)
        os.mkdir(container_name)
        os.chdir(container_name)
        # code below lists all the blobs in the container and
        # downloads them one after another
        for blob in generator:
            # check if the path contains a folder structure, create the folder structure
            if "/" in "{}".format(blob.name):
                # extract the folder path and check if that folder exists locally,
                # and if not create it
                head, tail = os.path.split("{}".format(blob.name))
                if os.path.isdir(os.getcwd() + "/" + head):
                    # download the files to this directory
                    try:
                        session.get_blob_to_path(container_name, blob.name,
                                                 os.getcwd() + "/" + head + "/" + tail)
                    except azure.common.AzureMissingResourceHttpError:
                        self.log.error("the specified container or blobs do not exist")
                else:
                    # create the diretcory and download the file to it
                    os.makedirs(os.getcwd() + "/" + head, exist_ok=True)
                    try:
                        session.get_blob_to_path(container_name, blob.name,
                                                 os.getcwd() + "/" + head + "/" + tail)
                    except azure.common.AzureMissingResourceHttpError:
                        self.log.error("the specified container or blobs do not exist")
            else:
                try:
                    session.get_blob_to_path(container_name, blob.name, blob.name)
                except azure.common.AzureMissingResourceHttpError:
                    self.log.error("the specified container or blobs do not exist")
        os.chdir(self.automation_directory)

        """downloads the specified contents from Azure cloud
        Args :
            session       (obj)      --      Azure blob session object
            content       (list)     --      Part of the subclient content which has to be
            downloaded from the cloud
            dir_name      (str)      --      Name of the folder to which the specified contents
            are to be downloaded
            oop_flag      (bool)     --      flag to determine if it's an in place restore
            or an out of place restore
        Returns :
            None

        """
        if oop_flag is True:
            container_name = content.replace("/", "")
            self.download_container_azure(session, container_name)
            os.chdir(str(self.automation_directory))
        else:
            os.mkdir(dir_name)
            for item in content:
                os.chdir(dir_name)
                path_to_file = ("/".join(item.strip("/").split('/')[1:]))
                container_name = Path(item).parts[1]
                generator = session.list_blobs(container_name)
                blob_list = []
                for j in generator:
                    blob_list.append(j.name)
                if path_to_file == "":
                    self.download_container_azure(session, container_name)
                    os.chdir(self.automation_directory)
                elif path_to_file in blob_list:
                    if os.path.exists(container_name) is False:
                        os.mkdir(container_name)
                    os.chdir(container_name)
                    head, tail = os.path.split("{}".format(path_to_file))
                    if os.path.isdir(os.getcwd() + "/" + head):
                        try:
                            session.get_blob_to_path(
                                container_name, path_to_file,
                                os.getcwd() + "/" + head + "/" + tail)
                        except azure.common.AzureMissingResourceHttpError:
                            self.log.error("the specified container or blobs do not exist")
                    else:
                        # create the diretcory and download the file to it
                        os.makedirs(os.getcwd() + "/" + head, exist_ok=True)
                        try:
                            session.get_blob_to_path(
                                container_name, path_to_file,
                                os.getcwd() + "/" + head + "/" + tail)
                        except azure.common.AzureMissingResourceHttpError:
                            self.log.error("the specified container or blobs do not exist")
                else:
                    generator = session.list_blobs(container_name, path_to_file+'/', delimiter='/')
                    if os.path.exists(container_name) is False:
                        os.mkdir(container_name)
                    os.chdir(container_name)
                    # code below lists all the blobs in the container and downloads
                    # them one after another
                    for blob in generator:
                        # check if the path contains a folder structure,
                        # create the folder structure
                        if "/" in "{}".format(blob.name):
                            # extract the folder path and check if that folder exists locally,
                            # and if not create it
                            head, tail = os.path.split("{}".format(blob.name))
                            if tail:
                                if os.path.isdir(os.getcwd() + "/" + head):
                                    # download the files to this directory
                                    try:
                                        session.get_blob_to_path(
                                            container_name, blob.name,

                                        session.get_blob_to_path(

                                            os.getcwd() + "/" + head + "/" + tail)

                            else:
                                self.recur(session, container_name, blob.name)

[docs]    def recur(self, session, container_name, blobname):
        """ recursively traverses through the subfolders and downloads the blobs
        Args :
            session        (obj)       --     Azure Storage session object
            container_name (str)       --     name of the azure container
            blobname       (str)       --     name of the azure blob
        Returns :
            None

        """
        generator = session.list_blobs(container_name, blobname, delimiter='/')
        for i in generator:
            head, tail = os.path.split("{}".format(i.name))
            if tail:
                if os.path.isdir(os.getcwd() + "/" + head):
                    # download the files to this directory
                    try:
                        session.get_blob_to_path(container_name, i.name,
                                                 os.getcwd() + "/" + head + "/" + tail)
                    except azure.common.AzureMissingResourceHttpError:
                        self.log.error("the specified container or blobs do not exist")
                else:
                    # create the diretcory and download the file to it
                    os.makedirs(os.getcwd() + "/" + head, exist_ok=True)
                    try:
                        session.get_blob_to_path(container_name, i.name,
                                                 os.getcwd() + "/" + head + "/" + tail)
                    except azure.common.AzureMissingResourceHttpError:
                        self.log.error("the specified container or blobs do not exist")
            else:
                self.recur(session, container_name, i.name)
