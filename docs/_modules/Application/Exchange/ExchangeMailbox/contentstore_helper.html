[docs]    def read_eml_properties_from_file(self):
        """Read eml properties from file and stores in CvEML object
            Returns:
                eml_messages (List) -- returns list of CvEML properties object"""
        try:
            eml_messages = []
            output = self.host_machine.read_file(self.smtp_property_file)
            output_list = output.split("* File :")
            for i in range(1, len(output_list)):

                string = output_list[i].split('Attachement :')
                if len(string) == 1:
                    contentstore = string[0]
                else:
                    contentstore = string[0]
                    exchange = string[1]
                eachline = exchange.split('\n')
                eml_message = CvEML()

                for j, item in enumerate(eachline):

                    if (eachline[j].lstrip().find("Description :") == 0 and
                            "Description :" in eachline[j]):
                        eml_message.exch_description = eachline[j].lstrip().split(
                            "Description :", 1)[1]
                    elif (eachline[j].lstrip().find("ContentType :") == 0
                          and "ContentType :" in eachline[j]):
                        eml_message.exch_content_type = eachline[j].lstrip().split(
                            "ContentType :", 1)[1]
                    elif (eachline[j].lstrip().find("Received :") == 0 and
                          "Received :" in eachline[j]):
                        eml_message.exch_received = eachline[j].lstrip().split(
                            "Received :", 1)[1]
                    elif (eachline[j].lstrip().find("MIME-Version :") == 0 and
                          "MIME-Version :" in eachline[j]):
                        eml_message.eml_mime_version = eachline[j].lstrip().split(
                            "MIME-Version :", 1)[1]
                    elif (eachline[j].lstrip().find("From :") == 0 and
                          "From :" in eachline[j]):
                        eml_message.exch_from_address = eachline[j].lstrip().split("From :", 1)[1]
                    elif (eachline[j].lstrip().find("To :") == 0 and
                          "To :" in eachline[j]):
                        eml_message.exch_to_address.append(
                            eachline[j].lstrip().split("To :", 1)[1])
                    elif eachline[j].lstrip().find("CC :") == 0 and "CC :" in eachline[j]:
                        eml_message.exch_cc_address.append(
                            eachline[j].lstrip().split("CC :", 1)[1])
                    elif eachline[j].lstrip().find("Date :") == 0 and "Date :" in eachline[j]:
                        eml_message.exch_received_time = eachline[j].lstrip().split("Date :", 1)[1]
                    elif eachline[j].lstrip().find("Subject :") == 0 and "Subject :" in eachline[j]:
                        eml_message.exch_subject = eachline[j].lstrip().split("Subject :", 1)[1]
                    elif (eachline[j].lstrip().find("Message-ID :") == 0 and
                          "Message-ID :" in eachline[j]):
                        eml_message.exch_message_id = eachline[j].lstrip().split(
                            "Message-ID :", 1)[1]
                    elif (eachline[j].lstrip().find("Return-Path :") == 0 and
                          "Return-Path :" in eachline[j]):
                        eml_message.exch_return_path = eachline[j].lstrip().split(
                            "Return-Path :", 1)[1]
                    elif (eachline[j].lstrip().find(
                            "X-MS-Exchange-Organization-OriginalArrivalTime :") == 0
                          and "X-MS-Exchange-Organization-OriginalArrivalTime :" in eachline[j]):
                        eml_message.exch_original_arrival_time = eachline[j].lstrip().split(
                            "X-MS-Exchange-Organization-OriginalArrivalTime :", 1)[1]

                eachline = contentstore.split('\n')

                for j, item in enumerate(eachline):
                    if j == 0:
                        fil_name = eachline[j].split('\\')[-1]
                        eml_message.eml_file_name = fil_name.split(".")[0]
                    if eachline[j].find("Subject :") == 0 and "Subject :" in eachline[j]:
                        eml_message.eml_subject = eachline[j].split("Subject :", 1)[1]
                    elif eachline[j].find("Date :") == 0 and "Date :" in eachline[j]:
                        eml_message.eml_received_time = eachline[j].split("Date :", 1)[1]
                    elif eachline[j].find("From :") == 0 and "From :" in eachline[j]:
                        eml_message.eml_from_address = eachline[j].split("From :", 1)[1]
                    elif eachline[j].find("To :") == 0 and "To :" in eachline[j]:
                        eml_message.eml_to_address.append(eachline[j].split("To :", 1)[1])
                    elif eachline[j].find("Cc :") == 0 and "Cc :" in eachline[j]:
                        eml_message.eml_cc_address.append(eachline[j].split("Cc :", 1)[1])
                    elif (eachline[j].find("- Return-Path :") == 0 and
                          "- Return-Path :" in eachline[j]):
                        eml_message.eml_return_path = eachline[j].split("- Return-Path :", 1)[1]
                    elif eachline[j].find("- Received :") == 0 and "- Received :" in eachline[j]:
                        eml_message.eml_received = eachline[j].split("- Received :", 1)[1]
                    elif eachline[j].find("- Content-Type :") == 0 and "Date :" in eachline[j]:
                        eml_message.eml_content_type = eachline[j].split("- Content-Type :", 1)[1]
                    elif eachline[j].find("- Sender :") == 0 and "- Content-Type :" in eachline[j]:
                        eml_message.eml_sender = eachline[j].split("- Sender :", 1)[1]
                    elif (eachline[j].find("- Message-ID :") == 0 and
                          "- Message-ID :" in eachline[j]):
                        eml_message.eml_message_id = eachline[j].split("- Message-ID :", 1)[1]
                eml_messages.append(eml_message)
            return eml_messages

        except Exception as excp:
            self.log.exception("Exception in read_eml_properties_from_file(): " + str(excp))
            raise excp

            self.mailbox_prop["restore_path"] = cvmailbox_object



[docs]    def get_folder_paths(self):
        """Get folder path in mailbox and save it in self.mailbox_folder_dict"""
        for user in self.mail_users:
            path = os.path.join(self.contentstore_mailbox_path, str(user))
            output = self.host_machine.get_files_and_directory_path(path)
            output = output.split("\r\n\r\n")
            folder_list = []
            for line in output:
                name = line.strip().split("\n")
                if any(".eml" in x for x in name):
                    if name[0].strip().split(":", 1)[1] not in folder_list:
                        folder_list.append(name[0].strip().split(":", 1)[1])
            self.mailbox_folder_dict[user] = folder_list

[docs]    def get_folder_status(self):
        """Get status from _master.db3 file"""
        try:
            for user, folders_path in self.mailbox_folder_dict.items():
                self.log.info("Getting folder status from master db for mailbox %s", user)
                master_db_path = os.path.dirname(folders_path[0]).strip().split('\\')
                drive = master_db_path[0].split(":")[0]
                path = '\\'.join((master_db_path[1:]))
                master_db_path = (f'\\\\{self.host_machine.client_object.client_name}\\'
                                  f'{drive}$\\{path}')
                master_dat_file = os.path.join(master_db_path, "_master.db3")
                conn = sqlite3.connect(master_dat_file)
                cursor = conn.cursor()
                query = "select subfoldername, status from Info"
                master_result = cursor.execute(query)
                for row in master_result:
                    self.status[row[0]] = row[1]
        except Exception as excp:
            self.log.exception("Exception in get_folder_status(): %s", excp)
            raise excp



[docs]    def verify_datfile(self):
        """Verify _MessagesInfo.db3  dat file after content store archive job
            Raises:
                Exception:
                    If verification of _MessagesInfo.db3 was not successful"""
        messages = []
        try:
            self.log.info("Verifying .dat file after archive job")
            for user, folders_path in self.mailbox_folder_dict.items():
                self.log.info("Validating _MessagesInfo.db3 dat files for mailbox : %s", user)
                mailbox_guid = utils.get_mailbox_guid(
                    user, self.ex_object.tc_object.subclient.subclient_id, self.csdb)
                for folder in folders_path:
                    self.log.info("Checking folder : " + folder)
                    folder_split = folder.strip().split("\\")
                    for prop_folder in self.mailbox_prop[user].folders:
                        if prop_folder.foldername == folder[-1]:
                            messages = prop_folder.messages
                    drive = folder_split[0].split(":")[0]
                    path = '\\'.join(folder_split[1:])
                    temp = (f'\\\\{self.host_machine.client_object.client_name}\\'
                            f'{drive}$\\{path}')
                    dat_file = os.path.join(temp, "_MessagesInfo.db3")
                    conn = sqlite3.connect(dat_file)
                    cursor = conn.cursor()
                    query = "select id, size, backedup ,internetmessageid from MessagesInfo"
                    result = cursor.execute(query)
                    status = self.status[folder.split("\\")[-1]]
                    for row in result:
                        for message in messages:
                            if message.eml_file_name == row[0]:
                                intrenet_id = str(mailbox_guid) + message.eml_message_id.strip()

                                if row[2] == "true" and status == "Dump":
                                    self.log.error(
                                        "Message is backed up from current folder %s ", row[0])
                                    raise Exception("Message is backed up from current folder")
                                if row[3] != intrenet_id:
                                    self.log.error("Intranet id not saved properly.")
                                    raise Exception("Intranet id not saved properly")
                                if row[2] != "true" and status == "Archive":
                                    self.log.error("Message not backed up %s", row[0])
                                    raise Exception("Message not backed up")

        except Exception as excp:
            self.log.exception("Exception in verify_datfile(): %s", excp)
            raise excp



[docs]    def verify_after_cleanup(self):
        """Verify folders are deleted after cleanup job
            Raises:
                Exception:
                    If cleanup of mailbox folder is not successful"""
        try:
            for user, folders_path in self.mailbox_folder_dict.items():
                self.log.info("################################################")
                for folder in folders_path:
                    self.log.debug(" Checking for folder path: %s", folder)
                    status = self.status[folder.split("\\")[-1]]
                    if status.lower() == "cleanup":
                        if self.host_machine.check_directory_exists(folder):
                            self.log.error('Cleanup was not successful of folder : %s,'
                                           ' Mailbox %s', folder, user)
                            raise Exception("Cleanup Validation failed for mailbox.")
        except Exception as excp:
            self.log.exception("Exception in verify_after_cleanup(): %s", excp)
            raise excp
