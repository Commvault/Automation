    def __init__(self, ex_object):
        """Initializes the Testdata object

            Args:
                tc_object (object) -- instance of the testcase object
        """

        self.ex_object = ex_object
        self.testcase_id = str(ex_object.tc_object.id)
        self.pst_path = ex_object.tc_inputs.get(
            'PSTPath', None)
        self.contentstore_mailbox_path = ex_object.tc_inputs.get(
            'ContentStoreMailboxPath', None)

        self.log = self.ex_object.log

        self.host_machine = machine.Machine(
            ex_object.server_name, self.ex_object.commcell)
        self.json_value = self.read_data_from_json()
        self.json_value = getattr(self.json_value, ("t" + self.testcase_id))
        self.exchange_online_user = ex_object.exchange_online_user
        self.exchange_online_password = ex_object.exchange_online_password
        self.service_account_user = ex_object.service_account_user
        self.service_account_password = ex_object.service_account_password

[docs]    def clean_online_mailbox_contents(self):
        """Method to clean content from exchnage online mailboxes
        """
        try:
            self.log.info(
                'Deleting mailbox content using powershell.Powershell '
                'Path %s', CLEANUP_CONTENT)
            for _iterator in self.json_value.CleanupMailbox:
                mb_name = "{0}{1}{2}".format(_iterator.DISPLAYNAME,
                                             self.ex_object.server_name,
                                             self.testcase_id)
                self.log.info(
                    'Deleting mailbox content for mailbox %s', mb_name)
                prop_dict = {
                    "LoginPassword": self.exchange_online_password,
                    "LoginUser": self.exchange_online_user,
                    "aliasName": mb_name
                }

                output = self.host_machine._execute_script(CLEANUP_CONTENT, prop_dict)
                if output.exit_code != 0 and "Success:True" not in output.output.replace(" ", ""):
                    self.log.error(
                        'Deleting mailbox content for mailbox %s failed', mb_name)
                    raise Exception("Exception raised while deleting mailbox content")
            self.log.info(
                'Deleting mailbox content completed succesfully')

        except Exception as excp:
            self.log.exception("Exception in cleanMailboxContents(): " + str(excp))
            raise excp



[docs]    def send_email_online_mailbox(self):
        """Method to send emails to exchnage online mailboxes

            Returns:
                smtp_list(list)  --  List of smtp address
        """
        try:

            smtp_list = []

            self.log.info(
                'Sending emails using powershell. Powershell Path %s', SEND_EMAIL)

            self.log.info(
                'Send Script path " {}" '.format(CONSTANT.SEND_SCRIPT_PATH))
            userlistpath = os.path.join(CONSTANT.SEND_SCRIPT_PATH, "userlist.txt")
            userlist = open(userlistpath, 'w')
            self.log.info(
                'Sending emails to mailboxes')

            for _iterator in self.json_value.SendEmail:
                smtp = "{0}{1}{2}@{3}".format(_iterator.SMTP,
                                              self.ex_object.server_name,
                                              self.testcase_id,
                                              self.ex_object.domain_name)
                userlist.write(smtp)
                smtp_list.append(smtp.lower())
                self.log.info(
                    'mailbox smtp " {}" '.format(smtp))
                userlist.write("\n")
            userlist.close()

            admin_pwd = b64decode(CONSTANT.ADMIN_PWD.encode()).decode()

            prop_dict = {
                "SMTPSever": CONSTANT.SMTP_SERVER,
                "LoginPassword": admin_pwd,
                "LoginUser": CONSTANT.ADMIN_USER,
                "NumberOfEmails": 50,
                "RootPath": CONSTANT.SEND_SCRIPT_PATH
            }
            output = self.host_machine._execute_script(SEND_EMAIL, prop_dict)
            self.log.info('Send_Email powershell output %s', output.output)

            if output.exit_code != 0:
                self.log.error(
                    'Sending emails failed')
                return False
            return smtp_list
        except Exception as excp:
            self.log.exception("Exception in send_email_online_mailbox(): " + str(excp))
            raise excp



[docs]    def create_online_mailbox(self):
        """Method to create exchnage online mailboxes

            Returns:
                alias_name_list(list)  --  List of alias name
        """
        try:
            alias_name_list = []

            self.log.info(
                'Creating mailbox content using powershell. '
                'Powershell Path %s', CREATE_ONLINE_MAILBOX)
            for _mbiterator in self.json_value.Mailbox:
                alias_name = ""
                if _mbiterator.OPTYPE.lower() == "create":
                    alias_name = "{0}{1}{2}".format(_mbiterator.ALIASNAME,
                                                    self.ex_object.server_name,
                                                    self.testcase_id)

                    self.log.info(
                        'Creating mailbox %s', alias_name)
                    prop_dict = {
                        "LoginPassword": self.exchange_online_password,
                        "LoginUser": self.exchange_online_user,
                        "aliasName": alias_name,
                        "displayName": "{0}{1}{2}".format(_mbiterator.DISPLAYNAME,
                                                          self.ex_object.server_name,
                                                          self.testcase_id),
                        "SMTP": "{0}{1}{2}".format(_mbiterator.SMTP,
                                                   self.ex_object.server_name,
                                                   self.testcase_id),
                        "ExchangeServerDomain": self.ex_object.domain_name
                    }
                    output = self.host_machine._execute_script(CREATE_ONLINE_MAILBOX, prop_dict)
                    if output.exit_code != 0 and \
                            "available for logon in approximately 15 minutes" not in output.output:
                        self.log.error('Create mailbox %s failed', alias_name)
                        self.log.error("Error message: %s", output.output)
                        return False
                alias_name_list.append(alias_name)
            return alias_name_list
        except Exception as excp:
            self.log.exception("Exception in cleanMailboxContents(): " + str(excp))
            raise excp



[docs]    def create_mailbox(self):
        """Method to create exchnage on premise mailboxes

            Returns:
                alias_name_list(list)  --  List of alias name
        """
        try:
            if "Mailbox" not in self.json_value._fields:
                self.log("provide details for mailbox creation")

            alias_name_list = []

            self.log.info(
                'Creating mailbox content using powershell. '
                'Powershell Path %s ', CREATE_MAILBOX)
            for _mbiterator in self.json_value.Mailbox:
                alias_name = ""
                if _mbiterator.OPTYPE.lower() == "create":
                    alias_name = "{0}_{1}".format(_mbiterator.ALIASNAME, self.testcase_id)
                    self.log.info('Creating mailbox %s ', alias_name)
                    database_name = "{0}_{1}_{2}".format(_mbiterator.DATABASE,
                                                         self.ex_object.exchange_server[0],
                                                         self.testcase_id)

                    if not self.check_if_db_exists(
                            database_name, self.ex_object.exchange_server[0]):
                        self.create_database(database_name, self.ex_object.exchange_server[0])
                    prop_dict = {
                        "LoginPassword": self.service_account_password,
                        "LoginUser": self.service_account_user,
                        "aliasName": str(alias_name),
                        "displayName": "{0}_{1}".format(_mbiterator.DISPLAYNAME,
                                                        self.testcase_id),
                        "SMTP": "{0}_{1}".format(_mbiterator.SMTP, self.testcase_id),
                        "ExchangeServerDomain": self.ex_object.domain_name,
                        "ExchangeDatabase": database_name,
                        "ExchangeCAServer": self.ex_object.exchange_cas_server
                    }
                    output = self.host_machine._execute_script(CREATE_MAILBOX, prop_dict)
                    if output.exit_code != 0:
                        self.log.error(
                            'Create mailbox %s failed', alias_name)
                        self.log.error("Error message: %s", output.output)
                        return False
                alias_name_list.append(alias_name)
            return alias_name_list
        except Exception as excp:
            self.log.exception("Exception in cleanMailboxContents(): " + str(excp))
            raise excp

[docs]    def check_if_db_exists(self, dbname, exchange_server):
        """Method to check if database exists

        """
        self.log.info("Checking if DB exists")

        self.log.info(
            'Creating mailbox content using powershell. '
            'Powershell Path %s', GET_DATABASES)
        prop_dict = {
            "LoginPassword": self.service_account_password,
            "LoginUser": self.service_account_user,
            "ExchangeServer": exchange_server,
        }
        output = self.host_machine._execute_script(GET_DATABASES, prop_dict)
        retrieved_db_list = ((output.output).split("Name")[-1]).splitlines()
        for temp_db in retrieved_db_list:
            temp_db1 = temp_db.strip()
            temp_db1 = ''.join(temp_db1.split())
            if temp_db1 == dbname:
                return True

        return False

        """Method to import pst

            Returns:
                smtp_list(list)  --  List of Smtp addresses
        """
        try:

            smtp_list = []

            self.log.info(
                'Creating mailbox content using powershell. '

                        "LoginUser": self.service_account_user,

                        "PstPath": self.pst_path
                    }
                    output = self.host_machine._execute_script(IMPORT_PST, prop_dict)

[docs]    def import_data_contentstore_mailbox(self):
        """Method to copy mailboxes data to contentStore mailboxes

            Returns:
                mailboxes(list)  --  List of mailboxes
        """
        mailbox_list = []
        self.log.info('Import Mailbox data to ContentStore Mailboxes')

        for mailbox in self.json_value.Mailbox:
            mailbox_dict = {}
            mailbox_dict['smtpAdrress'] = "{0}_{1}@commvault.com".format(mailbox.SMTP,
                                                                         self.testcase_id)
            mailbox_dict['displayName'] = "{0}_{1}".format(mailbox.DISPLAYNAME,
                                                           self.testcase_id)
            self.log.info('Copying data to mailbox %s', mailbox_dict['displayName'])

            if isinstance(self.ex_object.smtp_mailbox_path, str):
                mailbox_path = os.path.join(self.ex_object.smtp_mailbox_path, "Mailboxes",
                                            (mailbox.DISPLAYNAME + "_" + str(self.testcase_id)))

            if self.host_machine.check_directory_exists(mailbox_path):
                self.host_machine.remove_directory(mailbox_path)
            self.host_machine.copy_folder(self.contentstore_mailbox_path, mailbox_path)

            mailbox_list.append(mailbox_dict.copy())
        return mailbox_list
