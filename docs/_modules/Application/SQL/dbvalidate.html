    def db_compare(file1, file2):
        """
        This function compares two files 
        if files match return True else False
        
        file1       : file to be compared to file2
        file2       : file to be compared to file1
        
        """
        try:
            log = logger.get_log()
            log.info("Comparing the files")
            if filecmp.cmp(file1, file2):
                log.info("Files %s and %s are identical" % (file1, file2))
                return True
            else:
                log.error("Files %s and %s differ!" % (file1, file2))
                return False

        except Exception as excp:
            raise Exception("Exception raised in dbCompare()\nError: '{0}'".format(excp))

    def get_random_dbnames_and_filegroups(randomization, noofdbs, nooffilegroupsdb, nooftablesfilegroup):
        """
        This function shuffles the database tables and returns some table names
        
        randomization          : randomization 0 to 100
        noofdbs                : No.of databases
        nooffilegroupsdb       : No.of file groups for each database
        nooftablesfilegroup    : No.of file for each file group
        
        returns (True,list1,list2,list3) for success else (False,list1,list2,list3)
        
        list1       : data base numbers
        list2       : File Group numbers
        list3       : table numbers

        """
        try:
            list1 = []
            list2 = []
            list3 = []
            for i in range(1, int(noofdbs) + 1):
                list1.append(i)
            for i in range(1, int(nooffilegroupsdb) + 1):
                list2.append(i)
            for i in range(1, int(nooftablesfilegroup) + 1):
                list3.append(i)
            r = randomization
            random.shuffle(list1)
            random.shuffle(list2)
            random.shuffle(list3)
            i = int(math.ceil((float(int(r)) / float(100)) * int(noofdbs)))
            l = []
            m = []
            n = []
            a = 0
            for j in range(1, i + 1):
                l.append(list1[a])
                a = a + 1
            i = int(math.ceil((float(int(r)) / float(100)) * int(nooffilegroupsdb)))
            a = 0
            for j in range(1, i + 1):
                m.append(list2[a])
                a = a + 1
            i = int(math.ceil((float(int(r)) / float(100)) * int(nooftablesfilegroup)))
            a = 0
            for j in range(1, i + 1):
                n.append(list3[a])
                a = a + 1
            return True, l, m, n

        except Exception as excp:
            raise Exception("Exception raised in getRandomDbNamesAndFileGroups()\nError: '{0}'" .format(excp))



[docs]    def dump_db_to_file(self, fname, databasename, l, m, n, incbackuptype, p=1):
        """
        This function writes the database tables to file
        
        fname                : file name
        databasename         : database name
        l                    : list contains the table data
        m                    : list contains the file group data
        n                    : list contains the no.of records in table
        incbackuptype        : is "incremental" or "differential" or other
        p                    : calling modify_for_inc function 2nd time we put p=2 ...
        
        returns True for Success else False
        
        """
        log = logger.get_log()
        filehandle = None
        sqlcon = None
        try:
            log.info("Writing the data to file:%s" % fname)
            f_name = fname
            filehandle = open(f_name, 'w')
            sqlcon = database_helper.MSSQL(self.sqlInstance, self.sqlUser, self.sqlPass, "master")

            for val1 in l:
                db = databasename + str(val1)
                sqlcon.execute("use [{0}]".format(db))
                for val2 in m:
                    for val3 in n:
                        dbn = str("tab") + str(val1) + str(val2) + str(val3)
                        cur1 = sqlcon.execute("select * from " + str(dbn))
                        for r in cur1.rows:
                            filehandle.write(str(r) + '\n')
            for val1 in l:
                db = databasename + str(val1)
                sqlcon.execute("use [{0}]".format(db))
                for val3 in n:
                    dbn = str("tabm") + str(val1) + str(val3)
                    cur3 = sqlcon.execute("select * from " + str(dbn))
                    for r in cur3.rows:
                        filehandle.write(str(r) + '\n')

            if incbackuptype == "INCREMENTAL":
                for i in l:
                    db = databasename + str(i)
                    sqlcon.execute("use [{0}]".format(db))
                    cur4 = sqlcon.execute("select * from [tab_inc" + repr(p) + "_ndf]")
                    for r in cur4.rows:
                        filehandle.write(str(r) + '\n')
                    cur5 = sqlcon.execute("select * from [tab_inc" + repr(p) + "_mdf]")
                    for r in cur5.rows:
                        filehandle.write(str(r) + '\n')

            if incbackuptype == "DIFFERENTIAL":
                for i in l:
                    db = databasename + str(i)
                    sqlcon.execute("use [{0}]".format(db))
                    cur4 = sqlcon.execute("select * from [tab_diff_ndf]")
                    for r in cur4.rows:
                        filehandle.write(str(r) + '\n')
                    cur5 = sqlcon.execute("select * from [tab_diff_mdf]")
                    for r in cur5.rows:
                        filehandle.write(str(r) + '\n')
            log.info("Successfully writing the data to file " + fname)
            return True

        except Exception as excp:
            raise Exception("Exception raised in dumpDbToFile()\nError: '{0}'".format(excp))
        finally:
            filehandle.close()
            sqlcon.close()
