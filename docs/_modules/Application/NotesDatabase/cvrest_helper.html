    def get_all_apis(cls, testcase):
        """Returns a list of all REST API extensions active on Domino Server

            Args:
                testcase     (Testcase)  --  instance of the testcase in operation


            Returns:
                services    (list)  -  list of services
        """

        path = '/api/cvrest'
        url = 'http://' + testcase.tcinputs.get("DominoServerHostName") + path
        testcase.log.info('URL formed: %s', url)
        response = requests.get(url)
        testcase.log.info('Response status code: %s', response.status_code)
        data = json.loads(response.text)
        services = {}
        data = data['cv rest services']
        for item in data:
            key_temp = href_temp = type_temp = None
            for i in item:
                if i == 'href':
                    href_temp = url + item[i]
                elif i == 'type':
                    type_temp = item[i]
                else:
                    key_temp = i
            services[key_temp] = dict({'href': href_temp, 'type': type_temp})

        return services

    def return_result_json(cls, testcase, operation, param=None, jstring=None):
        """Calls the API and returns a dictionary of the response received

            Args:
                testcase        (Testcase)  --  instance of the testcase in operation

                operation       (str)       --  API functionality called

                param           (str)       --  parameters to be passed to the API for GET requests

                jstring         (str)       --  parameters to be passed to the API for POST request


            Returns:
                result          (dict)  -   dictionary of the json received
        """

        result = {}
        try:
            response = None
            if operation in testcase.rest_ops:
                temp = testcase.rest_ops[operation]
                url = temp['href']
                num_of_parameters = 0
                while '{' in url:
                    url = url.replace(url[url.index('{'):url.index('}') + 1],
                                      param[num_of_parameters])
                    num_of_parameters = num_of_parameters + 1
                testcase.log.info(url)
                if temp['type'] == 'get':
                    response = requests.get(url, auth=HTTPBasicAuth(
                        testcase.tcinputs.get("DominoNotesUser"),
                        testcase.tcinputs.get("DominoNotesPassword")))
                else:
                    response = requests.post(url, json=jstring, auth=HTTPBasicAuth(
                        testcase.tcinputs.get("DominoNotesUser"),
                        testcase.tcinputs.get("DominoNotesPassword")))
                testcase.log.info('Response code: ' + str(response.status_code))
            data = json.loads(response.text)
            for i in data:
                try:
                    result = eval(' '.join(str(x) for x in data[i]))
                except BaseException:
                    result = data[i]
            return result
        except Exception as ex:
            testcase.log.error('Error {} on line {}. Error {}'.format(
                type(ex).__name__, sys.exc_info()[-1].tb_lineno, ex))
            return result

    def wait_for_job_to_complete(cls, job, log):
        """Handles logging for a job

            Args:
                job             (Job)       --  instance for the job

                log             (Log)       --  logger for the testcase

            Returns:
                string                      --  job status
        """
        try:
            log.info(job)
            st = 'ok'
            if not job.wait_for_completion():
                pending_reason = job.pending_reason
                log.error('Pending reason:' + pending_reason)
                raise Exception('Pending reason:' + pending_reason)
            else:
                log.info('Job completed')
                return st
        except Exception as ex:
            if 'empty' in str(ex):
                cls.wait_for_job_to_complete(job, log)
                st = str(ex)
                return st

    def delete_db_and_validate(cls, testcase, dbsinsubclient):
        """"Deletes LN databases and validates their successful/unsuccessful delete

            Args:
                testcase        (Testcase)  --  instance of the testcase in operation

                dbsinsubclient  (list)      --  databases to be removed

        """
        testcase.log.info(
            '***************************************DELETE****************************************')
        for database in dbsinsubclient:
            response = cls.return_result_json(testcase, 'delete database', param=[database])
            if 'removed' in response.get('status'):
                testcase.log.info('{} removal successful'.format(database))
            else:
                testcase.log.info('{} removal unsuccessful'.format(database))
        testcase.log.info(
            '*****************************VALIDATE**DELETE****************************************')
        for database in dbsinsubclient:
            response = cls.return_result_json(testcase, 'delete database', param=[database])
            testcase.log.info('{} {}'.format(database, response.get('status')))

    def run_backup(cls, testcase, level='FULL', sc=None):
        """"Runs backup operation on an LNDB Subclient

            Args:
                testcase        (Testcase)  --  instance of the testcase in operation

                level           (str)       --  type of backup

                sc              (Subclient) --  instance of the subclient to perform backup on

        """
        try:
            if sc is None:
                sc = testcase.subclient
            testcase.log.info(
                '**********************************BACKUP***************************')
            testcase.log.info('Running {} backup for {}'.format(level, sc))
            time.sleep(30)
            job = sc.backup(level)
            ob = cls.wait_for_job_to_complete(job, testcase.log)
            if ob is not None:
                if not ob == 'ok':
                    if 'empty' not in ob:
                        raise Exception(ob)
            return job
        except Exception as ex:
            testcase.log.error('Error {} on line {}. Error {}'.format(type(ex).__name__,
                                                                      sys.exc_info()[-1].tb_lineno,
                                                                      ex))
            testcase.result_string = str(ex)

    def run_restore(cls, testcase, type_of_restore='in', sc=None):
        """"Runs restore operation on an LNDB Subclient

            Args:
                testcase        (Testcase)  --  instance of the testcase in operation

                type_of_restore (str)       --  type of restore

                sc              (Subclient) --  instance of the subclient to perform backup on

        """
        try:
            if sc is None:
                sc = testcase.subclient
            testcase.log.info("common_options_dict: {}".format(testcase.common_options_dict))
            testcase.log.info("lndb_restore_options: {}".format(testcase.lndb_restore_options))
            if 'out' in type_of_restore:
                testcase.log.info(
                    '***************************OUT*RESTORE*****************************')
                path, di = sc.browse()
                if not path:
                    path = '\\'
                dest_path = testcase.domino_data_path + '\\OOPRestore_auto'
                testcase.log.info('Out of place restore path: {}'.format(dest_path))
                time.sleep(30)
                job = sc.restore_out_of_place(testcase.tcinputs.get('ClientName'), paths=path,
                                              destination_path=dest_path,
                                              lndb_restore_options=testcase.lndb_restore_options,
                                              common_options_dict=testcase.common_options_dict)
                ob = cls.wait_for_job_to_complete(job, testcase.log)
                if ob is not None:
                    if not ob == 'ok':
                        if 'empty' not in ob:
                            raise Exception(ob)
            else:
                testcase.log.info(
                    '*******************************IN*RESTORE*****************************')
                path, di = sc.browse()
                if not path:
                    path = '\\'
                job = sc.restore_in_place(paths=path,
                                          lndb_restore_options=testcase.lndb_restore_options,
                                          common_options_dict=testcase.common_options_dict)
                ob = cls.wait_for_job_to_complete(job, testcase.log)
                if ob is not None:
                    if not ob == 'ok':
                        if 'empty' not in ob:
                            raise Exception(ob)
            return job
        except Exception as ex:
            testcase.log.error('Error {} on line {}. Error {}'.format(type(ex).__name__,
                                                                      sys.exc_info()[-1].tb_lineno,
                                                                      ex))
            testcase.result_string = str(ex)

    def browse_processor(cls, sc, log, queue, dictionary):
        """"Helps in processing recursive browse for LNDBSubclient

                    Args:
                        sc          (Subclient)     --  instance of the subclient to perform browse on

                        log         (Log)           --  logger for the testcase

                        queue       (Queue)         --  queue object storing paths found through browse

                        dictionary  (dict)          --  dict mapping backed up database to its path

         """
        while True:
            element = queue.get()
            t_path, t_dict = sc.browse(path=element)
            for p in t_path:
                regex = re.compile('(.*nsf*)|(.*ntf*)')
                if re.match(regex, p):
                    abspath = os.path.splitext(p)[0]
                    log.info('Path found:{}'.format(abspath))
                    name = abspath.rsplit('\\', 1)[1]
                    if name in dictionary:
                        dictionary[name].append(abspath.replace('\\DATABASES\\', ''))
                    else:
                        dictionary[name] = [abspath.replace('\\DATABASES\\', '')]
                else:
                    queue.put(p)
            queue.task_done()

    def recursive_browse(cls, sc, log):
        """"performs browse recursively for an LNDBSubclient

            Args:
                sc          (Subclient)     --  instance of the subclient to perform browse on

                log         (Log)           --  logger for the testcase

            Returns:
                dictionary  (dict)          -   dict mapping backed up database to its path

        """
        queue = Queue()
        dictionary = dict()
        for _ in range(5):
            try:
                t = Thread(target=cls.browse_processor, args=(sc, log, queue, dictionary))
                t.setDaemon(True)
                t.start()
            except Exception as ex:
                log.info(ex)
                continue
        queue.put('\\')
        queue.join()
        log.info('Browse operation completed')
        return dictionary
