        """
        Connects to salesforce
        """
        try:
            self.log.info('sfuser: ' + self.sf_user)
            self.log.info('sfpassword: ********** ')
            self.log.info('sftoken: **********')
            self.log.info('sandbox: ' + str(self.sandbox))

            if self.sandbox:
                self._sfconn = Salesforce(
                    username=self.sf_user,
                    password=self.sf_password,
                    security_token=self.sf_token,
                    sandbox=self.sandbox)
            else:
                self._sfconn = Salesforce(
                    username=self.sf_user,
                    password=self.sf_password,
                    security_token=self.sf_token)

        except Exception as exp:
            raise Exception("Failed to login to salesforce: {0} ".format(str(exp)))

[docs]    def sf_bulk_delete(self, sf_object, where_condition, hard_delete=False):
        """
        Runs bulk delete operation
        Args:
            sf_object           (str)       --  salesforce object name
            where_condition     (str)       --  filter the records for deletion based on condition
            hard_delete         (bool)         --  hard delete the records even from recyle bin
        """
        try:
            query = "select id from " + sf_object + " where " + where_condition
            result = self.sf_bulk_query(sf_object, query)
            if len(result):
                delete_data = [
                    {
                        'Id': record['Id']
                    } for record in result
                ]
            else:
                self.log.info("no matched records")
                return False
            self.sf_bulk_cmd(sf_object)
            if hard_delete:
                result = self._bulk_cmd.hard_delete(delete_data)
            else:
                result = self._bulk_cmd.delete(delete_data)
            self.log.info("Bulk delete job succeeded ")

        except Exception as exp:
            raise Exception("sf_bulk_delete error:{0} ".format(str(exp)))
