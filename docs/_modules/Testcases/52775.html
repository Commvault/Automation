[docs]    def wait_for_reverse_replication_job(self):
        """ Wait for DROrchestration Job """

        try:
            max_tries = 5
            current_try = 0

            while current_try < max_tries:

                # gets the list of Running jobs in the commcell
                job_controller = JobController(self.commcell)
                if isinstance(job_controller, JobController):
                    active_jobs = job_controller.active_jobs(None, 1, job_filter="DROrchestration")
                    dr_job_id = list(active_jobs.keys())[0]

                    if dr_job_id:

                        # wait for job till it is finished
                        job_state = self.wait_for_job(dr_job_id)
                        self.log.info(" Job state: [" + str(job_state) + "]")
                        if job_state:
                            return True

                    # else
                    # Reverse replication didnt launch by scheduler
                    # wait for 30 minutes
                    else:
                        time.sleep(30 * 60)

                    current_try += 1

            raise Exception("DROrchestration ReverseReplication didn't launch. Please check CVD.log or EvMgrs.log \
                    for failure reason.")

        except Exception as e:
            self.log.error("Error occurred: " + str(e))
            raise Exception(str(e))




[docs]    def wait_for_job(self, job_id):
        """ Wait for Job with job Id"""

        try:

            # create job object and wait till it finished
            job = Job(self.commcell, job_id)
            job_state = job.wait_for_completion()
            self.log.info(" Job state: [" + str(job_state) + "]")
            if job_state:
                return True

            raise Exception(
                    " Job [" +
                    str(job_id) +
                    "] didn't launch. Please check testcase.log or CVD.log or EvMgrs.log \
                    for failure reason.")

        except Exception as e:
            self.log.error("Error occurred: " + str(e))
            raise Exception(str(e))
