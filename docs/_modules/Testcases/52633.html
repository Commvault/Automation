            download_location = r"C:\Users\Admin\Downloads"
            url = "https://cloud.commvault.com/webconsole/summary/index.do?page=sharedWithMeLi"
            loginurl = "https://cloud.commvault.com/webconsole/login/index.jsp?disableSSO"
            linkstoverify = ["Profile Dashboard", "DRBackup"]
            newcloudpassword = "######"
            clenup_enties = {"storagepolicy": []}

            if "CloudUser" in self.tcinputs:
                clouduser = self.tcinputs["CloudUser"]
            else:
                clouduser = ""

            if "CloudUserPassword" in self.tcinputs:
                clouduserpsw = self.tcinputs["CloudUserPassword"]
            else:
                clouduserpsw = ""

            profile = self.tcinputs["Profile"]
            self._log.info("Create Machine class object")
            client = Client(self.commcell, self.commcell.commserv_name)
            client_machine = WindowsMachine(client.client_name, self.commcell)
            drobject = self.commcell.disasterrecovery
            drhelperobject = drhelper.DRHelper(
                self.commcell, self._log, self.tcinputs, killdrjobs=True)

            if "RestoreLocation" in self.tcinputs and self.tcinputs["RestoreLocation"]:
                drrestorelocation = self.tcinputs["RestoreLocation"]
            else:
                install_path = drhelperobject.csclient.install_directory.split(os.path.sep)[
                    0]
                if not install_path.endswith(os.path.sep):
                    install_path += os.path.sep
                drrestorelocation = os.path.join(install_path, "Automation_test",
                                                 "drrestore_location")

            full_mount_path = os.path.join(
                drrestorelocation, "Full_MountPath_{}".format(randomint))
            initialdrpath = drhelperobject.drpath
            drhelperobject.client_machine = client_machine
            drhelperobject.uploadtocloud = True
            drhelperobject.drpath = initialdrpath
            drhelperobject.changedrsettings()
            commcellquery = "select CommcellId from LicCsl where id =2 "
            commcellid = drhelperobject.executequery(commcellquery)
            commcellid = hex(int(commcellid[0][0]))
            commcellid = str(commcellid)[2:]
            commcellid = commcellid.upper()
            linkstoverify.append(commcellid)
            self._log.info(
                "Start validating Cloud web with links %s", str(linkstoverify))

            def updatecredentials(oldpsw, newpsw):
                '''
                updates the cloud password with new password.
                @args:
                     oldpsw  (String) -- cloud user old password
                     newpsw  (String) -- cloud user new password
                '''

                browser = BrowserFactory().create_browser_object()
                browser.set_downloads_dir(download_location)
                browser.open(False)
                webconsole = WebConsole(
                    browser, self.commcell.webconsole_hostname)
                browser.driver.get(loginurl)
                webconsole.login(clouduser, oldpsw, auto_open_login_page=False)
                self._log.info(
                    "updating Cloud user name and password to New password")
                changepassword = browser.driver.find_elements_by_xpath(
                    "//div[@id='navbar-ex']//ul//li")
                changepassword[0].click()
                webconsole.wait_till_load_complete()
                time.sleep(2)

                for item in browser.driver.find_elements_by_xpath(
                        "//div[@id='navbar-ex']//ul//li[@class='rightMenu hideOnExportFriendly userNameValue vw-contains-drop-down open']//span//span"):
                    if item.text == "Preferences":
                        item.click()

                time.sleep(2)
                oldpassword = browser.driver.find_elements_by_xpath(
                    "//div[@class='subChangePasswordDiv']//div[@class='form-group']//input[@id='oldPassword']")[0]
                newpassword = browser.driver.find_elements_by_xpath(
                    "//div[@class='subChangePasswordDiv']//div[@class='form-group']//input[@id='newPassword']")[0]
                retypepassword = browser.driver.find_elements_by_xpath(
                    "//div[@class='subChangePasswordDiv']//div[@class='form-group']//input[@id='confirmNewPassword']")[0]
                oldpassword.send_keys(oldpsw)
                newpassword.send_keys(newpsw)
                retypepassword.send_keys(newpsw)

                browser.driver.find_elements_by_xpath(
                    "//div//input[@id='changePwdBtn']")[0].click()
                webconsole.wait_till_load_complete()
                browser.driver.close()
            self._log.info(
                'Running DR backup with incorrect cloud credentials')
            updatecredentials(clouduserpsw, newcloudpassword)
            self._log.info("Running FUll backup job after creating Lib/SP")

            drobject.backup_type = 'full'
            fulldrjob = drobject.disaster_recovery_backup()
            self._log.info("Full DR Job {}".format(str(fulldrjob)))

            if not fulldrjob.wait_for_completion():
                raise Exception(
                    "Failed to run {0} backup job with error: {1}".format(
                        drobject.backup_type, fulldrjob.delay_reason
                    )
                )

            status = fulldrjob.status

            if not status.find("Completed w/ one or more errors") >= 0:
                error = "DR backup job status is not completed with errors. job id {},\
                 current job type is {} ".format(str(fulldrjob), status)
                raise Exception(error)

            self._log.info('resetting the password to Old')
            updatecredentials(newcloudpassword, clouduserpsw)

            browser = BrowserFactory().create_browser_object()
            browser.set_downloads_dir(download_location)

            webconsole = WebConsole(browser, self.commcell.webconsole_hostname)
            browser.driver.get(loginurl)
            webconsole.login(

                for tag in tags:

                        break

            found = False
            deletelink = False

            for link in linkstoverify:
                rows = browser.driver.find_elements_by_xpath(
                    "//table[@class='innerBrowseTable']//tbody[@id='filetablebody']//tr")
                found = True
                i = 0

                    i = i + 1

            if deletelink:
                downloadlink = browser.driver.find_elements_by_xpath(
                    "//div[@id='buttonsDivHolder']//div[@id='aButtons']//div[@class='vw-action-menu-wrapper pull-left']//ul[@id='firstControl']//li//a[@id='moreActionsLink']")
                downloadlink[0].click()
                downloadlink = browser.driver.find_elements_by_xpath(
                    "//div[@id='buttonsDivHolder']//div[@id='aButtons']//div[@class='vw-action-menu-wrapper pull-left']//ul[@id='firstControl']//li//ul[@id='firstControlDropdown']//li//a[@id='deleteLink']")

            try:

                self._log.info(

            drhelperobject.changedrsettings()

            mediaagent = storage.MediaAgent(

            disklibraries = storage.DiskLibraries(self.commcell)

            if not disklibraries.has_library(full_library_name):
                fulldisklibrary = disklibraries.add(
                    full_library_name, mediaagent, full_mount_path)
            else:
                fulldisklibrary = disklibraries.get(full_library_name)
            storageobj = storage.StoragePolicies(self.commcell)
            self._log.info(
                "Creating Storage Policy with name {}".format(full_sp))

            if not storageobj.has_policy(full_sp):
                full_sp = storageobj.add(
                    full_sp,

            found = True

            for link in linkstoverify:

                i = 0

                while i < len(rows):

                    if not found:

                    if rows[i].text.find(link) >= 0:

                        if linkcount != len(linkstoverify) - 1:

                                taglink.click()
                                found = False
                                time.sleep(shortsleep)
                                browser.driver.refresh()
                                break
                        else:
                            rows[i].click()
                            break

                    i = i + 1

                linkcount = linkcount + 1
                webconsole.wait_till_load_complete()
                time.sleep(shortsleep)

            downloadlink = browser.driver.find_elements_by_xpath(
                "//div[@id='buttonsDivHolder']//div[@id='aButtons']//div[@class='vw-action-menu-wrapper pull-left']//ul[@id='firstControl']//li//a[@id='downloadLink']")
            downloadlink[0].click()
            time.sleep(shortsleep)
            found = True
            filename = ''
            zipcount = 0

            while found and zipcount <= 50:
                drlist = os.listdir(download_location)

                for file in drlist:

                    if file.find(".zip") >= 0 and (not file.find("zip.") >= 0):
                        found = False
                        filename = file
                        break

                time.sleep(shortsleep)

            zipfilename = os.path.join(download_location, filename)
            zip_ref = zipfile.ZipFile(zipfilename, 'r')
            zip_ref.extractall(download_location)

            clouddrfolder = os.path.join(

            retcode = client_machine.compare_folders(

            self._log.error(

            self._log.info(

            if not storageobj.delete(full_sp.storage_policy_name):
                error = "Failed to delete SP {}".format(

                raise Exception(error)

            self._log.info("Run Full DB restore with CS recovery tool")
            drhelperobject.db_dumplocation = clouddrfolder
            drhelperobject.restore_db_with_cvmigrationtool()
            self._log.info(
                "Full DB restore is successful with CS recovery tool")
            self._log.info("Start services with DRrestore.exe")
            time.sleep(shortsleep)
            drhelperobject.startservices()
            time.sleep(shortsleep * 6)

            self._log.info("Verifying Full DB SP/Lib")
            storageobj = storage.StoragePolicies(self.commcell)
            disklibraries = storage.DiskLibraries(self.commcell)

            if storageobj.has_policy(full_sp.storage_policy_name):
                self._log.info(
                    "Storage policy created for Full DR is available after DR restore {}".format(
                        full_sp.storage_policy_name))
            else:
                error = "Storage policy created for Full DR is not available \
                after DR restore {}".format(full_sp.storage_policy_name)
                raise Exception(error)

            if disklibraries.has_library(full_library_name):
                self._log.info(
                    "full_library_name created for Full DR is available after DR restore {}".format(full_library_name))
            else:
                error = "full_library_name created for Full DR is not available\
                 after DR restore {}".format(full_library_name)
                raise Exception(error)

            try:
                client_machine.remove_directory(full_mount_path)
            except Exception as err:
                self._log.info(
                    "Failed to delete library mount path dir {0}".format(err))

            try:
                client_machine.remove_directory(download_location)
            except Exception as err:
                self._log.info("Failed to restore location {0}".format(err))
            self._log.info("Testcase execution completed")
        except Exception as exp:
            self._log.error('Failed with error %s ' % str(exp))
            self.result_string = str(exp)
            self.status = constants.FAILED
        finally:
            try:
                drhelperobject.startservices()
                drhelperobject.cleanup_entities(clenup_enties)
            except Exception:
                self._log.error("Failed to start services")
