[docs]    def init_tc(self):
        """Initialize browser and redirect to required report page"""
        try:
            self.browser = BrowserFactory().create_browser_object()
            self.browser.open()
            self.admin_console = AdminConsole(self.browser, self.commcell.webconsole_hostname)
            self.admin_console.login()
            self.navigator = self.admin_console.navigator
            self.navigator.navigate_to_big_data()
            self.instances = Instances(self.admin_console)
            if self.instances.is_instance_exists(self.cassandra_server_name):
                self.delete_cassandra_instance()
            self.cassandra_server = CassandraServer(self.admin_console)
            self.overview = Overview(self.admin_console)
        except Exception as _exception:
            raise CVTestCaseInitFailure(_exception) from _exception


    @test_step
    def create_cassandra_instance(self):
        """verify cassandra instance is created"""
        self.log.info("Creating [%s] cassandra server instance", self.cassandra_server_name)
        _cassandra_server = self.instances.add_cassandra_server()
        _cassandra_server.set_name(name=self.cassandra_server_name)
        _cassandra_server.select_gateway_node(node=self.gateway_node)
        _cassandra_server.set_config_file_path(conf_file_path=self.config_file_path)
        _cassandra_server.set_cql_host(host_name=self.cql_host)
        _cassandra_server.select_plan(plan_name=self.plan)
        _cassandra_server.expand_authentication_settings()
        _cassandra_server.enable_cql(user_name=self.cql_username, password=self.cql_password)
        _cassandra_server.enable_jmx(user_name=self.jmx_username, password=self.jmx_password)
        _cassandra_server.save()
        if not self.instances.is_instance_exists(self.cassandra_server_name):
            raise CVTestStepFailure("[%s] Cassandra server is not getting created " %
                                    self.cassandra_server_name)
        self.log.info("Successfully Created [%s] cassandra server instance",
                      self.cassandra_server_name)

    @test_step
    def verify_backup(self):
        """Initiate the backup and verify backup is completed"""
        # Initiate backup job
        self.instances.access_instance(self.cassandra_server_name)
        self.overview.backup_now()

        # Wait till backup job to complete
        _job_id = self.overview.get_jobid_from_popup()
        job = self.commcell.job_controller.get(_job_id)
        self.log.info("Wait for [%s] job to complete", str(job))
        if job.wait_for_completion(300):  # wait for max 5 minutes
            self.log.info("Backup job completed with job id:[%s]", _job_id)
        else:
            err_str = "[%s] job id for backup job failed" % _job_id
            raise CVTestStepFailure(err_str)

        # Verify backup is completed by checking the node in configuration tab
        _configuration = self.overview.access_configuration()
        if _configuration.is_nodes_exists():
            self.log.info("Data center backed up successfully")
            return
        raise CVTestStepFailure("Nodes are not present in configuration tab for [%s] "
                                "instance" % self.cassandra_server_name)

    @test_step
    def verify_restore(self):
        """Verify restore"""
        # Initiate restore job
        self.navigator.navigate_to_big_data()
        self.instances.access_instance(self.cassandra_server_name)
        self.instances.access_restore()
        _restore = self.overview.restore_all()
        _restore.ok()

        # Wait for restore job to complete
        _job_id = self.overview.get_jobid_from_popup()
        job = self.commcell.job_controller.get(_job_id)
        self.log.info("Wait for [%s] restore job to complete", str(job))
        if job.wait_for_completion(300):  # wait for max 5 minutes
            self.log.info("Restore job completed with job id:[%s]", _job_id)
        else:
            err_str = "[%s] job id for restore job failed" % _job_id
            raise CVTestStepFailure(err_str)
        self.log.info("Restore completed successfully")

    @test_step
    def delete_cassandra_instance(self):
        """Delete cassandra instance"""
        self.log.info("deleting [%s] instance", self.cassandra_server_name)
        self.navigator.navigate_to_big_data()
        self.instances.delete_instance_name(self.cassandra_server_name)
        self.commcell.clients.refresh()
        self.commcell.clients.delete(self.cassandra_server_name)
        self.browser.driver.refresh()
        self.instances.wait_for_completion()
        if self.instances.is_instance_exists(self.cassandra_server_name):
            raise CVTestStepFailure("[%s] Cassandra server is not getting deleted" %
                                    self.cassandra_server_name)
        self.log.info("Deleted [%s] instance successfully", self.cassandra_server_name)
