[docs]    def create_alert_situation(self):
        # Set additional setting required to trigger particular alert situation
        # Add Additional Setting to the commcell
        self.log.info('Adding Additional setting for Min Free Space Percentage')
        self.commcell.add_additional_setting(category='QMachineMaint',
                                             key_name='nMinFreeSpacePercentage',
                                             data_type='INTEGER',
                                             value='98')
        self.log.info('Setting added successfully')
        self.log.info('Adding setting for Running System directory space check thread at given minutes')
        self.commcell.add_additional_setting(category='QMachineMaint',
                                             key_name='nTimer_SystemDirSpaceCheck',
                                             data_type='INTEGER',
                                             value='1')
        self.log.info('Setting added successfully')

        # Restart Client Services
        self.log.info('Restarting Client Services')
        self.client.restart_services()
        self.log.info('Services Restarted Successfully')



[docs]    def run(self):
        """Main function for test case execution"""
        try:
            # Initialize alerts object
            self.log.info('Initializing Alerts')
            alert_helper = AlertHelper(commcell_object=self.commcell,
                                       category='Configuration', alert_type='MediaAgents')

            # Creating Alert of given type
            # Feed alert data to create the right format
            alert_data = alert_helper.get_alert_details(name='TEST DISK SPACE LOW MA',
                                                        notif_type=['Email', 'Console Alerts'],
                                                        entities={'media_agents': self.tcinputs['ClientName']},
                                                        users=["admin"],
                                                        criteria=18,
                                                        mail_recipent="TestAutomation3@commvault.com",
                                                        )
            # Initialize Mailbox object
            alert_helper.initialize_mailbox()

            self.log.info('Creating Alert %s for testcase %s', alert_data.get("alert_name"), self.id)
            alert_helper.create_alert()

            # Creating the alert situation to trigger the alert
            self.create_alert_situation()

            # Function to read email and confirmation alert notification
            alert_helper.check_if_alert_mail_received(short_interval=100, patterns=[alert_data["entities"].get('media_agents'),
                                                                          alert_helper.alert_category,
                                                                          alert_helper.alert_type])


        except Exception as excp:
            self.log.info('Failed with error %s ', str(excp))
            # Set the Test-Case params : result_string, status
            self.result_string = str(excp)
            self.status = constants.FAILED
        finally:
            # Cleanup phase
            # Alerts Cleanup
            alert_helper.cleanup()
            # Misc cleanup
            self.misc_cleanup()
