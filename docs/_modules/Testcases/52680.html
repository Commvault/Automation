[docs]    def run(self):
        """Main function for test case execution"""

        log = self.log
        logdir = self._log_dir
        time1 = (datetime.datetime.now()).strftime("%H:%M:%S")
        clientname = self.client.client_name
        instancename = self.instance.instance_name
        subclientname = "Subclient{0}_{1}".format(self.id, time1)
        sqluser = self.tcinputs["SQLServerUser"]
        sqlpass = self.tcinputs["SQLServerPass"]
        storagepolicy = self.tcinputs["StoragePolicyName"]
        self.sqlmachine = Machine(clientname)
        self.sqlhelper = SQLHelper(self, clientname, instancename, sqluser, sqlpass)

        ransum = ""
        for i in range(1, 7):
            ransum = ransum + random.choice("abcdefghijklmnopqrstuvwxyz")
        self.dbname = ransum

        try:
            sqldump_file1 = "before_backup_full.txt"
            sqldump_file2 = "after_restore.txt"
            self.tcdir = os.path.normpath(os.path.join(logdir, self.id + '-' + ransum))

            noofdbs = 5
            nooffilegroupsforeachdb = 3
            nooffilesforeachfilegroup = 4
            nooftablesforeachfilegroup = 10
            noofrowsforeachtable = 50
            randomization = 100

            log.info("Started executing {0} testcase".format(self.id))
            if not self.sqlmachine.create_directory(self.tcdir):
                raise Exception("Failed to create staging directory. ")

            subcontent = []
            for i in range(1, int(noofdbs) + 1):
                db = self.dbname + repr(i)
                subcontent.append(db)

            # perform database check if exists, if so, drop it first.
            if self.sqlhelper.dbinit.check_database(self.dbname):
                if not self.sqlhelper.dbinit.drop_databases(self.dbname):
                    raise Exception("Unable to drop the database")

            # create databases
            log.info("*" * 10 + " Creating database [{0}] ".format(self.dbname) + "*" * 10)
            if not self.sqlhelper.dbinit.db_new_create(self.dbname, noofdbs, nooffilegroupsforeachdb,
                                             nooffilesforeachfilegroup, nooftablesforeachfilegroup,
                                             noofrowsforeachtable):
                raise Exception("Failed to create databases.")

            # create subclient
            log.info("*" * 10 + " Creating subclient [{0}] ".format(subclientname) + "*" * 10)
            if not self.sqlhelper.create_subclient(subclientname, subcontent, storagepolicy):
                self._subclient = self._instance.subclients.get(subclientname)
                raise Exception("Failed to create subclient.")
            else:
                self._subclient = self._instance.subclients.get(subclientname)

            # get table shuffled list
            returnstring, list1, list2, list3 = self.sqlhelper.dbvalidate.get_random_dbnames_and_filegroups(
                randomization, noofdbs, nooffilegroupsforeachdb, nooftablesforeachfilegroup)
            if not returnstring:
                raise Exception("Error in while generating the random number.")

            # run a TL1 backup - should convert to full
            job_id = self.sqlhelper.sql_backup('transaction_log')

            # check backup level for TL1 - should be full
            backuplevel = self.sqlhelper.dbvalidate.get_sql_backup_type(job_id, multidb=False)
            if str(backuplevel) != "Full":
                raise Exception("Wrong backup level of the database detected in the job")
            else:
                log.info("Backup level confirmed: " + str(backuplevel))            
                       
            # run a TL2 backup
            job_id = self.sqlhelper.sql_backup('transaction_log')

            # check backup level for TL2
            backuplevel = self.sqlhelper.dbvalidate.get_sql_backup_type(job_id, multidb=False)
            if str(backuplevel) != "Transaction Log":
                raise Exception("Wrong backup level of the database detected in the job")
            else:
                log.info("Backup level confirmed: " + str(backuplevel))

            # modify database before TL3
            if not self.sqlhelper.modifydatabase.modify_db_for_inc(self.dbname, list1, list2, list3):
                raise Exception("Databases couldnt be modified.")

            # Checking if databases are online
            if not self.sqlhelper.dbvalidate.is_db_online(self.dbname, noofdbs):
                raise Exception("Databases are not online.")
           
            # write the database to file for comparison before DIFF
            if not self.sqlhelper.dbvalidate.dump_db_to_file(os.path.join(self.tcdir, sqldump_file1),
                                                   self.dbname, list1, list2, list3, 'INCREMENTAL'):
                raise Exception("Failed to write database to file.")

            # run a TL3
            job_id = self.sqlhelper.sql_backup('transaction_log')
            
            # check backup level for TL3
            backuplevel = self.sqlhelper.dbvalidate.get_sql_backup_type(job_id, multidb=False)
            if str(backuplevel) != "Transaction Log":
                raise Exception("Wrong backup level of the database detected in the job")
            else:
                log.info("Backup level confirmed: " + str(backuplevel))

            # get backup end time of TL3 from CS db
            tl3end = self.sqlhelper.get_sql_backup_end_time(job_id, timeformat="datestringformat")
            log.info("Job End time - " + tl3end)
            
            # drop databases
            if not self.sqlhelper.dbinit.drop_databases(self.dbname):
                log.error("Unable to drop the dataBase")            

            # run restore in place job - based on TL3 time
            log.info("*" * 10 + " Run Restore in place " + "*" * 10)
            if not self.sqlhelper.sql_restore(self.subclient.content, to_time=tl3end):
                raise Exception("Restore was not successful!")

            # Checking if databases are online
            if not self.sqlhelper.dbvalidate.is_db_online(self.dbname, noofdbs):
                raise Exception("Databases are not online.")

            # write the restored database to file for comparison
            if not self.sqlhelper.dbvalidate.dump_db_to_file(os.path.join(self.tcdir, sqldump_file2),
                                                   self.dbname, list1, list2, list3, 'INCREMENTAL'):
                raise Exception("Failed to write database to file.")

            # compare original and restored databases
            log.info("*" * 10 + " Validating content " + "*" * 10)
            if not self.sqlhelper.dbvalidate.db_compare(os.path.join(self.tcdir, sqldump_file1),
                                              os.path.join(self.tcdir, sqldump_file2)):
                raise Exception("Failed to compare both files.")

            log.info("*" * 10 + " TestCase {0} successfully completed! ".format(self.id) + "*" * 10)
            self.status = constants.PASSED

        except Exception as excp:
            log.error('Failed with error: ' + str(excp))
            self.result_string = str(excp)
            self.status = constants.FAILED

        """Tear down function"""

        log = self.log

        # drop databases
        if not self.sqlhelper.dbinit.drop_databases(self.dbname):

        # delete directory
        if not self.sqlmachine.remove_directory(self.tcdir):
            log.error("Unable to delete temporary testcase folder: [{0}]".format(self.tcdir))
        # delete subclient
        self._instance.subclients.delete(self.subclient.subclient_name)
