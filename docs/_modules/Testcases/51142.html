'''
Test case to check the basic acceptance of Plans in Admin console.

functions over,
1. Creation of alerts based on different criteria's passed as
   arguments to the test case and base files.
2. Verifies the triggering is working fine for restore jobs.
3. Checks for the user permissions validation on Alerts is working properly.
4. Deletes the alerts created & verified in above steps.

'''

import datetime
from AutomationUtils.cvtestcase import CVTestCase
from Web.Common.cvbrowser import BrowserFactory
from Web.AdminConsole.Helper.LoginHelper import LoginMain
from Web.AdminConsole.Helper.AlertHelper import AlertMain
from Reports.utils import TestCaseUtils


[docs]class TestCase(CVTestCase):
    ''' Basic Acceptance test for Alerts '''
    def __init__(self):
        '''
       Initializing the Test case file
        '''
        super(TestCase, self).__init__()
        self.name = "Basic Acceptance Test Alerts in AdminConsole"
        self.product = self.products_list.COMMSERVER
        self.feature = self.features_list.ADMINCONSOLE
        self.show_to_user = False
        self.utils = TestCaseUtils(self)

        self.tcinputs = {
            'recipients': None,
            'client_name': None,
            'test_subclient_name': None,
            'plan_name': None,
            'backupdata': None,
            'restore_path': None,
            're_login_username':None,
            're_login_password':None,
            'alert_target':None
        }


[docs]    def run(self):

        try:

            self.log.info("Started executing %s testcase", self.id)
            self.log.info("*" * 10 + " Initialize browser objects " + "*" * 10)

            factory = BrowserFactory()
            browser = factory.create_browser_object()
            browser.open()
            driver = browser.driver

            self.log.info("Creating the login object")
            login_obj = LoginMain(driver, self.csdb)

            login_obj.login(self.inputJSONnode['commcell']['commcellUsername'],
                            self.inputJSONnode['commcell']['commcellPassword'])
            self.log.info("login successful")

            alert_obj = AlertMain(driver, self.commcell, self.csdb)

            alert_obj.alert_name = "Restore_Alert " + str(datetime.datetime.today())
            alert_obj.alert_criteria = "Restore Job Succeeded"
            alert_obj.recipients = self.tcinputs['recipients']
            alert_obj.client_name = self.tcinputs['client_name']
            alert_obj.subclient_name = self.tcinputs['test_subclient_name']
            alert_obj.plan_name = self.tcinputs['plan_name']
            alert_obj.backupdata = self.tcinputs['backupdata']
            alert_obj.restore_path = self.tcinputs['restore_path']
            alert_obj.alert_target = self.tcinputs['alert_target']

            alert_obj.add_alert_definition()
            self.log.info("alert definition created successfully")

            alert_obj.create_fs_subclient()
            self.log.info("Test subclient added to the given client, now initiating backup")

            alert_obj.perform_fs_subclient_backup()
            self.log.info("Backup Completed for test subclient")

            alert_obj.perform_fs_subclient_restore()
            self.log.info("disabling and validating succeeded. Now re-enabling it")

            alert_obj.validate_triggered_alert()
            self.log.info("Validation succeeded.")

            alert_obj.logout_and_navigate_to_login_page()
            self.log.info("Logged out of application and routed back to login page.\
            Logging back with user having 'View All' Permission")

            login_obj.re_login(self.tcinputs['re_login_username'],
                               self.tcinputs['re_login_password'])
            self.log.info("Logged in as user having 'View All' Permission")

            alert_obj.add_alert_definition()
            self.log.info("Validation for user having 'View All' Permission, \
            completed successfully and was not able to create alert")

            alert_obj.logout_and_navigate_to_login_page()
            self.log.info("Logged out of application and routed back to login page.\
            Logging back with user having 'admin' Permission")

            login_obj.re_login(self.inputJSONnode['commcell']['commcellUsername'],
                               self.inputJSONnode['commcell']['commcellPassword'])
            self.log.info("re-logged in as admin user")

        except Exception as exp:
            self.utils.handle_testcase_exception(exp)

        finally:

            alert_obj.delete_test_subclient()
            self.log.info("Test Subclient deleted successfully")

            alert_obj.delete_alert_definition()
            self.log.info("Alert definition deleted successfully")
            self.log.info("Test case completed. Exiting module.....Thanks")
            browser.close()
