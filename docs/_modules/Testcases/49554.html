    test_step = TestStep()

    def __init__(self):
        super(TestCase, self).__init__()
        self.name = "Custom Reports - Validate TextBox inputs"
        self.applicable_os = self.os_list.WINDOWS
        self.product = self.products_list.SOFTWARESTORE
        self.feature = self.features_list.WEBCONSOLE
        self.show_to_user = True
        self.browser = None
        self.webconsole = None
        self.navigator = None
        self.builder: ReportBuilder = None
        self.textbox: TextBoxController = None
        self.input_variable = None

        # TC constant inputs
        self.expected_data = {"No Column Name1": ("""||--~!@#$%^&*()_+{}:"\:{}""", )}
        self.input_display_name = "Test Input Display Name"
        self.input_variable_name = "TestInputVariableName"
        self.utils = TestCaseUtils(self)


[docs]    def init_tc(self):
        try:
            self.utils.cre_api.delete_custom_report_by_name(self.name, suppress=True)
            self.browser = BrowserFactory().create_browser_object()
            self.browser.open()
            self.webconsole = WebConsole(self.browser, self.commcell.webconsole_hostname)
            self.webconsole.login()
            self.navigator = Navigator(self.webconsole)
        except Exception as e:
            raise CVTestCaseInitFailure(e) from e


    @test_step
    def start_step1(self):
        """Create report input UI component"""
        self.navigator.goto_report_builder()
        self.builder = ReportBuilder(self.webconsole)
        self.builder.set_report_name(self.name)
        self.input_variable = StringInputVariable(self.input_variable_name)
        self.builder.add_input(self.input_variable)
        self.textbox = TextBoxController(self.input_display_name)
        self.input_variable.add_html_controller(self.textbox)
        self.input_variable.save()

    @test_step
    def start_step2(self):
        """Add HTML input and parameter to Dataset and preview"""
        dataset = DatabaseDataset()
        self.builder.add_dataset(dataset)
        dataset.set_dataset_name("Test Dataset")
        dataset.set_sql_query("SELECT @sql_variable")
        dataset.add_parameter("sql_variable", self.input_variable)
        preview_data = dataset.get_preview_data(as_json=True)
        if preview_data != {'No Column Name1': ('', )}:
            raise CVTestStepFailure(
                "Unexpected preview data [%s]" % str(preview_data))
        dataset.save()

    @test_step
    def start_step3(self):
        """Associate dataset with input to Table and validate data"""
        table = BuilderDataTable("Test Table")
        self.builder.add_component(table, "Test Dataset")
        table.add_column_from_dataset("No Column Name1")
        self.textbox.set_textbox_controller("Text Input Data")
        self.textbox.apply()
        self.builder.save_and_deploy()
        table_data = table.get_table_data(as_json=True)
        if table_data != {"No Column Name1": ("Text Input Data", )}:
            raise CVTestStepFailure(
                "Unexpected data after applying input [%s]" % str(table_data))
        self.builder.goto_report_manager()

    @test_step
    def start_step4(self):
        """Set textbox input, and verify data on report viewer"""
        self.navigator.goto_worldwide_report(self.name)
        html_textbox = TextBoxController(self.input_display_name)
        table = ViewerDataTable("Test Table")
        viewer = CustomReportViewer(self.webconsole)
        viewer.associate_input(html_textbox)
        viewer.associate_component(table, "Test Table")
        html_textbox.set_textbox_controller(self.expected_data["No Column Name1"][0])
        html_textbox.apply()
        data = table.get_table_data(as_json=True)
        if data != self.expected_data:
            raise CVTestStepFailure(
                "Unexpected data [%s], expected [%s]" % (
                    str(data), str(self.expected_data)))

    @test_step
    def start_step5(self):
        """Bookmark the report URL with input and validate data"""
        self.browser.driver.refresh()
        table = ViewerDataTable("Test Table")
        viewer = CustomReportViewer(self.webconsole)
        viewer.associate_component(table, "Test Table")
        data = table.get_table_data(as_json=True)
        if data != self.expected_data:
            raise CVTestStepFailure(
                "Unexpected data [%s], expected [%s]" % (
                    str(data), str(self.expected_data)))

    @test_step
    def start_step6(self):
        """Submit input with HTML characters

        All the HTML characters will have
        """
        table = ViewerDataTable("Test Table")
        textbox = TextBoxController(self.input_display_name)
        viewer = CustomReportViewer(self.webconsole)
        viewer.associate_component(table, "Test Table")
        viewer.associate_input(textbox)
        textbox.set_textbox_controller("<h1>Inside</h1>Outside")
        textbox.apply()
        data = table.get_table_data(as_json=True)
        if data != {"No Column Name1": ("Outside", )}:
            raise CVTestStepFailure(
                "Unexpected data [%s], expected [%s]" % (
                    str(data), str({"No Column Name1": ("Outside", )})))


[docs]    def run(self):

            self.start_step2()

            self.start_step4()
            self.start_step5()

        finally:
            WebConsole.logout_silently(self.webconsole)
            Browser.close_silently(self.browser)
