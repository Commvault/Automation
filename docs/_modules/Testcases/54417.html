[docs]    def init_tc(self):
        """ Initial configuration for the test case. """
        try:
            self.browser = BrowserFactory().create_browser_object()
            self.browser.open()
            self.webconsole = WebConsole(self.browser, self.commcell.webconsole_hostname)
            self.utils.webconsole = self.webconsole
            self.webconsole.login()
            self.webconsole.goto_reports()
            self.utils.cre_api.delete_custom_report_by_name(self.name, suppress=True)
            navigator = Navigator(self.webconsole)
            navigator.goto_report_builder()
        except Exception as exception:
            raise CVTestCaseInitFailure(exception)from exception


    @test_step
    def create_shared_dataset(self):
        """Creates Shared dataset"""
        self.report_builder = builder.ReportBuilder(self.webconsole)
        self.report_builder.set_report_name(self.name)

        self.shared_dataset = builder.Datasets.SharedDataset()
        self.report_builder.add_dataset(self.shared_dataset)
        self.shared_dataset.set_dataset_name("Shared DS")
        self.shared_dataset.select_dataset("CommcellList")
        self.shared_dataset.save()

    @test_step
    def verify_alias_name(self):
        """Verifying alias name from commcell datasource"""
        self.table = builder.DataTable("Automation Table 54417")
        self.report_builder.add_component(self.table, self.shared_dataset)
        self.table.add_column_from_dataset()
        col = self.table.Column('aliasName')
        self.table.associate_column_in_builder(col)
        elements = self.utils.get_commcell_datasources()
        col_list = self.table.get_column_data('aliasName')
        elements = [item.lower() for item in elements]
        col_list = [item.lower() for item in col_list]
        if col_list != sorted(elements):
            self.log.error("Expected : ", elements.sort(), "received : ", col_list)
            raise CVTestStepFailure("CommCell name received from dataset [CommcellList] is not as expected")
        self.report_builder.save(deploy=True)
