'''
Test case to check the basic acceptance of Editing Plan Details in Admin console for
laptop plan.

functions over,
1. Creation of laptop plan based on different criteria's passed as
    arguments to the test case and base files.
2. Edits values and checks if the updated values are retained correctly.
3. Deletes the plan created, edited and verified in above steps.

Pre-requisites :
1. Index Server should be configured, if using Edge Drive Feature for Laptop Plam.
2. Primary and secondary storage pools should be configured.
'''

from AutomationUtils.cvtestcase import CVTestCase
from Web.Common.cvbrowser import BrowserFactory
from Web.AdminConsole.Helper.LoginHelper import LoginMain
from Web.AdminConsole.Helper.PlanHelper import PlanMain
from Reports.utils import TestCaseUtils


[docs]class TestCase(CVTestCase):

    ''' Basic Acceptance Test for Editing Plan Details for laptop plan
        in AdminConsole '''

    def __init__(self):
        '''
       Initializing the Test case file
        '''
        super(TestCase, self).__init__()
        self.name = "Basic Acceptance Test for  Editing Plan Details in AdminConsole"
        self.product = self.products_list.COMMSERVER
        self.feature = self.features_list.ADMINCONSOLE
        self.show_to_user = False
        self.utils = TestCaseUtils(self)

        self.tcinputs = {
            'plan_name' : None,
            'username': None,
            'password': None,
            'primary_storage' : None,
            'secondary_storage': None,
            'user_usergroup_association' : None,
            'user_usergroup_de_association' : {
                'DeleteAll': None,
                'DeleteAllUsers': None,
                'DeleteAllUserGroups': None,
                'Delete_Specific_user_usergroup': None}
            }


[docs]    def run(self):

        try:

            self.log.info("Started executing {0} testcase".format(self.id))
            self.log.info("*" * 10 + " Initialize browser objects " + "*" * 10)

            factory = BrowserFactory()
            browser = factory.create_browser_object()
            browser.open()
            driver = browser.driver

            self.log.info("Creating the login object")
            login_obj = LoginMain(driver, self.csdb)

            login_obj.login(self.inputJSONnode['commcell']['commcellUsername'],
                            self.inputJSONnode['commcell']['commcellPassword'])
            self.log.info("login successful")

            self.log.info("Creating Plan Helper Object")
            plan_obj = PlanMain(driver)

            self.log.info("Creating new plan")

            plan_obj.plan_name = self.tcinputs['plan_name']
            plan_obj.secondary_storage = self.tcinputs['secondary_storage']
            plan_obj.storage_name = self.tcinputs['primary_storage']
            plan_obj.edit_plan_storage = {'Edit': True,
                                          'old_storage_name': self.tcinputs['primary_storage'],
                                          'new_storage_name':'Primary',
                                          'new_ret_period':'20'}
            plan_obj.user_usergroup_association = self.tcinputs['user_usergroup_association']
            plan_obj.user_usergroup_de_association = self.tcinputs['user_usergroup_de_association']

            plan_obj.add_plan()
            self.log.info("Plan Creation completed. Starting to Edit Plan values.")

            plan_obj.edit_laptop_plan()
            self.log.info("Plan Editing completed. validating plan")

            plan_obj.validate_plan_details()
            self.log.info("Plan validation completed and passed successfully.")

        except Exception as exp:
            self.utils.handle_testcase_exception(exp)

        finally:

            plan_obj.remove_associations()
            self.log.info("Plan associations removed successfully")

            plan_obj.delete_plans()
            self.log.info("Plan Deleted successfully. Test case completed. \
Exiting module.....Thanks")
            browser.close()
