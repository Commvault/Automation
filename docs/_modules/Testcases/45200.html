        """Run function of this test case"""
        try:
            self._log.info("Started executing {0} testcase".format(self.id))
            libraryName = str(self.id) + "_lib"
            storagePolicyName = str(self.id) + "_SP"
            backupsetName = str(self.id) + "_BS"
            subclientName = str(self.id) + "_SC"
            self._csdb = database_helper.CommServDatabase(self._commcell)
            dict_prim = {}
            dict_sec = {}
            self._log.info(self.name)

            #dedupe.setup_environment(libraryName,storagePolicyName)
            self._log.info("********* previous run cleaning up ***********")
            try:
                self._agent.backupsets.delete(backupsetName)
                self.commcell.storage_policies.delete(storagePolicyName)
            except:
                self._log.info("previous run(if any) cleanup errors")
                pass

            # create library
            self._log.info("check library: "+libraryName)
            if not self.commcell.disk_libraries.has_library(libraryName):
                self._log.info("adding Library...")
                self.commcell.disk_libraries.add(libraryName,self.tcinputs["MediaAgentName"],self.tcinputs["MountPath"],self.tcinputs["MediaAgentUsername"],self.tcinputs["MediaAgentPassword"]) #self.commcell.commserv_name.encode()
            else:
                self._log.info("Library exists!")
            self._log.info("Library Config done.")

            #create SP
            self._log.info("check SP: "+storagePolicyName)
            if not self.commcell.storage_policies.has_policy(storagePolicyName):
                self._log.info("adding Storage policy...")
                self.commcell.storage_policies.add(storagePolicyName,libraryName,self.tcinputs["MediaAgentName"],self.tcinputs["DedupeStorePath"]+str(time.time()))
            else:
                self._log.info("Storage policy exists!")
            self._log.info("Storage policy config done.")

            #create BS
            self._log.info("check BS: "+ backupsetName)
            #client = self.commcell.clients.get(self.tcinputs["ClientName"])
            #agent = client.agents.get(self.tcinputs["AgentName"])
            if not self._agent.backupsets.has_backupset(backupsetName):
                self._log.info("adding Backupset...")
                self._agent.backupsets.add(backupsetName)
            else:
                self._log.info("Backupset exists!")
            self._log.info("Backupset config done.")

            #create SC
            self._log.info("check SC: "+subclientName)
            self._log.info("creating backupset object: "+backupsetName)
            self._backupset = self._agent.backupsets.get(backupsetName)
            if not self._backupset.subclients.has_subclient(subclientName):
                self._log.info("adding Subclient...")
                self._subclient = self._backupset.subclients.add(subclientName,storagePolicyName)
            else:
                self._log.info("Subclient exists!")

            #add subclient content
            self._log.info("creating subclient object: "+subclientName )
            self._subclient = self._backupset.subclients.get(subclientName)
            self._log.info("setting subclient content to: "+ str([self.tcinputs["ContentPath"]]))
            self._subclient.content = [self.tcinputs["ContentPath"]]
            self._log.info("Subclient config done.")

            #enable encyption
            self._log.info("enabling encryption on client")
            self.client.set_encryption_property("ON_CLIENT", key="4", key_len="256")
            self._log.info("enabling encryption on client: Done")

            #enabling client side cache
            self._log.info("enabling client side cache")
            self.client.set_dedup_property("clientSideDeduplication","ON_CLIENT",True)
            self._log.info("enabling client side cache: Done")

            # initialize dedupehelper class
            dedupe = DedupeHelper(self)

            # Run FULL backup
            self._log.info("Running full backup...")
            for iterator in range(1,3):
                job = self._subclient.backup("FULL")
                self._log.info("Backup job: "+str(job.job_id))
                if not job.wait_for_completion():
                   raise Exception(
                       "Failed to run FULL backup with error: {0}".format(job.delay_reason)
                   )
                self._log.info("Backup job completed.")

                #validations
                self._log.info("*************** Validations ****************")
                log_file="clBackup.log"
                log_path = self.client.install_directory+'\\Log Files'
                config_strings_clbackup = ['sigWhere[3]','Cache-DB lookup - Yes','Compress-enabled - Yes','sigScheme[4]','compressWhere[0]','encType[2]','CVSingleInstTarget[1]']
                error_flag = []
                self._log.info("CASE 1: IS CLIENT SIDE DEDUPE ENABLED?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file, config_strings_clbackup[0],job.job_id)
                if matched_line:
                    self._log.info("Result: Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: "+config_strings_clbackup[0]]

                self._log.info("CASE 2: IS CACHE DB ENABLED?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  config_strings_clbackup[1], job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: "+config_strings_clbackup[1]]

                self._log.info("CASE 3: IS COMPRESSION ENABLED?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  config_strings_clbackup[2], job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + config_strings_clbackup[2]]

                self._log.info("CASE 4: IS SIGNATURE SCHEMA 4?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  config_strings_clbackup[3], job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + config_strings_clbackup[3]]

                self._log.info("CASE 5: IS COMPRESSION ON CLIENT?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  config_strings_clbackup[4], job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + config_strings_clbackup[4]]

                self._log.info("CASE 6: IS ENCRYPTION TYPE SERPENT?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  config_strings_clbackup[5], job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + config_strings_clbackup[5]]

                self._log.info("CASE 7: IS CV SINGLE INSTANCE ENABLED?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  config_strings_clbackup[6], job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + config_strings_clbackup[6]]

                if error_flag:
                    raise Exception(
                        error_flag
                    )

                #signatures processed and skipped validation
                self._log.info("*** Signature processed/skipped validations ***")
                error_flag = []
                self._log.info("CASE 1: SIGNATURES PROCESSED COUNT")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  'Signatures processed', job.job_id)
                if matched_line:
                    signature_processed = matched_line[0].split("-")[2].split("]")[0]
                    no = signature_processed
                    self._log.info("Signature processed =" + signature_processed)
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + 'Signatures processed']

                self._log.info("CASE 2: SIGNATURES FOUND IN CACHE COUNT")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  'Found in cache', job.job_id)
                if matched_line:
                    signature_processed = matched_line[0].split("-")[6].split("]")[0]
                    #self._log.info("Signature processed =" + signature_processed)
                    #self._log.info("found in cache %s" % (matched_string[0].split('-')[-1]))
                    no1 = signature_processed
                    self._log.info("signatures processed %s" % (no))
                    self._log.info("matched %s" % (no1))

                    if (iterator == 1 and signature_processed == ' 0') or (
                            iterator == 2 and no1 == no):
                        self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + 'Found in cache']

                self._log.info("CASE 3: SIGNATURES FOUND IN DDB COUNT")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  'Found in DDB', job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + 'Found in DDB']

                self._log.info("CASE 4: GET APPLICATION SIZE")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  'Size-Application', job.job_id)
                if matched_line:
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Failed")
                    error_flag += ["failed to find: " + 'Size-Application']

                self._log.info("CASE 5: IS SIZE COMPRESSED == SIZE TRANSFERRED?")
                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  'Size-processed', job.job_id)
                if matched_line:
                    compressed_size = matched_line[0].split("-")[5].split(" ")[1]
                    no2 = compressed_size
                    self._log.info("compressed_size %s" % (compressed_size))

                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  'Size-New Data-', job.job_id)
                if matched_line:
                    transferred_size = matched_line[0].split("-")[9].split("]")[0].split(" ")[1]
                    no3 = transferred_size
                    self._log.info("transferred_size %s" % (transferred_size))

                (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                                  'Size-Discarded', job.job_id)
                if matched_line:
                    discarded_size = matched_line[0].split("-")[7].split("]")[0].split(" ")[1]
                    no4 = discarded_size
                    self._log.info("discarded_size %s" % (discarded_size))

                if (iterator == 1 and no2 == no3) or (
                        iterator == 2 and no4 == no2):
                    self._log.info("Result :Pass")
                else:
                    self._log.error("Result: Fail")
                    error_flag += ["failed in check: Size-Compressed == Size-Transferred"]

                self._log.info(r"CASE 6: DID DEDUPE OCCUR? 2nd BACKUP SHOULD HAVE 0 PRIMARY OBJECTS")
                data = dedupe.get_primary_objects(job.job_id)
                self._log.info("Primary Objects :" + str(data))
                data1 = dedupe.get_secondary_objects(job.job_id)
                self._log.info("Secondary Objects :" + str(data1))
                dict_prim[iterator] = data
                dict_sec[iterator] = data1
                if iterator == 2 and dict_sec[iterator] == dict_prim[iterator - 1] and dict_prim[iterator] == dict_sec[
                            iterator - 1]:
                    self._log.info("Dedupe validation:SUCCESS")
                else:
                    if iterator == 1:
                        self._log.info("validation will be done at the end of next iterator")
                    else:
                        self._log.error("Dedupe validation:FAILED")
                        error_flag += ["failed to validate Dedupe"]

                if error_flag:
                    raise Exception(
                        error_flag
                    )

            #restore
            self._log.info("running restore job")
            restorejob = self._subclient.restore_out_of_place(self.tcinputs["MediaAgentName"],self.tcinputs["RestorePath"],[self.tcinputs["ContentPath"]],True,True)
            self._log.info("Restore job: "+restorejob.job_id)
            if not restorejob.wait_for_completion():
                raise Exception(
                    "Failed to run FULL backup with error: {0}".format(restorejob.delay_reason)
                )
            self._log.info("restore job completed")

            #incremental before synth full
            incrjob = self._subclient.backup("incremental")
            self._log.info("Incremental job: "+str(incrjob.job_id))
            if not incrjob.wait_for_completion():
                raise Exception(
                    "Failed to run Incr backup with error: {0}".format(incrjob.delay_reason)
                )
            self._log.info("Incremental Backup job completed.")

            #synth full
            sfjob = self._subclient.backup("Synthetic_full",False)
            self._log.info("Synth full job: "+str(sfjob.job_id))
            if not sfjob.wait_for_completion():
                raise Exception(
                    "Failed to run Synth full backup with error: {0}".format(sfjob.delay_reason)
                )
            self._log.info("Synth full Backup job completed.")

            #synth full backup validation
            # New signatures encounterd - 0
            errorcount = 0
            log_file = 'SynthFull.log'
            self._log.info("VALIDATING SYNTHFULL JOB: NEW SIGNATURES ENCOUNTERRED SHOULD BE 0")
            (matched_line, matched_string) = dedupe.parse_log(self.client.client_name, log_file,
                                                              'New signatures encounterd - 0', sfjob.job_id)
            if matched_line:
                self._log.info("Result :Pass")
            else:
                self._log.error("Result: Failed")
                errorcount += 1
            self._log.info("VALIDATING SYNTHFULL JOB: PRIMARY OBJECTS IN CSDB SHOULD BE 0")
            data = dedupe.get_primary_objects(sfjob.job_id)
            self._log.info("Primary Objects :" + str(data))
            if str(data) == "0":
                self._log.info("Result: Pass")
            else:
                self._log.error("Result: Fail")
                errorcount += 1
            if errorcount == 2:
                raise Exception(
                    "Failed in synthfull job validation"
                )
            self._log.info("VALIDATING SYNTHFULL JOB: SECONDARY OBJECTS IN CSDB SHOULD BE GREATER THAN 0")
            data = dedupe.get_secondary_objects(sfjob.job_id)
            self._log.info("Secondary Objects :" + str(data))
            if str(data) == "0":
                self._log.error("Result: Fail")
                errorcount += 1
            else:
                self._log.info("Result: Pass")
            if errorcount >= 2:
                raise Exception(
                    "Failed in synthfull job validation"
                )

            #age data
            #self._log.info("delete job - incomplete code here.")

            #cleanup
            try:
                self._log.info("********* cleaning up ***********")
                self._agent.backupsets.delete(backupsetName)
                self.commcell.storage_policies.delete(storagePolicyName)
            except:
                self._log.info("something went wrong while cleanup.")
                pass

        except Exception as exp:
            self._log.error('Failed to execute test case with error: ' + str(exp))
            self.result_string = str(exp)
            self.status = constants.FAILED
