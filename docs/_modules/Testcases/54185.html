    @test_step
    def validate_adminconsole_sla(self):
        """
         validate the clienta that are in admin console SLA
        """
        self.sla.access_missed_sla()
        self.SLAReportClientList.clear()
        self.log.info("Verifying for Failed laptop clients")
        self.SLAReportClientList.append(
            SLAReportDataExpected(self.FailedLaptopClient, ""))
        title = 'Unprotected servers'
        self.validate_adminconsole_sla_clients(title)
        self.admin_console.select_breadcrumb_link_using_text("SLA")
        self.log.info("Verifying successful laptop clients")
        self.SLAReportClientList.clear()
        self.SLAReportClientList.append(
            SLAReportDataExpected(self.SuccessLaptopClient, ""))
        self.sla.access_met_sla()
        title = 'Protected servers'
        self.validate_adminconsole_sla_clients(title)


    @test_step
    def validate_adminconsole_sla_clients(self, table_name):
        """
        validate the clients in Admin console SLA

        Args:
            table_name:  missed SLA table
            report_type: web or metrics report
        """

        report_viewer = viewer.CustomReportViewer(self.webconsole)
        table = viewer.DataTable(table_name)
        report_viewer.associate_component(table)
        sla_table_data = table.get_table_data()
        server_list = sla_table_data['Server']
        rowcount = len(server_list)
        if not rowcount:
            if self.SLAReportClientList:
                raise CVTestStepFailure(
                    "Mismatched SLA, table is empty but expected list is non empty")

        for sla_object in self.SLAReportClientList:
            expected_entry_found = False
            if sla_object.client_name in server_list:
                expected_entry_found = True
            if not expected_entry_found:
                raise CVTestStepFailure(
                    "Mismatched SLA, expected entry client [%s]  not found in table [%s]" % (
                        sla_object.client_name,
                        table_name
                    )
                )
        self.log.info("Client [{0}] is present in the table [{1}]".format(
            sla_object.client_name, table_name))

        for sla_object in self.SLAExcludedList:
            entry_found = False
            if sla_object.client_name in server_list:
                entry_found = True
            if entry_found:
                raise CVTestStepFailure(
                    "Mismatched SLA, Excluded entry client [%s] found in table [%s]" % (
                        sla_object.client_name,
                        table_name
                    )
                )

    @test_step
    def validate_sla_clients(self, table_name):
        """
        validate the clients  in web  SLA

        Args:
            table_name:  missed SLA table
            report_type: web or metrics report
        """

        table = MetricsTable(self.webconsole, table_name)

        rowcount = table.get_rows_count()
        if rowcount == 0:
            if self.SLAReportClientList:
                raise CVTestStepFailure(
                    "Mismatched SLA, table is empty but expected list is non empty")
        sla_table_data = table.get_data_from_column("Client")
        for sla_object in self.SLAReportClientList:
            expected_entry_found = False
            if sla_object.client_name in sla_table_data:
                expected_entry_found = True
            if not expected_entry_found:
                raise CVTestStepFailure(
                    "Mismatched SLA, expected entry client [%s] not found in table [%s]" % (
                        sla_object.client_name,
                        table_name
                    )
                )
        for sla_object in self.SLAExcludedList:
            entry_found = False
            if sla_object.client_name in sla_table_data:
                entry_found = True
            if entry_found:
                raise CVTestStepFailure(
                    "Mismatched SLA, Excluded entry client [%s] found in table [%s]" % (
                        sla_object.client_name,
                        table_name
                    )
                )

    @test_step
    def validate_web_sla(self):
        """ validate webconsole SLA report"""
        self.init_webconsole()
        self.webconsole.goto_reports()
        self.navigator.goto_commcell_reports('SLA', commcell_name=self.commcell.commserv_name)
        self.sla = WebSla(self.webconsole)
        self.sla.access_missed_sla()
        self.SLAReportClientList.clear()
        self.log.info("Verifying for Failed laptop clients")
        self.SLAReportClientList.append(
            SLAReportDataExpected(self.FailedLaptopClient, ""))
        title = 'Unprotected Clients'
        self.validate_sla_clients(title)
        self.log.info("Verifying for Successful laptop clients")
        self.SLAReportClientList.clear()
        self.SLAReportClientList.append(
            SLAReportDataExpected(self.SuccessLaptopClient, ""))
        self.navigator.goto_page_via_breadcrumb("SLA")
        self.sla.access_met_sla()
        title = 'Protected Clients'
        self.validate_sla_clients(title)

            self.init_tc()

        except Exception as err:
            self.utils.handle_testcase_exception(err)
        finally:
            WebConsole.logout_silently(self.webconsole)
            Browser.close_silently(self.browser)
