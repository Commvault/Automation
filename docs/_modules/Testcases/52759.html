[docs]    def run(self):
        """Run function of this test case"""

        try:
            for i in range(2):

                # if i == 0 then full backup, delete docs and restore
                # if i == 1 then full backup, create doc, inc backup, delete and restore

                if i == 0:
                    # Verify discovery once for every testcase
                    self.cvcloud_object.cvoperations.verify_discover()
                self.cvcloud_object.gdrive.delete_folder()
                self.cvcloud_object.gdrive.create_files(
                    unicode_data=False, pdf=True, google_docs=True, new_folder=True, attempt='full')
                self.cvcloud_object.gdrive.get_file_properties()
                self.cvcloud_object.cvoperations.check_job_status(self.subclient.backup('FULL'))
                if i == 1:
                    self.cvcloud_object.gdrive.create_files(
                        unicode_data=False, pdf=True, google_docs=True, new_folder=False, attempt='inc')
                    self.cvcloud_object.gdrive.get_file_properties()
                    self.cvcloud_object.cvoperations.check_job_status(
                        self.subclient.backup('INCREMENTAL'))

                self.cvcloud_object.gdrive.delete_files()
                self.cvcloud_object.cvoperations.restore_subclient()
                self.cvcloud_object.gdrive.compare_file_properties()

                if i == 1:
                    # This is OOP restore to disk
                    self.cvcloud_object.gdrive.get_file_properties()
                    self.cvcloud_object.cvoperations.restore_subclient(oop=True, to_disk=True)
                    self.cvcloud_object.gdrive.compare_file_properties(oop=True, to_disk=True)

                    self.cvcloud_object.cvoperations.restore_subclient(oop=True)
                    self.cvcloud_object.gdrive.compare_file_properties(oop=True)

        except Exception as exp:
            self.log.error('Failed with error: ' + str(exp))
            self.result_string = str(exp)
            self.status = constants.FAILED
