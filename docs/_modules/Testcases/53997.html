            "StoragePolicy": None

        self.subclient = None
        self.storage_policy = None
        self.cl_machine = None
        self.idx_tc = None
        self.idx_help = None

        self.indexing_level = None
        self.index_db = None
        self.job_stats = None

[docs]    def setup(self):
        """Setup function of this test case"""

        self.backupset_name = 'pruning_cycles_based'
        self.subclient_name = "{0}_sc1".format(self.backupset_name)
        self.storage_policy = self.tcinputs['StoragePolicy']

        self.log.info("Backupset name generated for this test run: {0}".format(self.backupset_name))
        self.log.info("Subclient name generated for this test run: {0}".format(self.subclient_name))

        self.client_machine_obj = Machine(self.client, self.commcell)
        self.cl_machine = self.client_machine_obj
        self.cl_delim = self.cl_machine.os_sep

        self.idx_tc = IndexingTestcase(self)
        self.idx_help = IndexingHelpers(self.commcell)

        self.indexing_level = self.idx_help.get_agent_indexing_level(self.agent)
        self.cs_db = get_csdb()

        self.backupset = self.idx_tc.create_backupset(self.backupset_name)

        self.subclient = self.idx_tc.create_subclient(
            name=self.subclient_name,
            backupset_obj=self.backupset,
            storage_policy=self.storage_policy
        )

        self.commcell.add_additional_setting("CommServDB.GXGlobalParam", "bEnableIndexPruning", "BOOLEAN", "true")
        self.commcell.add_additional_setting("CommServDB.GXGlobalParam", "IndexPruning_Type", "INTEGER", "0")
        self.commcell.add_additional_setting("CommServDB.Console", "IndexPruning_DBRetentionCycle", "INTEGER", "-1")
        self.commcell.add_additional_setting("CommServDB.Console", "IndexPruning_Type", "INTEGER", "-1")

        entity = self.subclient if self.indexing_level == 'subclient' else self.backupset
        entity.index_pruning_type = 'cycles_based'
        entity.index_pruning_cycles_retention = 2



[docs]    def run(self):
        """Main function for test case execution

            Steps:
                1) Run 3 backup cycles
                2) Run checkpointing
                3) Run compaction
                4) Run checkpointing
                5) Verify if indexcheckpointinfo table is populated
                6) Run compaction
                7) Verify if dbPruneTime is populated correctly (should be start time of the 2nd cycle first job).
                8) Do browse on 2nd and 3rd cycle to see if no checkpoint is restored
                9) Do browse on 1st cycle to see if pruned jobs are served by restoring the checkpoint
        """

        try:

            sc_content = self.subclient.content
            self.idx_tc.new_testdata(sc_content)

            # FULL - Cycle 1
            self.idx_tc.run_backup(self.subclient, 'full', verify_backup=False)

            # INC
            self.idx_tc.edit_testdata(sc_content)
            self.idx_tc.run_backup(self.subclient, 'incremental', verify_backup=False)

            # SFULL - Cycle 2
            self.idx_tc.run_backup(self.subclient, 'Synthetic_full', verify_backup=False)

            # INC
            self.idx_tc.edit_testdata(sc_content)
            self.idx_tc.run_backup(self.subclient, 'incremental', verify_backup=False)

            # SFULL - Cycle 3
            self.idx_tc.run_backup(self.subclient, 'Synthetic_full', verify_backup=False)

            # INC
            self.idx_tc.edit_testdata(sc_content)
            self.idx_tc.run_backup(self.subclient, 'incremental', verify_backup=False)

            self.job_stats = self.idx_tc.get_backup_jobs_stats(self.subclient)
            self.log.info('Job stats - {0}'.format(self.job_stats))

            self.log.info("********** Verifying all data **********")
            self.idx_tc.verify_browse_restore(self.backupset, {
                'operation': 'find',
                'subclient': self.subclient_name,
                'from_time': 0,
                'to_time': self.job_stats[5]['servEndDate'],
                'path': "/**/*",
                'show_deleted': False,
                'restore': {'do': False}
            })

            if self.indexing_level == 'backupset':
                self.index_db = index_db.get(self.backupset)
            else:
                self.index_db = index_db.get(self.subclient)

            self.log.info('Current DBInfo file:' + str(self.index_db.db_info))

            self.log.info('********* Checkpointing the DB first *********')
            if not self.index_db.checkpoint_db(False):
                raise Exception("Failed to complete index checkpoint")

            self.log.info('********* Running compaction to initialize prune time *********')
            if not self.index_db.compact_db():
                raise Exception('Failed to perform index compaction')

            self.log.info('Current DBInfo data:' + str(self.index_db.db_info))

            self.log.info('********* Checkpointing the DB second time *********')
            if not self.index_db.checkpoint_db(False):
                raise Exception('Failed to complete index checkpoint')

            self.log.info('********* Verifying IndexCheckPointInfo data *********')
            checkpoint = self.index_db.get_index_db_checkpoints()[-1]
            self.log.info('Checkpoint info - {0}'.format(checkpoint))

            if checkpoint['startTime'] == 'Null':
                raise Exception('Index checkpoint info is not populated')

            # Start time of first cycle first job and 3rd cycle first job
            if (checkpoint['startTime'] == self.job_stats[0]['servStartDate'] and
                    checkpoint['endTime'] == self.job_stats[4]['servStartDate']):
                self.log.info('IndexCheckPointInfo time is correct and verified')
            else:
                raise Exception('IndexCheckPointInfo time is incorrect')

            self.log.info('********* Running compaction to perform Index pruning *********')
            if not self.index_db.compact_db():
                raise Exception('Failed to perform index compaction')

            self.log.info('Current DBInfo data:' + str(self.index_db.db_info))

            self.log.info('********* Verifying dbPrune time *********')

            # Start time of first job of 2nd cycle
            if str(self.index_db.get_db_info_prop('dbPruneTime')) == self.job_stats[2]['servStartDate']:
                self.log.info('DB prune time calculated is incorrect')
            else:
                raise Exception('DB prune time calculated is incorrect')

            partial_db_guid = '_'.join([
                self.index_db.db_guid,
                checkpoint['commcellId'],
                checkpoint['startTime'],
                checkpoint['endTime']
            ])

            partial_db_path = self.index_db.isc_delim.join([
                self.index_db.backupset_path,
                partial_db_guid
            ])

            self.log.info('Expected partial DB on checkpoint restore [{0}]'.format(partial_db_path))

            self.log.info('********* Verifying that no unexpected jobs are pruned *********')
            self.log.info('********** Third cycle browse **********')

            self.idx_tc.verify_browse_restore(self.backupset, {
                'operation': 'find',
                'subclient': self.subclient_name,
                'from_time': self.job_stats[4]['servStartDate'],
                'to_time': self.job_stats[5]['servEndDate'],
                'path': "/**/*",
                'show_deleted': False,
                'restore': {'do': False}
            })

            if self.index_db.isc_machine.check_directory_exists(partial_db_path):
                raise Exception('Partial DB is restored upon browse of 3rd cycle')
            else:
                self.log.info('No checkpoint restore happened while browsing 3rd cycle')

            self.log.info("********** Second cycle browse **********")
            self.idx_tc.verify_browse_restore(self.backupset, {
                'operation': 'find',
                'subclient': self.subclient_name,
                'from_time': self.job_stats[2]['servStartDate'],
                'to_time': self.job_stats[3]['servEndDate'],
                'path': "/**/*",
                'show_deleted': False,
                'restore': {'do': False}
            })

            if self.index_db.isc_machine.check_directory_exists(partial_db_path):
                raise Exception('Partial DB is restored upon browse of 2nd cycle')
            else:
                self.log.info('No checkpoint restore happened while browsing 2nd cycle')

            self.log.info("********** Verifying if first cycle is pruned from index **********")
            self.log.info("********** First cycle browse **********")

            self.idx_tc.verify_browse_restore(self.backupset, {
                'operation': 'find',
                'subclient': self.subclient_name,
                'from_time': 0,
                'to_time': self.job_stats[1]['servEndDate'],
                'path': '/**/*',
                'show_deleted': False,
                'restore': {'do': True}
            })

            if not self.index_db.isc_machine.check_directory_exists(partial_db_path):
                raise Exception('Partial DB is not restored upon browse of 1st cycle')
            else:
                self.log.info('********** Verified checkpoint is restored for pruned jobs **********')

        except Exception as exp:
            self.log.error('Failed to execute test case with error: {0}'.format(str(exp)))
            self.result_string = str(exp)
            self.status = constants.FAILED
            self.log.error(str(traceback.format_exc()))
