[docs]    def HSConfig(self, user, pwd):
        """Connect to Hyperscale"""
        try:
            inputs = self.tcinputs
            self.browser.driver.get(inputs['url'])
            self.browser.driver.implicitly_wait(60)
            self.browser.driver.find_element_by_id("username").send_keys(user)
            self.browser.driver.implicitly_wait(60)
            self.browser.driver.find_element_by_id("password").send_keys(pwd)
            self.browser.driver.implicitly_wait(60)
            self.browser.driver.find_element_by_id("login_submit").click()
            self.browser.driver.implicitly_wait(100)
            self.log.info("logged in")
            return True
        except Exception as e:
            raise Exception(str(e))

        """
        To set the details for CommServ
        """
        inputs = self.tcinputs
        self.log.info("Configuring Network was successful")

        if self.browser.driver.find_element_by_id("clusterInfoError"):
            # if

            self.log.info("Inputs for CommServ..")
            self.browser.driver.find_element_by_id("hostname").clear()
            self.browser.driver.find_element_by_id("hostname").send_keys(inputs['hostname'])

            self.browser.driver.find_element_by_id("pwd").clear()
            self.browser.driver.find_element_by_id("pwd").send_keys(inputs['cspwd'])

            self.browser.driver.find_element_by_id("confirm_pwd").clear()
            self.browser.driver.find_element_by_id("confirm_pwd").send_keys(inputs['cspwd'])

            self.browser.driver.find_element_by_id("window_product").clear()
            self.browser.driver.find_element_by_id(
                "window_product").send_keys(inputs['winprodkey'])

            self.browser.driver.find_element_by_id("virtual_engine").clear()
            self.browser.driver.find_element_by_id("virtual_engine").send_keys(inputs['ovirteng'])

            self.browser.driver.implicitly_wait(60)

            self.browser.driver.find_element_by_id("submit_step").click()

            self.browser.driver.implicitly_wait(60)
            print("commverse_info")
            sleep(3600)
            if self.browser.driver.find_element_by_xpath('//div/div/div[1]/div[1]/div'):
                warning = self.browser.driver.find_element_by_xpath(
                    '//div/div/div[1]/div[1]/div').text
                if str(warning) == 'HyperScale configuration completed with warnings.':
                    self.log.info(str(warning))
                    return True
            elif self.browser.driver.find_element_by_xpath('//div/div/div[1]/div[2]/div'):
                success = self.browser.driver.find_element_by_xpath(
                    '//div/div/div[1]/div[2]/div').text
                if str(success) == 'HyperScale configuration completed successfully!':
                    self.log.info(str(success))
                    return True
            else:
                print("Could not find the Hyperscale configuration page")
                return False
        else:
            return False

            self.log.info(
                "-------------------Start of the case----------")

            if self.HSConfig('root', 'cvadmin'):
                self.log.info("Hello")
                if self.commvault_hyperscale():
                    if self.commserv_info():
                        self.log.info('completed')
                        self.browser.driver.close()
                        self.browser.driver.quit()

        except Exception as exp:
