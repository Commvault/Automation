[docs]    def create_subclient_obj(self):
        """
        Create Subclient object for respective client.1
        Return:
            (Object) subclient object
        """
        if self.agent.backupsets.has_backupset(self.backupset_name):
            self.log.info("Backup set %s already exists!! Deleting!!" % self.backupset_name)
            self.agent.backupsets.delete(self.backupset_name)
        backupset_obj = self.agent.backupsets.add(self.backupset_name)
        subclient_obj = backupset_obj.subclients.add(
            self.subclient_name, self.tcinputs['StoragePolicy'])
        self.log.info("Successfully created subclient: %s" % subclient_obj.subclient_name)
        subclient_obj.content = [self.tcinputs['FileServerLocalTestDataPath']]
        subclient_obj.update_properties(cs.FSO_SUBCLIENT_PROPS)
        return subclient_obj

[docs]    def run(self):
        """Main function for test case execution"""
        try:
            self.job_manager = JobManager(commcell=self.commcell)
            self.backupset_name = f"{self.id}_backupset_fso"
            self.subclient_name = f"{self.id}_subclient_fso"
            self.job_manager.job = self.create_subclient_obj().backup(backup_level="Full")
            self.job_manager.wait_for_state('completed', time_limit=180)
            self.init_tc()
            self.perform_cleanup()
            self.create_inventory()
            self.create_plan()
            self.create_fso_client()
            self.fso_helper.create_sqlite_db_connection(self.tcinputs['TestDataSQLiteDBPath'])
            self.gdpr_obj.file_server_lookup_obj.add_file_server(
                self.client.client_name, 'Client name',
                self.file_server_display_name, self.country_name,
                agent_installed=True, backup_data_import=True,
                fso_server=True, crawl_type='Quick')
            if not self.gdpr_obj.file_server_lookup_obj.wait_for_data_source_status_completion(
                    self.file_server_display_name):
                raise Exception("Could not complete Data Source Scan")
            self.log.info("Sleeping for: '%d' seconds" % self.wait_time)
            time.sleep(self.wait_time)
            self.navigator.navigate_to_governance_apps()
            self.gdpr_obj.inventory_details_obj.select_file_storage_optimization()
            self.fso_helper.analyze_client_quickview(self.tcinputs['FsoClientName'])
            try:
                self.fso_helper.analyze_client_details(
                    self.tcinputs['FsoClientName'],
                    self.file_server_display_name,
                    self.fso_helper.get_fso_file_count_db() + self.fso_helper.get_fso_dir_count_db(),
                    self.plan_name,
                    is_backed_up=True
                )
            except Exception as error_status:
                self.error_dict["Analyze client details page"] = \
                    str(error_status)
                self.test_case_error = str(error_status)
            self.fso_obj.select_client(self.file_server_display_name)
            self.gdpr_obj.verify_data_source_name()
            try:
                self.fso_helper.review_size_distribution_dashboard(crawl_type='Backup')
            except Exception as exp:
                self.error_dict['Size Distribution Dashboard'] = str(exp)
                self.test_case_error = str(exp)

            try:
                self.fso_helper.review_file_duplicates_dashboard()
            except Exception as exp:
                self.error_dict['Duplicates Dashboard'] = str(exp)
                self.test_case_error = str(exp)

            flag = None
            exp_flag = False
            try:
                flag = self.fso_helper.verify_fso_time_data()
            except Exception as exp:
                self.error_dict['Access/Modified/Created Time'] = str(exp)
                self.test_case_error = str(exp)
                flag = True
            if not flag and not exp_flag:
                self.error_dict['Access/Modified/Created Time'] = \
                    "Access Time or Modified Time or Created Time Match Failed"
                self.test_case_error = "Access Time or Modified Time or Created Time Match Failed"

            if self.test_case_error is not None:
                raise CVTestStepFailure(str(self.error_dict))
            self.perform_cleanup()

        except Exception as exp:
            if self.test_case_error is not None and len(self.error_dict.keys()) > 0:
                self.log.info("***Following Error Occurred in the Automation Testcase******")
                for key, value in self.error_dict.items():
                    self.log.info('{%s}  {%s} \n' % (key, value))
            handle_testcase_exception(self, exp)

        finally:
            AdminConsole.logout_silently(self.admin_console)
            Browser.close_silently(self.browser)
