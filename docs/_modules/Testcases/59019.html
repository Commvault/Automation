        self.show_to_user = True
        self.test_individual_status = True
        self.tcinputs = {}


[docs]    def run(self):
        """Main function for test case execution"""
        log = logger.get_log()

        try:
            log.info("Started executing %s testcase", self.id)
            log.info(
                "-------------------Initialize helper objects------------------------------------")
            auto_commcell = VirtualServerhelper.AutoVSACommcell(self.commcell, self.csdb)
            auto_client = VirtualServerhelper.AutoVSAVSClient(auto_commcell, self.client)
            auto_instance = VirtualServerhelper.AutoVSAVSInstance(
                auto_client, self.agent, self.instance)
            auto_backupset = VirtualServerhelper.AutoVSABackupset(auto_instance, self.backupset)
            auto_subclient = VirtualServerhelper.AutoVSASubclient(auto_backupset, self.subclient)
            try:
                VirtualServerUtils.decorative_log("Creating proxy machines object")
                self.proxy_list = self.subclient.subclient_proxy
                self.proxymachine1 = Client(self.commcell, self.proxy_list[0], client_id=None)
                self.proxymachine2 = Client(self.commcell, self.proxy_list[1], client_id=None)
                VirtualServerUtils.decorative_log("created proxy machine object successfully")
            except Exception as err:
                self.log.error("--Failed to create proxy machine object--")
                raise Exception
            VirtualServerUtils.decorative_log("Starting Backup Job")
            try:
                backup_options = OptionsHelper.BackupOptions(auto_subclient)
                backup_options.backup_type = "INCREMENTAL"
                bkpobj = VirtualServerhelper.AutoVSASubclient(auto_backupset, self.subclient)
                bkpobj.backup_folder_name = backup_options.backup_type
                if not backup_options.testdata_path:
                    bkpobj.testdata_path = VirtualServerUtils.get_testdata_path(
                        bkpobj.controller_machine)
                    backup_options.testdata_path = bkpobj.testdata_path
                bkpobj.timestamp = bkpobj.testdata_path.rpartition('\\')[-1]
                self.timestamp = bkpobj.timestamp
                bkpobj.cleanup_testdata(backup_options)
                bkpobj.vsa_discovery(backup_options, extra_options={})
                _backup_jobs = self.subclient.backup()
                VirtualServerUtils.decorative_log("Backup job triggered successfully")
            except Exception as err:
                self.log.exception("--Triggering backup job failed--")
                raise Exception
            VirtualServerUtils.decorative_log('checking Database for child job complete status')
            time.sleep(120)
            try:
                jobvmdict = {}
                for each_vm in auto_subclient.vm_list:
                    query1 = "select jobId from JMBkpJobInfo where applicationId in" \
                       " (select id from APP_Application where subclientName = '"+self.tcinputs.get('SubclientName')+"' " \
                       "and clientid in (select id from APP_Client where name = '"+each_vm+"')) "
                    self.csdb.execute(query1)
                    query1output = self.csdb.fetch_all_rows()
                    jobvmdict[each_vm] = query1output[0][0]
                    for key in sorted(jobvmdict):
                        print('-sorting dict-')
                VirtualServerUtils.decorative_log('sorted in dictionary format'+str(jobvmdict))
                joblist = jobvmdict.values()
                VirtualServerUtils.decorative_log(
                    'Successfully got job IDs for individual VMs from database')
            except Exception as err:
                self.log.exception(
                    "-----Failed to get individual job IDs of VMs from database-----")
                raise Exception
            try:
                VirtualServerUtils.decorative_log('getting child job completed status from DB')
                joblist1 = list(jobvmdict.values())
                i = 0
                while i <= 1:
                    for eachjob in joblist1:
                        query2 = "select status from JMBkpStats where jobid = "+eachjob+""
                        self.csdb.execute(query2)
                        query2output = self.csdb.fetch_all_rows()
                        query2output = query2output[0][0]
                        if query2output != None and query2output != '':
                            i += 1
                            VirtualServerUtils.decorative_log(
                                "successfully got child job completed status from DB-"+str(eachjob))
                            joblist1.remove(eachjob)
                            break
                time.sleep(1)
            except Exception as err:
                self.log.error(
                    "-----Failed to get child job completed status from DB-----")
                raise Exception
            VirtualServerUtils.decorative_log(
                "Restart services on proxies while job in progress")
            try:
                if query2output == '1' or query2output == '3':
                    self.proxymachine1.restart_service(service_name='GxCVD(Instance001)')
                    self.proxymachine2.restart_service(service_name='GxCVD(Instance001)')
                time.sleep(30)
                commandlist = ['GxClMgrS(Instance001)', 'GXMMM(Instance001)', 'GxBlr(Instance001)']
                for each_command in commandlist:
                    self.proxymachine1.start_service(service_name=each_command)
                    self.proxymachine2.start_service(service_name=each_command)
                time.sleep(10)
                VirtualServerUtils.decorative_log("Services restarted successfully")
            except Exception as err:
                self.log.error(
                    "-----Failed to restart services-----")
                raise Exception
            if not _backup_jobs.wait_for_completion():
                raise Exception("Failed to run VM   job with error: "
                                +str(_backup_jobs.delay_reason))
            VirtualServerUtils.decorative_log("Back up job completed successfully")
            VirtualServerUtils.decorative_log(
                'Validating if job completed VMs attempted to backup on other proxy with different job')
            jobvmdict1 = {}
            for each_vm in auto_subclient.vm_list:
                query = "select TOP 1 PERCENT jobId from JMBkpStats where appid =" \
                       " (select id from APP_Application where subclientName = '"+self.tcinputs.get('SubclientName')+"' " \
                       "and clientid in (select id from APP_Client where name = '"+each_vm+"')) " \
                       "ORDER by jobid DESC"
                self.csdb.execute(query)
                queryoutput = self.csdb.fetch_all_rows()
                jobvmdict1[each_vm] = queryoutput[0][0]
            for key in sorted(jobvmdict1):
                VirtualServerUtils.decorative_log('sorting dict')
            VirtualServerUtils.decorative_log(
                'JOB and VM details sorted in dictionary format '+str(jobvmdict1))
            try:
                VirtualServerUtils.decorative_log('Validating if child jobs ran as incremental')
                jobtypevmlist = []
                for eachjob in joblist:
                    query3 = "select bkpLevel from JMBkpStats where jobId ="+eachjob+""
                    self.csdb.execute(query3)
                    query3output = self.csdb.fetch_all_rows()
                    jobtypevmlist.append(query3output)
                    if all(i == [['2']] for i in jobtypevmlist):
                        VirtualServerUtils.decorative_log('All child jobs ran as incremental jobs')
            except Exception as err:
                self.log.error(
                    "-----All child jobs didnt ran as incremental jobs which is not expected -----")
                raise Exception
            VirtualServerUtils.decorative_log(
                'Validating if new job triggered after restart services for completed VMs')
            try:
                if jobvmdict == jobvmdict1:
                    VirtualServerUtils.decorative_log(
                        'No new jobs triggered for job completed VMs after restart')
                time.sleep(120)
            except Exception as err:
                self.log.error(
                    "-----New jobs triggered for job completed VMs after restart-----")
                raise Exception
            VirtualServerUtils.decorative_log('All failover validations succeeded')
            try:
                VirtualServerUtils.decorative_log(
                    "-" * 25 + " FULL VM out of Place restores " + "-" * 25)
                vm_restore_options = OptionsHelper.FullVMRestoreOptions(auto_subclient, self)
                vm_restore_options.power_on_after_restore = True
                vm_restore_options.unconditional_overwrite = True
                auto_subclient.testdata_path = bkpobj.testdata_pat
                auto_subclient.timestamp = self.timestamp
                auto_subclient.backup_folder_name = bkpobj.backup_folder_name
                auto_subclient.virtual_machine_restore(vm_restore_options)
            except Exception as exp:
                self.test_individual_status = False
                self.test_individual_failure_message = str(exp)
        except Exception as exp:
            self.log.error('Failed with error: '+str(exp))
            self.result_string = str(exp)
            self.status = constants.FAILED
        finally:
            if auto_subclient and backup_options:
                auto_subclient.cleanup_testdata(backup_options)
            if not self.test_individual_status:
                self.result_string = self.test_individual_failure_message
                self.status = constants.FAILED
