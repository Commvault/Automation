[docs]    def run(self):
        """Main function for test case execution"""
        log = logger.get_log()
        try:
            _client = self.commcell.clients.get(self.client.client_name)
            entities  = CVEntities(self)
            entity_properties = entities.create(['backupset', 'subclient', 'storagepolicy', 'disklibrary'])
            
            # Associate data with client.
            log.info("Associating back up data with the given client.")
            Subclientobject = entity_properties['subclient']['object']
            Idautils = CommonUtils(self)
            Idautils.subclient_backup(Subclientobject)
            
            # Perform the Retire client Operation and wait for the job to complete.           
            log.info("Will perform Retire operation now on client %s.", self.client.client_name)
            job = _client.retire()
            if not job.wait_for_completion():
                raise Exception("Uninstall job failed with error: {0}".format(job.delay_reason))
            
            #Refreshing the clients associated with the commcell Object
            self.commcell.clients.refresh()
            
            # Validate that client is NOT deleted
            if not self.commcell.clients.has_client(self.client.client_name):
                raise Exception("Client has been deleted. Check logs to make sure Retire Operation succeeded.")
            
            log.info("Test case to retire a client when when data is associated with it completed successfully.")
            
        except Exception as exp:
            self.log.error('Failed with error: %s', exp)
            self.result_string = str(exp)
            self.status = constants.FAILED
