    """Testcase to verify if index pruning  does not take place when disabled"""

    def __init__(self):
        """Initializes test case class object"""
        super(TestCase, self).__init__()
        self.name = "Indexing - Pruning test case - backupset level indexing - Infinite retention"
        self.tcinputs = {
            "StoragePolicy": None
        }

        self.backupset = None
        self.subclient = None
        self.storage_policy = None
        self.cl_machine = None


[docs]    def setup(self):
        """Setup function of this test case"""

        self.backupset_name = "pruning_test_infinite_backupset"
        self.subclient_name = "{0}_sc1".format(self.backupset_name)
        self.storage_policy = self.tcinputs['StoragePolicy']
        self.log.info("Backupset name generated for this test run:{0}".format(self.backupset_name))
        self.log.info("Subclient name generated for this test run:{0}".format(self.subclient_name))

        self.idx_tc = None

        self.client_machine_obj = Machine(self.client, self.commcell)
        self.cl_machine = self.client_machine_obj
        self.cl_delim = self.cl_machine.os_sep

        self.idx_tc = IndexingTestcase(self)

        self.backupset = self.idx_tc.create_backupset(self.backupset_name)

        self.subclient = self.idx_tc.create_subclient(
            name=self.subclient_name,
            backupset_obj=self.backupset,
            storage_policy=self.storage_policy
        )

        self.commcell.add_additional_setting("CommServDB.GXGlobalParam", "bEnableIndexPruning",
                                             "BOOLEAN", "true")
        self.commcell.add_additional_setting("CommServDB.GXGlobalParam", "IndexPruning_Type",
                                             "INTEGER", "0")
        self.commcell.add_additional_setting("CommServDB.Console", "IndexPruning_DBRetentionCycle",
                                             "INTEGER", "-1")
        self.commcell.add_additional_setting("CommServDB.Console", "IndexPruning_Type",
                                             "INTEGER", "-1")



[docs]    def run(self):
        """Main function for test case execution

        Steps:
                1) Run 3 backup cycles
                2) Run checkpointing
                3) Run compaction
                4) Verify no jobs are pruned

        """

        try:

            sc_content = self.subclient.content
            self.idx_tc.new_testdata(sc_content)

            full_job_cycle_1 = self.idx_tc.run_backup(self.subclient, 'full', verify_backup=False)

            self.idx_tc.edit_testdata(sc_content)
            inc_job_1_cycle_1 = self.idx_tc.run_backup(self.subclient, 'incremental',
                                                       verify_backup=False)

            self.idx_tc.edit_testdata(sc_content)
            inc_job_2_cycle_1 = self.idx_tc.run_backup(self.subclient, 'incremental',
                                                       verify_backup=False)
            cycle_1_etime = commonutils.convert_to_timestamp(inc_job_2_cycle_1.end_time)

            full_job_cycle_2 = self.idx_tc.run_backup(self.subclient, 'full', verify_backup=False)

            self.idx_tc.edit_testdata(sc_content)
            inc_job_1_cycle_2 = self.idx_tc.run_backup(self.subclient, 'incremental',
                                                       verify_backup=False)

            self.idx_tc.edit_testdata(sc_content)
            inc_job_2_cycle_2 = self.idx_tc.run_backup(self.subclient, 'incremental',
                                                       verify_backup=False)
            cycle_2_etime = commonutils.convert_to_timestamp(inc_job_2_cycle_2.end_time)

            synth_full_cycle3 = self.idx_tc.run_backup(self.subclient, 'Synthetic_full',
                                                       verify_backup=False)

            self.idx_tc.edit_testdata(sc_content)
            inc_job_1_cycle_3 = self.idx_tc.run_backup(self.subclient, 'incremental',
                                                       verify_backup=False)

            self.idx_tc.edit_testdata(sc_content)
            inc_job_2_cycle_3 = self.idx_tc.run_backup(self.subclient, 'incremental',
                                                       verify_backup=False)
            cycle_3_etime = commonutils.convert_to_timestamp(inc_job_2_cycle_3.end_time)

            self.log.info("********** Verifying first cycle browse **********")
            self.idx_tc.verify_browse_restore(self.backupset, {
                'operation': 'find',
                'subclient': self.subclient_name,
                'from_time': 0,
                'to_time': cycle_1_etime,
                'path': "/**/*",
                'show_deleted': False,
                'restore': {'do': False}
            })

            checkpoint_reg_keys = {
                'CHKPOINT_ITEMS_ADDED_MIN': 0
            }

            indexdb_obj = index_db.get(self.backupset)

            self.log.info('Current DBInfo file:' + str(indexdb_obj.db_info))

            self.log.info('********* Checkpointing the DB first *********')

            idx_chkpoint_status = indexdb_obj.checkpoint_db(False, checkpoint_reg_keys)
            if not idx_chkpoint_status:
                raise Exception("Failed to complete index checkpoint")

            self.log.info('********* Running compaction *********')
            idx_compact_status = indexdb_obj.compact_db()
            if not idx_compact_status:
                raise Exception(idx_compact_status)

            self.log.info(indexdb_obj.db_info)

            dbinfo_dict = indexdb_obj.db_info
            if '@dbPruneTime' in dbinfo_dict['Indexing_DbProps']:
                self.log.info("Index db pruned at {0}".format(dbinfo_dict['Indexing_DbProps']
                                                              ['@dbPruneTime']))
                raise Exception("Index db unexpectedly being pruned")

            self.log.info('********* Verifying that no unexpected jobs are pruned *********')
            self.log.info("********** Third cycle browse **********")

            self.idx_tc.verify_browse_restore(self.backupset, {
                'operation': 'find',
                'subclient': self.subclient_name,
                'from_time': synth_full_cycle3.start_time,
                'to_time': cycle_3_etime,
                'path': "/**/*",
                'show_deleted': False,
                'restore': {'do': False}
            })

            self.log.info("********** Second cycle browse **********")

            self.idx_tc.verify_browse_restore(self.backupset, {
                'operation': 'find',
                'subclient': self.subclient_name,
                'from_time': full_job_cycle_2.start_time,
                'to_time': cycle_2_etime,
                'path': "/**/*",
                'show_deleted': False,
                'restore': {'do': False}
            })

            self.log.info("********** First cycle browse **********")

            resp, dummy = self.subclient.browse({
                'from_time': 0,
                'to_time': cycle_1_etime,
                'path': self.cl_delim,
                'show_deleted': False,
                '_raw_response': True
            })

            self.log.info("***********")

            if dummy['browseResponses'][0]['respType'] == 2:
                self.log.info(dummy['browseResponses'][0]['messages'][0]['errorMessage'])
                self.log.info("First cycle unexpectedly pruned from index")
                raise Exception("First cycle should not be pruned from index")

        except Exception as exp:
            self.log.error('Failed to execute test case with error: {0}'.format(str(exp)))
            self.result_string = str(exp)
            self.status = constants.FAILED
            self.log.error(str(traceback.format_exc()))
