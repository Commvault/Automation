[docs]    def run_backup(self, backup_type):
        """Initiates backup job and waits for completion"""
        log = logger.get_log()
        log.info(
            " Starting Subclient %s Backup ", backup_type)
        job = self.subclient.backup(backup_type)
        log.info("Started %s backup with Job ID: %s", backup_type, str(job.job_id))
        if not job.wait_for_completion():
            raise Exception(
                "Failed to run {0} backup job with error: {1}".format(
                    backup_type, job.delay_reason
                )
            )

        log.info("Successfully finished %s backup job", backup_type)
        return job

        """Gets the password of the mysql instance

                Raises:
                    Exception:
                        if failed to get the password of the instance
        """

        try:
            query = "Select attrVal from app_instanceprop where componentNameId = {0} and \
                attrName = 'MySQL SA Password'".format(str(self.mysql_instance.instance_id))
            self.csdb.execute(query)
            password = self.csdb.fetch_one_row()

            mysql_db_password = cvhelper.format_string(self.commcell, password)

[docs]    def get_databases_size(self, database_name):
        """ This function gets the database size"""

        db_name = database_name
        try:
            self.log.info("Getting the size for database: %s", db_name)
            self.db_connect = database_helper.MySQL(
                self.host_name, self.usr, self.pwd, self.port, db_name)
            # This command gives the database Size in MB
            cmd_get_db_size = "SELECT CONCAT(sum(DATA_LENGTH + INDEX_LENGTH ) \
                / 1024 / 1024) AS Size FROM  INFORMATION_SCHEMA.TABLES where TABLE_SCHEMA \
                like '{0}';".format(db_name)
            db_size_in_mb = self.db_connect.execute(cmd_get_db_size)
            db_size_in_mb = db_size_in_mb.rows[0][0]
            return db_size_in_mb

        except Exception as exp:
            raise Exception("Could not get the database size. " + str(exp))

        finally:
            if self.db_connect is not None:
                self.db_connect.close()



[docs]    def get_database_and_table_sizes(self, sub_client_content_list):
        """This function calls get_databases_size() and get_tables_list_for_db() and returns"""

        sb_content_list = sub_client_content_list
        source_db_size = {}
        source_tables_size = {}
        table_list = []
        each_db_in_strip = ""
        try:
            for each_db in sb_content_list:
                db_size_in_mb = ""
                if each_db.startswith('\\'):
                    each_db_in_strip = each_db.lstrip("\\")
                else:
                    each_db_in_strip = each_db.lstrip('/')

                if each_db_in_strip == 'mysql' or each_db_in_strip == 'sys':
                    continue
                else:
                    db_size_in_mb = self.get_databases_size(each_db_in_strip)
                    source_db_size[each_db_in_strip] = db_size_in_mb

                    table_list = self.get_tables_list_for_db(each_db_in_strip)
                    tables_size = {}

                    for each_table in table_list:
                        self.log.info(each_table)
                        size = self.get_table_size(each_db_in_strip, each_table)
                        tables_size[each_table] = size
                        source_tables_size[each_db_in_strip] = tables_size

            return (source_db_size, source_tables_size)

        except Exception as exp:
            raise Exception("Could not get database and table sizes. " + str(exp))



[docs]    def compare_db_and_table_sizes(
            self,
            init_dbsize_dict,
            init_tablesize_dict,
            final_dbsize_dict,
            final_tablesize_dict):
        """This function compare the database size and then compare tables sizes"""

        init_db_size = init_dbsize_dict
        init_table_size = init_tablesize_dict
        final_db_size = final_dbsize_dict
        final_table_size = final_tablesize_dict
        try:
            self.log.info("Comparing the DataBase Size")
            if len(init_db_size) != len(final_db_size):
                self.log.warning("DataBase Count mismatch")
            for each_db in init_db_size.keys():
                if init_db_size[each_db] != final_db_size[each_db]:
                    self.log.warning("DataBase: %s size failed", each_db)

            for each_db in init_table_size.keys():
                for each_table in init_table_size[each_db].keys():
                    ini_total_rows, ini_avg_length, ini_data_length, ini_table_size = (
                        init_table_size[each_db][each_table]).split("|")
                    fin_total_rows, fin_avg_length, fin_data_length, fin_table_size = (
                        final_table_size[each_db][each_table]).split("|")

                    if (ini_total_rows != fin_total_rows or
                            ini_avg_length != fin_avg_length or
                            ini_data_length != fin_data_length or
                            ini_table_size != fin_table_size):
                        self.log.error(
                            "Mismatch in TotalRows, AvgRowLength, DataLength, Size for table: %s \
                            in DataBase: %s",
                            each_table,
                            each_db)
                        self.log.error(
                            "Initial TotalRows: %s, AvgRowLenght: %s, DataLength: %s, Size: %s",
                            ini_total_rows,
                            ini_avg_length,
                            ini_data_length,
                            ini_table_size)
                        self.log.error(
                            "Final TotalRows: %s, AvgRowLenght: %s, DataLength: %s, Size: %s",
                            fin_total_rows,
                            fin_avg_length,
                            fin_data_length,
                            fin_table_size)
                        raise Exception("Validation failure")

            self.log.info("********* Successful Compared database and tables properties *********")

        except Exception:
            self.log.exception("Exception raised at compare_db_and_table_sizes")

        """ This Function drop the tables"""

        if database_name.startswith('\\'):
            db_name = database_name.lstrip("\\")
            tb_name = table_name.lstrip("\\")
        else:
            db_name = database_name.lstrip("/")
            tb_name = table_name.lstrip("/")

        try:
            self.log.info("Dropping the table:%s from dataBase:%s", tb_name, db_name)
            self.db_connect = database_helper.MySQL(
                self.host_name, self.usr, self.pwd, self.port, db_name)
            cmd = "drop table IF EXISTS `{0}`;".format(tb_name)
            self.db_connect.execute(cmd)

        except Exception as exp:
            raise Exception("Could not drop the table. " + str(exp))

        finally:
            if self.db_connect is not None:
                self.db_connect.close()

[docs]    def get_table_size(self, database_name, table_name):
        """ This Function get the Table Size for the given table and the output
            is dispalyed in this format.
            TableRows|AVG_ROW_LENGTH|DATA_LENGTH
        """

        db_name = database_name
        tb_name = table_name
        try:
            avg_row_length = "0"
            self.log.info(
                "Getting the Table Details : %s for the database: %s", tb_name, db_name)
            self.db_connect = database_helper.MySQL(
                self.host_name, self.usr, self.pwd, self.port, db_name)
            query_engine = "SELECT ENGINE FROM information_schema.`TABLES` WHERE `TABLE_SCHEMA`= \
                '{0}' and `TABLE_NAME`='{1}';".format(db_name, tb_name)
            engine = self.db_connect.execute(query_engine)
            engine = engine.rows[0][0]
            # Fetching number of rows in table
            query_rows = "SELECT COUNT(*) FROM `{0}`.`{1}`;".format(db_name, tb_name)
            tb_rows = self.db_connect.execute(query_rows)
            tb_rows = tb_rows.rows[0][0]

            # Fetching Average Row Length of Table
            query_avg_row_length = "SELECT AVG_ROW_LENGTH FROM information_schema.`TABLES` WHERE \
                `TABLE_SCHEMA`= '{0}' and `TABLE_NAME`='{1}';".format(db_name, tb_name)
            avg_row_length = self.db_connect.execute(query_avg_row_length)
            avg_row_length = avg_row_length.rows[0][0]

            # Fetching Data Length of Table
            query_data_length = "SELECT DATA_LENGTH FROM information_schema.`TABLES` WHERE \
                `TABLE_SCHEMA`= '{0}' and `TABLE_NAME`='{1}';".format(db_name, tb_name)
            data_length = self.db_connect.execute(query_data_length)
            data_length = data_length.rows[0][0]

            # Fetching Size Of Table (In MB)
            query_table_size = "SELECT ((DATA_LENGTH + INDEX_LENGTH) / 1024 / 1024) 'Size in MB' \
                FROM information_schema.`TABLES` WHERE `TABLE_SCHEMA`= '{0}' and \
                `TABLE_NAME`='{1}';".format(db_name, tb_name)
            table_size = self.db_connect.execute(query_table_size)
            table_size = table_size.rows[0][0]
            table_details = "{0}|{1}|{2}|{3}".format(
                tb_rows, avg_row_length, data_length, table_size)
            return table_details

        except Exception as exp:
            raise Exception("Could not get the content list. " + str(exp))

        finally:
            if self.db_connect is not None:
                self.db_connect.close()

            status = output.rows[0][1]
            self.log.info("Output: %s", str(status))
            if status == ('log_bin', 'ON'):
                pass
            elif status == ('log_bin', 'OFF'):
                raise Exception("Exception raised at check_log_bin_on_mysql_server. ")
        except Exception:
            self.log.exception(

            self,
            database_name,
            table_name="AutomTable",
            no_of_tables=500,
            column_in_each_table=7,
            drop_table_before_create=False):
        """This function created the table with the givenName and insert data"""

        if database_name.startswith('\\'):
            database_name = database_name.lstrip("\\")
        else:
            database_name = database_name.lstrip("/")

        table_name = table_name
        tables_created = []
        try:
            if database_name != "mysql" and database_name != "sys":
                self.db_connect = database_helper.MySQL(
                    self.host_name, self.usr, self.pwd, self.port, database_name)

                for each_table in range(0, no_of_tables):
                    each_table_name = "{0}_{1}_txtdt_{2}" .format(
                        database_name, table_name, each_table)

                    if drop_table_before_create:
                        self.drop_table(database_name,
                                        each_table_name)

                    sql_query = """CREATE TABLE IF NOT EXISTS `%s`(id int unsigned primary key \
                        AUTO_INCREMENT,column_var CHAR(254), column_string VARCHAR(254),column_tinytext \
                        TINYTEXT NOT NULL, column_text TEXT NOT NULL, column_mediumtext MEDIUMTEXT NOT NULL, \
                        column_longtext LONGTEXT NOT NULL)""" % (each_table_name)

                    self.log.info(sql_query)
                    self.db_connect.execute(sql_query)
                    #**************** Insert the values into the database*********************
                    sample_data_for_text_data_type = self.create_data("Text")

                    for i in range(0, column_in_each_table):
                        self.db_connect.execute(
                            "insert into `%s`(`id`,`column_var`, `column_string`, `column_tinytext`\
                            , `column_text`, `column_mediumtext`, `column_longtext`) values \
                            (NULL, '%s','%s','%s','%s','%s','%s')" %
                            (each_table_name,
                             sample_data_for_text_data_type[0],
                             sample_data_for_text_data_type[0],
                             sample_data_for_text_data_type[0],
                             sample_data_for_text_data_type[0],
                             sample_data_for_text_data_type[0],
                             sample_data_for_text_data_type[0]))

                    tables_created.append(each_table_name)

                return tables_created
            else:
                return tables_created

        except Exception as exp:
            raise Exception("Exception raised at create_table_with_text_db " + str(exp))

        finally:
            if self.db_connect is not None:
                self.db_connect.close()
