    """Class for Creating Custom Package xml needed to create custom package from cloud"""

    def __init__(self, input_answers):
        """Initialize the CustomPackageCloudXML class instance for generating XML

            Args:
                input_dict (dict) -- inputs for Installation

                     Supported keywords:
                        commserv_client_name: Commserve client name for which package
                                                needs to be created.

                        commserv_host_name: Commserve host name for client name

                        commserver_version: Commserve's current service pack version

                        commcell_username: User name for logging in commcell

                        commcell_password: Password for logging in commcell

                        authcode: Organization authcode for which to create package
                                (optional)

                        SP revision: Service pack revision for which package needs to
                                        be created. (Optional. If not provided, this
                                        would not be populated with package creation)

                        proxy_list: Proxy string list for clients in given format
                                    [cl1:cl1_hostname:4333, cl2:cl2_hostname:4333]

            Returns:
                object  -   instance of the CustomPackageCloudXML class

        """
        self.input_answers = input_answers
        self.log = logger.get_log()


[docs]    def generate_xml(self):
        """generates the xml for CustomPackageCloudXML class

            Returns:
                str  -   xml in String Format

        """
        self.log.info("Generating xml from inputs [{0}]".format(pformat(self.input_answers)))
        simxml = self._create_root()
        simxml = ET.tostring(simxml).decode('utf-8')
        return simxml


    def _create_root(self):
        """
        This Method creates Root Tag and appends child tags to it
            Returns:
                Object            --  XML Root Object

            Raises:
                SDKException:
                    if any exception raised while generating the XML
        """
        try:
            root = Element('inputs')
            root.append(self._input_xml())
            return root

        except Exception as excp:
            self.log.exception("Failed to create root tag: %s"%excp)
            return None

    def _request_id(self):
        """Creates requestId tag and appends its child Tags"""
        request_id_component = Element('requestId')
        request_id_component.text = self.input_answers['request_id']
        return request_id_component

    def _input_xml(self):
        """Creates inputXML tag and appends its child Tags"""
        input_xml_component = Element('inputXML')
        input_xml_component.append(self._custom_install_package())
        return input_xml_component

    def _custom_install_package(self):
        """Creates Custom_Install_Package tag and appends its child Tags"""
        custom_package_component = Element('Custom_Install_Package')
        custom_package_component.append(self._package_info())
        custom_package_component.append(self._proxy_info())
        custom_package_component.append(self._admin_info())
        custom_package_component.append(self._advanced_info())
        return custom_package_component

    def _package_info(self):
        """Creates Package_Info tag and appends its child Tags"""
        package_info_component = Element('Package_Info')
        if 'commserv_client_name' in self.input_answers.keys():
            package_info_component.set('shortNameCS', self.input_answers['commserv_client_name'])
        else:
            package_info_component.set('shortNameCS', '')
        if 'commserv_host_name' in self.input_answers.keys():
            package_info_component.set('hostNameCS', self.input_answers['commserv_host_name'])
        else:
            package_info_component.set('hostNameCS', '')
        if 'proxy_list' in self.input_answers.keys() and self.input_answers['proxy_list']:
            package_info_component.set('useProxy', 'true')
        else:
            package_info_component.set('useProxy', 'false')
        if 'commserver_version' in self.input_answers.keys():
            package_info_component.set('servicePack',
                                       'SP'+str(self.input_answers['commserver_version']))
            rev = self.input_answers.get('SP revision')
            if rev is not None and rev != 0:
                package_info_component.set('revision', 'Rev'+str(rev))
        else:
            package_info_component.set('servicePack', 'SP10')
        package_info_component.set('multiRules', 'false')
        package_info_component.set('requireCredentials', 'true')
        return  package_info_component

    def _proxy_info(self):
        """Creates Proxy_Info tag and appends its child Tags"""
        proxy_info_component = Element('Proxy_Info')
        if 'proxy_list' in self.input_answers.keys() and self.input_answers['proxy_list']:
            proxy_client_name_list = []
            proxy_hostname_list = []
            proxy_port_number_list = []
            for each_proxy in self.input_answers['proxy_list']:
                proxy_client_name_list.append(each_proxy.split(':')[0])
                proxy_hostname_list.append(each_proxy.split(':')[1])
                proxy_port_number_list.append(each_proxy.split(':')[2])
            proxy_client_name = "[" + ",".join(str(name) for name in proxy_client_name_list) + "]"
            proxy_host_name = "[" + ",".join(str(name) for name in proxy_hostname_list) + "]"
            proxy_port = "[" + ",".join(str(name) for name in proxy_port_number_list) + "]"
            proxy_info_component.set('proxyClientName', proxy_client_name)
            proxy_info_component.set('proxyHostname', proxy_host_name)
            proxy_info_component.set('proxyPort', proxy_port)
        else:
            proxy_info_component.set('proxyClientName', "[]")
            proxy_info_component.set('proxyHostname', "[]")
            proxy_info_component.set('proxyPort', "[]")
        proxy_info_component.set('httpProxyHostname', '')
        proxy_info_component.set('httpProxyPort', '')
        proxy_info_component.set('autoHTTPProxy', 'false')
        return proxy_info_component

    def _admin_info(self):
        """Creates Admin_Info tag and appends its child Tags"""
        admin_info_component = Element('Admin_Info')
        if 'commcell_username' in self.input_answers.keys():
            admin_info_component.set('username', self.input_answers['commcell_username'])
        else:
            admin_info_component.set('username', '')
        if 'commcell_password' in self.input_answers.keys():
            admin_info_component.set('password', self.input_answers['commcell_password'])
        else:
            admin_info_component.set('password', '')
        if 'authcode' in self.input_answers.keys():
            admin_info_component.set('authCode', self.input_answers['authcode'])
        else:
            admin_info_component.set('authCode', '')
        admin_info_component.set('authLater', 'false')
        return admin_info_component

    def _advanced_info(self):
        """Creates Advanced_Info tag and appends its child Tags"""
        advanced_info_component = Element('Advanced_Info')
        advanced_info_component.set('editAdvancedOptions', 'false')
        advanced_info_component.set('clientGroup', "[]")
        advanced_info_component.set('windowsSubclientPolicy', "")
        advanced_info_component.set('macSubclientPolicy', "")
        advanced_info_component.set('showInstallDirectory', 'false')
        advanced_info_component.set('hideApps', 'false')
        advanced_info_component.set('includeVPN', 'false')
        advanced_info_component.set('laptopClient', 'true')
        advanced_info_component.set('backupMonitor', 'false')
        advanced_info_component.set('migrationAssistant', 'false')
        advanced_info_component.set('processManager', 'false')
        advanced_info_component.set('subClientPlan', '')
        return advanced_info_component
