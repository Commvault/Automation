"""Main file that manages basic acceptance cases for hana client

BasicAcceptance is the only class defined in this file

BasicAcceptance:
    __init__()              --  initializes basicacceptance object

    _run_backup()           --  starts the backup job

    run()                   --  runs the basic acceptance test case
"""

import datetime

from AutomationUtils import logger
from .hanahelper import HANAHelper


[docs]class BasicAcceptance(object):
    """Helper class to run basic acceptance test case for hana client"""

    def __init__(self, test_case_obj):
        """Initializes basicacceptance object"""
        self._inputs = test_case_obj.tcinputs
        self._commcell = test_case_obj.commcell
        self._client = test_case_obj.client
        self._log = logger.get_log()
        self._subclient = test_case_obj.subclient
        self._commserver_name = self._commcell.commserv_name
        self._instance = test_case_obj.instance
        self._backupset = self._subclient._backupset_object
        self._hana_helper = HANAHelper(self._commcell, self._instance)

    def _run_backup(self, backup_type):
        """Starts backup job"""
        self._log.info("*" * 10 + " Starting Subclient {0} Backup ".format(backup_type) + "*" * 10)
        job = self._subclient.backup(backup_type)
        self._log.info("Started {0} backup with Job ID: {1}".format(backup_type, str(job.job_id)))
        if not job.wait_for_completion():
            raise Exception(
                "Failed to run {0} backup job with error: {1}".format(
                    backup_type, job.delay_reason
                )
            )

        return job


[docs]    def run(self):
        """Executes basic acceptance test case"""
        self._log.info(
            "Will run below test case on: {0} subclient".format(str(self._inputs['SubclientName']))
        )

        self._log.info("Creating test tables in the database")
        self._hana_helper.create_test_tables()

        job = self._run_backup("FULL")
        restore_data_prefix = job.job_id + "_COMPLETE_DATA_BACKUP"

        if not job.wait_for_completion():
            raise Exception("Failed to run FULL backup job with error: " + str(job.delay_reason))

        self._log.info("Creating test tables before incremental backup")
        self._hana_helper.create_test_tables()

        job = self._run_backup("INCREMENTAL")
        if not job.wait_for_completion():
            raise Exception("Failed to run FULL backup job with error: " + str(job.delay_reason))

        inc_job_end_time = datetime.datetime.strptime(
            str(job.end_time), '%a %b %d %H:%M:%S %Y'
        ).strftime('%Y-%m-%d %H:%M:%S')

        self._log.info(inc_job_end_time)

        self._log.info("Creating test tables before differential backup")
        self._hana_helper.create_test_tables()

        self._run_backup("DIFFERENTIAL")
        if not job.wait_for_completion():
            raise Exception("Failed to run FULL backup job with error: " + str(job.delay_reason))


        self._log.info("Successfully ran all three types of backup")
        self._log.info("Starting a current time restore")

        self._log.info("*" * 10 + " Run current time restore to same Client " + "*" * 10)

        job = self._backupset.restore(
            self._subclient._subClientEntity['clientName'].lower(), self._instance.instance_name
        )
        self._log.info(
            "Started Current time restore to same client job with Job ID: " + str(job.job_id)
        )

        if not job.wait_for_completion():
            raise Exception(
                "Failed to run current time restore job with error: " + str(job.delay_reason)
            )

        self._log.info("Successfully finished Current time restore to same client")

        self._hana_helper.test_tables_validation()

        self._log.info("Successfully validated the restored content")

        self._log.info("*" * 10 + " Run Point In Time restore to same client" + "*" * 10)

        job = self._backupset.restore(
            self._subclient._subClientEntity['clientName'].lower(), self._instance.instance_name,
            point_in_time=inc_job_end_time, initialize_log_area=True,
            check_access=True
        )
        self._log.info(
            "Started point in time restore to same client job with Job ID: " + str(job.job_id)
        )

        if not job.wait_for_completion():
            raise Exception(
                "Failed to run point in time restore with error: " + str(job.delay_reason)
            )

        self._log.info("Successfully finished point in time restore to same client")

        self._hana_helper.test_tables_validation(point_in_time=True)
        self._log.info("Successfully validated restored content")

        self._log.info("*" * 10 + " Run Recover data only restore " + "*" * 10)
        job = self._backupset.restore(
            self._subclient._subClientEntity['clientName'].lower(), self._instance.instance_name,
            restore_data_prefix, check_access=True
        )
        self._log.info("Started recover data only restore job with Job ID: {0}".format(
            str(job.job_id)))

        if not job.wait_for_completion():
            raise Exception(
                "Failed to run restore in place job with error: {0}".format(job.delay_reason)
            )

        self._log.info("Successfully finished recover data only restore job")

        self._hana_helper.test_tables_validation(recover_data=True)

        self._log.info("Successfully validated restored content")
