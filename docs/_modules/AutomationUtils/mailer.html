"""Helper file for sending Automation Report emails.

Mailer is the only class defined in this file.

Mailer: Class that makes a request to the server to send email report to specified users.

Mailer:
    __init__()                  --  initialize objects of Mailer class

    mail()                      --  Sends the email to specified receivers

    parse_template_config()     --  checks for valid json file and correcponding keys

    parse_receiver()            -- checks for receivers and replaces the delimiters in the
                                   mail addresses by ','.

"""

import os
import json
from smtplib import SMTP
from email.mime.text import MIMEText

from . import logger
from .constants import AUTOMATION_EMAIL
from .constants import CONFIG_FILE_PATH

    """Class that makes a request to the server to send email report to specified users."""

    RECEIVERS = None

    def __init__(self, mailing_inputs=None, commcell_object=None):
        """Initialize required attributes of the mailer object.

        Args:
            mailing_inputs      (dict)    --  required inputs to send an email
                    default: None

                    Example:
                        {
                            "receiver": "user1@test.com,user2@test.com"

                        }

            commcell_object     (object)  --  object reference to this commcell
                    default: None

        **Note:**  If mailing_inputs is not defined we pick the mailing inputs from
                    static attribute (Mailer.RECEIVERS)

        """
        self._log = logger.get_log()
        self._commcell_object = commcell_object
        self._receiver = None
        self.config_data = None

        self.parse_template_config()

        if commcell_object is None:
            error_message = "Commcell object is not initialized."
            self._log.error(error_message)

        self.parse_receiver(mailing_inputs)


[docs]    def mail(self, subject, body):
        """ Sends the mail notification to specified recipients via REST API if commcell
            object is initialized or via SMTPlib when commcell object is not initialized

        Args:
                subject   (str)     --  subject of the mail to be sent

                body      (str)     --  content to be sent in the mail

        """
        try:
            if self._commcell_object:
                receiver = self._receiver.split(',')
                self._log.info("Sending email to: %s", receiver)
                self._commcell_object.send_mail(
                    receiver, subject, body,
                    copy_sender=self.config_data.get('email', {}).get('copySender', True))
                self._log.info('Successfully sent email to: %s', receiver)

            else:
                message = MIMEText(body, 'html')
                message['Subject'] = subject
                message['From'] = AUTOMATION_EMAIL
                message['To'] = self._receiver

                mail = SMTP(self.config_data['email']['server'])
                try:
                    if self.config_data['email']['username']:
                        mail.login(self.config_data['email']['username'],
                                   self.config_data['email']['password'])

                    self._log.info("Sending email to: %s", self._receiver)
                    mail.sendmail(message['From'], message['To'], message.as_string())
                    self._log.info("E-mail sent via SMTP server.")
                finally:
                    mail.quit()

        except Exception as exp:
            exp_msg = "Failed to send mail with error: {0}".format(str(exp))
            self._log.error(exp_msg)
            raise Exception(exp_msg)



[docs]    def parse_template_config(self):
        """ Checks whether the path to the template_config.json is valid  and parses
            the template_config.json to get the values of server, username and password.

        Raises:
            OSError: When the json file does not exist.
                     Permission is not granted to execute os.stat() on the requested file,
                     even if the path physically exists.

            IOError: Unable to read the file.

        """
        if not os.path.exists(CONFIG_FILE_PATH):
            error_message = "Failed to read configuration file. Either the user doesn’t have"
            " permission or configuration file is not present. Skip sending email via SMTP."
            self._log.error(error_message)
            raise OSError("Failed to send email.")

        try:
            with open(CONFIG_FILE_PATH, 'r') as config_file:
                self.config_data = json.load(config_file)

        except IOError:
            error_message = "Unable to read template_config.json."
            self._log.error(error_message)
            raise IOError(error_message)

        if not self.config_data['email']['server']:
            error_message = "email server field is empty."
            self._log.error(error_message)
            raise KeyError(error_message)



[docs]    def parse_receiver(self, mailing_inputs):
        """ checks for list of receiver email and also replaces the delimiters ':' and ';'
            with ','.

        Args:
                mailing_inputs   (str)     --  email_addresses to which email has to be sent.

        Raises:
                Exception: When the email receiver list is empty.

        """
        if mailing_inputs and 'receiver' in mailing_inputs:
            self._receiver = str(mailing_inputs['receiver'])
        elif Mailer.RECEIVERS is not None:
            self._receiver = Mailer.RECEIVERS
        else:
            error_message = "Email receiver's list is empty. Please specify the receiver's."
            self._log.error(error_message)
            raise Exception(error_message)

        if ':' in self._receiver:
            self._receiver = self._receiver.replace(':', ',')
        elif ';' in self._receiver:
            self._receiver = self._receiver.replace(';', ',')
