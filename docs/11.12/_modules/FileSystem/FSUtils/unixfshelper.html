[docs]    def verify_dc(self, jobid, log_dir=None):
        """Verifies whether the job used Scan optimization

             Args:
                jobid    (int)  -- jobid to be checked
                                 
                log_dir  (str)  -- path of the log directory
                    default:None

            Returns:
                bool   -   Returns True if dc scan was successful or else false
                
            Raises:
                Exception:
                    if any error occurred while verifying the scan status
        """  
        if log_dir is None:
            log_dir = self.log_files_dir
        script_arguments = "-logdir " + log_dir + " -jobid " + str(jobid) 
        output = self.testcase.client_machine.execute(
            UNIX_VERIFY_DC, script_arguments
            )
        
        if(output.exit_code != 0):
            self.log.error(
                "Error occurred while verifying the scan status " 
                + output.output 
                + output.exception
                )
            raise Exception(
                "Error occurred while verifying the scan status " 
                + output.output 
                + output.exception
                )
        else:
            result = output.output.strip('\n')
            if(result == "success"):
                return True
            else:
                return False
