[docs]    def check_page_load_time(self, time_out=100):
        """Collect page load time for the report"""
        try:
            start_time = time.time()
            self.webconsole.wait_till_load_complete(timeout=time_out)
            self.load_time = str((time.time() - start_time))
        except CVTimeOutException:
            pass

[docs]    def set_report_status(self):
        if self.notifications:
            self.failure_reason = "Notification Error"
            return
        if self.console_errors:
            self.failure_reason = "Console Error"
            return
        if self.is_page_no_data or self.is_page_blank is True:
            self.status = self.ReportStatus.WARNING
            self.failure_reason = "No data available"
            return
        if self.load_time is "NA":
            self.failure_reason = "Page is not loaded within timout[%s sec] period" % \
                                  str(self.load_time)
            self.status = self.ReportStatus.CRITICAL



[docs]    def verify_report_page(self):

        self.check_console_error()
        self.check_notification_error()
        self.set_report_status()

    @test_step
    def collect_reports_access_details(self, report_details):
        """Collect reports access details """
        reports = self.reports_home_page.get_all_report_details()
        _temp_report_names = []
        self.webconsole.clear_all_notifications()
        for each_report in reports:
            if reportsutils.is_custom_report(each_report['href']):  # skip the custom report
                continue
            #  skip report if its already present in different tag.
            if each_report['name'] in _temp_report_names:
                continue
            # if report is present different tags maintain temp for skipping the report.
            _temp_report_names.append(each_report['name'])
            report_object = ReportDetails(self.browser, self.webconsole, self.utils,
                                          each_report['name'],
                                          each_report['href'])
            self.browser.driver.get(report_object.url)
            self.log.info("Verifying page for the report [%s] with url [%s]",
                          each_report['name'],
                          each_report['href'])
            report_object.verify_report_page()
            report_details.append(report_object)

        """ Get console error as html table"""

[docs]    def get_reports_html(self, reports_objects):
        """ Get reports as html table"""
        _list = []
        html = ""
        for each_report in reports_objects:
            if not each_report.console_errors and not each_report.notifications:
                if each_report.status == ReportDetails.ReportStatus.SUCCESS:
                    continue
                _list.append(("<a href=%s >%s</a>" % (each_report.url, each_report.name)))
                _list.append(each_report.status)
                _list.append(each_report.failure_reason)
                _list.append(each_report.load_time)
        if _list:
            _list.insert(0, "LoadTime")
            _list.insert(0, "Comments")
            _list.insert(0, "Status")
            _list.insert(0, "Report")
            html += ''.join(self.make_html_table((self.prep_page_status_list(_list, 4))))
        return html

        prepared_list = [list[i:i + sub_length] for i in range(0, len(list), sub_length)]

[docs]    def make_html_table(self, prepared_list):
        """ Prepares the html table for the mail """
        yield '    <table border="1">'
        headings_list = ['Report', 'Console Error', 'Notification Error', 'Status', 'Comments',
                         'LoadTime']
        for sublist in prepared_list:
            self.log.info("Making html for:%s", sublist)
            if sublist[1] == "Critical":
                yield '  <tr><td bgcolor="Red">'
                yield '    </td><td bgcolor="Red">'.join(sublist)
            elif sublist[1] == "Warning":
                yield '  <tr><td bgcolor="Orange">'
                yield '    </td><td bgcolor="Orange">'.join(sublist)
            else:
                if sublist[0] in headings_list:
                    yield '  <tr><td><b>'
                    yield '    </td><td><b>'.join(sublist)
                else:
                    yield '  <tr><td>'
                    yield '    </td><td>'.join(sublist)
            yield '  </b></td></tr>'
        yield '    </table><br><br>'



[docs]    def run(self):
        try:
            self.init_tc()
            self.collect_reports_access_details(self.worldwide_reports)
            self.navigator.goto_commcell_dashboard(self.commcell_name)
            self.navigator.goto_commcell_reports()
            self.collect_reports_access_details(self.commcell_reports)
            html = self.generate_html_page()
            self.send_mail(html_content=html)
            self.log.info("Survey completed")
            self.set_testcase_status()
        except Exception as err:
            self.utils.handle_testcase_exception(err)
        finally:
            WebConsole.logout_silently(self.webconsole)
            Browser.close_silently(self.browser)
