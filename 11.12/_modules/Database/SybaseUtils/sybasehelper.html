[docs]    def execute_sybase_command(self, cmd, temp_input_file_name):
        """Method to create temp file with sybase command and execute the same on remote machine
        Args:
            cmd    (str)    --  sybase command to be executed on remote client

            temp_input_file_name    (str)   --  temporary input file to be created on controller and copied to remote machine

        Returns:
            (str)    -    processed output

            (str)    -    formatted output

        Raises:
            Exception:
                if unable to open input file
        """

        exit_cmd = "go\nexit\n"
        sybase_server_name = self.instance.instance_name
        local_filename = self.common_dir_path + temp_input_file_name

        # Create local temp file and write the the command on the controller to local file
        self.log.info(
            "Create local temp file and write the the command on the controller to local file")
        platform = self.machine_object.os_info
        if platform == 'UNIX':
            fp = open(local_filename, 'w', newline='\n')
        else:
            fp = open(local_filename, 'w')
        if cmd is not None:
            fp.write(cmd)
            fp.write(exit_cmd)
        else:
            cmd = "exit\n"
            fp.write(cmd)
        fp.close()

        # Copy the local file from controller to remote machine
        self.log.info("Copy the local file from controller to remote machine")
        copy_status = self.machine_object.copy_from_local(
            local_filename, self.remote_path)

        # Execute the sybase command on remote machine using copied temp file
        remote_file = self.remote_path + temp_input_file_name
        self.log.info(
            "Execute the sybase command on remote machine using copied temp file")
        cmd = self.sybasecmd + "-U " + self.sybase_sa_username + " -P " + \
            self.sybase_sa_userpassword + " -S " + \
            sybase_server_name + " -i " + "\"" + remote_file + "\""
        self.log.info(cmd)
        cmd_output = self.machine_object.execute_command(cmd)

        # Fetching output of remote command execution
        output = cmd_output.output

        # Fetching formatted output for list items
        formatted_output = cmd_output.formatted_output

        # Checking if input file is valid for sybase command execution
        if output.find("Unable to open input file") >= 0:
            raise Exception(
                "Failed to open the input file given to sybase ISQL command. Exiting ")

        # Cleanup temp file on controller and on remote machine
        self.log.info("Cleanup temp file on controller and on remote machine")
        self.controller_object.delete_file(local_filename)
        self.machine_object.delete_file(
            self.remote_path + temp_input_file_name)

        return output, formatted_output

[docs]    def get_all_database_tables(self):
        """Method to return dictionary containing all tables lists of all sybase databases
        Returns:
            dict(list)  -   dictionary of list containing all table list of each database in sybase server

        Raises:
            Exception:
                if any of user database state not online
                if table list  fetch fails
        """
        AllTableList = {}
        status, db_list = self.sybase_helper.get_database_list()
        for eachDB in db_list:
            self.log.info("Check database state before fetching table lists")
            status = self.sybase_helper.check_database_state(eachDB)
            if status is False:
                time.sleep(20)
                status = self.sybase_helper.check_database_state(eachDB)
                if status is False:
                    raise Exception(
                        "Database state of {0} is not online to fetch table lists".format(eachDB))
            (retCode, table_list) = self.sybase_helper.get_table_list(eachDB)
            if retCode is False:
                raise Exception(
                    "Error in getting table list for database " + str(eachDB))
            AllTableList[eachDB] = table_list
        return AllTableList



[docs]    def comparing_dict_of_list(self, d1, d2):
        """Method to compare two dictionary of lists items
        Args:
            d1      (dict)  --  first dictionary of lists for comparison

            d2      (dict)  --  second dictionary of lists for comparison

        Returns:
                bool    -   boolean value whether comparison succeeded or not

        """
        status = False
        dl1 = len(d1)
        dl2 = len(d2)
        count = 0
        if dl1 == dl2:
            self.log.info("dict length equal")
            for eachkey in d2:
                l1 = d1[eachkey]
                l2 = d2[eachkey]
                list_status = (l1 == l2)
                if list_status is False:
                    self.log.info(
                        "List of key {0} not matching ".format(eachkey))
                    break
                else:
                    count = count + 1
                    continue
            if count == dl1:
                self.log.info("All dict list are equal")
                status = True
        else:
            self.log.info("Dictionary length doesn't match")
        return status
