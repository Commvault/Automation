[docs]class ActivityControl(object):
    """
    class that act as wrapper for SDK and Testcase

    Methods:
            run_backup()             - runs backup job and waits for completion

            run_backup_without_wait() - runs backup job without waiting for completion

            run_backup_with_backup_activity_disabled()  - runs backup when backup
                                                          activity is disabled

            set_all_activity()           - Sets all the activity control options
                                        (default: enable)
    """

    def __init__(self, testcase):
        """
        Initialize the  SDK objects

        Args:
            testcase  (obj)    -- Testcase object
        """
        self.log = testcase.log
        self._subclient = testcase.subclient
        self._activity_control = activitycontrol.ActivityControl(
            testcase.commcell)


[docs]    def run_backup(self, backup_type, wait=True):
        """
        Initiates backup job and waits for completion

        Args:
                backup_type        (str)   --  level of backup the user wish to run
                        Full / Incremental / Differential / Synthetic_full
                    default: Incremental

        Return:
                job   (obj)       - instance of the Job class for this backup job
                
        Exception:
                When failed to run backup job.
        """

        self.log.info(
            "*" *
            10 +
            " Starting Subclient {0} Backup ".format(backup_type) +
            "*" *
            10)
        job = self._subclient.backup(backup_type)
        self.log.info(
            "Started {0} backup with Job ID: {1}".format(
                backup_type, str(job.job_id)))
        
        if not wait:        
            if not job.wait_for_completion():
                raise Exception(
                    "Failed to run {0} backup job with error: {1}".format(
                        backup_type, job.delay_reason
                    )
                )

        self.log.info("Successfully finished {0} backup job".format(backup_type))

        return job



[docs]    def run_backup_with_backup_activity_disabled(self, backup_type):
        """
        runs backup when backup activity is disabled

        Args:
                backup_type        (str)   --  level of backup the user wish to run
                        Full / Incremental / Differential / Synthetic_full
                    default: Incremental

        Return:
                job   (obj)       - instance of the Job class for this backup job
                True   (Bool)       - True if job triggered but failed only if
                                             Data Management activity disabled

        Exception:
                if job failed for any other reason
        """

        try:
            self.run_backup(backup_type, False)
            
        except Exception as excp:
            if re.search("Data Management activity on CommServe is disabled.",
                         str(excp), re.IGNORECASE):
                self.log.info(str(excp))
                return True
            else:
                raise Exception(str(excp))            



[docs]    def set_all_activity(self, action="Enable"):
        """
        Sets all the activity control options

        Args:
                action        (str)   --  Enable or Disable all the activities
                        Enable/Disable
                    default: Enable
        """

        for activity_type in self._activity_control._activity_type_dict.keys():
            self.log.info(
                'Setting Activity Type: "{0}" to "{1}"'.format(
                    activity_type, action))
            self._activity_control.set(activity_type, action)

        self.log.info("Enabling all activity back.")
